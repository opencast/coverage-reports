82    public void testAddAndGetGroup() throws Exception {
83      Set<JpaRole> authorities = new HashSet<JpaRole>();

194   @Test
195   public void testDuplicateGroup() throws UnauthorizedException {
196     Set<JpaRole> roles1 = set(new JpaRole("ROLE_ASTRO_101_SPRING_2011_STUDENT", org1));
197     Set<JpaRole> roles2 = set(new JpaRole("ROLE_ASTRO_101_SPRING_2011_STUDENT", org2));
198     Set<String> members = set("admin");
199     provider.addGroup(new JpaGroup("test1", org1, "Test", "Test group", roles1, members));
200     provider.addGroup(new JpaGroup("test1", org2, "Test 2", "Test group 2", roles2, members));
201     assertEquals("Test", provider.loadGroup("test1", org1.getId()).getName());
202     // duplicate group, but add group does an update so it will pass
203     provider.addGroup(new JpaGroup("test1", org1, "Test 1", "Test group 1", roles1, members));
204     assertEquals("Test 1", provider.loadGroup("test1", org1.getId()).getName());
205   }

240     // findRoles() should return a role per group, not the included roles for each group
241     Set<JpaRole> authorities = new HashSet<JpaRole>();
242     authorities.add(new JpaRole("ROLE_ASTRO_101_SPRING_2011_STUDENT", org1));

247     JpaGroup group = new JpaGroup("test", org1, "Test", "Test group", authorities, members);

249 
250     Role role = provider.findRoles("%test%", Role.Target.ALL, 0, 0).next();
251     Assert.assertEquals("ROLE_GROUP_TEST", role.getName());

258     provider.addGroup(group2);
259 
260     authorities = new HashSet<JpaRole>();
261     authorities.add(new JpaRole("ROLE_ASTRO_134_SPRING_2011_STUDENT", org2));
262     authorities.add(new JpaRole("ROLE_ASTRO_144_SPRING_2012_STUDENT", org2));
263 
264     JpaGroup group3 = new JpaGroup("test2", org2, "Test2", "Test 2 group", authorities, members);

273   @Test
274   public void testRemoveUserFromAllGroups() throws UnauthorizedException {
275 
276     Set<String> members = new HashSet<String>();
277     members.add("user");
278 
279     JpaGroup group1 = new JpaGroup("test1", org1, "Test1", "Test 1 group",
280             new HashSet<JpaRole>(), members);
281     JpaGroup group2 = new JpaGroup("test2", org1, "Test2", "Test 2 group",
282             new HashSet<JpaRole>(), members);
283     JpaGroup group3 = new JpaGroup("test3", org1, "Test3", "Test 3 group",
284             new HashSet<JpaRole>(), members);
285 
286     provider.addGroup(group1);
287     provider.addGroup(group2);
288     provider.addGroup(group3);
289 
290     List<Role> groupRoles = provider.getRolesForUser("user");
291     Assert.assertEquals("There should be three groupRoles added to the user",3, groupRoles.size());
292     Assert.assertTrue("GroupRole for group1 should be added", groupRoles.contains(new JpaRole(group1.getRole(), org1)));
293     Assert.assertTrue("GroupRole for group2 should be added", groupRoles.contains(new JpaRole(group2.getRole(), org1)));
294     Assert.assertTrue("GroupRole for group3 should be added", groupRoles.contains(new JpaRole(group3.getRole(), org1)));
295 
296     provider.removeMemberFromAllGroups("user", "org1");
297 
298     groupRoles = provider.getRolesForUser("user");
299     Assert.assertEquals("There should be no more groupRoles for the user",0, groupRoles.size());
300     Assert.assertFalse("GroupRole for group1 should be removed",
301             groupRoles.contains(new JpaRole(group1.getRole(), org1)));
302     Assert.assertFalse("GroupRole for group2 should be removed",
303             groupRoles.contains(new JpaRole(group2.getRole(), org1)));
304     Assert.assertFalse("GroupRole for group3 should be removed",
305             groupRoles.contains(new JpaRole(group3.getRole(), org1)));
306 
307     provider.removeMemberFromAllGroups("user", "org1");
308 
309     groupRoles = provider.getRolesForUser("user");
310     Assert.assertEquals("Make sure there is no issue with users that are not part of any group",
311             0, groupRoles.size());
312   }
