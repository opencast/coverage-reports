81    public void testAddAndGetUser() throws Exception {
82      JpaUser user = createUserWithRoles(org1, "user1", "ROLE_ASTRO_101_SPRING_2011_STUDENT");
83      provider.addUser(user);

85      User loadUser = provider.loadUser("user1");

91      assertEquals(user.getRoles(), loadUser.getRoles());

93      assertNull("Loading 'does not exist' should return null", provider.loadUser("does not exist"));
94      assertNull("Loading 'does not exist' should return null", provider.loadUser("user1", org2.getId()));
95  
96      loadUser = provider.loadUser("user1", org1.getId());
97      assertNotNull(loadUser);
98  
99      assertEquals(user.getUsername(), loadUser.getUsername());
100     assertTrue(passwordEncoder.isPasswordValid(loadUser.getPassword(), user.getPassword(), null));

102     assertEquals(user.getRoles(), loadUser.getRoles());

164   @Test
165   public void testDeleteUser() throws Exception {
166     JpaUser user1 = createUserWithRoles(org1, "user1", "ROLE_ASTRO_101_SPRING_2011_STUDENT");
167     JpaUser user2 = createUserWithRoles(org1, "user2", "ROLE_ASTRO_101_SPRING_2011_STUDENT");
168     JpaUser user3 = createUserWithRoles(org1, "user3", "ROLE_ASTRO_101_SPRING_2011_STUDENT");
169     JpaUser user4 = createUserWithRoles(org1, "user4", "ROLE_ASTRO_101_SPRING_2011_STUDENT");
170     provider.addUser(user1);
171     provider.addUser(user2);
172     provider.addUser(user3);
173     provider.addUser(user4);
174 
175     User loadUser = provider.loadUser("user1");
176 
177     assertNotNull(loadUser);
178 
179     provider.deleteUser("user1", user1.getOrganization().getId());
180     provider.deleteUser("user2", user1.getOrganization().getId());
181     provider.deleteUser("user3", user1.getOrganization().getId());
182 
183     assertNull(provider.loadUser("user1", org1.getId()));
184     assertNull(provider.loadUser("user2", org1.getId()));
185     assertNull(provider.loadUser("user3", org1.getId()));
186     assertNotNull(provider.loadUser("user4", org1.getId()));
187 
188     try {
189       provider.deleteUser("user1", user1.getOrganization().getId());
190       fail("Should throw a NotFoundException");
191     } catch (NotFoundException e) {
192       assertTrue("User not found.", true);
193     }
194   }

248   public void testUpdateUserForbiddenForNonAdminUsers() throws Exception {
249     JpaUser adminUser = createUserWithRoles(org1, "admin", SecurityConstants.GLOBAL_ADMIN_ROLE);
250     JpaUser user = createUserWithRoles(org1, "user", "ROLE_USER");
251 
252     provider.addUser(adminUser);
253     provider.addUser(user);
254     provider.setSecurityService(mockSecurityServiceWithUser(user));
255 
256     // try to add ROLE_USER
257     Set<JpaRole> updatedRoles = Collections.set(
258             new JpaRole("ROLE_USER", org1),
259             new JpaRole(SecurityConstants.GLOBAL_ADMIN_ROLE, org1));
260 
261     try {
262       provider.updateUser(new JpaUser(adminUser.getUsername(), adminUser.getPassword(), org1,
263               adminUser.getName(), true, updatedRoles));
264       fail("The current user may not edit an admin user");
265     } catch (UnauthorizedException e) {
266       // pass
267     }
268 
269     // try to remove ROLE_ADMIN
270     updatedRoles = Collections.set(new JpaRole("ROLE_USER", org1));
271     try {
272       provider.updateUser(new JpaUser(adminUser.getUsername(), adminUser.getPassword(), org1,
273               adminUser.getName(), true, updatedRoles));
274       fail("The current user may not remove the admin role on other user");
275     } catch (UnauthorizedException e) {
276       // pass
277     }

334   @Test
335   public void testRolesForUser() {
336     JpaRole astroRole = new JpaRole("ROLE_ASTRO_105_SPRING_2013_STUDENT", org1, "Astro role");
337 
338     provider.addRole(astroRole);
339 
340     JpaUser userOne = createUserWithRoles(org1, "user1", "ROLE_ONE", "ROLE_TWO");
341     try {
342       provider.addUser(userOne);
343     } catch (UnauthorizedException e) {
344       fail("User should be created");
345     }
346 
347     // Provider not authoritative for these roles
348     assertEquals(
349         "There should be zero roles",
350         0,
351         IteratorUtils.toList(provider.findRoles("%", Role.Target.ALL, 0, 0)).size()
352     );
353 
354     List<String> rolesForUser = provider.getRolesForUser("user1").stream()
355             .map(Role::getName)
356             .sorted()
357             .collect(Collectors.toList());
358     assertEquals("There should be two roles", 2, rolesForUser.size());
359     assertEquals("ROLE_ONE", rolesForUser.get(0));
360     assertEquals("ROLE_TWO", rolesForUser.get(1));
361   }

398   private static SecurityService mockSecurityServiceWithUser(User currentUser) {
399     // Set the security sevice
400     SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
401     EasyMock.expect(securityService.getOrganization()).andReturn(currentUser.getOrganization()).anyTimes();
402     EasyMock.expect(securityService.getUser()).andReturn(currentUser).anyTimes();
403     EasyMock.replay(securityService);
404     return securityService;
405   }

407   private static JpaUser createUserWithRoles(JpaOrganization org, String username, String... roles) {
408     Set<JpaRole> userRoles = new HashSet<>();
409     for (String adminRole : roles) {
410       userRoles.add(new JpaRole(adminRole, org));
411     }
412     return new JpaUser(username, "pass1", org, "opencast", true, userRoles);
413   }
