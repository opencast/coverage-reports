50  /**
51   * Tests for ExecuteOnceWorkflowOperationHandler
52   */
53  public class ExecuteOnceWorkflowOperationHandlerTest {
54  
55    private static ExecuteService executeService;
56    private static Workspace workspaceService;
57    private static ExecuteOnceWorkflowOperationHandler execOnceWOH;
58    private static WorkflowInstance workflowInstance;
59    private static Catalog catalog;
60  
61    /**
62     * @throws java.lang.Exception
63     */
64    @Before
65    public void setUp() throws Exception {
66      // Mocking just about everything, just testing the mediapackage parse
67      String expectedTypeString = "catalog";
68      String catalogId = "catalog-id";
69  
70      URI catUri = new URI("http://api.com/catalog");
71      catalog = CatalogImpl.newInstance();
72      catalog.addTag("engage-download");
73      catalog.setIdentifier(catalogId);
74      catalog.setURI(catUri);
75  
76      WorkflowOperationInstance operation = EasyMock.createMock(WorkflowOperationInstance.class);
77  
78      EasyMock.expect(operation.getId()).andReturn(123L).anyTimes();
79      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.EXEC_PROPERTY)).andReturn(null)
80              .anyTimes();
81      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.PARAMS_PROPERTY)).andReturn(null)
82              .anyTimes();
83      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.LOAD_PROPERTY)).andReturn("123")
84              .anyTimes();
85      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.TARGET_FLAVORS_PROPERTY))
86          .andReturn(null).anyTimes();
87      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.TARGET_FLAVOR_PROPERTY))
88              .andReturn(null).anyTimes();
89      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.TARGET_TAGS_PROPERTY))
90              .andReturn(null).anyTimes();
91      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.TARGET_TAG_PROPERTY))
92          .andReturn(null).anyTimes();
93      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.OUTPUT_FILENAME_PROPERTY))
94              .andReturn(null).anyTimes();
95      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.SET_WF_PROPS_PROPERTY))
96              .andReturn("false").anyTimes();
97      // these two need to supply a real string
98      EasyMock.expect(operation.getConfiguration(ExecuteOnceWorkflowOperationHandler.EXPECTED_TYPE_PROPERTY))
99              .andReturn(expectedTypeString).anyTimes();
100     EasyMock.replay(operation);
101 
102     Id mpId = EasyMock.createMock(Id.class);
103 
104     MediaPackage mediaPackage = EasyMock.createMock(MediaPackage.class);
105     mediaPackage.add((MediaPackageElement) EasyMock.anyObject());
106     EasyMock.expect(mediaPackage.getIdentifier()).andReturn(mpId).anyTimes();
107     EasyMock.replay(mediaPackage);
108 
109     workspaceService = EasyMock.createMock(Workspace.class);
110     EasyMock.expect(workspaceService.moveTo((URI) EasyMock.anyObject(), (String) EasyMock.anyObject(),
111             (String) EasyMock.anyObject(), (String) EasyMock.anyObject())).andReturn(catUri).anyTimes();
112     EasyMock.replay(workspaceService);
113 
114     workflowInstance = EasyMock.createMock(WorkflowInstance.class);
115     EasyMock.expect(workflowInstance.getMediaPackage()).andReturn(mediaPackage).anyTimes();
116     EasyMock.expect(workflowInstance.getCurrentOperation()).andStubReturn(operation);
117     EasyMock.replay(workflowInstance);
118 
119     // Override the waitForStatus method to not block the jobs
120     execOnceWOH = new ExecuteOnceWorkflowOperationHandler() {
121       @Override
122       protected Result waitForStatus(long timeout, Job... jobs) {
123         var map = new HashMap<Job, Status>();
124         for (var job: jobs) {
125           map.put(job, Status.FINISHED);
126         }
127         return new Result(map);
128       }
129     };
130     execOnceWOH.setWorkspace(workspaceService);
131   }
132 
133   private void setEmptyPayload() throws ExecuteException, MediaPackageException {
134     Job catalogJob = EasyMock.createNiceMock(Job.class);
135     EasyMock.expect(catalogJob.getQueueTime()).andReturn(123L).anyTimes();
136     EasyMock.expect(catalogJob.getPayload()).andReturn("").anyTimes();
137     EasyMock.replay(catalogJob);
138 
139     executeService = EasyMock.createMock(ExecuteService.class);
140     EasyMock.expect(
141             executeService.execute(EasyMock.anyString(), EasyMock.anyString(), (MediaPackage) EasyMock.anyObject(),
142                     EasyMock.anyString(), (MediaPackageElement.Type) EasyMock.anyObject(), EasyMock.anyLong()))
143             .andReturn(catalogJob).anyTimes();
144     EasyMock.replay(executeService);
145 
146     execOnceWOH.setExecuteService(executeService);
147 
148   }
149 
150   private void setSomePayload() throws ExecuteException, MediaPackageException {
151     Job catalogJob = EasyMock.createNiceMock(Job.class);
152     EasyMock.expect(catalogJob.getQueueTime()).andReturn(123L).anyTimes();
153     EasyMock.expect(catalogJob.getPayload()).andReturn(MediaPackageElementParser.getAsXml(catalog)).anyTimes();
154     EasyMock.replay(catalogJob);
155 
156     executeService = EasyMock.createMock(ExecuteService.class);
157     EasyMock.expect(
158             executeService.execute(EasyMock.anyString(), EasyMock.anyString(), (MediaPackage) EasyMock.anyObject(),
159                     EasyMock.anyString(), (MediaPackageElement.Type) EasyMock.anyObject(), EasyMock.anyLong()))
160             .andReturn(catalogJob).anyTimes();
161     EasyMock.replay(executeService);
162 
163     execOnceWOH.setExecuteService(executeService);
164 
165   }
166 
167   @Test
168   public void startParseSomePayloadTest() throws ExecuteException, MediaPackageException {
169     try {
170       setSomePayload();
171       execOnceWOH.start(workflowInstance, null);
172     } catch (WorkflowOperationException ex) {
173       fail("Should not throw exception" + ex.getMessage());
174     }
175   }
176 
177   @Test
178   public void startParseEmptyPayloadTest() throws ExecuteException, MediaPackageException {
179     try {
180       setEmptyPayload();
181       execOnceWOH.start(workflowInstance, null);
182     } catch (WorkflowOperationException ex) {
183       fail("Should not throw exception" + ex.getMessage());
184     }
185   }
