106   @Test
107   public void testDerivates() {
108     try {
109 
110       // Create a media package and add an element
111       MediaPackage mediaPackage = mediaPackageBuilder.createNew();
112       MediaPackageElementFlavor sourceFlavor = new MediaPackageElementFlavor("some", "source");
113       MediaPackageElement dcCatalog = mediaPackage.add(dcFile.toURI(), Type.Catalog, sourceFlavor);
114 
115       // Add the "derived" catalog
116       MediaPackageElementFlavor derivedFlavor = new MediaPackageElementFlavor("dublincore", "specialedition");
117       MediaPackageElement derivedElement = mediaPackageElementBuilder.elementFromURI(dcFile.toURI(), Type.Catalog,
118               derivedFlavor);
119       mediaPackage.addDerived(derivedElement, dcCatalog);
120 
121       // Test the whole thing
122       assertEquals(1, mediaPackage.getElementsByFlavor(derivedFlavor).length);
123       assertEquals(2, mediaPackage.getCatalogs().length);
124       assertEquals(1, mediaPackage.getCatalogs(derivedFlavor).length);
125       assertEquals(1, mediaPackage.getDerived(dcCatalog, derivedFlavor).length);
126       assertEquals(derivedElement, mediaPackage.getDerived(dcCatalog, derivedFlavor)[0]);
127 
128     } catch (MediaPackageException e) {
129       fail("Media package excpetion while reading media package from manifest: " + e.getMessage());
130     } catch (ConfigurationException e) {
131       fail("Configuration exception while reading media package from manifest: " + e.getMessage());
132     } catch (UnsupportedElementException e) {
133       fail("Error while creating media package: " + e.getMessage());
134     }
135   }

137   @Test
138   public void testPublicationElement() throws Exception {
139     final MediaPackage mp = mediaPackageBuilder.createNew();
140     mp.add(publication("1", "engage", new URI("http://localhost/1.html"), mimeType("text", "html")));
141     assertEquals("Number of media package elements", 1, mp.getElements().length);
142     final String xml = MediaPackageParser.getAsXml(mp);
143     logger.info(xml);
144     assertEquals("Media package identifier", mp.getIdentifier().toString(), from(xml).get("mediapackage.@id"));
145     assertEquals("Publication channel name", "engage", from(xml).get("mediapackage.publications.publication.@channel"));
146   }

150     final MediaPackage mp = loadFromClassPath("/manifest.xml");
151     assertEquals("Number of publication elements", 1, mp.getPublications().length);
152     assertEquals("Publication channel name in deserialized mediapackage", "engage",
153             mp.getPublications()[0].getChannel());
154     final String xml = MediaPackageParser.getAsXml(mp);
155     assertEquals("Publication channel name in serialized mediapackage", "engage",
156             from(xml).get("mediapackage.publications.publication.@channel"));
157   }
