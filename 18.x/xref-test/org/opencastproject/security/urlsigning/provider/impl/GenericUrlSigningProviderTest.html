115   @Test
116   public void testPropertiesUpdated() throws ConfigurationException {
117     // Handles empty properties
118     assertFalse(signer.accepts(RESOURCE_PATH));
119     assertEquals(0, signer.getUris().size());
120 
121     // Incomplete entries (no secret configured)
122     properties = new Hashtable<>();
123     properties.put(
124         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.URL),
125         MATCHING_URI);
126     boolean exceptionThrown = false;
127     try {
128       signer.updated(properties);
129     } catch (ConfigurationException e) {
130       exceptionThrown = true;
131     }
132     assertTrue(exceptionThrown);
133     assertFalse(signer.accepts(RESOURCE_PATH));
134     assertEquals(0, signer.getUris().size());
135 
136     // Non-Matching secret
137     properties = new Hashtable<>();
138     properties.put(
139         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.URL),
140         NON_MATCHING_URI);
141     properties.put(
142         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.SECRET),
143         NON_MATCHING_SECRET);
144     signer.updated(properties);
145     assertFalse(signer.accepts(RESOURCE_PATH));
146     assertEquals(1, signer.getUris().size());
147 
148     // Matching secret
149     properties = new Hashtable<>();
150     properties.put(
151         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.URL),
152         MATCHING_URI);
153     properties.put(
154         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.SECRET),
155         SECRET);
156     signer.updated(properties);
157     assertTrue(signer.accepts(RESOURCE_PATH));
158     assertEquals(1, signer.getUris().size());
159 
160     // Matching secret and Unrelated Key
161     properties = new Hashtable<>();
162     properties.put(
163         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.URL),
164         MATCHING_URI);
165     properties.put(
166         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.SECRET),
167         SECRET);
168 
169     properties.put(
170         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID_2, GenericUrlSigningProvider.URL),
171         NON_MATCHING_URI);
172     properties.put(
173         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID_2, GenericUrlSigningProvider.SECRET),
174         NON_MATCHING_SECRET);
175 
176     signer.updated(properties);
177     assertTrue(signer.accepts(RESOURCE_PATH));
178     assertEquals(2, signer.getUris().size());
179 
180     // Organization set to "any" organization works
181     properties = new Hashtable<>();
182     properties.put(
183         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.URL),
184         MATCHING_URI);
185     properties.put(
186         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.SECRET),
187         SECRET);
188     properties.put(
189         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID,
190             GenericUrlSigningProvider.ORGANIZATION),
191         GenericUrlSigningProvider.ANY_ORGANIZATION);
192 
193     signer.updated(properties);
194     assertTrue(signer.accepts(RESOURCE_PATH));
195     assertEquals(1, signer.getUris().size());
196   }

198   @Test
199   public void testSign() throws UrlSigningException, ConfigurationException {
200     properties.put(
201         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.URL),
202         MATCHING_URI);
203     properties.put(
204         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.SECRET),
205         SECRET);
206 
207     properties.put(
208         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID_2, GenericUrlSigningProvider.URL),
209         RTMP_MATCHER);
210     properties.put(
211         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID_2, GenericUrlSigningProvider.SECRET),
212         SECRET);
213 
214     signer.updated(properties);
215 
216     DateTime before = new DateTime(2020, 03, 01, 00, 46, 17, 0, DateTimeZone.UTC);
217     // Handles a policy without query parameters.
218     Policy withoutQuery = Policy.mkSimplePolicy(RESOURCE_PATH, before);
219     String result = signer.sign(withoutQuery);
220     logger.info(result);
221     assertEquals(
222         "http://www.opencast.org/path/to/resource.mp4?policy=eyJTdGF0ZW1lbnQiOnsiQ29uZGl0aW9uIjp7Ik"
223             + "RhdGVMZXNzVGhhbiI6MTU4MzAyMzU3NzAwMH0sIlJlc291cmNlIjoiaHR0cDpcL1wvd3d3Lm9wZW5jYXN0Lm"
224             + "9yZ1wvcGF0aFwvdG9cL3Jlc291cmNlLm1wNCJ9fQ&keyId=theId&signature=5b45e678275e6bc7b06a5"
225             + "79f7f42e9a7ea5c58f1da130701db532f121e363e98",
226         result);
227     // Handles a policy with additional query parameters.
228     Policy withQuery = Policy.mkSimplePolicy(RESOURCE_PATH + "?queryparam=this", before);
229     result = signer.sign(withQuery);
230     logger.info(result);
231     assertEquals(
232         "http://www.opencast.org/path/to/resource.mp4?queryparam=this&policy=eyJTdGF0ZW1lbnQiOnsiQ2"
233             + "9uZGl0aW9uIjp7IkRhdGVMZXNzVGhhbiI6MTU4MzAyMzU3NzAwMH0sIlJlc291cmNlIjoiaHR0cDpcL1wvd3"
234             + "d3Lm9wZW5jYXN0Lm9yZ1wvcGF0aFwvdG9cL3Jlc291cmNlLm1wND9xdWVyeXBhcmFtPXRoaXMifX0&keyId="
235             + "theId&signature=0f9cab4393a5d0ebe2683124a92094c3ccf06ed07e87b8f60fa2bab3963bd462",
236         result);
237     // Handles rtmp protocol
238     Policy withRtmp = Policy.mkSimplePolicy("rtmp://www.opencast.org/path/to/resource.mp4", before);
239     result = signer.sign(withRtmp);
240     logger.info(result);
241     assertEquals(
242         "rtmp://www.opencast.org/path/to/resource.mp4?policy=eyJTdGF0ZW1lbnQiOnsiQ29uZGl0aW9uIjp7Ik"
243             + "RhdGVMZXNzVGhhbiI6MTU4MzAyMzU3NzAwMH0sIlJlc291cmNlIjoicnRtcDpcL1wvd3d3Lm9wZW5jYXN0Lm"
244             + "9yZ1wvcGF0aFwvdG9cL3Jlc291cmNlLm1wNCJ9fQ&keyId=theSecondId&signature=0464d9672fa5cb6"
245             + "2ed82e7a6c46db5552dcd76590cf11efa5ba5222f53f5bbaa",
246         result);
247   }

265   @Test
266   public void testMultitenantSign() throws UrlSigningException, ConfigurationException {
267 
268     properties.put(
269         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.URL),
270         MATCHING_URI);
271     properties.put(
272         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.SECRET),
273         SECRET);
274 
275     properties.put(
276         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, ORGANIZATION_A_KEY_ID,
277             GenericUrlSigningProvider.URL),
278         ORGANIZATION_A_MATCHING_URI);
279     properties.put(
280         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, ORGANIZATION_A_KEY_ID,
281             GenericUrlSigningProvider.SECRET),
282         ORGANIZATION_A_SECRET);
283     properties.put(
284         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, ORGANIZATION_A_KEY_ID,
285             GenericUrlSigningProvider.ORGANIZATION),
286         ORGANIZATION_A_ID);
287 
288     properties.put(
289         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, ORGANIZATION_B_KEY_ID,
290             GenericUrlSigningProvider.URL),
291         ORGANIZATION_B_MATCHING_URI);
292     properties.put(
293         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, ORGANIZATION_B_KEY_ID,
294             GenericUrlSigningProvider.SECRET),
295         ORGANIZATION_B_SECRET);
296     properties.put(
297         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, ORGANIZATION_B_KEY_ID,
298             GenericUrlSigningProvider.ORGANIZATION),
299         ORGANIZATION_B_ID);
300 
301     signerA.updated(properties);
302     signerB.updated(properties);
303 
304     DateTime before = new DateTime(2020, 03, 01, 00, 46, 17, 0, DateTimeZone.UTC);
305     Policy policy;
306     String result;
307     boolean exceptionThrown;
308 
309     // Organization A can sign its URLs using its key
310     policy = Policy.mkSimplePolicy(ORGANIZATION_A_RESOURCE_PATH, before);
311     result = signerA.sign(policy);
312     logger.info(result);
313     assertEquals(
314         "http://organization-a.opencast.org/path/to/resource.mp4?policy=eyJTdGF0ZW1lbnQiOnsiQ29uZGl"
315             + "0aW9uIjp7IkRhdGVMZXNzVGhhbiI6MTU4MzAyMzU3NzAwMH0sIlJlc291cmNlIjoiaHR0cDpcL1wvb3JnYW5"
316             + "pemF0aW9uLWEub3BlbmNhc3Qub3JnXC9wYXRoXC90b1wvcmVzb3VyY2UubXA0In19&keyId=key-id-organ"
317             + "ization-a&signature=7dce4419738aba73b20a3e9aea02dc8921532243e87635465f67cedac8a7b993",
318         result);
319 
320     // Organization A can sign URLs not specific to any organization
321     policy = Policy.mkSimplePolicy(RESOURCE_PATH, before);
322     result = signerA.sign(policy);
323     logger.info(result);
324     assertEquals(
325             "http://www.opencast.org/path/to/resource.mp4?policy=eyJTdGF0ZW1lbnQiOnsiQ29uZGl0aW9uIj"
326                 + "p7IkRhdGVMZXNzVGhhbiI6MTU4MzAyMzU3NzAwMH0sIlJlc291cmNlIjoiaHR0cDpcL1wvd3d3Lm9wZW"
327                 + "5jYXN0Lm9yZ1wvcGF0aFwvdG9cL3Jlc291cmNlLm1wNCJ9fQ&keyId=theId&signature=5b45e6782"
328                 + "75e6bc7b06a579f7f42e9a7ea5c58f1da130701db532f121e363e98",
329             result);
330 
331     // Organization A cannot sign URLs of organization B
332     exceptionThrown = false;
333     policy = Policy.mkSimplePolicy(ORGANIZATION_B_RESOURCE_PATH, before);
334     try {
335       result = signerA.sign(policy);
336     } catch (UrlSigningException e) {
337       exceptionThrown = true;
338     }
339     assertTrue(exceptionThrown);
340 
341     // Organization B can sign its URLs using its key
342     policy = Policy.mkSimplePolicy(ORGANIZATION_B_RESOURCE_PATH, before);
343     result = signerB.sign(policy);
344     logger.info(result);
345     assertEquals(
346         "http://organization-b.opencast.org/path/to/resource.mp4?policy=eyJTdGF0ZW1lbnQiOnsiQ29uZGl"
347             + "0aW9uIjp7IkRhdGVMZXNzVGhhbiI6MTU4MzAyMzU3NzAwMH0sIlJlc291cmNlIjoiaHR0cDpcL1wvb3JnYW5"
348             + "pemF0aW9uLWIub3BlbmNhc3Qub3JnXC9wYXRoXC90b1wvcmVzb3VyY2UubXA0In19&keyId=key-id-organ"
349             + "izatino-b&signature=331c925e17e7bfc85e8be412955e4c37945c552e9a9198166b0b843a290b67ea",
350         result);
351 
352     // Organization B can sign URLs not specific to any organization
353     policy = Policy.mkSimplePolicy(RESOURCE_PATH, before);
354     result = signerB.sign(policy);
355     logger.info(result);
356     assertEquals(
357         "http://www.opencast.org/path/to/resource.mp4?policy=eyJTdGF0ZW1lbnQiOnsiQ29uZGl0aW9uIjp7Ik"
358             + "RhdGVMZXNzVGhhbiI6MTU4MzAyMzU3NzAwMH0sIlJlc291cmNlIjoiaHR0cDpcL1wvd3d3Lm9wZW5jYXN0Lm"
359             + "9yZ1wvcGF0aFwvdG9cL3Jlc291cmNlLm1wNCJ9fQ&keyId=theId&signature=5b45e678275e6bc7b06a5"
360             + "79f7f42e9a7ea5c58f1da130701db532f121e363e98",
361         result);
362 
363     // Organization B cannot sign URLs of organization B
364     exceptionThrown = false;
365     policy = Policy.mkSimplePolicy(ORGANIZATION_A_RESOURCE_PATH, before);
366     try {
367       result = signerB.sign(policy);
368     } catch (UrlSigningException e) {
369       exceptionThrown = true;
370     }
371     assertTrue(exceptionThrown);
372   }

374   @Test
375   public void testMultipleUrlsPerKey() throws UrlSigningException, ConfigurationException {
376 
377     properties.put(
378         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.URL + ".1"),
379         MATCHING_URI);
380     properties.put(
381         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.URL + ".2"),
382         MATCHING_URI_2);
383     properties.put(
384         String.join(".", AbstractUrlSigningProvider.KEY_PROPERTY_PREFIX, KEY_ID, GenericUrlSigningProvider.SECRET),
385         SECRET);
386 
387     signer.updated(properties);
388     assertEquals(2, signer.getUris().size());
389 
390     DateTime before = new DateTime(2020, 03, 01, 00, 46, 17, 0, DateTimeZone.UTC);
391     // Uses KEY_ID for MATCHING_URI
392     Policy policy = Policy.mkSimplePolicy(RESOURCE_PATH, before);
393     String result = signer.sign(policy);
394     logger.info(result);
395     assertEquals(
396         "http://www.opencast.org/path/to/resource.mp4?policy=eyJTdGF0ZW1lbnQiOnsiQ29uZGl0aW9uIjp7Ik"
397             + "RhdGVMZXNzVGhhbiI6MTU4MzAyMzU3NzAwMH0sIlJlc291cmNlIjoiaHR0cDpcL1wvd3d3Lm9wZW5jYXN0Lm"
398             + "9yZ1wvcGF0aFwvdG9cL3Jlc291cmNlLm1wNCJ9fQ&keyId=theId&signature=5b45e678275e6bc7b06a5"
399             + "79f7f42e9a7ea5c58f1da130701db532f121e363e98",
400         result);
401 
402     // Uses KEY_ID for MATCHING_URI_2, too
403     policy = Policy.mkSimplePolicy(RESOURCE_PATH_2, before);
404     result = signer.sign(policy);
405     logger.info(result);
406     assertEquals(
407         "http://docs.opencast.org/path/to/resource.mp4?policy=eyJTdGF0ZW1lbnQiOnsiQ29uZGl0aW9uIjp7I"
408             + "kRhdGVMZXNzVGhhbiI6MTU4MzAyMzU3NzAwMH0sIlJlc291cmNlIjoiaHR0cDpcL1wvZG9jcy5vcGVuY2Fzd"
409             + "C5vcmdcL3BhdGhcL3RvXC9yZXNvdXJjZS5tcDQifX0&keyId=theId&signature=94a0a7c2e660dc5eaaf"
410             + "ca857e49a42a5b3857fe53353093873c8b41fa8d2b9b1",
411         result);
412   }
