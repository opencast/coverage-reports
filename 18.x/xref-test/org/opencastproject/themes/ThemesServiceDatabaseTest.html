49  /**
50   * Tests persistence: storing, merging, retrieving and removing.
51   */
52  public class ThemesServiceDatabaseTest {
53  
54    private ThemesServiceDatabaseImpl themesDatabase;
55  
56    /**
57     * @throws java.lang.Exception
58     */
59    @Before
60    public void setUp() throws Exception {
61      // Mock up a security service
62      SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
63      User user = new JaxbUser("admin", "test", new DefaultOrganization(), new JaxbRole(
64              SecurityConstants.GLOBAL_ADMIN_ROLE, new DefaultOrganization()));
65      EasyMock.expect(securityService.getOrganization()).andReturn(new DefaultOrganization()).anyTimes();
66      EasyMock.expect(securityService.getUser()).andReturn(user).anyTimes();
67      EasyMock.replay(securityService);
68  
69      UserDirectoryService userDirectoryService = EasyMock.createNiceMock(UserDirectoryService.class);
70      EasyMock.expect(userDirectoryService.loadUser(EasyMock.anyString())).andReturn(user).anyTimes();
71      EasyMock.replay(userDirectoryService);
72  
73      SearchResult result = EasyMock.createMock(SearchResult.class);
74      EasyMock.expect(result.getDocumentCount()).andReturn(0L).anyTimes();
75      EasyMock.replay(result);
76  
77      ElasticsearchIndex index = EasyMock.createNiceMock(ElasticsearchIndex.class);
78      EasyMock.expect(index.getIndexName()).andReturn("adminui").anyTimes();
79      EasyMock.expect(index.getByQuery(EasyMock.anyObject(ThemeSearchQuery.class))).andReturn(result).anyTimes();
80      EasyMock.replay(index);
81  
82      themesDatabase = new ThemesServiceDatabaseImpl();
83      themesDatabase.setEntityManagerFactory(newEntityManagerFactory(ThemesServiceDatabaseImpl.PERSISTENCE_UNIT));
84      themesDatabase.setDBSessionFactory(getDbSessionFactory());
85      themesDatabase.setSecurityService(securityService);
86      themesDatabase.setUserDirectoryService(userDirectoryService);
87      themesDatabase.setIndex(index);
88      themesDatabase.activate(null);
89    }
90  
91    @Test
92    public void testStoreUpdateAndDelete() throws Exception {
93      JaxbOrganization org = new DefaultOrganization();
94      JaxbUser creator = new JaxbUser("admin", "test", org);
95  
96      Theme theme = new Theme(Option.<Long> none(), new Date(), true, creator, "New");
97      Theme updateTheme = themesDatabase.updateTheme(theme);
98      Assert.assertEquals("New", updateTheme.getName());
99      Assert.assertEquals(1, themesDatabase.countThemes());
100     theme = new Theme(updateTheme.getId(), new Date(), true, creator, "Updated");
101 
102     updateTheme = themesDatabase.updateTheme(theme);
103     Assert.assertEquals("Updated", updateTheme.getName());
104     Assert.assertEquals(1, themesDatabase.countThemes());
105 
106     try {
107       theme = themesDatabase.getTheme(updateTheme.getId().get());
108       Assert.assertNotNull(theme);
109     } catch (NotFoundException e) {
110       Assert.fail("Existing theme has not been found");
111     }
112 
113     themesDatabase.deleteTheme(updateTheme.getId().get());
114     Assert.assertEquals(0, themesDatabase.countThemes());
115 
116     try {
117       themesDatabase.getTheme(updateTheme.getId().get());
118       Assert.fail("Deleted theme has been found");
119     } catch (NotFoundException e) {
120       Assert.assertNotNull(e);
121     }
122   }
