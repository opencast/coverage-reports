33  import com.mysema.query.jpa.impl.JPADeleteClause;
34  import com.mysema.query.jpa.impl.JPAQuery;
35  import com.mysema.query.types.EntityPath;
36  import com.mysema.query.types.Predicate;
37  
38  import javax.persistence.EntityManager;
39  
40  public class AssetManagerDeleteTestBase extends AssetManagerTestBase implements EntityPaths {
41  
42    protected DBSession db;
43  
44    @Override
45    public AssetManagerImpl makeAssetManager() throws Exception {
46      db = DBTestEnv.newDBSession(PERSISTENCE_UNIT);
47      // empty database
48      db.execTx(em -> {
49        nativeQuery.delete("delete from oc_assets_asset").apply(em);
50        nativeQuery.delete("delete from oc_assets_properties").apply(em);
51        nativeQuery.delete("delete from oc_assets_snapshot").apply(em);
52        nativeQuery.delete("delete from oc_assets_version_claim").apply(em);
53      });
54  
55      final Database database = new Database(db);
56      am = super.makeAssetManager();
57      am.setDatabase(database);
58      return am;
59    }
60  
61    long runCount(final JPAQuery q) {
62      return db.execTx(em -> {
63        return q.clone(em, Database.TEMPLATES).count();
64      });
65    }
66  
67    void assertPropertiesTotal(long count) {
68      assertEquals(format("[SQL] There should be %d properties total", count),
69                   count,
70                   runCount(new JPAQuery().from(Q_PROPERTY)));
71    }
72  
73    void assertAssetsTotal(long count) {
74      assertEquals(format("[SQL] There should be %d assets total", count),
75                   count,
76                   runCount(new JPAQuery().from(Q_ASSET)));
77    }
78  
79    void assertSnapshotsTotal(long count) {
80      assertEquals("[AssetManager] There should be " + count + " snapshots total",
81                   count,
82                   q.select(q.snapshot()).run().getSize());
83      assertEquals(format("[SQL] There should be %d snapshots total", count),
84                   count,
85                   runCount(new JPAQuery().from(Q_SNAPSHOT)));
86    }
87  
88    void assertTotals(long snapshots, long assets, long properties) {
89      assertSnapshotsTotal(snapshots);
90      assertAssetsTotal(assets);
91      assertPropertiesTotal(properties);
92    }
93  
94    static Function<EntityManager, JPADeleteClause> deleteFrom(final EntityPath<?> from) {
95      return new Function<EntityManager, JPADeleteClause>() {
96        @Override public JPADeleteClause apply(EntityManager em) {
97          return new JPADeleteClause(em, from, Database.TEMPLATES);
98        }
99      };
100   }
101 
102   static final Function<JPADeleteClause, Long> execute = new Function<JPADeleteClause, Long>() {
103     @Override public Long apply(JPADeleteClause delete) {
104       return delete.execute();
105     }
106   };
107 
108   static Function<JPADeleteClause, JPADeleteClause> where(final Predicate... p) {
109     return new Function<JPADeleteClause, JPADeleteClause>() {
110       @Override public JPADeleteClause apply(JPADeleteClause delete) {
111         return delete.where(p);
112       }
113     };
114   }
115 
116   static java.util.function.Function<EntityManager, Long> deleteFromWhere(final EntityPath<?> from,
117       final Predicate... where) {
118     return em -> deleteFrom(from)
119         .then(where(where))
120         .then(execute)
121         .apply(em);
122   }
123 
124   long delete(final EntityPath<?> from, final Predicate... where) {
125     return db.execTx(deleteFromWhere(from, where));
126   }
127 }
