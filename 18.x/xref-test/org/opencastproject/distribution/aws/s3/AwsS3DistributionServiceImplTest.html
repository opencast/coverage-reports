76    @Before
77    public void setUp() throws Exception {
78      // Set up the service
79      s3 = EasyMock.createNiceMock(AmazonS3Client.class);
80      tm = EasyMock.createNiceMock(TransferManager.class);
81      // Replay will be called in each test
82      File baseDir = FileSupport.getTempDirectory("s3distribution");
83      File srcFile = new File(baseDir, "presenter-m3u8/video-presenter-delivery.m3u8");
84      FileUtils.copyURLToFile(this.getClass().getResource("/video-presenter-delivery.m3u8"), srcFile);
85      srcFile = new File(baseDir, "presenter-mp4/video-presenter-delivery.mp4");
86      FileUtils.copyURLToFile(this.getClass().getResource("/video-presenter-delivery.mp4"), srcFile);
87      srcFile = new File(baseDir, "video-presenter-delivery.mp4");
88      FileUtils.copyURLToFile(this.getClass().getResource("/video-presenter-delivery.mp4"), srcFile);
89  
90      workspace = EasyMock.createNiceMock(Workspace.class);
91      EasyMock.expect(workspace.get((URI) EasyMock.anyObject())).andAnswer(new IAnswer<File>() {
92        @Override
93        public File answer() throws Throwable {
94          String name;
95          URI uri = (URI) EasyMock.getCurrentArguments()[0];
96          name = uri.getPath();
97          return new File(baseDir, name);
98        }
99      }).anyTimes();
100     EasyMock.replay(workspace);
101 
102     serviceRegistry = EasyMock.createNiceMock(ServiceRegistry.class);
103 
104     defaultOrganization = new DefaultOrganization();
105     SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
106     EasyMock.expect(securityService.getOrganization()).andReturn(defaultOrganization).anyTimes();
107     EasyMock.replay(securityService);
108 
109     service = new AwsS3DistributionServiceImpl();
110     service.setServiceRegistry(serviceRegistry);
111     service.setBucketName(defaultOrganization.getId(), BUCKET_NAME);
112     service.setOpencastDistributionUrl(DOWNLOAD_URL);
113     service.setS3(s3);
114     service.setS3TransferManager(tm);
115     service.setStorageTmp(baseDir.getAbsolutePath());
116     service.setWorkspace(workspace);
117     service.setSecurityService(securityService);
118 
119     MediaPackageBuilder builder = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder();
120     storageDir = new File(baseDir.getAbsolutePath() + AwsS3DistributionServiceImpl.DEFAULT_TEMP_DIR);
121 
122     URI mpURI = AwsS3DistributionServiceImpl.class.getResource("/mediapackage.xml").toURI();
123     mp = builder.loadFromXml(mpURI.toURL().openStream());
124 
125     mpURI = AwsS3DistributionServiceImpl.class.getResource("/mediapackage_distributed.xml").toURI();
126     distributedMp = builder.loadFromXml(mpURI.toURL().openStream());
127   }

133   @Test
134   public void testDistributeJob() throws Exception {
135     Set<String> mpeIds = new LinkedHashSet<String>();
136     mpeIds.add("presenter-delivery");
137 
138     boolean checkAvailability = true;
139 
140     List<String> args = new LinkedList<String>();
141     args.add("channelId");
142     args.add(MediaPackageParser.getAsXml(mp));
143     args.add(gson.toJson(mpeIds));
144     args.add(Boolean.toString(checkAvailability));
145 
146     EasyMock.expect(
147             serviceRegistry.createJob(
148                     AwsS3DistributionServiceImpl.JOB_TYPE,
149                     AwsS3DistributionServiceImpl.Operation.Distribute.toString(),
150                     args, AwsS3DistributionServiceImpl.DEFAULT_DISTRIBUTE_JOB_LOAD
151             )).andReturn(null).once();
152     EasyMock.replay(serviceRegistry);
153 
154     service.distribute("channelId", mp, "presenter-delivery");
155 
156     EasyMock.verify(serviceRegistry);
157   }

271   @Test
272   public void testDistributeHLSElement() throws Exception {
273     MediaPackageBuilder builder = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder();
274 
275     URI mpURI = AwsS3DistributionServiceImpl.class.getResource("/hls_mediapackage.xml").toURI();
276     mp = builder.loadFromXml(mpURI.toURL().openStream());
277 
278     mpURI = AwsS3DistributionServiceImpl.class.getResource("/distributed_hls_mediapackage.xml").toURI();
279     distributedMp = builder.loadFromXml(mpURI.toURL().openStream());
280     Upload upload = EasyMock.createNiceMock(Upload.class);
281     EasyMock.expect(tm.upload(EasyMock.anyObject(String.class), EasyMock.anyObject(String.class),
282             EasyMock.anyObject(File.class))).andReturn(upload).anyTimes();
283     EasyMock.replay(upload, tm);
284 
285     Set<String> mpeIds = new LinkedHashSet<String>();
286     mpeIds.add("presenter-mp4");
287     mpeIds.add("presenter-m3u8");
288 
289     MediaPackageElement[] mpes = service.distributeElements("channelId", mp, mpeIds, false);
290     MediaPackageElement mpe = mpes[0];
291 
292     Assert.assertEquals(new URI(
293         "http://XYZ.cloudfront.net/"
294             + defaultOrganization.getId()
295             + "/channelId/efd6e4df-63b6-49af-be5f-15f598778877/"
296             + "presenter-mp4/video-presenter-delivery.mp4"),
297         mpe.getURI());
298     // Test that temp directory is removed
299     Path tempfile = storageDir.toPath().resolve(mp.getIdentifier().toString());
300     Assert.assertFalse(Files.exists(tempfile));
301   }
