290   /**
291    * Setup for the video editor service, including creation of a mock workspace and all dependencies.
292    *
293    * @throws Exception
294    *           if setup fails
295    */
296   @Before
297   public void setUp() throws Exception {
298 
299     File tmpDir = folder.newFolder(getClass().getName());
300     tempFile1 = new File(tmpDir, "testoutput.mp4"); // output file
301 
302     /* mock the workspace for the input/output file */
303     // workspace.get(new URI(sourceTrackUri));
304     Workspace workspace = EasyMock.createMock(Workspace.class);
305     EasyMock.expect(workspace.rootDirectory()).andReturn(tmpDir.getAbsolutePath()).anyTimes();
306     EasyMock.expect(workspace.get(track1.getURI())).andReturn(new File(track1.getURI())).anyTimes();
307     EasyMock.expect(workspace.get(track2.getURI())).andReturn(new File(track2.getURI())).anyTimes();
308     EasyMock.expect(workspace.get(trackSubtitle.getURI())).andReturn(new File(trackSubtitle.getURI())).anyTimes();
309     EasyMock.expect(workspace.putInCollection(EasyMock.anyString(), EasyMock.anyString(),
310         EasyMock.anyObject(InputStream.class))).andAnswer(() -> {
311           Object[] things = EasyMock.getCurrentArguments();
312           InputStream in = (InputStream) EasyMock.getCurrentArguments()[2];
313           IOUtils.copy(in, new FileOutputStream(tempFile1));
314           return tempFile1.toURI();
315         }).anyTimes();
316 
317     /* mock the role/org/security dependencies */
318     User anonymous = new JaxbUser("anonymous", "test", new DefaultOrganization(), new JaxbRole(
319             DefaultOrganization.DEFAULT_ORGANIZATION_ANONYMOUS, new DefaultOrganization()));
320     UserDirectoryService userDirectoryService = EasyMock.createMock(UserDirectoryService.class);
321     EasyMock.expect(userDirectoryService.loadUser((String) EasyMock.anyObject())).andReturn(anonymous).anyTimes();
322 
323     Organization organization = new DefaultOrganization();
324     OrganizationDirectoryService organizationDirectoryService = EasyMock.createMock(OrganizationDirectoryService.class);
325     EasyMock.expect(organizationDirectoryService.getOrganization((String) EasyMock.anyObject()))
326     .andReturn(organization).anyTimes();
327 
328     SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
329     EasyMock.expect(securityService.getUser()).andReturn(anonymous).anyTimes();
330     EasyMock.expect(securityService.getOrganization()).andReturn(organization).anyTimes();
331 
332     /* mock the osgi init for the video editor itself */
333     BundleContext bc = EasyMock.createNiceMock(BundleContext.class);
334     File storageDir = folder.newFolder();
335     logger.info("storageDir: {}", storageDir);
336     EasyMock.expect(bc.getProperty("org.opencastproject.storage.dir")).andReturn(storageDir.getPath()).anyTimes();
337     EasyMock.expect(bc.getProperty("org.opencastproject.composer.ffmpegpath")).andReturn(FFMPEG_BINARY).anyTimes();
338     EasyMock.expect(bc.getProperty(FFmpegAnalyzer.FFPROBE_BINARY_CONFIG)).andReturn("ffprobe").anyTimes();
339     ComponentContext cc = EasyMock.createNiceMock(ComponentContext.class);
340     EasyMock.expect(cc.getBundleContext()).andReturn(bc).anyTimes();
341     EasyMock.replay(bc,cc,workspace,userDirectoryService,organizationDirectoryService,securityService);
342 
343     /* mock inspector output so that the job will alway pass */
344     String sourceTrackXml = "<?xml version='1.0' encoding='UTF-8' standalone='yes'?>"
345         + "<track xmlns=\"http://mediapackage.opencastproject.org\" type='presentation/source' "
346         + "id='deadbeef-a926-4ba9-96d9-2fafbcc30d2a'>"
347         + "<audio id='audio-1'><encoder type='MP3 (MPEG audio layer 3)'/><channels>2</channels>"
348         + "<bitrate>96000.0</bitrate></audio><video id='video-1'><device/>"
349         + "<encoder type='FLV / Sorenson Spark / Sorenson H.263 (Flash Video)'/>"
350         + "<bitrate>512000.0</bitrate><framerate>15.0</framerate>"
351         + "<resolution>854x480</resolution></video>"
352         + "<mimetype>video/mpeg</mimetype><url>video.mp4</url></track>";
353 
354     inspectedTrack = (Track) MediaPackageElementParser.getFromXml(sourceTrackXml);
355     veditor = new VideoEditorServiceImpl() {
356       @Override
357       protected Job inspect(Job job, URI workspaceURI) throws MediaInspectionException, ProcessFailedException {
358         Job inspectionJob = EasyMock.createNiceMock(Job.class);
359         try {
360           EasyMock.expect(inspectionJob.getPayload()).andReturn(MediaPackageElementParser.getAsXml(inspectedTrack));
361         } catch (MediaPackageException e) {
362           throw new MediaInspectionException(e);
363         }
364         EasyMock.replay(inspectionJob);
365         return inspectionJob;
366       }
367     };
368 
369     /* set up video editor */
370     veditor.activate(cc);
371     veditor.setWorkspace(workspace);
372     veditor.setSecurityService(securityService);
373     veditor.setUserDirectoryService(userDirectoryService);
374     veditor.setSmilService(smilService);
375     veditor.setOrganizationDirectoryService(organizationDirectoryService);
376 
377     serviceRegistry = EasyMock.createMock(ServiceRegistry.class);
378     final Capture<String> type = EasyMock.newCapture();
379     final Capture<String> operation = EasyMock.newCapture();
380     final Capture<List<String>> args = EasyMock.newCapture();
381     EasyMock.expect(serviceRegistry.createJob(capture(type), capture(operation), capture(args), EasyMock.anyFloat()))
382             .andAnswer(() -> {
383               Job job = new JobImpl(0);
384               logger.error("type: {}", type.getValue());
385               job.setJobType(type.getValue());
386               job.setOperation(operation.getValue());
387               job.setArguments(args.getValue());
388               job.setPayload(veditor.process(job));
389               return job;
390             }).anyTimes();
391     EasyMock.replay(serviceRegistry);
392 
393     veditor.setServiceRegistry(serviceRegistry);
394 
395   }

412       assertTrue("Merged video is is not empty", tempFile1.length() > 0);
413       logger.info("Resulting file size {} ", tempFile1.length());
414     }
