48  public class TransferMetadataOperationHandlerTest {
49  
50    private TransferMetadataWorkflowOperationHandler handler;
51    private WorkflowInstance instance;
52    private WorkflowOperationInstance operation;
53    private DublinCoreCatalog resultCatalog = null;

55    @Before
56    public void setUp() throws Exception {
57      handler = new TransferMetadataWorkflowOperationHandler();
58  
59      operation = new WorkflowOperationInstance("test", WorkflowOperationInstance.OperationState.RUNNING);
60  
61      String dc = IOUtils.toString(getClass().getResourceAsStream("/dublincore.xml"), StandardCharsets.UTF_8);
62      Workspace workspace = EasyMock.createMock(Workspace.class);
63      EasyMock.expect(workspace.read(EasyMock.anyObject()))
64              .andAnswer(() -> getClass().getResourceAsStream("/dublincore.xml")).anyTimes();
65      EasyMock.expect(workspace.put(EasyMock.anyString(), EasyMock.anyString(), EasyMock.anyString(), EasyMock.anyObject()))
66              .andAnswer(() -> {
67                resultCatalog = DublinCores.read((InputStream) EasyMock.getCurrentArguments()[3]);
68                return new URI("http://example.opencast.org/xy");
69              }).once();
70      EasyMock.replay(workspace);
71      handler.setWorkspace(workspace);
72  
73      MediaPackage mediaPackage = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder().createNew();
74      Catalog catalog = CatalogImpl.fromURI(new URI("http://example.com"));
75      catalog.setFlavor(MediaPackageElementFlavor.parseFlavor("dublincore/episode"));
76      mediaPackage.add(catalog);
77      instance = EasyMock.createMock(WorkflowInstance.class);
78      EasyMock.expect(instance.getCurrentOperation()).andReturn(operation).anyTimes();
79      EasyMock.expect(instance.getMediaPackage()).andReturn(mediaPackage).anyTimes();
80      EasyMock.replay(instance);
81    }

83    @Test
84    public void testNoConfiguration() throws WorkflowOperationException {
85      try {
86        handler.start(instance, null);
87        Assert.fail("Operation should fail");
88      } catch (IllegalArgumentException e) {
89        // this is expected
90      }
91  
92      operation.setConfiguration("source-flavor", "dublincore/episode");
93      operation.setConfiguration("target-flavor", "lk/episode");
94  
95      try {
96        handler.start(instance, null);
97        Assert.fail("Operation should fail");
98      } catch (IllegalArgumentException e) {
99        // this is expected
100     }
101   }
102 
103   @Test
104   public void testExistingElement() throws WorkflowOperationException {
105     // setup
106     operation.setConfiguration("source-flavor", "dublincore/episode");
107     operation.setConfiguration("target-flavor", "dublincore/episode");
108     operation.setConfiguration("source-element", "{http://purl.org/dc/terms/}title");
109     operation.setConfiguration("target-element", "{http://purl.org/dc/terms/}description");
110 
111     // execution
112     try {
113       handler.start(instance, null);
114       Assert.fail("Operation should have failed");
115     } catch (WorkflowOperationException e) {
116       // expected
117     }
118 
119     // force
120     operation.setConfiguration("force", "true");
121     WorkflowOperationResult result = handler.start(instance, null);
122 
123     // checks
124     Assert.assertEquals(WorkflowOperationResult.Action.CONTINUE, result.getAction());
125     Assert.assertNotNull(resultCatalog);
126     EName src = EName.mk("http://purl.org/dc/terms/", "title");
127     EName dest = EName.mk("http://purl.org/dc/terms/", "description");
128     Assert.assertEquals(resultCatalog.get(src).size(), resultCatalog.get(dest).size());
129     Assert.assertEquals(resultCatalog.getFirst(src), resultCatalog.getFirst(dest));
130   }

132   @Test
133   public void testSkip() throws WorkflowOperationException {
134     // setup
135     operation.setConfiguration("source-flavor", "does-not/exist");
136     operation.setConfiguration("target-flavor", "dublincore/episode");
137     operation.setConfiguration("source-element", "does-not-exist");
138     operation.setConfiguration("target-element", "{http://purl.org/dc/terms/}description");
139 
140     // skip since source catalog does not exists
141     WorkflowOperationResult result = handler.start(instance, null);
142     Assert.assertEquals(WorkflowOperationResult.Action.SKIP, result.getAction());
143   }
