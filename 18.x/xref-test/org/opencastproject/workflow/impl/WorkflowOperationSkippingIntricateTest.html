210   @Test
211   public void testIfSuccessful() throws Exception {
212 
213     EasyMock.expect(property.getId()).andReturn(new PropertyId(mediapackage1.getIdentifier().toString(),
214             OpencastConstants.WORKFLOW_PROPERTIES_NAMESPACE, "executecondition")).anyTimes();
215     EasyMock.expect(property.getValue()).andReturn(new Value.StringValue("a")).once();
216 
217     EasyMock.replay(property);
218 
219     Map<String, String> properties1 = new HashMap<>();
220     properties1.put("executecondition", "a");
221 
222     WorkflowInstance instance = startAndWait(workingDefinition, mediapackage1, properties1);
223 
224     // See if the skip operation has been executed
225     WorkflowInstance instanceFromDb = service.getWorkflowById(instance.getId());
226     assertNotNull(instanceFromDb);
227     assertEquals(OperationState.SUCCEEDED, instanceFromDb.getOperations().get(0).getState());
228   }

244     // See if the skip operation has been executed
245     WorkflowInstance instanceFromDb = service.getWorkflowById(instance.getId());
246     assertNotNull(instanceFromDb);
247     assertEquals(OperationState.SKIPPED, instanceFromDb.getOperations().get(0).getState());

250   private WorkflowInstance startAndWait(WorkflowDefinition definition, MediaPackage mp, Map<String, String> properties) throws Exception {
251 
252     WorkflowStateListener stateListener = new WorkflowStateListener(WorkflowState.SUCCEEDED);
253     service.addWorkflowListener(stateListener);
254     WorkflowInstance instance;
255     instance = service.start(definition, mp, properties);
256     WorkflowTestSupport.poll(stateListener, 1);
257     service.removeWorkflowListener(stateListener);
258 
259     return instance;
260   }

262   class SucceedingWorkflowOperationHandler extends AbstractWorkflowOperationHandler {
263 
264     private MediaPackage mp;
265 
266     SucceedingWorkflowOperationHandler(MediaPackage mp) {
267       this.mp = mp;
268     }
269 
270     @Override
271     public String getId() {
272       return this.getClass().getName();
273     }
274 
275     @Override
276     public String getDescription() {
277       return "ContinuingWorkflowOperationHandler";
278     }
279 
280     @Override
281     public WorkflowOperationResult start(WorkflowInstance workflowInstance, JobContext context) {
282       return createResult(mp, Action.CONTINUE);
283     }
