21  package org.opencastproject.terminationstate.api;
22  
23  import org.opencastproject.job.api.Job;
24  import org.opencastproject.serviceregistry.impl.ServiceRegistryJpaImpl;
25  
26  import org.easymock.EasyMock;
27  import org.junit.Assert;
28  import org.junit.Before;
29  import org.junit.Test;
30  
31  public class AbstractJobTerminationStateServiceTest {
32    private static final String LOCALHOST = "http://localhost:8080";
33  
34    private ServiceRegistryJpaImpl serviceRegistry;
35  
36    @Before
37    public void setUp() throws Exception {
38      serviceRegistry = EasyMock.createMock(ServiceRegistryJpaImpl.class);
39      EasyMock.expect(serviceRegistry.getRegistryHostname()).andReturn(LOCALHOST).anyTimes();
40      EasyMock.expect(serviceRegistry.countByHost(null, LOCALHOST, Job.Status.RUNNING)).andReturn(3L).once();
41      EasyMock.expect(serviceRegistry.countByHost(null, LOCALHOST, Job.Status.RUNNING)).andReturn(0L).once();
42      EasyMock.replay(serviceRegistry);
43    }
44  
45    @Test
46    public void testReadyToTerminate() throws Exception {
47      AbstractJobTerminationStateService service = new AbstractJobTerminationStateService() { };
48      service.setServiceRegistry(serviceRegistry);
49  
50      // jobs running, state NONE
51      Assert.assertEquals(false, service.readyToTerminate());
52      // jobs running state WAIT
53      service.setState(TerminationStateService.TerminationState.WAIT);
54      Assert.assertEquals(false, service.readyToTerminate());
55      // no jobs running state WAIT
56      Assert.assertEquals(true, service.readyToTerminate());
57      Assert.assertEquals(TerminationStateService.TerminationState.READY, service.getState());
58    }
59  }
