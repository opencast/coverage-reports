75    /**
76     * Tests {@link AbstractCoverImageService#parseXsl(String)}
77     */
78    @Test
79    public void testParseXsl() throws Exception {
80  
81      InputStream xslIn = this.getClass().getResourceAsStream("/metadata2svg.xsl");
82      StringWriter writer = new StringWriter();
83      IOUtils.copy(xslIn, writer, "UTF-8");
84      String xslString = writer.toString();
85      Document xslDoc = AbstractCoverImageService.parseXsl(xslString);
86      assertNotNull(xslDoc);
87  
88      NamespaceContext usingNamespaces = new SimpleNamespaceContext().withBinding("xsl",
89              "http://www.w3.org/1999/XSL/Transform").withBinding("svg", "http://www.w3.org/2000/svg");
90      assertThat(the(xslDoc),
91              hasXPath("/xsl:stylesheet/xsl:template/svg:svg/svg:defs/svg:linearGradient[@id='lgGray']",
92                usingNamespaces));
93    }

122   /**
123    * Tests {@link AbstractCoverImageService#transformSvg(Result, InputSource, Document, int, int, String)} 
124    */
125   @Test
126   public void testTransformSvg() throws Exception {
127     Writer svgWriter = new StringWriter();
128     Result svg = new StreamResult(svgWriter);
129 
130     InputStream isXml = CoverImageServiceTest.class.getResourceAsStream("/metadata.xml");
131     InputSource xmlSource = new InputSource(isXml);
132 
133     InputStream isXsl = CoverImageServiceTest.class.getResourceAsStream("/metadata2svg.xsl");
134     Document xslDoc = AbstractCoverImageService.parseXsl(IOUtils.toString(isXsl));
135 
136     AbstractCoverImageService.transformSvg(svg, xmlSource, xslDoc, 1600, 900, null);
137 
138     String svgString = svgWriter.toString();
139     NamespaceContext nsContext = new SimpleNamespaceContext().withBinding("svg", "http://www.w3.org/2000/svg");
140     assertThat(the(svgString), hasXPath("//svg:svg[@width='1600']", nsContext));
141     assertThat(the(svgString), hasXPath("//svg:svg/svg:text/svg:tspan", nsContext));
142   }
