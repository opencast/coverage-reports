51  /**
52   * Take look in specified catalog for specified term, if the value matches the specified value add the target-tags
53   */
54  @Component(
55      immediate = true,
56      service = WorkflowOperationHandler.class,
57      property = {
58          "service.description=Configure by Dublin Core Term Workflow Handler",
59          "workflow.operation=configure-by-dcterm"
60      }
61  )
62  public class ConfigureByDublinCoreTermWOH extends ResumableWorkflowOperationHandlerBase {
63  
64    private static final Logger logger = LoggerFactory.getLogger(ConfigureByDublinCoreTermWOH.class);
65  
66    /** Name of the configuration option that provides the catalog to examine */
67    public static final String DCCATALOG_PROPERTY = "dccatalog";
68  
69    /** Name of the configuration option that provides Dublin Core term/element */
70    public static final String DCTERM_PROPERTY = "dcterm";
71  
72    /** Name of the configuration option that provides term's default value if not present */
73    public static final String DEFAULT_VALUE_PROPERTY = "default-value";
74  
75    /** Name of the configuration option that provides value to match */
76    public static final String MATCH_VALUE_PROPERTY = "match-value";
77  
78    /** Name of the configuration option that provides the copy boolean we are looking for */
79    public static final String COPY_PROPERTY = "copy";
80  
81    /** The local workspace */
82    private Workspace workspace = null;
83  
84    /**
85     * Callback for declarative services configuration that will introduce us to the local workspace service.
86     * Implementation assumes that the reference is configured as being static.
87     *
88     * @param workspace
89     *          an instance of the workspace
90     */
91    @Reference
92    public void setWorkspace(Workspace workspace) {
93      this.workspace = workspace;
94    }
95  
96    /**
97     * {@inheritDoc}
98     *
99     * @see org.opencastproject.workflow.api.WorkflowOperationHandler#start(org.opencastproject.workflow.api.WorkflowInstance,
100    *      JobContext)
101    */
102   @Override
103   public WorkflowOperationResult start(WorkflowInstance workflowInstance, JobContext context)
104           throws WorkflowOperationException {
105 
106     MediaPackage mediaPackage = workflowInstance.getMediaPackage();
107     WorkflowOperationInstance currentOperation = workflowInstance.getCurrentOperation();
108 
109     String configuredCatalog = StringUtils.trimToEmpty(currentOperation.getConfiguration(DCCATALOG_PROPERTY));
110     String configuredDCTerm = StringUtils.trimToEmpty(currentOperation.getConfiguration(DCTERM_PROPERTY));
111     String configuredDefaultValue = StringUtils.trimToNull(currentOperation.getConfiguration(DEFAULT_VALUE_PROPERTY));
112     String configuredMatchValue = StringUtils.trimToEmpty(currentOperation.getConfiguration(MATCH_VALUE_PROPERTY));
113 
114     // Find Catalog
115     Catalog[] catalogs = mediaPackage
116             .getCatalogs(new MediaPackageElementFlavor("dublincore", StringUtils.lowerCase(configuredCatalog)));
117 
118     if (catalogs != null && catalogs.length > 0) {
119       Boolean foundValue = false;
120       EName dcterm = new EName(TERMS_NS_URI, configuredDCTerm);
121 
122       // Find DCTerm
123       for (Catalog catalog : catalogs) {
124         DublinCoreCatalog dc = DublinCoreUtil.loadDublinCore(workspace, catalog);
125         // Match Value
126         List<DublinCoreValue> values = dc.get(dcterm);
127         if (values.isEmpty()) {
128           // Use default
129           if (configuredDefaultValue != null) {
130             foundValue = configuredDefaultValue.equals(configuredMatchValue);
131           }
132         } else {
133           foundValue = values.contains(DublinCoreValue.mk(configuredMatchValue));
134         }
135       }
136 
137       if (foundValue) {
138         Map<String, String> properties = new HashMap<>();
139 
140         for (String key : currentOperation.getConfigurationKeys()) {
141           // Ignore this operations configuration
142           if (DCCATALOG_PROPERTY.equals(key) || DCTERM_PROPERTY.equals(key) || DEFAULT_VALUE_PROPERTY.equals(key)
143                   || MATCH_VALUE_PROPERTY.equals(key)) {
144             continue;
145           }
146 
147           String value = currentOperation.getConfiguration(key);
148           properties.put(key, value);
149           logger.info("Configuration key '{}' of workflow {} is set to value '{}'", key, workflowInstance.getId(),
150                   value);
151         }
152 
153         return createResult(mediaPackage, properties, Action.CONTINUE, 0);
154 
155       } // if foundValue
156     } // if catalogs
157 
158     return createResult(mediaPackage, Action.CONTINUE);
159   }
