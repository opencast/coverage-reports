56    @Test
57    public void testWorkflowFields() throws Exception {
58      // Workflows should obtain information from the definition used in the constructor
59      WorkflowDefinitionImpl def = new WorkflowDefinitionImpl();
60      def.setId("123");
61  
62      Map<String, String> props = new HashMap<String, String>();
63      props.put("key1", "value1");
64      WorkflowInstance instance = new WorkflowInstance(def, null, null, null, props);
65      Assert.assertEquals(def.getId(), instance.getTemplate());
66      Assert.assertEquals("value1", instance.getConfiguration("key1"));
67      def.setTitle("a title");
68    instance = new WorkflowInstance(def, null, null, null, null);
69      Assert.assertEquals(def.getTitle(), instance.getTitle());
70    }

72    @Test
73    public void testMediaPackageSerializationInWorkflowInstance() throws Exception {
74      WorkflowInstance workflow = new WorkflowInstance();
75      MediaPackage src = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder().createNew();
76      Track track = (Track) MediaPackageElementBuilderFactory.newInstance().newElementBuilder()
77              .elementFromURI(new URI("http://sample"), Track.TYPE, MediaPackageElements.PRESENTER_SOURCE);
78      src.add(track);
79      MediaPackage deserialized = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder()
80              .loadFromXml(MediaPackageParser.getAsXml(src));
81      workflow.setMediaPackage(deserialized);
82      Assert.assertEquals(1, workflow.getMediaPackage().getTracks().length);
83    }

121   @Test
122   public void testFlavorMarshalling() throws Exception {
123     URI uri = new URI("http://testing");
124     Track track = TrackImpl.fromURI(uri);
125     track.setFlavor(MediaPackageElements.PRESENTATION_SOURCE);
126 
127     MediaPackage mp = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder().createNew();
128     mp.add(track);
129 
130     WorkflowInstance workflow = new WorkflowInstance();
131     workflow.setMediaPackage(mp);
132 
133     // Marshall the workflow to xml
134     String xml = XmlWorkflowParser.toXml(workflow);
135 
136     // Get it back from xml
137     WorkflowInstance instance2 = XmlWorkflowParser.parseWorkflowInstance(xml);
138     Assert.assertEquals(workflow.getMediaPackage().getTracks()[0].getFlavor(),
139             instance2.getMediaPackage().getTracks()[0].getFlavor());
140 
141     String namespaceXml = "<workflow xmlns=\"http://workflow.opencastproject.org\" xmlns:mp=\"http://mediapackage.opencastproject.org\"><parent/><mp:mediapackage><mp:media><mp:track type=\"presentation/source\" id=\"track-1\"><mp:url>http://testing</mp:url></mp:track></mp:media></mp:mediapackage></workflow>";
142     WorkflowInstance instance3 = XmlWorkflowParser.parseWorkflowInstance(namespaceXml);
143     Assert.assertEquals(workflow.getMediaPackage().getTracks()[0].getFlavor(),
144             instance3.getMediaPackage().getTracks()[0].getFlavor());
145   }
