23  package org.opencastproject.workflow.impl;
24  
25  import org.opencastproject.util.NotFoundException;
26  import org.opencastproject.workflow.api.JaxbWorkflowInstance;
27  import org.opencastproject.workflow.api.WorkflowDefinition;
28  import org.opencastproject.workflow.api.WorkflowInstance;
29  import org.opencastproject.workflow.api.WorkflowService;
30  import org.opencastproject.workflow.endpoint.WorkflowRestService;
31  
32  import org.easymock.EasyMock;
33  import org.junit.Assert;
34  import org.junit.Before;
35  import org.junit.Test;
36  
37  import java.util.ArrayList;
38  
39  public class WorkflowRestEndpointTest {
40  
41    private WorkflowRestService restService;
42    private WorkflowInstance workflow;
43  
44    @Before
45    public void setUp() throws Exception {
46      // Create a workflow for the service to return
47      workflow = new WorkflowInstance();
48      workflow.setTitle("a workflow instance");
49      workflow.setId(1);
50  
51      // Mock up the behavior of the workflow service
52      WorkflowService service = EasyMock.createNiceMock(WorkflowService.class);
53      EasyMock.expect(service.listAvailableWorkflowDefinitions()).andReturn(new ArrayList<WorkflowDefinition>());
54      EasyMock.expect(service.getWorkflowById(EasyMock.anyLong())).andThrow(new NotFoundException()).times(2)
55              .andReturn(workflow);
56      EasyMock.replay(service);
57  
58      // Set up the rest endpoint
59      restService = new WorkflowRestService();
60      restService.setService(service);
61      restService.activate(null);
62    }
63  
64    @Test
65    public void testGetWorkflowInstance() throws Exception {
66      try {
67        restService.getWorkflowAsJson(-1);
68        Assert.fail("This should have thrown a not found exception");
69      } catch (NotFoundException e) {
70        // expected
71      }
72  
73      try {
74        restService.getWorkflowAsXml(-1);
75        Assert.fail("This should have thrown a not found exception");
76      } catch (NotFoundException e) {
77        // expected
78      }
79  
80      JaxbWorkflowInstance xmlResponse = restService.getWorkflowAsXml(1);
81      Assert.assertNotNull(xmlResponse);
82    }
83  }
