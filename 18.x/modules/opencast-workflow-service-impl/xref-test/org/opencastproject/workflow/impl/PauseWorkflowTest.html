189   @Test
190   public void testHoldAndResume() throws Exception {
191 
192     // Start a new workflow
193     WorkflowStateListener pauseListener = new WorkflowStateListener(WorkflowState.PAUSED);
194     service.addWorkflowListener(pauseListener);
195     workflow = service.start(def, mp, null);
196     WorkflowTestSupport.poll(pauseListener, 1);
197 
198     // Ensure that the first operation handler was called, but not the second
199     Assert.assertTrue(firstHandler.isStarted());
200     Assert.assertTrue(!secondHandler.isStarted());
201 
202     // The workflow should be in the paused state
203     Assert.assertEquals(WorkflowState.PAUSED, service.getWorkflowById(workflow.getId()).getState());
204 
205     // Resume the workflow with some properties
206     Map<String, String> properties = new HashMap<String, String>();
207     properties.put("key", "value");
208 
209     service.addWorkflowListener(pauseListener);
210     service.resume(workflow.getId(), properties);
211     WorkflowTestSupport.poll(pauseListener, 1);
212     //Workflow takes some time internally, so poll until things match, or we time out (and the assert fails below)
213     // sleep time, and counter size are set arbitrarily.  Should be large enough to work, but not force a huge wait
214     for (int counter = 0; counter < 100 && !properties.equals(firstHandler.getProperties()); counter++) {
215       Thread.sleep(100);
216     }
217     service.removeWorkflowListener(pauseListener);
218 
219     // The handler should have picked up the properties
220     Assert.assertEquals(properties, firstHandler.getProperties());
221   }
