204   @Test
205   public void testHoldAndResume() throws Exception {
206     // Wait for all workflows to be in paused state
207     WorkflowStateListener listener = new WorkflowStateListener(WorkflowState.PAUSED);
208     service.addWorkflowListener(listener);
209 
210     Map<String, String> initialProps = new HashMap<String, String>();
211     initialProps.put("testproperty", "foo");
212     WorkflowInstance workflow1 = null;
213     workflow1 = service.start(def, mp, initialProps);
214     WorkflowTestSupport.poll(listener, 1);
215     mp.setIdentifier(IdImpl.fromUUID());
216     service.start(def, mp, initialProps);
217     WorkflowTestSupport.poll(listener, 2);
218     service.removeWorkflowListener(listener);
219 
220     // Test for two paused workflows in "op1"
221     assertEquals(2, service.countWorkflowInstances());
222     assertEquals(2, service.countWorkflowInstances(WorkflowState.PAUSED));
223     assertEquals(0, service.countWorkflowInstances(WorkflowState.SUCCEEDED));
224 
225     // Continue one of the two workflows, waiting for success
226     listener = new WorkflowStateListener(WorkflowState.SUCCEEDED);
227     service.addWorkflowListener(listener);
228     service.resume(workflow1.getId());
229     WorkflowTestSupport.poll(listener, 1);
230     service.removeWorkflowListener(listener);
231 
232     // Make sure one workflow is still on hold, the other is finished.
233     assertEquals(2, service.countWorkflowInstances());
234     assertEquals(1, service.countWorkflowInstances(WorkflowState.PAUSED));
235     assertEquals(1, service.countWorkflowInstances(WorkflowState.SUCCEEDED));
236   }
