238   @Test
239   public void testMergeSegments() throws Exception {
240     List<SegmentData> segments = new ArrayList<>();
241     segments.add(new SegmentData(0L, 2449L));
242     segments.add(new SegmentData(4922L, 11284L));
243     segments.add(new SegmentData(14000L, 15000L));
244     List<SegmentData> segments2 = new ArrayList<>();
245     segments2.add(new SegmentData(1449L, 3449L));
246     segments2.add(new SegmentData(11285L, 11290L));
247     segments2.add(new SegmentData(15000L, 16000L));
248     List<SegmentData> mergedSegments = editorService.mergeSegments(segments, segments2);
249     assertEquals(5, mergedSegments.size());
250   }

252   @Test
253   public void testDeletedSegments() throws Exception {
254     List<SegmentData> segments = new ArrayList<>();
255     segments.add(new SegmentData(0L, 2449L));
256     segments.add(new SegmentData(4922L, 11284L));
257 
258     MediaPackage mediaPackage = EasyMock.createNiceMock(MediaPackage.class);
259     EasyMock.expect(mediaPackage.getDuration()).andReturn(14000L).anyTimes();
260     EasyMock.replay(mediaPackage);
261 
262     List<SegmentData> deleted = editorService.getDeletedSegments(mediaPackage, segments);
263     assertNotNull(deleted);
264     assertEquals(2, deleted.size());
265     assertTrue(deleted.contains(new SegmentData(2449L, 4922L, true)));
266     assertTrue(deleted.contains(new SegmentData(11284L, 14000L, true)));
267   }

269   @Test
270   public void testAddSmilToArchive() throws Exception {
271     final String mpId = UUID.randomUUID().toString();
272     final URI archiveElementURI = new URI("http://host.tld/archive/cut.smil");
273     final String smilId = "s-afe311c6-9161-41f4-98d0-e951fe66d89e";
274 
275     Workspace workspace = createNiceMock(Workspace.class);
276     expect(workspace.put(same(mpId), same(smilId), same("cut.smil"), anyObject(InputStream.class)))
277             .andReturn(archiveElementURI);
278     replay(workspace);
279     editorService.setWorkspace(workspace);
280 
281     AssetManager assetManager = createNiceMock(AssetManager.class);
282     replay(assetManager);
283     editorService.setAssetManager(assetManager);
284 
285     MediaPackage mp = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder().createNew(new IdImpl(mpId));
286 
287     editorService.addSmilToArchive(mp, smil);
288 
289     assertEquals(1, mp.getCatalogs().length);
290     assertEquals(smil.getId(), mp.getCatalogs()[0].getIdentifier());
291     assertEquals("smil/cutting", mp.getCatalogs()[0].getFlavor().toString());
292   }
