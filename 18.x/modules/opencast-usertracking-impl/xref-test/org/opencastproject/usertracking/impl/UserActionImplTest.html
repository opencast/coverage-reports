22  package org.opencastproject.usertracking.impl;
23  
24  import org.opencastproject.usertracking.api.UserAction;
25  import org.opencastproject.usertracking.api.UserSession;
26  
27  import org.junit.Assert;
28  import org.junit.Test;
29  
30  public class UserActionImplTest {
31  
32    /**
33     * Ensures a positive length user action calculates its length correctly.
34     */
35    @Test
36    public void testForwardLength() {
37      UserSession us = new UserSessionImpl();
38      us.setSessionId("4");
39      us.setUserId("testing user");
40      us.setUserIp("127.0.0.1");
41  
42      UserAction ua = new UserActionImpl();
43      ua.setId(4L);
44      ua.setInpoint(90);
45      ua.setIsPlaying(true);
46      ua.setMediapackageId("false");
47      ua.setOutpoint(100);
48      ua.setSession(us);
49      ua.setType("test");
50  
51      Assert.assertEquals(new Long(4), ua.getId());
52      Assert.assertEquals(90, ua.getInpoint());
53      Assert.assertEquals(true, ua.getIsPlaying());
54      Assert.assertEquals("false", ua.getMediapackageId());
55      Assert.assertEquals(100, ua.getOutpoint());
56      Assert.assertEquals("4", ua.getSession().getSessionId());
57      Assert.assertEquals("test", ua.getType());
58      Assert.assertEquals("testing user", ua.getSession().getUserId());
59      Assert.assertEquals("127.0.0.1", ua.getSession().getUserIp());
60      Assert.assertEquals(10, ua.getLength());
61    }
62  
63    /**
64     * Ensures a negative length user action calculates its length correctly.
65     */
66    @Test
67    public void testBackwardLength() {
68      UserSession us = new UserSessionImpl();
69      us.setSessionId("4");
70      us.setUserId("testing user");
71      us.setUserIp("127.0.0.1");
72  
73      UserAction ua = new UserActionImpl();
74      ua.setId(4L);
75      ua.setInpoint(100);
76      ua.setIsPlaying(true);
77      ua.setMediapackageId("false");
78      ua.setOutpoint(90);
79      ua.setSession(us);
80      ua.setType("test");
81  
82      Assert.assertEquals(new Long(4), ua.getId());
83      Assert.assertEquals(100, ua.getInpoint());
84      Assert.assertEquals(true, ua.getIsPlaying());
85      Assert.assertEquals("false", ua.getMediapackageId());
86      Assert.assertEquals(90, ua.getOutpoint());
87      Assert.assertEquals("4", ua.getSession().getSessionId());
88      Assert.assertEquals("test", ua.getType());
89      Assert.assertEquals("testing user", ua.getSession().getUserId());
90      Assert.assertEquals("127.0.0.1", ua.getSession().getUserIp());
91      Assert.assertEquals(-10, ua.getLength());
92    }

94  }
