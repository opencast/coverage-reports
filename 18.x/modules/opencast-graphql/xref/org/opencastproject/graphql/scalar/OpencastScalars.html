22  package org.opencastproject.graphql.scalar;
23  
24  import graphql.schema.GraphQLScalarType;
25  
26  /**
27   * This class provides custom scalar types for the Opencast GraphQL API.
28   * Scalar types are primitive data types that GraphQL uses to validate and serialize data.
29   * In this case, a custom scalar type for Duration is provided.
30   */
31  public final class OpencastScalars {
32  
33    private OpencastScalars() {
34    }
35  
36    /**
37     * An ISO 8601 compliant duration scalar that accepts string values like `PT1H30M10S` and produces
38     * `java.time.Duration` objects at runtime.
39     * <p>
40     * Its {@link graphql.schema.Coercing#serialize(java.lang.Object)}
41     * and {@link graphql.schema.Coercing#parseValue(java.lang.Object)} methods
42     * accept OffsetDateTime, ZoneDateTime and formatted Strings as valid objects.
43     *
44     * @see java.time.Duration
45     */
46    public static final GraphQLScalarType Duration = DurationScalar.INSTANCE;
47  
48  }
