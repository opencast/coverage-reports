207   @Activate
208   public void activate(ComponentContext cc) {
209     workflowService.addWorkflowListener(new WorkflowListener() {
210       @Override
211       public void stateChanged(WorkflowInstance workflow) {
212         if (!workflow.getTemplate().equals(copyWorkflowId)) {
213           return;
214         }
215 
216         if (workflow.getState().equals(SUCCEEDED)) {
217           final String publishWorkflowName = "publish";
218           logger.info("workflow '{}' succeeded for media package '{}', starting workflow '{}'", workflow.getTemplate(),
219                   workflow.getMediaPackage().getIdentifier(), publishWorkflowName);
220           final WorkflowDefinition wfd;
221           try {
222             wfd = workflowService.getWorkflowDefinitionById(publishWorkflowName);
223             final Workflows workflows = new Workflows(assetManager, workflowService);
224             final ConfiguredWorkflow newWorkflow = workflow(wfd);
225             final String targetMpId = workflow.getConfiguration(NEW_MP_ID_KEY);
226             final List<WorkflowInstance> workflowInstances = workflows
227                     .applyWorkflowToLatestVersion(Collections.singleton(targetMpId), newWorkflow);
228             if (workflowInstances.isEmpty()) {
229               throw new RuntimeException(
230                       String.format("couldn't start workflow '%s' for event %s", publishWorkflowName, targetMpId));
231             }
232           } catch (WorkflowDatabaseException e) {
233             logger.error(String.format("couldn't instantiate workflow '%s'", publishWorkflowName), e);
234           } catch (NotFoundException e) {
235             logger.error(String.format("couldn't find media package while starting workflow workflow '%s'",
236                     publishWorkflowName), e);
237           }
238         }
239       }
240     });
241     updated(cc.getProperties());
242   }

249   @Override
250   public List<LtiJob> listJobs(String seriesId) {
251     final User user = securityService.getUser();
252     final EventSearchQuery query = new EventSearchQuery(securityService.getOrganization().getId(), user)
253             .withSeriesId(StringUtils.trimToNull(seriesId));
254     if (!listAllJobsInSeries) {
255       query.withCreator(user.getName());
256     }
257     try {
258       SearchResult<Event> results = this.searchIndex.getByQuery(query);
259       ZonedDateTime startOfDay = ZonedDateTime.now().truncatedTo(ChronoUnit.DAYS);
260       return Arrays.stream(results.getItems())
261               .map(SearchResultItem::getSource)
262               .filter(e -> ZonedDateTime.parse(e.getCreated()).compareTo(startOfDay) > 0)
263               .map(e -> new LtiJob(e.getTitle(), e.getDisplayableStatus(workflowService.getWorkflowStateMappings())))
264               .collect(Collectors.toList());
265     } catch (SearchIndexException e) {
266       throw new RuntimeException("search index exception", e);
267     }
268   }

270   @Override
271   public void upsertEvent(
272           final LtiFileUpload file,
273           final String captions,
274           final String captionFormat,
275           final String captionLanguage,
276           final String eventId,
277           final String seriesId,
278           final String metadataJson) throws UnauthorizedException, NotFoundException {
279     if (eventId != null) {
280       updateEvent(eventId, metadataJson);
281       return;
282     }
283     if (workflow == null || workflowConfiguration == null) {
284       throw new RuntimeException("No workflow configured, cannot upload");
285     }
286     try {
287       MediaPackage mediaPackage = ingestService.createMediaPackage();
288       if (mediaPackage == null) {
289         throw new RuntimeException("Unable to create media package for event");
290       }
291 
292       if (captions != null) {
293         final MediaPackageElementFlavor captionsFlavor = new MediaPackageElementFlavor(
294             "captions", "source"
295         );
296         final MediaPackageElementBuilder elementBuilder =
297             MediaPackageElementBuilderFactory.newInstance().newElementBuilder();
298         final MediaPackageElement captionsMediaPackage = elementBuilder
299                 .newElement(MediaPackageElement.Type.Track, captionsFlavor);
300         if (!"vtt".equals(captionFormat)) {
301           throw new IllegalArgumentException("Subtitle format must be vtt, but was " + captionFormat);
302         }
303         captionsMediaPackage.setMimeType(mimeType("text", captionFormat));
304 
305         captionsMediaPackage.addTag("lang:" + captionLanguage);
306         mediaPackage.add(captionsMediaPackage);
307         final URI captionsUri = workspace
308                 .put(
309                         mediaPackage.getIdentifier().toString(),
310                         captionsMediaPackage.getIdentifier(),
311                         "captions." + captionFormat,
312                         new ByteArrayInputStream(captions.getBytes(StandardCharsets.UTF_8)));
313         captionsMediaPackage.setURI(captionsUri);
314       }
315 
316       final EventCatalogUIAdapter adapter = getEventCatalogUIAdapter();
317 
318       final DublinCoreMetadataCollection collection = adapter.getRawFields();
319 
320       JSONArray metadataJsonArray = (JSONArray) new JSONParser().parse(metadataJson);
321 
322       JSONArray collectionJsonArray = MetadataJson.extractSingleCollectionfromListJson(metadataJsonArray);
323       MetadataJson.fillCollectionFromJson(collection, collectionJsonArray);
324 
325       replaceField(collection, "isPartOf", seriesId);
326       adapter.storeFields(mediaPackage, collection);
327 
328       AccessControlList accessControlList = null;
329 
330       // If series is set and it's ACL is not empty, use series' ACL as default
331       if (StringUtils.isNotBlank(seriesId)) {
332         accessControlList = seriesService.getSeriesAccessControl(seriesId);
333       }
334 
335       if (accessControlList == null || accessControlList.getEntries().isEmpty()) {
336         accessControlList = new AccessControlList(
337           new AccessControlEntry("ROLE_ADMIN", "write", true),
338           new AccessControlEntry("ROLE_ADMIN", "read", true),
339           new AccessControlEntry("ROLE_USER", "read", true));
340       }
341 
342       this.authorizationService.setAcl(mediaPackage, AclScope.Episode, accessControlList);
343       mediaPackage = ingestService.addTrack(
344             file.getStream(),
345             file.getSourceName(),
346             MediaPackageElements.PRESENTER_SOURCE,
347             mediaPackage
348       );
349 
350       final Map<String, String> configuration = gson.fromJson(workflowConfiguration, Map.class);
351       configuration.put("workflowDefinitionId", workflow);
352       ingestService.ingest(mediaPackage, workflow, configuration);
353     } catch (Exception e) {
354       throw new RuntimeException("unable to create event", e);
355     }
356   }

358   private static Map<String, String> createCopyWorkflowConfig(final String seriesId, final String newMpId) {
359     final Map<String, String> result = new HashMap<>();
360     result.put("numberOfEvents", "1");
361     result.put("noSuffix", "true");

367   @Override
368   public void copyEventToSeries(final String eventId, final String seriesId) {
369     final String workflowId = copyWorkflowId;
370     try {
371       final WorkflowDefinition wfd = workflowService.getWorkflowDefinitionById(workflowId);
372       final Workflows workflows = new Workflows(assetManager, workflowService);
373       final ConfiguredWorkflow workflow
374           = workflow(wfd, createCopyWorkflowConfig(seriesId, UUID.randomUUID().toString()));
375       final List<WorkflowInstance> workflowInstances = workflows
376               .applyWorkflowToLatestVersion(Collections.singleton(eventId), workflow);
377       if (workflowInstances.isEmpty()) {
378         throw new RuntimeException(String.format("Couldn't start workflow '%s' for event %s", workflowId, eventId));
379       }
380     } catch (WorkflowDatabaseException | NotFoundException e) {
381       logger.error("Unable to get workflow definition {}", workflowId, e);
382       throw new RuntimeException(e);
383     }
384   }

386   private EventCatalogUIAdapter getEventCatalogUIAdapter() {
387     final MediaPackageElementFlavor flavor = new MediaPackageElementFlavor("dublincore", "episode");
388     final EventCatalogUIAdapter adapter = catalogUIAdapters.stream()
389         .filter(e -> e.getFlavor().equals(flavor))
390         .findAny()
391         .orElseThrow(() -> new RuntimeException("no adapter found"));
392     return adapter;
393   }

395   private void updateEvent(final String eventId, final String metadata)
396           throws NotFoundException, UnauthorizedException {
397     final EventCatalogUIAdapter adapter = getEventCatalogUIAdapter();
398     final DublinCoreMetadataCollection collection = adapter.getRawFields();
399     try {
400       final MetadataList metadataList = new MetadataList();
401       metadataList.add(adapter, collection);
402       MetadataJson.fillListFromJson(metadataList, (JSONArray) new JSONParser().parse(metadata));
403       this.indexService.updateEventMetadata(eventId, metadataList, searchIndex);
404       republishMetadata(eventId);
405     } catch (IndexServiceException | SearchIndexException | ParseException e) {
406       throw new RuntimeException(e);
407     }
408   }

410   private void republishMetadata(final String eventId) {
411     final String workflowId = "republish-metadata";
412     try {
413       final WorkflowDefinition wfd = workflowService.getWorkflowDefinitionById(workflowId);
414       final Workflows workflows = new Workflows(assetManager, workflowService);
415       final ConfiguredWorkflow workflow = workflow(wfd, Collections.emptyMap());
416       if (workflows.applyWorkflowToLatestVersion(Collections.singleton(eventId), workflow).isEmpty()) {
417         throw new RuntimeException(String.format("couldn't start workflow '%s' for event %s", workflowId, eventId));
418       }
419     } catch (WorkflowDatabaseException | NotFoundException e) {
420       logger.error("Unable to get workflow definition {}", workflowId, e);
421       throw new RuntimeException(e);
422     }
423   }

425   @Override
426   public String getEventMetadata(final String eventId) throws NotFoundException, UnauthorizedException {
427     final Opt<Event> optEvent;
428     try {
429       optEvent = indexService.getEvent(eventId, searchIndex);
430     } catch (SearchIndexException e) {
431       throw new RuntimeException(e);
432     }
433 
434     if (optEvent.isNone()) {
435       throw new NotFoundException("cannot find event with id '" + eventId + "'");
436     }
437 
438     final Event event = optEvent.get();
439 
440     final MetadataList metadataList = new MetadataList();
441     final List<EventCatalogUIAdapter> catalogUIAdapters = this.indexService.getEventCatalogUIAdapters();
442     catalogUIAdapters.remove(this.indexService.getCommonEventCatalogUIAdapter());
443     final MediaPackage mediaPackage;
444     try {
445       mediaPackage = this.indexService.getEventMediapackage(event);
446     } catch (IndexServiceException e) {
447       if (e.getCause() instanceof NotFoundException) {
448         throw new NotFoundException("Cannot retrieve metadata for event with id '" + eventId + "'");
449       } else if (e.getCause() instanceof UnauthorizedException) {
450         throw new UnauthorizedException("Not authorized to access event with id '" + eventId + "'");
451       }
452       throw new RuntimeException(e);
453     }
454     for (EventCatalogUIAdapter catalogUIAdapter : catalogUIAdapters) {
455       metadataList.add(catalogUIAdapter, catalogUIAdapter.getFields(mediaPackage));
456     }
457 
458     final DublinCoreMetadataCollection metadataCollection;
459     try {
460       metadataCollection = EventUtils.getEventMetadata(event, this.indexService.getCommonEventCatalogUIAdapter());
461     } catch (final Exception e) {
462       throw new RuntimeException(e);
463     }
464     metadataList.add(this.indexService.getCommonEventCatalogUIAdapter(), metadataCollection);
465 
466     final String wfState = event.getWorkflowState();
467     if (wfState != null && WorkflowUtil.isActive(WorkflowInstance.WorkflowState.valueOf(wfState))) {
468       metadataList.setLocked(MetadataList.Locked.WORKFLOW_RUNNING);
469     }
470     return new SimpleSerializer().toJson(MetadataJson.listToJson(metadataList, true));
471   }

473   @Override
474   public String getNewEventMetadata() {
475     final MetadataList metadataList = this.indexService.getMetadataListWithAllEventCatalogUIAdapters();
476     final DublinCoreMetadataCollection collection = metadataList
477             .getMetadataByAdapter(this.indexService.getCommonEventCatalogUIAdapter());
478     if (collection != null) {
479       if (collection.getOutputFields().containsKey(DublinCore.PROPERTY_CREATED.getLocalName())) {
480         collection.removeField(collection.getOutputFields().get(DublinCore.PROPERTY_CREATED.getLocalName()));
481       }
482       if (collection.getOutputFields().containsKey("duration")) {
483         collection.removeField(collection.getOutputFields().get("duration"));
484       }
485       if (collection.getOutputFields().containsKey(DublinCore.PROPERTY_IDENTIFIER.getLocalName())) {
486         collection.removeField(collection.getOutputFields().get(DublinCore.PROPERTY_IDENTIFIER.getLocalName()));
487       }
488       if (collection.getOutputFields().containsKey(DublinCore.PROPERTY_SOURCE.getLocalName())) {
489         collection.removeField(collection.getOutputFields().get(DublinCore.PROPERTY_SOURCE.getLocalName()));
490       }
491       if (collection.getOutputFields().containsKey("startDate")) {
492         collection.removeField(collection.getOutputFields().get("startDate"));
493       }
494       if (collection.getOutputFields().containsKey("startTime")) {
495         collection.removeField(collection.getOutputFields().get("startTime"));
496       }
497       if (collection.getOutputFields().containsKey("location")) {
498         collection.removeField(collection.getOutputFields().get("location"));
499       }
500 
501       if (collection.getOutputFields().containsKey(DublinCore.PROPERTY_PUBLISHER.getLocalName())) {
502         final MetadataField publisher
503             = collection.getOutputFields().get(DublinCore.PROPERTY_PUBLISHER.getLocalName());
504         final Map<String, String> users
505             = publisher.getCollection() == null ? new HashMap<>() : publisher.getCollection();
506         final String loggedInUser = this.securityService.getUser().getName();
507         if (!users.containsKey(loggedInUser)) {
508           users.put(loggedInUser, loggedInUser);
509         }
510         publisher.setValue(loggedInUser);
511       }
512 
513       metadataList.add(this.indexService.getCommonEventCatalogUIAdapter(), collection);
514     }
515     return new SimpleSerializer().toJson(MetadataJson.listToJson(metadataList, true));
516   }

534   @Override
535   public void delete(String id) {
536     try {
537       final Opt<Event> event = indexService.getEvent(id, searchIndex);
538       if (event.isNone()) {
539         throw new RuntimeException("Event '" + id + "' not found");
540       }
541       final IndexService.EventRemovalResult eventRemovalResult = indexService.removeEvent(event.get(),
542               retractWorkflowId);
543       if (eventRemovalResult == IndexService.EventRemovalResult.GENERAL_FAILURE) {
544         throw new RuntimeException("Error deleting event: " + eventRemovalResult);
545       }
546     } catch (WorkflowDatabaseException | SearchIndexException | UnauthorizedException | NotFoundException e) {
547       throw new RuntimeException("Error deleting event", e);
548     }
549   }

551   /** OSGi callback if properties file is present */
552   private void updated(Dictionary<String, ?> properties) {
553     // Ensure properties is not null
554     if (properties == null) {
555       throw new IllegalArgumentException("No configuration specified for events endpoint");
556     }
557     String workflowStr = (String) properties.get("workflow");
558     if (workflowStr == null) {
559       throw new IllegalArgumentException("Configuration is missing 'workflow' parameter");
560     }
561     String workflowConfigurationStr = (String) properties.get("workflow-configuration");
562     if (workflowConfigurationStr == null) {
563       throw new IllegalArgumentException("Configuration is missing 'workflow-configuration' parameter");
564     }
565     try {
566       gson.fromJson(workflowConfigurationStr, Map.class);
567       workflowConfiguration = workflowConfigurationStr;
568       workflow = workflowStr;
569       this.retractWorkflowId = Objects.toString(properties.get("retract-workflow-id"), "retract");
570       this.copyWorkflowId = Objects.toString(properties.get("copy-workflow-id"), "copy-event-to-series");
571     } catch (JsonSyntaxException e) {
572       throw new IllegalArgumentException("Invalid JSON specified for workflow configuration");
573     }
574     String listAllJobsInSeriesStr = Objects.toString(properties.get("list-all-jobs-in-series"), "false");
575     this.listAllJobsInSeries = Boolean.parseBoolean(listAllJobsInSeriesStr);
576   }
