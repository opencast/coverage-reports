51    private JpaGroupRoleProvider provider = null;
52    private GroupRoleEndpoint endpoint = null;
53    private static JpaOrganization org1 = new JpaOrganization("org1", "org1", "localhost", 80, "admin", "anon", null);
54    private static JpaOrganization org2 = new JpaOrganization("org2", "org2", "127.0.0.1", 80, "admin", "anon", null);

56    @Before
57    public void setUp() throws Exception {
58      JpaUser adminUser = new JpaUser("admin", "pass1", org1, "Admin", "admin@localhost", "opencast", true,
59              Collections.set(new JpaRole(SecurityConstants.GLOBAL_ADMIN_ROLE, org1)));
60  
61      // Set the security sevice
62      SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
63      EasyMock.expect(securityService.getUser()).andReturn(adminUser).anyTimes();
64      EasyMock.expect(securityService.getOrganization()).andReturn(org1).anyTimes();
65      EasyMock.replay(securityService);
66  
67      provider = new JpaGroupRoleProvider();
68      provider.setSecurityService(securityService);
69      provider.setEntityManagerFactory(newEntityManagerFactory(JpaGroupRoleProvider.PERSISTENCE_UNIT));
70      provider.setDBSessionFactory(getDbSessionFactory());
71      provider.activate(null);
72  
73      endpoint = new GroupRoleEndpoint();
74      endpoint.setJpaGroupRoleProvider(provider);
75    }

77    @Test
78    public void testUpdateGroupNotAllowedAsNonAdminUser() throws UnauthorizedException {
79      JpaGroup group = new JpaGroup("test", org1, "Test", "Test group", Collections.set(
80              new JpaRole(SecurityConstants.GLOBAL_ADMIN_ROLE, org1)));
81      try {
82        provider.addGroup(group);
83        Group loadGroup = provider.loadGroup(group.getGroupId(), group.getOrganization().getId());
84        assertNotNull(loadGroup);
85        assertEquals(loadGroup.getGroupId(), loadGroup.getGroupId());
86      } catch (Exception e) {
87        fail("The group schould be added");
88      }
89  
90      JpaUser user = new JpaUser("user", "pass1", org1, "User", "user@localhost", "opencast", true,
91              Collections.set(new JpaRole("ROLE_USER", org1)));
92  
93      // Set the security sevice
94      SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
95      EasyMock.expect(securityService.getUser()).andReturn(user).anyTimes();
96      EasyMock.expect(securityService.getOrganization()).andReturn(org1).anyTimes();
97      EasyMock.replay(securityService);
98      provider.setSecurityService(securityService);
99  
100     try {
101       // try add ROLE_USER
102       Response updateGroupResponse = endpoint.updateGroup(group.getGroupId(), group.getName(), group.getDescription(),
103               "ROLE_USER, " + SecurityConstants.GLOBAL_ADMIN_ROLE, null);
104       assertNotNull(updateGroupResponse);
105       assertEquals(HttpStatus.SC_FORBIDDEN, updateGroupResponse.getStatus());
106 
107       // try remove ROLE_ADMIN
108       updateGroupResponse = endpoint.updateGroup(group.getGroupId(), group.getName(), group.getDescription(),
109               "ROLE_USER", null);
110       assertNotNull(updateGroupResponse);
111       assertEquals(HttpStatus.SC_FORBIDDEN, updateGroupResponse.getStatus());
112     } catch (NotFoundException e) {
113       fail("The existing group isn't found");
114     }
115   }

145   @Test
146   public void testDuplicateGroupCreation() {
147     Response response = endpoint.createGroup("Test 1", "Test group", "ROLE_ASTRO_101_SPRING_2011_STUDENT", "admin");
148     assertEquals(HttpStatus.SC_CREATED, response.getStatus());
149     response = endpoint.createGroup("Test 1", "Test group 2", "ROLE_ASTRO_101_SPRING_2011_STUDENT", "admin");
150     assertEquals(HttpStatus.SC_CONFLICT, response.getStatus());
151   }
