48  /**
49   * Generic implementation to query {@link org.opencastproject.assetmanager.impl.persistence.SnapshotDto} fields.
50   *
51   * @param <A>
52   *         The business type of the field, e.g. {@link org.opencastproject.assetmanager.api.Version}
53   * @param <B>
54   *         The JPA internal type of the field, e.g. {@link Long}
55   */
56  public abstract class AbstractSnapshotField<A, B extends Comparable> implements Field<A>, EntityPaths {
57    private final ComparableExpressionBase<B> path;
58  
59    /**
60     * Create a new snapshot field.
61     *
62     * @param path a path to a snapshot field
63     */
64    public AbstractSnapshotField(ComparableExpressionBase<B> path) {
65      this.path = path;
66    }
67  
68    /**
69     * Extract database type <code>B</code> from business type <code>A</code>.
70     */
71    protected abstract B extract(A a);
72  
73    @Override public Predicate eq(final A right) {
74      return mkComparison(Ops.EQ, right);
75    }
76  
77    @Override public Predicate eq(PropertyField<A> right) {
78      return mkComparison(Ops.EQ, right);
79    }
80  
81    @Override public Predicate lt(A right) {
82      return mkComparison(Ops.LT, right);
83    }
84  
85    @Override public Predicate lt(PropertyField<A> right) {
86      return mkComparison(Ops.LT, right);
87    }
88  
89    @Override public Predicate le(A right) {
90      return mkComparison(Ops.LOE, right);
91    }
92  
93    @Override public Predicate le(PropertyField<A> right) {
94      return mkComparison(Ops.LOE, right);
95    }
96  
97    @Override public Predicate gt(A right) {
98      return mkComparison(Ops.GT, right);
99    }
100 
101   @Override public Predicate gt(PropertyField<A> right) {
102     return mkComparison(Ops.GT, right);
103   }
104 
105   @Override public Predicate ge(A right) {
106     return mkComparison(Ops.GOE, right);
107   }
108 
109   @Override public Predicate ge(PropertyField<A> right) {
110     return mkComparison(Ops.GOE, right);
111   }
112 
113   @Override public Predicate exists() {
114     return mkPredicate(path.isNotNull());
115   }
116 
117   @Override public Predicate notExists() {
118     return mkPredicate(path.isNull());
119   }
120 
121   @Override public Order desc() {
122     return new AbstractOrder() {
123       @Override public SelectQueryContribution contributeSelect(JPAQueryFactory f) {
124         return SelectQueryContribution.mk().order($(path.desc()));
125       }
126     };
127   }
128 
129   @Override public Order asc() {
130     return new AbstractOrder() {
131       @Override public SelectQueryContribution contributeSelect(JPAQueryFactory f) {
132         return SelectQueryContribution.mk().order($(path.asc()));
133       }
134     };
135   }
136 
137   protected static Predicate mkPredicate(final BooleanExpression where) {
138     return new AbstractPredicate() {
139       @Override public SelectQueryContribution contributeSelect(JPAQueryFactory f) {
140         return SelectQueryContribution.mk().from($Q_SNAPSHOT).where(where);
141       }
142 
143       @Override public DeleteQueryContribution contributeDelete(String owner) {
144         return DeleteQueryContribution.mk().where(new Fn<EntityPath<?>, BooleanExpression>() {
145           @Override public BooleanExpression apply(EntityPath<?> from) {
146             if (from instanceof QSnapshotDto) {
147               return where;
148             } else if (from instanceof QPropertyDto) {
149               return new JPASubQuery().from(Q_SNAPSHOT)
150                   .where(Q_SNAPSHOT.mediaPackageId.eq(Q_PROPERTY.mediaPackageId).and(where))
151                   .exists();
152             } else {
153               throw new RuntimeException("BUG");
154             }
155           }
156         });
157       }
158     };
159   }
160 
161   /**
162    * Create a predicate for comparisons with a constant value.
163    */
164   private Predicate mkComparison(final Operator<? super Boolean> op, A right) {
165     return mkPredicate(BooleanOperation.create(op, path, ConstantImpl.create(extract(right))));
166   }
167 
168   /**
169    * Create a predicate for comparisons with a property field.
170    */
171   private Predicate mkComparison(final Operator<? super Boolean> op, final PropertyField<A> right) {
172     return mkPredicate(PropertyPredicates.mkWhereSelect(
173         right.name(),
174         new Function<QPropertyDto, Opt<BooleanExpression>>() {
175           @Override public Opt<BooleanExpression> apply(QPropertyDto qPropertyDto) {
176             return Opt.some(BooleanOperation.create(op, path, RuntimeTypes.convert(right).getPath(qPropertyDto)));
177           }
178         }));
179   }
180 }
