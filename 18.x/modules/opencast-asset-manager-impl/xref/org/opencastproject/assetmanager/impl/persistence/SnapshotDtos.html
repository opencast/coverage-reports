21  package org.opencastproject.assetmanager.impl.persistence;
22  
23  import org.opencastproject.assetmanager.api.Availability;
24  
25  import com.entwinemedia.fn.Fn;
26  import com.mysema.query.Tuple;
27  import com.mysema.query.jpa.impl.JPAQuery;
28  import com.mysema.query.types.Expression;
29  
30  import javax.persistence.EntityManager;
31  
32  /**
33   * Collection of non-JPA DTOs to tuple up the {@link SnapshotDto} JPA entity with some additional joined data.
34   */
35  public final class SnapshotDtos {
36    private SnapshotDtos() {
37    }
38  
39    /**
40     * Create base query for a {@link SnapshotDto} query.
41     */
42    public static JPAQuery baseQuery(EntityManager em) {
43      final QSnapshotDto snapshotDto = QSnapshotDto.snapshotDto;
44      return new JPAQuery(em, Database.TEMPLATES)
45              .from(snapshotDto);
46    }
47  
48    public static class Medium {
49      private final SnapshotDto snapshotDto;
50      private final String availability;
51      private final String storageId;
52      private final String organizationId;
53      private final String owner;
54  
55      public Medium(
56          SnapshotDto snapshotDto,
57          Availability availability,
58          String storageId,
59          String organizationId,
60          String owner
61      ) {
62        this.snapshotDto = snapshotDto;
63        this.availability = availability.name();
64        this.storageId = storageId;
65        this.organizationId = organizationId;
66        this.owner = owner;
67      }
68  
69      public SnapshotDto getSnapshotDto() {
70        return snapshotDto;
71      }
72  
73      public Availability getAvailability() {
74        return Availability.valueOf(availability);
75      }
76  
77      public String getStorageId() {
78        return storageId;
79      }
80  
81      public String getOrganizationId() {
82        return organizationId;
83      }
84  
85      public String getOwner() {
86        return owner;
87      }
88  
89  
90  
91      public static final Fn<Tuple, Medium> fromTuple = new Fn<Tuple, Medium>() {
92        @Override public Medium apply(Tuple result) {
93          return new Medium(
94                  result.get(QSnapshotDto.snapshotDto),
95                  Availability.valueOf(result
96                          .get(QSnapshotDto.snapshotDto.availability)),
97                  result.get(QSnapshotDto.snapshotDto.storageId),
98                  result.get(QSnapshotDto.snapshotDto.organizationId),
99                  result.get(QSnapshotDto.snapshotDto.owner));
100       }
101     };
102 
103     /**
104      * Parameter for query execution methods like
105      * {@link com.mysema.query.jpa.impl.JPAQuery#singleResult(com.mysema.query.types.Expression[])} or
106      * {@link com.mysema.query.jpa.impl.JPAQuery#list(Expression[])}.
107      */
108     public static final Expression<?>[] select = new Expression[] {
109         QSnapshotDto.snapshotDto,
110         QSnapshotDto.snapshotDto.storageId,
111         QSnapshotDto.snapshotDto.availability,
112         QSnapshotDto.snapshotDto.organizationId
113     };
114   }
115 }
