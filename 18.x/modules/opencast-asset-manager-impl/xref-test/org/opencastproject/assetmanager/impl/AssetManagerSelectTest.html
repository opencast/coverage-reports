78    public void testSelectSnapshots() throws Exception {
79      final MediaPackage mp = mkMediaPackage();
80      final MediaPackageElement mpe = mkCatalog();
81      mp.add(mpe);
82      final Snapshot snapshot = am.takeSnapshot(OWNER, mp);
83      final Version version = snapshot.getVersion();
84      assertThat("Archival date should not be in the future", snapshot.getArchivalDate(), lessThanOrEqualTo(new Date()));
85      assertThat("Snapshot should be available", snapshot.getAvailability(), equalTo(Availability.ONLINE));
86      assertThat("Snapshot should belong to the default organization",
87                 snapshot.getOrganizationId(), equalTo(DefaultOrganization.DEFAULT_ORGANIZATION_ID));
88      final Optional<Asset> asset = am.getAsset(version, mp.getIdentifier().toString(), mpe.getIdentifier());
89      assertTrue("Asset should be found", asset.isPresent());
90      assertEquals("Media package element part of the asset ID should equal the element's ID",
91                   mpe.getIdentifier(), asset.get().getId().getMediaPackageElementId());
92      assertEquals("Mime types should equal", mpe.getMimeType(), asset.get().getMimeType().get());
93      assertFalse("Asset should not be found", am.getAsset(version, "id", "id").isPresent());
94      // try to find the catalog of the media package by checksum
95      final MediaPackage mpCopy = MediaPackageSupport.copy(mp);
96      am.calcChecksumsForMediaPackageElements(AssetManagerImpl.assetsOnly(mpCopy));
97      assertEquals("Media package should be set up with a single catalog", 1, mpCopy.getCatalogs().length);
98      final String checksum = mpCopy.getCatalogs()[0].getChecksum().toString();
99      assertTrue("Media package element should be retrievable by checksum", am.getDatabase().findAssetByChecksum(checksum).isPresent());
100     // issue some queries
101     {
102       logger.info("Run a failing query");
103       assertEquals("The result should not contain any records", 0,
104                    q.select(q.snapshot())
105                            .where(q.mediaPackageId(mp.getIdentifier().toString()).and(q.availability(Availability.ONLINE)))
106                            .where(q.mediaPackageId("12"))
107                            .run().getSize());
108     }
109     {
110       logger.info("Run query to find snapshot");
111       final AResult r = q.select(q.snapshot())
112               .where(q.mediaPackageId(mp.getIdentifier().toString()).and(q.availability(Availability.ONLINE)))
113               .run();
114       assertEquals("The result set should contain exactly one record", 1, r.getSize());
115       assertEquals("The media package IDs should be equal", mp.getIdentifier().toString(), r.getRecords().stream().findFirst().get().getMediaPackageId());
116       assertTrue("The snapshot should be contained in the record", r.getRecords().stream().findFirst().get().getSnapshot().isPresent());
117       assertEquals("The media package IDs should be equal", mp.getIdentifier(), r.getRecords().stream().findFirst().get().getSnapshot().get().getMediaPackage().getIdentifier());
118     }
119     {
120       final AResult r = q.select().where(q.mediaPackageId(mp.getIdentifier().toString()).and(q.availability(Availability.ONLINE))).run();
121       assertEquals("The result should contain one record", 1, r.getSize());
122       assertTrue("The result should not contain a snapshot", r.getRecords().stream().findFirst().get().getSnapshot().isEmpty());
123     }
124   }
