22  package org.opencastproject.serviceregistry.impl.endpoint;
23  
24  import static javax.servlet.http.HttpServletResponse.SC_OK;
25  
26  import org.opencastproject.rest.OsgiAbstractJobProducerEndpoint;
27  import org.opencastproject.serviceregistry.api.NopService;
28  import org.opencastproject.serviceregistry.api.ServiceRegistry;
29  import org.opencastproject.serviceregistry.impl.NopServiceImpl;
30  import org.opencastproject.util.RestUtil;
31  import org.opencastproject.util.doc.rest.RestQuery;
32  import org.opencastproject.util.doc.rest.RestResponse;
33  import org.opencastproject.util.doc.rest.RestService;
34  
35  import org.osgi.service.component.annotations.Component;
36  import org.osgi.service.component.annotations.Reference;
37  import org.osgi.service.jaxrs.whiteboard.propertytypes.JaxrsResource;
38  
39  import javax.ws.rs.GET;
40  import javax.ws.rs.Path;
41  import javax.ws.rs.core.Response;
42  
43  @Path("/nop")
44  @RestService(
45          name = "nopservice",
46          title = "Nop Service",
47          notes = {},
48          abstractText = "No operation service. Creates empty jobs for testing purposes.")
49  @Component(
50      immediate = true,
51      service = NopServiceEndpoint.class,
52      property = {
53          "service.description=No operation service REST endpoint",
54          "opencast.service.type=org.opencastproject.nop",
55          "opencast.service.path=/nop",
56          "opencast.service.jobproducer=true"
57      }
58  )
59  @JaxrsResource
60  public class NopServiceEndpoint extends OsgiAbstractJobProducerEndpoint<NopServiceImpl> {
61    @GET
62    @Path("nop")
63    @RestQuery(
64            name = "nop",
65            description = "Create an empty job for testing purposes.",
66            returnDescription = "The service statistics.",
67            responses = { @RestResponse(responseCode = SC_OK, description = "OK") })
68    public Response nop() {
69      return RestUtil.R.ok(getSvc().nop());
70    }
71  
72    @Reference
73    @Override
74    public void setServiceRegistry(ServiceRegistry serviceRegistry) {
75      super.setServiceRegistry(serviceRegistry);
76    }
77  
78    @Reference
79    public void setService(NopService nopService) {
80      super.setService((NopServiceImpl)nopService);
81    }
82  
83  }
