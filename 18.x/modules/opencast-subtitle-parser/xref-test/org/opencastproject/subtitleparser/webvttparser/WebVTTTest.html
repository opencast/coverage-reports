22  package org.opencastproject.subtitleparser.webvttparser;
23  
24  import org.opencastproject.subtitleparser.SubtitleParsingException;
25  
26  import org.junit.Assert;
27  import org.junit.Before;
28  import org.junit.Test;
29  
30  import java.io.File;
31  import java.io.FileInputStream;
32  import java.io.FileOutputStream;
33  import java.io.IOException;
34  import java.net.URISyntaxException;
35  
36  /**
37   * Test the subtitle parsing
38   */
39  public class WebVTTTest {
40    protected String inputFilePath;
41    protected String inputFileWithBom;
42    protected String outputFilePath;
43  
44    public WebVTTTest() throws URISyntaxException {
45      inputFilePath = new File(getClass().getResource("/testresources/example.vtt").toURI()).getAbsolutePath();
46      inputFileWithBom = new File(getClass().getResource("/testresources/example-bom.vtt").toURI()).getAbsolutePath();
47      outputFilePath = new File("target/testoutput/output.vtt").getAbsolutePath();
48    }
49  
50    @Before
51    public void setUp() {
52      if (new File(outputFilePath).exists())  {
53        new File(outputFilePath).delete();
54      } else if (!new File(outputFilePath).getParentFile().exists()) {
55        new File(outputFilePath).getParentFile().mkdir();
56      }
57    }
58  
59    @Test
60    public void parseWithoutException() throws IOException, SubtitleParsingException {
61      WebVTTParser parser = new WebVTTParser();
62      parser.parse(new FileInputStream(inputFilePath));
63    }
64  
65    @Test
66    public void parseBomWithoutException() throws IOException, SubtitleParsingException {
67      WebVTTParser parser = new WebVTTParser();
68      parser.parse(new FileInputStream(inputFileWithBom));
69    }
70  
71    @Test
72    public void parseCorrectly() throws IOException, SubtitleParsingException {
73      WebVTTParser parser = new WebVTTParser();
74      WebVTTSubtitle subtitle = parser.parse(new FileInputStream(inputFilePath));
75  
76      assertExampleVtt(subtitle);
77    }
78  
79    @Test
80    public void parseBomCorrectly() throws IOException, SubtitleParsingException {
81      WebVTTParser parser = new WebVTTParser();
82      WebVTTSubtitle subtitle = parser.parse(new FileInputStream(inputFileWithBom));
83  
84      assertExampleVtt(subtitle);
85    }
86  
87    @Test
88    public void writeCorrectly() throws IOException, SubtitleParsingException {
89      WebVTTParser parser = new WebVTTParser();
90      WebVTTSubtitle subtitle = parser.parse(new FileInputStream(inputFilePath));
91  
92      WebVTTWriter writer = new WebVTTWriter();
93      writer.write(subtitle, new FileOutputStream(outputFilePath));
94  
95      subtitle = parser.parse(new FileInputStream(inputFilePath));
96  
97      assertExampleVtt(subtitle);
98    }
99  
100   private void assertExampleVtt(WebVTTSubtitle subtitle) {
101     Assert.assertEquals(subtitle.getRegions().size(), 2);
102     Assert.assertEquals(subtitle.getStyle().size(), 1);
103     Assert.assertEquals(subtitle.getCues().size(), 4);
104 
105     WebVTTSubtitleCue cue = subtitle.getCues().get(0);
106     Assert.assertNotNull(cue.getCueSettingsList());
107     Assert.assertNotNull(cue.getId());
108     Assert.assertEquals(cue.getLines().size(), 1);
109     Assert.assertEquals(cue.getStartTime(), 0);
110   }
111 
112 }
