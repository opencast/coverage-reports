143   @Test
144   public void testTwoShapeLayoutCompose4to3And16to9() {
145     final Dimension canvas16To9 = dimension(1600, 900);
146     final Dimension upper4To3 = dimension(400, 300);
147     final Dimension lower16To9 = dimension(960, 540);
148     final double upperCoverage = 0.3;
149     final double lowerCoverage = 0.7;
150     final TwoShapeLayout layout = LayoutManager.twoShapeLayout(
151             canvas16To9,
152             upper4To3,
153             lower16To9,
154             new TwoShapeLayouts.TwoShapeLayoutSpec(new HorizontalCoverageLayoutSpec(anchorOffset(Anchors.TOP_LEFT,
155                                                                                                  Anchors.TOP_LEFT, 0, 0), upperCoverage), new HorizontalCoverageLayoutSpec(anchorOffset(
156                     Anchors.BOTTOM_RIGHT, Anchors.BOTTOM_RIGHT, 0, 0), lowerCoverage)));
157     assertEquals(dimension(1120, 630), layout.getLower().getDimension());
158     assertEquals(ASPECT_RATIO_16_9, aspectRatio(lower16To9), TOLERANCE);
159     assertEquals(ASPECT_RATIO_16_9, aspectRatio(layout.getLower().getDimension()), TOLERANCE);
160     assertEquals(dimension(480, 360), layout.getUpper().getDimension());
161     assertEquals(ASPECT_RATIO_4_3, aspectRatio(upper4To3), TOLERANCE);
162     assertEquals(ASPECT_RATIO_4_3, aspectRatio(layout.getUpper().getDimension()), TOLERANCE);
163   }

234   @Test
235   public void testWatermarkLayout() {
236     final List<Tuple<Offset, String>> fixtures = list(
237             // top left
238             tuple(offset(20, 20), "{\"anchorOffset\":{\"referring\":{\"left\":0.0,\"top\":0.0},\"offset\":{\"y\":20,\"x\":20},\"reference\":{\"left\":0.0,\"top\":0.0}}}"),
239             // top right
240             tuple(offset(1340, 20), "{\"anchorOffset\":{\"referring\":{\"left\":1.0,\"top\":0.0},\"offset\":{\"y\":20,\"x\":-20},\"reference\":{\"left\":1.0,\"top\":0.0}}}"),
241             // bottom left
242             tuple(offset(20, 340), "{\"anchorOffset\":{\"referring\":{\"left\":0.0,\"top\":1.0},\"offset\":{\"y\":-20,\"x\":20},\"reference\":{\"left\":0.0,\"top\":1.0}}}"),
243             // bottom right
244             tuple(offset(1340, 340), "{\"anchorOffset\":{\"referring\":{\"left\":1.0,\"top\":1.0},\"offset\":{\"y\":-20,\"x\":-20},\"reference\":{\"left\":1.0,\"top\":1.0}}}"));
245     for (final Tuple<Offset, String> fixture : fixtures) {
246       final AbsolutePositionLayoutSpec spec = Serializer.absolutePositionLayoutSpec(JsonObj.jsonObj(fixture.getB()));
247       final MultiShapeLayout layout = LayoutManager.absoluteMultiShapeLayout(
248               Dimension.dimension(1900, 1080),
249               list(tuple(Dimension.dimension(540, 720), spec)));
250       assertEquals(1, layout.getShapes().size());
251       assertEquals(fixture.getA(), layout.getShapes().get(0).getOffset());
252     }
