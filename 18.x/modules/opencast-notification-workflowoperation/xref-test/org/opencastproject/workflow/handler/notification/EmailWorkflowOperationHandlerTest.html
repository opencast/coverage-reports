71    @Before
72    public void setUp() throws Exception {
73      MediaPackageBuilder builder = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder();
74  
75      uriMP = EmailWorkflowOperationHandlerTest.class.getResource("/email_mediapackage.xml").toURI();
76      mp = builder.loadFromXml(uriMP.toURL().openStream());
77  
78      operationHandler = new EmailWorkflowOperationHandler();
79  
80      EmailTemplateService emailTemplateService = EasyMock.createMock(EmailTemplateService.class);
81      EasyMock.expect(emailTemplateService.applyTemplate("DCE_workflow_null_body",
82              "This is the media package: ${mediaPackage.identifier}", workflowInstance))
83              .andReturn("This is the media package: 3e7bb56d-2fcc-4efe-9f0e-d6e56422f557");
84      EasyMock.expect(emailTemplateService.applyTemplate("DCE_workflow_null_body_html",
85              "<h1>This is the media package: ${mediaPackage.identifier}</h1>", workflowInstance))
86              .andReturn("<h1>This is the media package: 3e7bb56d-2fcc-4efe-9f0e-d6e56422f557</h1>");
87      EasyMock.expect(emailTemplateService.applyTemplate("template1", null, workflowInstance))
88              .andReturn("This is the media package: 3e7bb56d-2fcc-4efe-9f0e-d6e56422f557");
89      EasyMock.expect(emailTemplateService.applyTemplate("template1html", null, workflowInstance))
90          .andReturn("<h1>This is the media package: 3e7bb56d-2fcc-4efe-9f0e-d6e56422f557</h1>");
91      EasyMock.expect(emailTemplateService.applyTemplate("templateNotFound", null, workflowInstance))
92              .andReturn("TEMPLATE NOT FOUND!");
93      EasyMock.replay(emailTemplateService);
94      operationHandler.setEmailTemplateService(emailTemplateService);
95  
96      SmtpService smtpService = EasyMock.createMock(SmtpService.class);
97      capturedTo = Capture.newInstance();
98      capturedCC = Capture.newInstance();
99      capturedBCC = Capture.newInstance();
100     capturedSubject = Capture.newInstance();
101     capturedBodyText = Capture.newInstance();
102     capturedBodyHTML = Capture.newInstance();
103     smtpService.send(EasyMock.capture(capturedTo), EasyMock.capture(capturedCC), EasyMock.capture(capturedBCC),
104             EasyMock.capture(capturedSubject), EasyMock.capture(capturedBodyText), EasyMock.capture(capturedBodyHTML));
105     EasyMock.expectLastCall().once();
106     EasyMock.replay(smtpService);
107     operationHandler.setSmtpService(smtpService);
108 
109     User user = EasyMock.createNiceMock(User.class);
110     EasyMock.expect(user.getEmail()).andReturn(DEFAULT_TO).anyTimes();
111     User emailUser = EasyMock.createNiceMock(User.class);
112     EasyMock.expect(emailUser.getEmail()).andReturn(DEFAULT_TO).anyTimes();
113     User noEmailUser = EasyMock.createNiceMock(User.class);
114     EasyMock.expect(noEmailUser.getEmail()).andReturn(null).anyTimes();
115     UserDirectoryService userDirectoryService = EasyMock.createNiceMock(UserDirectoryService.class);
116     EasyMock.expect(userDirectoryService.loadUser(USER_NAME)).andReturn(user).anyTimes();
117     EasyMock.expect(userDirectoryService.loadUser(USER_NAME_EMAIL)).andReturn(emailUser).anyTimes();
118     EasyMock.expect(userDirectoryService.loadUser(USER_NAME_NO_EMAIL)).andReturn(noEmailUser).anyTimes();
119     EasyMock.replay(user, emailUser, noEmailUser, userDirectoryService);
120     operationHandler.setUserDirectoryService(userDirectoryService);
121 
122     workflowInstance.setId(1);
123     workflowInstance.setState(WorkflowState.RUNNING);
124     workflowInstance.setTemplate("DCE-workflow");
125     workflowInstance.setMediaPackage(mp);
126     WorkflowOperationInstance failedOperation1 = new WorkflowOperationInstance("operation1",
127             OperationState.FAILED);
128     failedOperation1.setFailOnError(true);
129     WorkflowOperationInstance failedOperation2 = new WorkflowOperationInstance("operation2",
130             OperationState.FAILED);
131     failedOperation2.setFailOnError(false);
132     operation = new WorkflowOperationInstance("email", OperationState.RUNNING);
133     List<WorkflowOperationInstance> operationList = new ArrayList<WorkflowOperationInstance>();
134     operationList.add(failedOperation1);
135     operationList.add(failedOperation2);
136     operationList.add(operation);
137     workflowInstance.setOperations(operationList);
138   }

140   @Test
141   public void testDefaultBody() throws Exception {
142     operation.setConfiguration(EmailWorkflowOperationHandler.TO_PROPERTY, DEFAULT_TO);
143     operation.setConfiguration(EmailWorkflowOperationHandler.CC_PROPERTY, DEFAULT_CC);
144     operation.setConfiguration(EmailWorkflowOperationHandler.BCC_PROPERTY, DEFAULT_BCC);
145     operation.setConfiguration(EmailWorkflowOperationHandler.SUBJECT_PROPERTY, DEFAULT_SUBJECT);
146 
147     WorkflowOperationResult result = operationHandler.start(workflowInstance, null);
148 
149     Assert.assertEquals(Action.CONTINUE, result.getAction());
150     Assert.assertArrayEquals(new String[] {DEFAULT_TO}, capturedTo.getValue());
151     Assert.assertArrayEquals(new String[] {DEFAULT_CC}, capturedCC.getValue());
152     Assert.assertArrayEquals(new String[] {DEFAULT_BCC}, capturedBCC.getValue());
153     Assert.assertEquals(DEFAULT_SUBJECT, capturedSubject.getValue());
154     Assert.assertEquals("Test Media Package (3e7bb56d-2fcc-4efe-9f0e-d6e56422f557)", capturedBodyText.getValue());
155   }
