72  /**
73   * The workflow definition for converting a smil containing cut marks into a legal smil for cutting
74   */
75  @Component(
76      immediate = true,
77      service = WorkflowOperationHandler.class,
78      property = {
79          "service.description=Cut Marks To Smil Operation Handler",
80          "workflow.operation=cut-marks-to-smil"
81      }
82  )
83  public class CutMarksToSmilWorkflowOperationHandler extends AbstractWorkflowOperationHandler {
84  
85    /** Workflow configuration keys */
86    private static final String SOURCE_MEDIA_FLAVORS = "source-media-flavors";
87    private static final String SOURCE_JSON_FLAVOR = "source-json-flavor";
88  
89    private static final String TARGET_SMIL_FLAVOR = "target-smil-flavor";
90    private static final String TARGET_TAGS = "target-tags";
91  
92    private static final String CUTTING_SMIL_NAME = "prepared_cutting_smil";
93  
94    /** The logging facility */
95    private static final Logger logger = LoggerFactory.getLogger(CutMarksToSmilWorkflowOperationHandler.class);

122   /** JSON Parser */
123   private static final Gson gson = new Gson();
124   private static final Type timesListType = new TypeToken<List<Times>>() { }.getType();
125 
126   /** Stores information read from JSON */
127   static class Times {
128     private Long begin;
129     private Long duration;
130   }
131 
132   /**
133    * {@inheritDoc}
134    *
135    * @see org.opencastproject.workflow.api.WorkflowOperationHandler#start(
136    *        org.opencastproject.workflow.api.WorkflowInstance,
137    *        JobContext)
138    */
139   @Override
140   public WorkflowOperationResult start(final WorkflowInstance workflowInstance, JobContext context)
141           throws WorkflowOperationException {
142     logger.info("Running cut marks to smil workflow operation on workflow {}", workflowInstance.getId());
143 
144     WorkflowOperationInstance operation = workflowInstance.getCurrentOperation();
145     final MediaPackage mediaPackage = (MediaPackage) workflowInstance.getMediaPackage().clone();
146 
147     // Read config options
148     final MediaPackageElementFlavor jsonFlavor = MediaPackageElementFlavor.parseFlavor(
149             getConfig(operation, SOURCE_JSON_FLAVOR));
150     final MediaPackageElementFlavor targetSmilFlavor = MediaPackageElementFlavor.parseFlavor(
151             getConfig(operation, TARGET_SMIL_FLAVOR));
152 
153     String flavorNames = operation.getConfiguration(SOURCE_MEDIA_FLAVORS);
154     final List<MediaPackageElementFlavor> flavors = new ArrayList<>();
155     for (String flavorName : asList(flavorNames)) {
156       flavors.add(MediaPackageElementFlavor.parseFlavor(flavorName));
157     }
158 
159     // Target tags
160     String targetTagsOption = StringUtils.trimToNull(operation.getConfiguration(TARGET_TAGS));
161     List<String> targetTags = asList(targetTagsOption);
162 
163     // Is there a catalog?
164     MediaPackageElement[] cutMarksElements = mediaPackage.getAttachments(jsonFlavor);
165     if (cutMarksElements.length < 1) {
166       logger.debug("No cut marks found as attachment. Falling back to catalogs…");
167       cutMarksElements = mediaPackage.getCatalogs(jsonFlavor);
168     }
169     if (cutMarksElements.length < 1) {
170       logger.warn("No cut marks with source flavor {} found. Skipping…", jsonFlavor);
171       return createResult(mediaPackage, WorkflowOperationResult.Action.SKIP);
172     } else if (cutMarksElements.length > 1) {
173       // Remember Highlander? There can be only one!
174       throw new WorkflowOperationException(String.format(
175           "More than one cut marks element with source flavor %s found! Make sure there is only one.", jsonFlavor));
176     }
177 
178     // Parse JSON
179     List<Times> cutMarks;
180     MediaPackageElement jsonWithTimes = cutMarksElements[0];
181     try (BufferedReader reader = new BufferedReader(new FileReader(getMediaPackageElementPath(jsonWithTimes)))) {
182       cutMarks = gson.fromJson(reader, timesListType);
183     } catch (Exception e) {
184       throw new WorkflowOperationException("Could not read JSON", e);
185     }
186 
187     // If the catalog was empty, give up
188     if (cutMarks.size() < 1) {
189       logger.warn("Source JSON did not contain any timestamps! Skipping...");
190       return createResult(mediaPackage, WorkflowOperationResult.Action.SKIP);
191     }
192 
193     // Check parsing results
194     for (Times entry : cutMarks) {
195       logger.debug("Entry begin={}, duration={}", entry.begin, entry.duration);
196       if (entry.begin < 0 || entry.duration < 0) {
197         throw new WorkflowOperationException("Times may not be negative.");
198       }
199     }
200 
201     // Get video tracks
202     logger.info("Get tracks from media package");
203     ArrayList<Track> tracksFromFlavors = new ArrayList<>();
204     for (MediaPackageElementFlavor flavor : flavors) {
205       logger.debug("Trying to get tracks with flavor {}", flavor);
206       TrackSelector trackSelector = new TrackSelector();
207       trackSelector.addFlavor(flavor);
208       Collection<Track> tracks = trackSelector.select(mediaPackage, false);
209       logger.debug("Found {} tracks with flavor {}", tracks.size(), flavor);
210       tracksFromFlavors.addAll(tracks);
211     }
212 
213     // Are there actually any tracks?
214     if (tracksFromFlavors.isEmpty()) {
215       logger.warn("No track with given flavors. Skipping…");
216       return createResult(mediaPackage, WorkflowOperationResult.Action.SKIP);
217     }
218 
219     // Check for cut marks that would lead to errors with the given tracks and remove them
220     // Possible TODO: Instead of removing, only apply cut marks to tracks with a long enough duration?
221     // Get the shortest duration of all tracks
222     long shortestDuration = Long.MAX_VALUE;
223     for (Track track : tracksFromFlavors) {
224       if (track.getDuration() < shortestDuration) {
225         shortestDuration = track.getDuration();
226       }
227     }
228     // Remove all timestamps that begin after the shortest duration
229     ListIterator<Times> iter = cutMarks.listIterator();
230     while (iter.hasNext()) {
231       long begin = iter.next().begin;
232       if (begin > shortestDuration) {
233         logger.info("Skipped mark with begin: {}, ", begin);
234         iter.remove();
235       }
236     }
237     // If the timestamp list is now empty, give up
238     if (cutMarks.size() < 1) {
239       logger.warn("No timestamps are valid for the given tracks! Skipping...");
240       return createResult(mediaPackage, WorkflowOperationResult.Action.SKIP);
241     }
242 
243     // Create the new SMIL document
244     Smil smil;
245     try {
246       SmilResponse smilResponse = smilService.createNewSmil(mediaPackage);
247 
248       logger.info("Start adding tracks");
249       for (Times mark : cutMarks) {
250         smilResponse = smilService.addParallel(smilResponse.getSmil());
251         SmilMediaContainer par = (SmilMediaContainer) smilResponse.getEntity();
252         logger.debug("Segment begin: {}; Segment duration: {}", mark.begin, mark.duration);
253         // Add tracks (as array) to par
254         smilResponse = smilService
255                 .addClips(smilResponse.getSmil(),
256                         par.getId(),
257                         tracksFromFlavors.toArray(new Track[0]),
258                         mark.begin,
259                         mark.duration);
260       }
261 
262       smil = smilResponse.getSmil();
263       logger.info("Done adding tracks");
264     } catch (SmilException e) {
265       throw new WorkflowOperationException("Failed to create SMIL Catalog", e);
266     }
267 
268     // Put new SMIL into workspace and add it to media package
269     try (InputStream is = IOUtils.toInputStream(smil.toXML(), "UTF-8")) {
270       URI smilURI = workspace.put(mediaPackage.getIdentifier().toString(), smil.getId(), CUTTING_SMIL_NAME, is);
271       MediaPackageElementBuilder mpeBuilder = MediaPackageElementBuilderFactory.newInstance().newElementBuilder();
272       Catalog catalog = (Catalog) mpeBuilder
273               .elementFromURI(smilURI, MediaPackageElement.Type.Catalog, targetSmilFlavor);
274       catalog.setIdentifier(smil.getId());
275       for (String tag : targetTags) {
276         catalog.addTag(tag);
277       }
278       mediaPackage.add(catalog);
279     } catch (JAXBException | SAXException | IOException e) {
280       throw new WorkflowOperationException("Failed to parse crated SMIL Catalog", e);
281     }
282 
283     final WorkflowOperationResult result = createResult(mediaPackage, WorkflowOperationResult.Action.CONTINUE);
284     logger.debug("Cut marks to smil operation completed");
285     return result;
286   }
287 
288   /**
289    * Returns the absolute path for a given MediaPackageElement
290    * @param mpe
291    *          The MediaPackageElement we want to know the absolute path for
292    * @return
293    *          The absolute path
294    * @throws WorkflowOperationException
295    */
296   private String getMediaPackageElementPath(MediaPackageElement mpe) throws WorkflowOperationException {
297     File mediaFile;
298     try {
299       mediaFile = workspace.get(mpe.getURI());
300     } catch (NotFoundException | IOException e) {
301       throw new WorkflowOperationException(
302               "Error finding the media file in the workspace", e);
303     }
304 
305     return mediaFile.getAbsolutePath();
306   }
