49    @Before
50    public void setUp() throws Exception {
51      // Set up a mock security service that always returns "me" as the current user
52      DefaultOrganization organization = new DefaultOrganization();
53      JaxbRole role = new JaxbRole(DefaultOrganization.DEFAULT_ORGANIZATION_ANONYMOUS, organization, "");
54      HashSet<JaxbRole> roles = new HashSet<JaxbRole>();
55      roles.add(role);
56      User me = new JaxbUser("me", "test", organization, roles);
57      SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
58      EasyMock.expect(securityService.getUser()).andReturn(me).anyTimes();
59      EasyMock.replay(securityService);
60  
61      // Set up the annotation service
62      annotationService = new AnnotationServiceJpaImpl();
63      annotationService.setEntityManagerFactory(newEntityManagerFactory(AnnotationServiceJpaImpl.PERSISTENCE_UNIT));
64      annotationService.setDBSessionFactory(getDbSessionFactory());
65      annotationService.setSecurityService(securityService);
66      annotationService.activate();
67    }

69    @Test
70    public void testAnnotationPersistence() throws Exception {
71      // Add an annotation
72      AnnotationImpl a = new AnnotationImpl();
73      a.setInpoint(10);
74      a.setOutpoint(100);
75      a.setMediapackageId("mp123");
76      a.setSessionId("session123");
77      a.setType("ugc");
78      a.setValue("This is some user generated content");
79      annotationService.addAnnotation(a);
80  
81      // Ensure that by persisting the annotation, we now have an ID
82      Assert.assertNotNull(a.getAnnotationId());
83  
84      // Ensure that the annotation was saved and retrieved properly
85      Annotation a1FromDb = annotationService.getAnnotation(a.getAnnotationId());
86      Assert.assertEquals(a.getType(), a1FromDb.getType());
87      Assert.assertEquals(a.getValue(), a1FromDb.getValue());
88      Assert.assertEquals(a.getMediapackageId(), a1FromDb.getMediapackageId());
89      Assert.assertEquals(a.getSessionId(), a1FromDb.getSessionId());
90      Assert.assertEquals(a.getUserId(), a1FromDb.getUserId());
91      Assert.assertEquals(a.getUserId(), annotationService.securityService.getUser().getUsername());
92    }

119   @Test
120   public void testChangeAnnotation() throws Exception {
121     // Add an annotation
122     AnnotationImpl a1 = new AnnotationImpl();
123     a1.setInpoint(10);
124     a1.setOutpoint(100);
125     a1.setMediapackageId("mp123");
126     a1.setSessionId("session123");
127     a1.setType("ugc");
128     a1.setValue("This is some user generated content");
129     annotationService.addAnnotation(a1);
130     Long a1Id = a1.getAnnotationId();
131 
132     AnnotationImpl a2 = new AnnotationImpl();
133     a2.setAnnotationId(a1Id);
134     a2.setValue("This is some other user generated content");
135     Annotation a3 = annotationService.changeAnnotation(a2);
136     Assert.assertNotNull(a3);
137 
138     Long a2Id = a2.getAnnotationId();
139     Assert.assertEquals(a1Id, a2Id);
140 
141     Annotation a1FromDb = annotationService.getAnnotation(a1Id);
142     Assert.assertEquals(a1.getAnnotationId(), a1FromDb.getAnnotationId());
143     Assert.assertEquals(a2.getValue(), a1FromDb.getValue());
144     Assert.assertEquals(a1.getSessionId(), a1FromDb.getSessionId());
145     Assert.assertEquals(a1.getUserId(), a1FromDb.getUserId());
146     Assert.assertEquals(a1.getUserId(), annotationService.securityService.getUser().getUsername());
147 
148     // remove annotation
149     Assert.assertTrue(annotationService.removeAnnotation(a1));
150 
151   }

153   @Test
154   public void testGetAnnotationsByTypeAndMediapackageId() throws Exception {
155     String type = "a type of annotation, such as 'bookmark' or 'note'";
156 
157     // Add an annotation
158     AnnotationImpl a1 = new AnnotationImpl();
159     a1.setType(type);
160     a1.setInpoint(10);
161     a1.setOutpoint(100);
162     a1.setMediapackageId("mp");
163     a1.setSessionId("session");
164     a1.setValue("This is some user generated content");
165     annotationService.addAnnotation(a1);
166 
167     // Add another annotation of the same type to a different mediapackage
168     AnnotationImpl a2 = new AnnotationImpl();
169     a2.setType(type);
170     a2.setInpoint(10);
171     a2.setOutpoint(100);
