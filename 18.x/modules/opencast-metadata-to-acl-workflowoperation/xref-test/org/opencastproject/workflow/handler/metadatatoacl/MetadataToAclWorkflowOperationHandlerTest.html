46  /**
47   * Test class for Metadata to ACL operation
48   */
49  public class MetadataToAclWorkflowOperationHandlerTest {
50  
51    @Test
52    public void testAcl() throws Exception {
53      // Prepare workflow
54      var mediaPackage = new MediaPackageBuilderImpl().createNew();
55      var catalog = CatalogImpl.newInstance();
56      catalog.setFlavor(MediaPackageElements.EPISODE);
57      mediaPackage.add(catalog);
58      var operation = new WorkflowOperationInstance("test-id", WorkflowOperationInstance.OperationState.RUNNING);
59      var workflowInstance = new WorkflowInstance();
60      workflowInstance.setMediaPackage(mediaPackage);
61      workflowInstance.setOperations(Collections.singletonList(operation));
62  
63      // Mock workspace
64      Workspace workspace = EasyMock.createMock(Workspace.class);
65      EasyMock.expect(workspace.read(EasyMock.anyObject()))
66              .andAnswer(() -> this.getClass().getResourceAsStream("/dublincore.xml"))
67              .anyTimes();
68  
69      // Mock user directory
70      UserDirectoryService userDirectoryService = EasyMock.createMock(UserDirectoryService.class);
71      var janeDoe = new JaxbUser("jane.doe", null, new DefaultOrganization());
72      EasyMock.expect(userDirectoryService.loadUser("jane.doe")).andReturn(janeDoe).anyTimes();
73      EasyMock.expect(userDirectoryService.loadUser("john.doe")).andReturn(null).anyTimes();
74  
75      // Mock authorization service
76      AuthorizationService authorizationService = EasyMock.createMock(AuthorizationService.class);
77      final Capture<AccessControlList> acl = EasyMock.newCapture();
78      EasyMock.expect(authorizationService.setAcl(EasyMock.anyObject(), EasyMock.anyObject(), EasyMock.capture(acl)))
79              .andReturn(null)
80              .once();
81      EasyMock.expect(authorizationService.getActiveAcl(EasyMock.anyObject(MediaPackage.class)))
82              .andReturn(new Tuple<>(new AccessControlList(), AclScope.Global))
83              .once();
84  
85      EasyMock.replay(workspace, userDirectoryService, authorizationService);
86  
87      // Initialize operation
88      var service = new MetadataToAclWorkflowOperationHandler();
89      service.setWorkspace(workspace);
90      service.setUserDirectory(userDirectoryService);
91      service.setAuthorizationService(authorizationService);
92  
93      // Run operation
94      service.start(workflowInstance, null);
95  
96      // We should have two entries for Jane
97      Assert.assertEquals(2, acl.getValue().getEntries().size());
98    }
99  }
