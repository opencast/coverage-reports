22  package org.opencastproject.transcription.microsoft.azure;
23  
24  import org.junit.Assert;
25  import org.junit.Before;
26  import org.junit.Test;
27  import org.slf4j.Logger;
28  import org.slf4j.LoggerFactory;
29  
30  import java.util.Calendar;
31  import java.util.Date;
32  import java.util.TimeZone;
33  
34  public class MicrosoftAzureAuthorizationTest {
35  
36    private static final Logger logger = LoggerFactory.getLogger(MicrosoftAzureAuthorizationTest.class);
37  
38    private MicrosoftAzureAuthorization azureAuthorization;
39  
40    @Before
41    public void setUp() throws MicrosoftAzureStorageClientException {
42      String azureStorageAccountName = "azurestorageaccount";
43      String azureAccountAccessKey = "mysecretaccesskey";
44      azureAuthorization = new MicrosoftAzureAuthorization(azureStorageAccountName, azureAccountAccessKey);
45    }
46  
47    @Test
48    public void generateAccountSASToken() {
49      Calendar c = Calendar.getInstance();
50      c.set(2023,Calendar.JANUARY,15,12,0,0);
51      c.setTimeZone(TimeZone.getTimeZone("GMT+1"));
52      Date start = c.getTime();
53      c.set(2023,Calendar.JANUARY,16,12,0,0);
54      Date end = c.getTime();
55      String sasToken = azureAuthorization.generateAccountSASToken("r", "c", start, end, null, null);
56      Assert.assertNotNull(sasToken);
57      Assert.assertTrue(sasToken.contains("sp=r"));
58      Assert.assertTrue(sasToken.contains("srt=c"));
59      Assert.assertTrue(sasToken.contains("st=2023-01-15T11:00:00Z"));
60      Assert.assertTrue(sasToken.contains("se=2023-01-16T11:00:00Z"));
61      Assert.assertTrue(sasToken.contains("spr=https"));
62      Assert.assertTrue(sasToken.contains("sig=1oAO19BF8zsHJ7xKO1AzVtgSacN8xgqM6dFBZoQMZUM%3D"));
63    }
64  }
