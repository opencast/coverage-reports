25  /**
26   * Contains all well-known definitions, names and symbols REPLAY relies on as constants for an easy usage and as a
27   * documentation.
28   */
29  public interface MediaPackageElements {
30  
31    /** The manifest file name */
32    String MANIFEST_FILENAME = "index.xml";
33  
34    /** Cover art flavor */
35    MediaPackageElementFlavor MEDIAPACKAGE_COVER_FLAVOR = new MediaPackageElementFlavor("cover", "source");
36  
37    // Track flavors
38  
39    /** Track containing the presenter/s */
40    MediaPackageElementFlavor PRESENTER_SOURCE = new MediaPackageElementFlavor("presenter", "source");
41  
42    /** Track containing presentational material */
43    MediaPackageElementFlavor PRESENTATION_SOURCE = new MediaPackageElementFlavor("presentation", "source");
44  
45    /** Track containing the presenter/s partial material */
46    MediaPackageElementFlavor PRESENTER_SOURCE_PARTIAL = new MediaPackageElementFlavor("presenter", "source+partial");
47  
48    /** Track containing presentational partial material */
49    MediaPackageElementFlavor PRESENTATION_SOURCE_PARTIAL = new MediaPackageElementFlavor("presentation",
50            "source+partial");
51  
52    /** Track capturing the audience */
53    MediaPackageElementFlavor AUDIENCE_SOURCE = new MediaPackageElementFlavor("audience", "source");
54  
55    /** Track capturing the contents of a document camera */
56    MediaPackageElementFlavor DOCUMENTS_SOURCE = new MediaPackageElementFlavor("documents", "source");
57  
58    /** Track without any known semantics */
59    MediaPackageElementFlavor INDEFINITE_SOURCE = new MediaPackageElementFlavor("indefinite", "source");
60  
61    // Dublin core catalog flavors
62  
63    /** Dublin core catalog describing an episode */
64    MediaPackageElementFlavor EPISODE = new MediaPackageElementFlavor("dublincore", "episode");
65  
66    /** Dublin core catalog describing a series */
67    MediaPackageElementFlavor SERIES = new MediaPackageElementFlavor("dublincore", "series");
68  
69    // Mpeg-7 catalog flavors
70  
71    /** The flavor produced by video segmentation */
72    MediaPackageElementFlavor SEGMENTS = new MediaPackageElementFlavor("mpeg-7", "segments");
73  
74    /** The flavor produced by text extraction */
75    MediaPackageElementFlavor TEXTS = new MediaPackageElementFlavor("mpeg-7", "text");
76  
77    /** The flavor produced by speech recognition */
78    MediaPackageElementFlavor SPEECH = new MediaPackageElementFlavor("mpeg-7", "speech");
79  
80    /** A flavor for MPEG-7 chapter catalogs */
81    MediaPackageElementFlavor CHAPTERING = new MediaPackageElementFlavor("mpeg-7", "chapter");
82  
83    // Engage flavors
84  
85    /** Presenter player preview image flavor */
86    MediaPackageElementFlavor PRESENTER_PLAYER_PREVIEW = new MediaPackageElementFlavor("presenter", "player+preview");
87  
88    /** Presentation player preview image flavor */
89    MediaPackageElementFlavor PRESENTATION_PLAYER_PREVIEW = new MediaPackageElementFlavor("presentation",
90      "player+preview");
91  
92    /** Presenter search result preview image flavor */
93    MediaPackageElementFlavor PRESENTER_SEARCHRESULT_PREVIEW = new MediaPackageElementFlavor("presenter",
94            "search+preview");
95  
96    /** Presentation search result preview image flavor */
97    MediaPackageElementFlavor PRESENTATION_SEARCHRESULT_PREVIEW = new MediaPackageElementFlavor("presentation",
98            "search+preview");
99  
100   /** Presenter segment preview image flavor */
101   MediaPackageElementFlavor PRESENTER_SEGMENT_PREVIEW = new MediaPackageElementFlavor("presenter", "segment+preview");
102 
103   /** Presentation segment preview image flavor */
104   MediaPackageElementFlavor PRESENTATION_SEGMENT_PREVIEW = new MediaPackageElementFlavor("presentation",
105           "segment+preview");
106 
107   // Feed flavors
108 
109   /** Presenter feed preview image flavor */
110   MediaPackageElementFlavor PRESENTER_FEED_PREVIEW = new MediaPackageElementFlavor("presenter", "feed+preview");
111 
112   /** Presentation feed preview image flavor */
113   MediaPackageElementFlavor PRESENTATION_FEED_PREVIEW = new MediaPackageElementFlavor("presentation", "feed+preview");
114 
115   // Security flavors
116 
117   /** Episode bound XACML policy flavor */
118   MediaPackageElementFlavor XACML_POLICY_EPISODE = new MediaPackageElementFlavor("security", "xacml+episode");
119 
120   /** Series bound XACML policy flavor */
121   MediaPackageElementFlavor XACML_POLICY_SERIES = new MediaPackageElementFlavor("security", "xacml+series");
122 
123   // Other flavors
124 
125   /** A default flavor for DFXP captions catalogs" */
126   MediaPackageElementFlavor CAPTION_GENERAL = new MediaPackageElementFlavor("captions", "timedtext");
127 
128   /** A flavor for DFXP caption files */
129   MediaPackageElementFlavor CAPTION_DFXP_FLAVOR = new MediaPackageElementFlavor("caption", "dfxp");
130 
131   /** OAI-PMH subtype flavor */
132   MediaPackageElementFlavor OAIPMH = new MediaPackageElementFlavor("*", "oaipmh");
133 
134   /** Comments metadata flavor */
135   MediaPackageElementFlavor COMMENTS = new MediaPackageElementFlavor("metadata", "comments");
136 
137   /** Notes metadata flavor */
138   MediaPackageElementFlavor NOTES = new MediaPackageElementFlavor("metadata", "notes");
139 
140   /** Partial SMIL flavor */
141   MediaPackageElementFlavor SMIL = new MediaPackageElementFlavor("smil", "source+partial");
142 
143   /** Processing properties flavor */
144   MediaPackageElementFlavor PROCESSING_PROPERTIES = new MediaPackageElementFlavor("processing", "defaults");
145 
146 }
