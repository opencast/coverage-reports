22  package org.opencastproject.mediapackage;
23  
24  import org.opencastproject.mediapackage.track.SubtitleStreamImpl;
25  
26  import javax.xml.bind.annotation.adapters.XmlAdapter;
27  import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;
28  
29  /**
30   * A stream containing subtitle data.
31   */
32  @XmlJavaTypeAdapter(SubtitleStream.Adapter.class)
33  public interface SubtitleStream extends Stream {
34  
35    String getFormat();
36  
37    String getFormatVersion();
38  
39    String getEncoderLibraryVendor();
40  
41    class Adapter extends XmlAdapter<SubtitleStreamImpl, Stream> {
42      @Override
43      public SubtitleStreamImpl marshal(Stream s) throws Exception {
44        return (SubtitleStreamImpl) s;
45      }
46  
47      @Override
48      public Stream unmarshal(SubtitleStreamImpl s) throws Exception {
49        return s;
50      }
51    }
52  }
