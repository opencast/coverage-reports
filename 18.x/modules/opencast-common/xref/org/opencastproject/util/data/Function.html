23  package org.opencastproject.util.data;
24  
25  import static org.opencastproject.util.data.functions.Misc.chuck;
26  
27  import org.opencastproject.util.data.functions.Functions;
28  
29  import com.entwinemedia.fn.Fn;
30  
31  /**
32   * Function of arity 1.
33   *
34   * A general note on function implementations: Each function has an associated X version with the only difference that a
35   * checked exception may be thrown on application. This behaviour could be nicely combined into the main function class
36   * but with the cost of having no abstract method. Some IDEs such as IntelliJ support nice code folding for SAM (single
37   * abstract method) classes.
38   *
39   * @see X
40   */
41  public abstract class Function<A, B> {
42    /** Apply function to <code>a</code>. */
43    public abstract B apply(A a);
44  
45    /** Currying. */
46    public Function0<B> curry(final A a) {
47      return Functions.curry(this, a);
48    }
49  
50    /** Currying. */
51    public Function<A, Function0<B>> curry() {
52      return Functions.curry(this);
53    }
54  
55    /** Function composition. <code>g.o(f).apply(x) == g(f(x))</code> */
56    public <C> Function<C, B> o(final Function<? super C, ? extends A> f) {
57      return Functions.o(this, f);
58    }
59  
60    /** Function composition. <code>g.o(f).apply() == g(f())</code> */
61    public Function0<B> o(final Function0<? extends A> f) {
62      return Functions.o(this, f);
63    }
64  
65    /** Left to right composition: <code>f then g = g(f(x))</code> */
66    public <C> Function<A, C> then(final Function<? super B, ? extends C> f) {
67      return Functions.then(this, f);
68    }
69  
70    /** @see Functions#rethrow(Function, Function) */
71    public Function<A, B> rethrow(final Function<? super Exception, ? extends Exception> transformer) {
72      return Functions.rethrow(this, transformer);
73    }
74  
75    /** @see Functions#handle(Function, Function) */
76    public Function<A, B> handle(final Function<? super Exception, B> handler) {
77      return Functions.handle(this, handler);
78    }
79  
80    /** @see Functions#either(Function, Function) */
81    public <X> Function<A, Either<X, B>> either(final Function<Exception, X> handler) {
82      return Functions.either(this, handler);
83    }
84  
85    /** Turn this function into an effect by discarding its result. */
86    public Effect<A> toEffect() {
87      return Functions.toEffect(this);
88    }
89  
90    public Fn<A, B> toFn() {
91      return new Fn<A, B>() {
92        @Override
93        public B apply(A a) {
94          return Function.this.apply(a);
95        }
96      };
97    }
98  
99    /** Version of {@link Function} that allows for throwing a checked exception. */
100   public abstract static class X<A, B> extends Function<A, B> {
101     @Override
102     public final B apply(A a) {
103       try {
104         return xapply(a);
105       } catch (Exception e) {
106         return chuck(e);
107       }
108     }
109 
110     /**
111      * Apply function to <code>a</code>. Any thrown exception gets "chucked" so that you may catch them as is. See
112      * {@link Functions#chuck(Throwable)} for details.
113      */
114     protected abstract B xapply(A a) throws Exception;
115 
116   }

118 }
