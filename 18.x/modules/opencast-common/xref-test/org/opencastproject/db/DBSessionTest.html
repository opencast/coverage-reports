41  public class DBSessionTest {
42    private DBSession db;
43  
44    @Before
45    public void before() {
46      db = newDBSession("test");
47    }
48  
49    @After
50    public void after() {
51      db.close();
52    }

54    @Test
55    public void testPersistAndFind() {
56      final long id = db.execTx(namedQuery.persist(TestDto.create("key", "value"))).getId();
57      assertEquals("value", db.execTx(namedQuery.findById(TestDto.class, id)).getValue());
58    }
59  
60    @Test
61    public void testPersistOrUpdateUpdate() {
62      assertTrue(db.execTx(TestDto.findAll).isEmpty());
63      final TestDto dto = db.execTx(namedQuery.persist(TestDto.create("key", "value")));
64      assertEquals("value", db.execTx(namedQuery.findById(TestDto.class, dto.getId())).getValue());
65      dto.setValue("new-value");
66      db.execTx(namedQuery.persistOrUpdate(dto));
67      assertEquals("new-value", db.execTx(namedQuery.findById(TestDto.class, dto.getId())).getValue());
68    }
69  
70    @Test
71    public void testPersistOrUpdatePersist() {
72      assertTrue(db.execTx(TestDto.findAll).isEmpty());
73      final TestDto dto = db.execTx(namedQuery.persistOrUpdate(TestDto.create("key", "value")));
74      assertEquals("value", db.execTx(namedQuery.findById(TestDto.class, dto.getId())).getValue());
75      dto.setValue("new-value");
76      db.execTx(namedQuery.persistOrUpdate(dto));
77      assertEquals("new-value", db.execTx(namedQuery.findById(TestDto.class, dto.getId())).getValue());
78    }

80    @Test
81    public void testCloseEntityManager() {
82      // this should not throw an exception in db.execTx()
83      db.execTx(EntityManager::close);
84    }
85  
86    @Test(expected = RuntimeException.class)
87    public void testException() {
88      db.execTx((Consumer<EntityManager>) em -> {
89        throw new RuntimeException("error");
90      });
91    }
92  
93    @Test
94    public void testRetries() {
95      AtomicInteger calls = new AtomicInteger();
96  
97      calls.set(0);
98      db.execTx(em -> {
99        calls.incrementAndGet();
100       if (calls.get() == 1) {
101         throw new RollbackException("error");
102       }
103     });
104     assertEquals(2, calls.get());
105 
106     calls.set(0);
107     db.execTx(em -> {
108       calls.incrementAndGet();
109       if (calls.get() == 1) {
110         SQLException sqlEx = new SQLException("error", SqlState.TRANSACTION_ROLLBACK_SERIALIZATION_FAILURE);
111         throw DatabaseException.sqlException(sqlEx);
112       }
113     });
114     assertEquals(2, calls.get());
115   }
116 
117   @Test
118   public void testTransactionPropagation() {
119     long id = db.execTx(em -> {
120       final TestDto dto = TestDto.create("key", "A");
121       em.persist(dto);
122       // nested transaction. if transaction propagation would fail a duplicate key exception would be thrown
123       return save(dto);
124     });
125     assertEquals("dto value", "B", db.execTx(namedQuery.findById(TestDto.class, id)).getValue());
126   }
127 
128   private long save(TestDto dto) {
129     dto.setValue("B");
130     return db.execTx(namedQuery.persist(dto)).getId();
131   }
