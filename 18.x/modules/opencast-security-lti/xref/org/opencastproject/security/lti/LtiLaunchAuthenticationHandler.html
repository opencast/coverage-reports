208   @Activate
209   protected void activate(ComponentContext cc) {
210     logger.info("Activating LtiLaunchAuthenticationHandler");
211     componentContext = cc;
212     Dictionary<String, Object> properties = cc.getProperties();
213 
214     logger.debug("Updating LtiLaunchAuthenticationHandler");
215 
216     highlyTrustedConsumerKeys.clear();
217     usernameBlacklist.clear();
218 
219     if (properties == null) {
220       logger.warn("LtiLaunchAuthenticationHandler is not configured");
221       return;
222     }
223 
224     // Highly trusted OAuth consumer keys
225     for (int i = 1; true; i++) {
226       logger.debug("Looking for configuration of {}", HIGHLY_TRUSTED_CONSUMER_KEY_PREFIX + i);
227       String consumerKey = StringUtils.trimToNull((String) properties.get(HIGHLY_TRUSTED_CONSUMER_KEY_PREFIX + i));
228       if (consumerKey == null) {
229         break;
230       }
231       highlyTrustedConsumerKeys.add(consumerKey);
232     }
233 
234     // User blacklist
235     if (!BooleanUtils.toBoolean(StringUtils.trimToNull((String) properties.get(ALLOW_SYSTEM_ADMINISTRATOR_KEY)))) {
236       String adminUsername = StringUtils.trimToNull((String)
237               properties.get(SecurityConstants.GLOBAL_ADMIN_USER_PROPERTY));
238       if (adminUsername != null) {
239         usernameBlacklist.add(adminUsername);
240       }
241     }
242     if (!BooleanUtils.toBoolean(StringUtils.trimToNull((String) properties.get(ALLOW_DIGEST_USER_KEY)))) {
243       usernameBlacklist.add(SecurityUtil.getSystemUserName(componentContext));
244     }
245 
246     for (int i = 1; true; i++) {
247       logger.debug("Looking for configuration of {}", BLACKLIST_USER_PREFIX + i);
248       String username = StringUtils.trimToNull((String) properties.get(BLACKLIST_USER_PREFIX + i));
249       if (username == null) {
250         break;
251       }
252       usernameBlacklist.add(username);
253     }
254 
255     String createJpaUserRefStr = (String) properties.get(CREATE_JPA_USER_REFERENCE_KEY);
256     createJpaUserReference = BooleanUtils.toBooleanDefaultIfNull(
257             BooleanUtils.toBooleanObject(StringUtils.trimToNull(createJpaUserRefStr)),
258             true);
259 
260     ltiRolesForUserCreation = extractLtiRolesForUserCreation(
261             Objects.toString(properties.get(LTI_ROLES_TO_CREATE_JPA_USER_REFERENCES_FROM), "*"));
262 
263     for (int i = 1; true; i++) {
264       logger.debug("Looking for custom role configuration of {}", CUSTOM_ROLE_NAME_PREFIX + i);
265       String customRoleName = StringUtils.trimToNull((String) properties.get(CUSTOM_ROLE_NAME_PREFIX + i));
266       if (customRoleName == null) {
267         break;
268       }
269       String customRolesString = StringUtils.trimToNull((String) properties.get(CUSTOM_ROLES_LIST_PREFIX + i));
270       if (customRolesString == null) {
271         continue;
272       }
273       enrichLtiCustomRoles(customRoleName, customRolesString);
274     }
275 
276     // Add support for legacy custom_role_name
277     String customRoleName = StringUtils.trimToNull((String) properties.get(CUSTOM_ROLE_NAME));
278 
279     if (customRoleName != null) {
280       String customRolesString = StringUtils.trimToNull((String) properties.get(CUSTOM_ROLES));
281       enrichLtiCustomRoles(customRoleName, customRolesString);
282     }
283 
284     // Allow configuring prefixes for certain consumer
285     for (String key: Collections.list(properties.keys())) {
286       if (key.startsWith(ROLE_PREFIX_KEY)) {
287         final String consumerKey = key.substring(ROLE_PREFIX_KEY.length());
288         final String prefix = Objects.toString(properties.get(key), "");
289         logger.debug("Adding role prefix '{}' for consumer using OAuth key '{}'", prefix, consumerKey);
290         rolePrefixes.put(consumerKey, prefix);
291       }
292     }
293 
294   }

296   /**
297    * {@inheritDoc}
298    *
299    * @see org.springframework.security.oauth.provider.OAuthAuthenticationHandler#createAuthentication(
300    *      javax.servlet.http.HttpServletRequest,
301    *      org.springframework.security.oauth.provider.ConsumerAuthentication,
302    *      org.springframework.security.oauth.provider.token.OAuthAccessProviderToken)
303    */
304   @Override
305   public Authentication createAuthentication(HttpServletRequest request, ConsumerAuthentication authentication,
306           OAuthAccessProviderToken authToken) {
307     // The User ID must be provided by the LTI consumer
308     String userIdFromConsumer = request.getParameter(LTI_USER_ID_PARAM);
309 
310     if (StringUtils.isBlank(userIdFromConsumer)) {
311       logger.warn("Received authentication request without user id ({})", LTI_USER_ID_PARAM);
312       return null;
313     }
314 
315     // Get the consumer guid if provided
316     String consumerGUID = request.getParameter(LTI_CONSUMER_GUID);
317     // This is an optional field, so it could be blank
318     if (StringUtils.isBlank(consumerGUID)) {
319       consumerGUID = "UnknownConsumer";
320     }
321 
322     // We need to construct a complex ID to avoid confusion
323     String username = LTI_USER_ID_PREFIX + LTI_ID_DELIMITER + consumerGUID + LTI_ID_DELIMITER + userIdFromConsumer;
324 
325     final String oaAuthKey = request.getParameter("oauth_consumer_key");
326 
327     final String rolePrefix = rolePrefixes.getOrDefault(oaAuthKey, "");
328 
329     // if this is a trusted consumer we trust their details
330     if (highlyTrustedConsumerKeys.contains(oaAuthKey)) {
331       logger.debug("{} is a trusted key", oaAuthKey);
332 
333       // If supplied we use the human readable name coming from:
334       //   1. ext_user_username    (optional Moodle-only field)
335       //   2. lis_person_sourcedid (optional standard field)
336       String ltiUsername = request.getParameter("ext_user_username");
337       if (StringUtils.isBlank(ltiUsername)) {
338         ltiUsername = request.getParameter("lis_person_sourcedid");
339         if (StringUtils.isBlank(ltiUsername)) {
340           // If no eid is set we use the supplied ID
341           ltiUsername = userIdFromConsumer;
342         }
343       }
344 
345       // Check if the provided username should be trusted
346       if (usernameBlacklist.contains(ltiUsername)) {
347         // Do not trust the username
348         logger.debug("{} is blacklisted", ltiUsername);
349       } else {
350         username = ltiUsername;
351       }
352     }
353 
354     if (logger.isDebugEnabled()) {
355       logger.debug("LTI user id is : {}", username);
356     }
357 
358     UserDetails userDetails;
359     Collection<GrantedAuthority> userAuthorities;
360     try {
361       userDetails = userDetailsService.loadUserByUsername(username);
362 
363       // userDetails returns a Collection<? extends GrantedAuthority>, which cannot be directly casted to a
364       // Collection<GrantedAuthority>.
365       // On the other hand, one cannot add non-null elements or modify the existing ones in a Collection<? extends
366       // GrantedAuthority>. Therefore, we *must* instantiate a new Collection<GrantedAuthority> (an ArrayList in this
367       // case) and populate it with whatever elements are returned by getAuthorities()
368       userAuthorities = new HashSet<>(userDetails.getAuthorities());
369 
370       // we still need to enrich this user with the LTI Roles
371       String roles = request.getParameter(ROLES);
372       String context = request.getParameter(CONTEXT_ID);
373       enrichRoleGrants(roles, context, rolePrefix, userAuthorities);
374     } catch (UsernameNotFoundException e) {
375       logger.trace("This user is known to the tool consumer only. Creating an Opencast user on the fly.", e);
376 
377       userAuthorities = new HashSet<>();
378       // We should add the authorities passed in from the tool consumer?
379       String roles = request.getParameter(ROLES);
380       String context = request.getParameter(CONTEXT_ID);
381       enrichRoleGrants(roles, context, rolePrefix, userAuthorities);
382 
383       logger.debug("Returning user with {} authorities", userAuthorities.size());
384 
385       userDetails = new User(username, "oauth", true, true, true, true, userAuthorities);
386     }
387 
388     // All users need the OAUTH, USER and ANONYMOUS roles
389     userAuthorities.add(new SimpleGrantedAuthority(ROLE_OAUTH_USER));
390     userAuthorities.add(new SimpleGrantedAuthority("ROLE_USER"));
391     userAuthorities.add(new SimpleGrantedAuthority("ROLE_ANONYMOUS"));
392 
393     // Create/Update the user reference
394     if (createJpaUserReference && ltiRolesForUserCreation.size() > 0) {
395       if (activePersistenceTransactions.putIfAbsent(username, Boolean.TRUE) != null) {
396         logger.debug("Concurrent access of user {}. Ignoring.", username);
397       } else if (!ltiRolesForUserCreation.contains("*") && !requestContainsMatchingRoles(request)) {
398         logger.debug("No JpaUserReference will be created for LTI user {}.", username);
399         // Remove user from the cache; next time, the user may have additional/different roles.
400         activePersistenceTransactions.remove(username);
401       } else {
402         try {
403           JpaOrganization organization = fromOrganization(securityService.getOrganization());
404           JpaUserReference jpaUserReference = userReferenceProvider.findUserReference(username, organization.getId());
405           Set<JpaRole> jpaRoles = new HashSet<>();
406           for (GrantedAuthority authority : userAuthorities) {
407             jpaRoles.add(new JpaRole(authority.getAuthority(), organization));
408           }
409 
410           Date loginDate = new Date();
411 
412           // Get some user details
413           String name = request.getParameter("lis_person_name_full");
414           if (name == null) {
415             final String familyName = Objects.toString(request.getParameter("lis_person_name_family"), "");
416             final String givenName = Objects.toString(request.getParameter("lis_person_name_given"), "");
417             name = String.format("%s %s", givenName, familyName).trim();
418             if (name.isEmpty()) {
419               name = username;
420             }
421           }
422           final String email = request.getParameter("lis_person_contact_email_primary");
423 
424           // Create new JpaUserReference if none exists or update existing
425           if (jpaUserReference == null) {
426             final String jpaContext = Objects.toString(request.getParameter(CONTEXT_ID), DEFAULT_CONTEXT);
427             JpaUserReference userReference = new JpaUserReference(username, name, email, jpaContext, loginDate,
428                 organization, jpaRoles);
429             userReferenceProvider.addUserReference(userReference, jpaContext);
430           } else {
431             jpaUserReference.setLastLogin(loginDate);
432             jpaUserReference.setName(name);
433             jpaUserReference.setEmail(email);
434             jpaUserReference.setRoles(jpaRoles);
435             userReferenceProvider.updateUserReference(jpaUserReference);
436           }
437         } catch (RollbackException e) {
438           // In the unlikely case that concurrency happens at the same time on multiple servers, we catch the
439           // rollback, assuming that the user is already persisted and let the user pass. Worst case, this means that
440           // the user is only temporarily authenticated.
441           logger.warn("Could not store reference since database was changed during update by another process", e);
442         } finally {
443           activePersistenceTransactions.remove(username);
444         }
445       }
446     }
447     //Create/Update UserReference End
448 
449     Authentication ltiAuth = new PreAuthenticatedAuthenticationToken(userDetails, authentication.getCredentials(),
450             userAuthorities);
451     SecurityContextHolder.getContext().setAuthentication(ltiAuth);
452     return ltiAuth;
453   }

455   /**
456    * Enrich A collection of role grants with specified LTI memberships.
457    *
458    * @param roles
459    *          String of LTI roles.
460    * @param context
461    *          LTI context ID.
462    * @param userAuthorities
463    *          Collection to append to.
464    */
465   private void enrichRoleGrants(String roles, String context, final String rolePrefix,
466       Collection<GrantedAuthority> userAuthorities) {
467     if (roles != null) {
468       // Roles could be a list
469       String[] roleList = roles.split(",");
470 
471       // Use a generic context and learner if none is given:
472       context = StringUtils.isBlank(context) ? DEFAULT_CONTEXT : context;
473 
474       for (final String ltiRole : roleList) {
475         // Build the role
476         for (Pattern rolePattern : customRolePatterns.keySet()) {
477           logger.debug("Matching role pattern '{}' against '{}'", rolePattern.pattern(), ltiRole);
478           if (rolePattern.matcher(ltiRole).matches()) {
479             for (String roleName : customRolePatterns.get(rolePattern)) {
480               userAuthorities.add(new SimpleGrantedAuthority(roleName));
481             }
482           }
483         }
484 
485         final String normalizedLtiRole = StringUtils.defaultIfBlank(ltiRole, DEFAULT_LEARNER);
486         final String role = rolePrefix + context + "_" + normalizedLtiRole;
487 
488         // Make sure to not accept ROLE_…
489         if (role.trim().toUpperCase().startsWith("ROLE_")) {
490           logger.warn("Discarding attempt to acquire role “{}”", role);
491           continue;
492         }
493 
494         // Add this role
495         logger.debug("Adding role: {}", role);
496         userAuthorities.add(new SimpleGrantedAuthority(role));
497       }
498     }
499   }

501   private void enrichLtiCustomRoles(String roleName, String customRolesString) {
502     String[] customRoles = customRolesString.split(",");
503     Pattern customRolePattern = Pattern.compile(roleName);
504     customRolePatterns.put(customRolePattern, customRoles);
505   }

522   /**
523    * Extracts LTI roles from a cfg key. This roles determine whether a JpaUserReference
524    * shall be created on a LTI login or not.
525    * If no roles are given, no JPA references will be created.
526    * The default value is '*', which means JPA references will be created always.
527    *
528    * @param ltiRoles The roles from the cfg.
529    * @return The list of the LTI roles where a jpaUserReference will be created.
530    */
531   private List<String> extractLtiRolesForUserCreation(String ltiRoles) {
532     ltiRoles = StringUtils.trimToNull(ltiRoles);
533     if ("*".equals(ltiRoles.trim())) {
534       return Collections.singletonList("*");
535     }
536     ltiRoles = ltiRoles.toLowerCase();
537     return Arrays.asList(ltiRoles.trim().split("\\s*,\\s*"));
538   }
539 
540   /**
541    * Gets the LTI roles from the request and checks if one role is allowed
542    * for jpaUserReference creation.
543    *
544    * @param request The HttpServletRequest with the roles
545    * @return create JpaUserReference or not
546    */
547   private boolean requestContainsMatchingRoles(HttpServletRequest request) {
548     String roles = request.getParameter(ROLES);
549     if (roles != null) {
550       List<String> requestRoleList = Arrays.asList(roles.toLowerCase().trim().split("\\s*,\\s*"));
551       return !Collections.disjoint(requestRoleList, ltiRolesForUserCreation);
552     }
553     return false;
554   }
