128   /**
129    * {@inheritDoc}
130    *
131    * @see org.opencastproject.silencedetection.api.SilenceDetectionService#detect(
132    * org.opencastproject.mediapackage.Track, org.opencastproject.mediapackage.Track[])
133    */
134   @Override
135   public Job detect(Track sourceTrack, Track[] referenceTracks) throws SilenceDetectionFailedException {
136     try {
137       if (sourceTrack == null) {
138         throw new SilenceDetectionFailedException("Source track is null!");
139       }
140       List<String> arguments = new LinkedList<String>();
141       // put source track as job argument
142       arguments.add(0, MediaPackageElementParser.getAsXml(sourceTrack));
143 
144       // put reference tracks as second argument
145       if (referenceTracks != null) {
146         arguments.add(1, MediaPackageElementParser.getArrayAsXml(Arrays.asList(referenceTracks)));
147       }
148 
149       return serviceRegistry.createJob(
150               getJobType(),
151               Operation.SILENCE_DETECTION.toString(),
152               arguments,
153               jobload);
154 
155     } catch (ServiceRegistryException ex) {
156       throw new SilenceDetectionFailedException("Unable to create job! " + ex.getMessage());
157     } catch (MediaPackageException ex) {
158       throw new SilenceDetectionFailedException("Unable to serialize track!");
159     }
160   }
161 
162   @Override
163   protected String process(Job job) throws SilenceDetectionFailedException, SmilException, MediaPackageException {
164     if (Operation.SILENCE_DETECTION.toString().equals(job.getOperation())) {
165       // get source track
166       String sourceTrackXml = StringUtils.trimToNull(job.getArguments().get(0));
167       if (sourceTrackXml == null) {
168         throw new SilenceDetectionFailedException("Track not set!");
169       }
170       Track sourceTrack = (Track) MediaPackageElementParser.getFromXml(sourceTrackXml);
171 
172       // run detection on source track
173       MediaSegments segments = runDetection(sourceTrack);
174 
175       // get reference tracks if any
176       List<Track> referenceTracks = null;
177       if (job.getArguments().size() > 1) {
178         String referenceTracksXml = StringUtils.trimToNull(job.getArguments().get(1));
179         if (referenceTracksXml != null) {
180           referenceTracks = (List<Track>) MediaPackageElementParser.getArrayFromXml(referenceTracksXml);
181         }
182       }
183 
184       if (referenceTracks == null) {
185         referenceTracks = Arrays.asList(sourceTrack);
186       }
187 
188       // create smil XML as result
189       try {
190         return generateSmil(segments, referenceTracks).toXML();
191       } catch (Exception ex) {
192         throw new SmilException("Failed to create smil document.", ex);
193       }
194     }
195 
196     throw new SilenceDetectionFailedException("Can't handle this operation: " + job.getOperation());
197   }

199   /**
200    * Run silence detection on the source track and returns
201    * {@link org.opencastproject.silencedetection.api.MediaSegments}
202    * XML as string. Source track should have an audio stream. All detected
203    * {@link org.opencastproject.silencedetection.api.MediaSegment}s
204    * (one or more) are non silent sequences.
205    *
206    * @param track track where to run silence detection
207    * @return {@link MediaSegments} Xml as String
208    * @throws SilenceDetectionFailedException if an error occures
209    */
210   protected MediaSegments runDetection(Track track) throws SilenceDetectionFailedException {
211     try {
212       FFmpegSilenceDetector silenceDetector = new FFmpegSilenceDetector(properties, track, workspace);
213       return silenceDetector.getMediaSegments();
214     } catch (Exception ex) {
215       throw new SilenceDetectionFailedException(ex.getMessage());
216     }
217   }
218 
219   /**
220    * Create a smil from given parameters.
221    *
222    * @param segments media segment list with timestamps
223    * @param referenceTracks tracks to put as media segment source files
224    * @return generated smil
225    * @throws SmilException if smil creation failed
226    */
227   protected Smil generateSmil(MediaSegments segments, List<Track> referenceTracks) throws SmilException {
228     SmilResponse smilResponse = smilService.createNewSmil();
229     Track[] referenceTracksArr = referenceTracks.toArray(new Track[0]);
230 
231     for (MediaSegment segment : segments.getMediaSegments()) {
232       smilResponse = smilService.addParallel(smilResponse.getSmil());
233       String parId = smilResponse.getEntity().getId();
234 
235       smilResponse = smilService.addClips(smilResponse.getSmil(), parId, referenceTracksArr,
236               segment.getSegmentStart(), segment.getSegmentStop() - segment.getSegmentStart());
237     }
238     return smilResponse.getSmil();
239   }

261   @Activate
262   @Modified
263   public void activate(BundleContext bundleContext, ComponentContext context) {
264     logger.debug("Loading configuration");
265     super.activate(context);
266 
267     this.properties = new Properties();
268     if (bundleContext == null) {
269       logger.info("No configuration available, using defaults");
270       return;
271     }
272     FFmpegSilenceDetector.init(bundleContext);
273     Dictionary<String, Object> properties = context.getProperties();
274     Enumeration<String> keys = properties.keys();
275     while (keys.hasMoreElements()) {
276       final String key = keys.nextElement();
277       logger.debug("{} = {}", key, properties.get(key));
278       this.properties.put(key, properties.get(key));
279     }
280     logger.debug("Properties updated!");
281 
282     jobload = LoadUtil.getConfiguredLoadValue(properties, JOB_LOAD_KEY, DEFAULT_JOB_LOAD, serviceRegistry);
283   }
