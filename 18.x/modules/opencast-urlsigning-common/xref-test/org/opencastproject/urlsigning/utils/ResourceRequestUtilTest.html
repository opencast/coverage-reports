54    @Test
55    public void testQueryStringParsing() {
56      String policyValue = "{policy:'value'}";
57      String signatureValue = "randomString";
58  
59      String queryString = "?" + ResourceRequest.POLICY_KEY + "=" + policyValue + "&" + ResourceRequest.SIGNATURE_KEY
60              + "=" + signatureValue + "&" + ResourceRequest.ENCRYPTION_ID_KEY + "=" + keyId;
61  
62      List<NameValuePair> parameters = ResourceRequestUtil.parseQueryString(queryString);
63  
64      boolean foundOrg = false;
65      boolean foundPolicy = false;
66      boolean foundSignature = false;
67  
68      for (NameValuePair nameValuePair : parameters) {
69        if (ResourceRequest.ENCRYPTION_ID_KEY.equals(nameValuePair.getName())) {
70          assertEquals(keyId, nameValuePair.getValue());
71          foundOrg = true;
72        }
73        if (ResourceRequest.POLICY_KEY.equals(nameValuePair.getName())) {
74          assertEquals(policyValue, nameValuePair.getValue());
75          foundPolicy = true;
76        }
77        if (ResourceRequest.SIGNATURE_KEY.equals(nameValuePair.getName())) {
78          assertEquals(signatureValue, nameValuePair.getValue());
79          foundSignature = true;
80        }
81      }
82  
83      assertTrue("Didn't find the organization value.", foundOrg);
84      assertTrue("Didn't find the policy value.", foundPolicy);
85      assertTrue("Didn't find the signature value.", foundSignature);
86    }

88    @Test
89    public void testAuthenticateDuplicateProperties() {
90      // Test duplicate query properties.
91      String twoOrgs = ResourceRequest.ENCRYPTION_ID_KEY + "=org1&" + ResourceRequest.ENCRYPTION_ID_KEY + "=org2";
92  
93      assertEquals(Status.BadRequest,
94              ResourceRequestUtil.resourceRequestFromQueryString(twoOrgs, clientIp, null, properties, true).getStatus());
95  
96      String twoPolicies = ResourceRequest.POLICY_KEY + "=policy1&" + ResourceRequest.POLICY_KEY + "=policy2";
97      assertEquals(Status.BadRequest, ResourceRequestUtil
98              .resourceRequestFromQueryString(twoPolicies, clientIp, null, properties, true).getStatus());
99  
100     String twoSignatures = ResourceRequest.SIGNATURE_KEY + "=signature1&" + ResourceRequest.SIGNATURE_KEY
101             + "=signature1";
102     assertEquals(Status.BadRequest, ResourceRequestUtil
103             .resourceRequestFromQueryString(twoSignatures, clientIp, null, properties, true).getStatus());
104   }

124   @Test
125   public void testAuthenticatePolicyMatchesSignature() throws Exception {
126     DateTime after = new DateTime(DateTimeZone.UTC);
127     after = after.minus(2 * 60 * 60 * 1000L);
128     DateTime before = new DateTime(DateTimeZone.UTC);
129     before = before.plus(2 * 60 * 60 * 1000L);
130     String nonMatchingResource = "http://other.com";
131     Policy nonMatchingPolicy = Policy.mkSimplePolicy(nonMatchingResource, before);
132     String matchingResource = "http://mh-allinone/";
133     Policy matchingPolicy = Policy.mkSimplePolicy(matchingResource, before);
134     String signature = PolicyUtils.getPolicySignature(matchingPolicy, key);
135 
136     // Test non-existant encryption key is forbidden.
137     String wrongEncryptionKeyId = ResourceRequest.ENCRYPTION_ID_KEY + "=" + "WrongId" + "&" + ResourceRequest.POLICY_KEY
138             + "=" + PolicyUtils.toBase64EncodedPolicy(matchingPolicy) + "&" + ResourceRequest.SIGNATURE_KEY + "="
139             + signature;
140     assertEquals(Status.Forbidden,
141             ResourceRequestUtil
142                     .resourceRequestFromQueryString(wrongEncryptionKeyId, clientIp, matchingResource, properties, true)
143                     .getStatus());
144 
145     // Test non matching resource results is forbidden.
146     String nonMatching = ResourceRequest.ENCRYPTION_ID_KEY + "=organization&" + ResourceRequest.POLICY_KEY + "="
147             + PolicyUtils.toBase64EncodedPolicy(nonMatchingPolicy) + "&" + ResourceRequest.SIGNATURE_KEY + "="
148             + signature;
149     assertEquals(Status.Forbidden, ResourceRequestUtil
150             .resourceRequestFromQueryString(nonMatching, clientIp, matchingResource, properties, true).getStatus());
151 
152     // Test non-matching client ip results in forbidden.
153     Policy wrongClientPolicy = Policy.mkPolicyValidWithIP(matchingResource, before, "10.0.0.255");
154     String wrongClient = ResourceRequestUtil.policyToResourceRequestQueryString(wrongClientPolicy, keyId, key);
155     assertEquals(Status.Forbidden, ResourceRequestUtil
156             .resourceRequestFromQueryString(wrongClient, clientIp, matchingResource, properties, true).getStatus());
157 
158     // Test matching client ip results in ok.
159     Policy rightClientPolicy = Policy.mkPolicyValidWithIP(matchingResource, before, clientIp);
160     String rightClient = ResourceRequestUtil.policyToResourceRequestQueryString(rightClientPolicy, keyId, key);
161     assertEquals(Status.Ok, ResourceRequestUtil
162             .resourceRequestFromQueryString(rightClient, clientIp, matchingResource, properties, true).getStatus());
163 
164     // Test not yet DateGreaterThan results in gone
165     Policy wrongDateGreaterThanPolicy = Policy.mkPolicyValidFrom(matchingResource, before, before);
166     String wrongDateGreaterThan = ResourceRequestUtil.policyToResourceRequestQueryString(wrongDateGreaterThanPolicy,
167             keyId, key);
168     assertEquals(Status.Gone,
169             ResourceRequestUtil
170                     .resourceRequestFromQueryString(wrongDateGreaterThan, clientIp, matchingResource, properties, true)
171                     .getStatus());
172 
173     // Test after DateGreaterThan results in ok
174     Policy rightDateGreaterThanPolicy = Policy.mkPolicyValidFrom(matchingResource, before, after);
175     String rightDateGreaterThan = ResourceRequestUtil.policyToResourceRequestQueryString(rightDateGreaterThanPolicy,
176             keyId, key);
177     assertEquals(Status.Ok,
178             ResourceRequestUtil
179                     .resourceRequestFromQueryString(rightDateGreaterThan, clientIp, matchingResource, properties, true)
180                     .getStatus());
181 
182     // Test before DateLessThan results in gone
183     Policy wrongDateLessThanPolicy = Policy.mkSimplePolicy(matchingResource, after);
184     String wrongDateLessThan = ResourceRequestUtil.policyToResourceRequestQueryString(wrongDateLessThanPolicy, keyId,
185             key);
186     assertEquals(Status.Gone,
187             ResourceRequestUtil
188                     .resourceRequestFromQueryString(wrongDateLessThan, clientIp, matchingResource, properties, true)
189                     .getStatus());
190 
191     // Test matching results in ok.
192     String matching = ResourceRequest.ENCRYPTION_ID_KEY + "=" + keyId + "&" + ResourceRequest.POLICY_KEY + "="
193             + PolicyUtils.toBase64EncodedPolicy(matchingPolicy) + "&" + ResourceRequest.SIGNATURE_KEY + "=" + signature;
194     assertEquals(Status.Ok, ResourceRequestUtil
195             .resourceRequestFromQueryString(matching, clientIp, matchingResource, properties, true).getStatus());
196   }

198   @Test
199   public void testIsSigned() throws URISyntaxException {
200     String noQueryString = "http://notsigned.com";
201     String wrongQueryString = "http://notsigned.com?irrelevant=value";
202     String signed = "http://notsigned.com?signature=theSignature&keyId=theKey&policy=thePolicy";
203     assertFalse(ResourceRequestUtil.isSigned(new URI(noQueryString)));
204     assertFalse(ResourceRequestUtil.isSigned(new URI(wrongQueryString)));
205     assertTrue(ResourceRequestUtil.isSigned(new URI(signed)));
206   }

208   @Test
209   public void testNonStrictResourceChecking() throws Exception {
210     DateTime before = new DateTime(DateTimeZone.UTC);
211     before = before.plus(2 * 60 * 60 * 1000L);
212     String hostname = "signed.host.com";
213     String path = "/path/to/resource";
214     String rtmpResource = "rtmp://" + hostname + path;
215     String httpResource = "http://" + hostname + path;
216     String portResource = "rtmp://" + hostname + ":8080" + path;
217     String differentHostnameResource = "rtmp://different.host.com" + path;
218 
219     Policy differentScheme = Policy.mkSimplePolicy(rtmpResource, before);
220     String signature = PolicyUtils.getPolicySignature(differentScheme, key);
221     String differentSchemeQueryString = ResourceRequest.ENCRYPTION_ID_KEY + "=default&"
222         + ResourceRequest.POLICY_KEY + "="
223         + PolicyUtils.toBase64EncodedPolicy(differentScheme) + "&" + ResourceRequest.SIGNATURE_KEY + "="
224         + signature;
225 
226     assertEquals(Status.Ok, ResourceRequestUtil.resourceRequestFromQueryString(
227         differentSchemeQueryString, clientIp, httpResource, properties, false).getStatus());
228     assertEquals(Status.Forbidden, ResourceRequestUtil.resourceRequestFromQueryString(
229         differentSchemeQueryString, clientIp, httpResource, properties, true).getStatus());
230 
231     assertEquals(Status.Ok, ResourceRequestUtil.resourceRequestFromQueryString(
232         differentSchemeQueryString, clientIp, portResource, properties, false).getStatus());
233     assertEquals(Status.Forbidden, ResourceRequestUtil.resourceRequestFromQueryString(
234         differentSchemeQueryString, clientIp, portResource, properties, true).getStatus());
235 
236     assertEquals(Status.Ok, ResourceRequestUtil.resourceRequestFromQueryString(
237         differentSchemeQueryString, clientIp, differentHostnameResource, properties, false).getStatus());
238     assertEquals(Status.Forbidden, ResourceRequestUtil.resourceRequestFromQueryString(
239         differentSchemeQueryString, clientIp, differentHostnameResource, properties, true).getStatus());
240   }
