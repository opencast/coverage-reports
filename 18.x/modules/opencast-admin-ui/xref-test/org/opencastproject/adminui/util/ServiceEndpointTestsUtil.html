22  package org.opencastproject.adminui.util;
23  
24  import org.json.simple.JSONArray;
25  import org.json.simple.JSONObject;
26  import org.junit.Assert;
27  
28  import java.util.Iterator;
29  import java.util.Set;
30  
31  public final class ServiceEndpointTestsUtil {
32  
33    private static final String COUNT = "count";
34    private static final String TOTAL = "total";
35    private static final String LIMIT = "limit";
36    private static final String OFFSET = "offset";
37    private static final String RESULTS = "results";
38  
39    private ServiceEndpointTestsUtil() {
40    }
41  
42    public static void testJSONObjectEquality(JSONObject expected, JSONObject actual) {
43      Assert.assertEquals(expected.size(), actual.size());
44      testSimpleProperty(COUNT, expected, actual);
45      testSimpleProperty(TOTAL, expected, actual);
46      testSimpleProperty(LIMIT, expected, actual);
47      testSimpleProperty(OFFSET, expected, actual);
48      testArrayProperty(RESULTS, expected, actual);
49    }
50  
51    private static void testSimpleProperty(String key, JSONObject expected, JSONObject actual) {
52      Assert.assertEquals(expected.get(key), actual.get(key));
53    }
54  
55    private static void testArrayProperty(String key, JSONObject expected, JSONObject actual) {
56      JSONArray expectedArray = (JSONArray) expected.get(key);
57      JSONArray actualArray = (JSONArray) actual.get(key);
58  
59      Assert.assertEquals(expectedArray.size(), actualArray.size());
60      JSONObject exObject;
61      JSONObject acObject;
62  
63      for (int i = 0; i < expectedArray.size(); i++) {
64        exObject = (JSONObject) expectedArray.get(i);
65        acObject = (JSONObject) actualArray.get(i);
66        Set<String> exEntrySet = exObject.keySet();
67        Assert.assertEquals(exEntrySet.size(), acObject.size());
68        Iterator<String> exIter = exEntrySet.iterator();
69  
70        while (exIter.hasNext()) {
71          String item = exIter.next();
72          Object exValue = exObject.get(item);
73          Object acValue = acObject.get(item);
74          Assert.assertEquals(exValue, acValue);
75        }
76      }
77    }
78  }
