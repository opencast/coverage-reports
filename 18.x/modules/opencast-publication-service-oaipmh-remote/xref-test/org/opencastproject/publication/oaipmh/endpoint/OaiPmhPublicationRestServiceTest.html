64    @Test
65    public void testPublishBrokenMediaPackage() throws Exception {
66      // this should yield an error
67      given().formParam("mediapackage", "bla").expect().statusCode(500).when().post(host("/"));
68    }
69  
70    @Test
71    public void testPublish() throws Exception {
72      final MediaPackage mp = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder().createNew();
73      mp.addCreator(CREATOR);
74      final String mpXml = MediaPackageParser.getAsXml(mp);
75      given().formParam("mediapackage", mpXml).expect().statusCode(200).when().post(host("/"));
76    }
77  
78    @Test
79    public void testPublishUsingRemoteService() throws Exception {
80      final MediaPackage mp = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder().createNew();
81      mp.addCreator(CREATOR);
82      //
83      final ServiceRegistry registry = EasyMock.createNiceMock(ServiceRegistry.class);
84      List<HostRegistration> hostRegistrations = List.of(
85          new HostRegistrationInMemory("http://localhost", "127.0.0.1", "Fake",
86              1.0f, 1, 1024));
87      EasyMock.expect(registry.getHostRegistrations()).andReturn(hostRegistrations).anyTimes();
88      final ServiceRegistration registration = EasyMock.createNiceMock(ServiceRegistration.class);
89      EasyMock.expect(registration.getHost())
90          .andReturn(rt.host(""))
91          .anyTimes();
92      EasyMock.expect(registration.getPath()).andReturn("").anyTimes();
93      EasyMock.expect(registry.getServiceRegistrationsByLoad(EasyMock.anyString()))
94          .andReturn(Arrays.asList(registration))
95          .anyTimes();
96      EasyMock.replay(registry, registration);
97      final OaiPmhPublicationServiceRemoteImpl remote = new OaiPmhPublicationServiceRemoteImpl();
98      remote.setTrustedHttpClient(new TestHttpClient(registry));
99      remote.setRemoteServiceManager(registry);
100     //
101     final Job job = remote.publish(mp, "mmp", new HashSet<String>(), new HashSet<String>(), false);
102     assertEquals(job.getUri(), JOB_URI);
103   }

109   private static final class TestHttpClient extends TrustedHttpClientImpl {
110     TestHttpClient(ServiceRegistry registry) {
111       super("user", "pass");
112       setSecurityService(EasyMock.createNiceMock(SecurityService.class));
113       OrganizationDirectoryService orgDirSvc = EasyMock.createMock(OrganizationDirectoryService.class);
114       Organization org = new DefaultOrganization();
115       EasyMock.expect(orgDirSvc.getOrganizations()).andReturn(List.of(org));
116       EasyMock.replay(orgDirSvc);
117       setOrganizationDirectoryService(orgDirSvc);
118       setServiceRegistry(registry);
119     }

121     /**
122      * Override method with a no-op. In a test environment where the Opencast servlet filter chain is not in place
123      * this will result in an inadvertent call to the REST endpoint which will most likely cause exceptions.
124      */
125     @Override protected String[] getRealmAndNonce(HttpRequestBase request) throws TrustedHttpClientException {
126       return null;
127     }
128   }
