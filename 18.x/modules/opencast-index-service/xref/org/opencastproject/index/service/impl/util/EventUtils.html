60    /**
61     * Loads the metadata for the given event
62     *
63     * @param event
64     *          the source {@link Event}
65     * @param eventCatalogUIAdapter
66     *          the catalog definition
67     * @return a {@link DublinCoreMetadataCollection} instance with all the event metadata
68     *
69     * @throws ParseException
70     */
71    public static DublinCoreMetadataCollection getEventMetadata(Event event, EventCatalogUIAdapter eventCatalogUIAdapter)
72            throws ParseException {
73      DublinCoreMetadataCollection eventMetadata = new DublinCoreMetadataCollection(eventCatalogUIAdapter.getRawFields());
74      setEventMetadataValues(event, eventMetadata);
75      return eventMetadata;
76    }
77  
78    /**
79     * Loads the metadata for the given event
80     *
81     * @param event
82     *          the source {@link Event}
83     * @param eventCatalogUIAdapter
84     *          the catalog definition
85     * @param collectionQueryOverride
86     *          a custom list provider query mapped to every metadata field.
87     *
88     * @return a {@link DublinCoreMetadataCollection} instance with all the event metadata
89     *
90     * @throws ParseException
91     */
92    public static DublinCoreMetadataCollection getEventMetadata(Event event, EventCatalogUIAdapter eventCatalogUIAdapter,
93        ResourceListQuery collectionQueryOverride) throws ParseException {
94      DublinCoreMetadataCollection eventMetadata = new DublinCoreMetadataCollection(
95          eventCatalogUIAdapter.getRawFields(collectionQueryOverride));
96      setEventMetadataValues(event, eventMetadata);
97      return eventMetadata;
98    }

100   /**
101    * Set values of metadata fields from event.
102    *
103    * @param event
104    *          the {@link Event} from the index
105    * @param eventMetadata
106    *          a {@link DublinCoreMetadataCollection} to be modified
107    *
108    * @throws ParseException
109    */
110   public static void setEventMetadataValues(Event event, DublinCoreMetadataCollection eventMetadata)
111           throws ParseException {
112     for (MetadataField field: eventMetadata.getOutputFields().values()) {
113       if (field.getOutputID().equals(DublinCore.PROPERTY_TITLE.getLocalName())) {
114         field.setValue(event.getTitle());
115       }
116       else if (field.getOutputID().equals(DublinCore.PROPERTY_SUBJECT.getLocalName())) {
117         field.setValue(event.getSubject());
118       }
119       else if (field.getOutputID().equals(DublinCore.PROPERTY_DESCRIPTION.getLocalName())) {
120         field.setValue(event.getDescription());
121       }
122       else if (field.getOutputID().equals(DublinCore.PROPERTY_LANGUAGE.getLocalName())) {
123         field.setValue(event.getLanguage());
124       }
125       else if (field.getOutputID().equals(DublinCore.PROPERTY_RIGHTS_HOLDER.getLocalName())) {
126         field.setValue(event.getRights());
127       }
128       else if (field.getOutputID().equals(DublinCore.PROPERTY_LICENSE.getLocalName())) {
129         field.setValue(event.getLicense());
130       }
131       else if (field.getOutputID().equals(DublinCore.PROPERTY_IS_PART_OF.getLocalName())) {
132         field.setValue(event.getSeriesId());
133       }
134       else if (field.getOutputID().equals(DublinCore.PROPERTY_CREATOR.getLocalName())) {
135         field.setValue(event.getPresenters());
136       }
137       else if (field.getOutputID().equals(DublinCore.PROPERTY_CONTRIBUTOR.getLocalName())) {
138         field.setValue(event.getContributors());
139       }
140       else if (field.getOutputID().equals(DublinCore.PROPERTY_SOURCE.getLocalName())) {
141         field.setValue(event.getSource());
142       }
143       else if (field.getOutputID().equals(DublinCore.PROPERTY_CREATED.getLocalName())) {
144 
145         String createdDate = event.getCreated();
146         if (StringUtils.isNotBlank(createdDate)) {
147           field.setValue(new Date(DateTimeSupport.fromUTC(createdDate)));
148         }
149       }
150       else if (field.getOutputID().equals(DublinCore.PROPERTY_IDENTIFIER.getLocalName())) {
151         field.setValue(event.getIdentifier());
152       }
153       else if (field.getOutputID().equals(DublinCore.PROPERTY_PUBLISHER.getLocalName())) {
154         field.setValue(event.getPublisher());
155       }
156       else if (field.getOutputID().equals("duration")) {
157         Long duration = event.getDuration();
158         if (duration != null) {
159           field.setValue(event.getDuration().toString());
160         }
161       }
162       else if (field.getOutputID().equals("location")) {
163         field.setValue(event.getLocation());
164       }
165       else if (field.getOutputID().equals("startDate")) {
166         String recordingStartDate = event.getRecordingStartDate();
167         if (StringUtils.isNotBlank(recordingStartDate)) {
168           Date startDateTime = new Date(DateTimeSupport.fromUTC(recordingStartDate));
169           SimpleDateFormat sdf = new SimpleDateFormat(field.getPattern());
170           sdf.setTimeZone(TimeZone.getTimeZone("UTC"));
171           field.setValue(sdf.format(startDateTime));
172         }
173       }
174     }
175   }

177   /**
178    * A filter to remove all internal channel publications.
179    */
180   public static final Fn<Publication, Boolean> internalChannelFilter = new Fn<Publication, Boolean>() {
181     @Override
182     public Boolean apply(Publication a) {
183       if (InternalPublicationChannel.CHANNEL_ID.equals(a.getChannel()))
184         return false;
185       return true;
186     }
187   };
