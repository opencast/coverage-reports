22  package org.opencastproject.index.service.resources.list.query;
23  
24  import org.opencastproject.index.service.resources.list.provider.JobsListProvider;
25  import org.opencastproject.index.service.resources.list.provider.ServersListProvider;
26  import org.opencastproject.index.service.util.FiltersUtils;
27  import org.opencastproject.list.api.ResourceListFilter;
28  import org.opencastproject.list.api.ResourceListQuery;
29  import org.opencastproject.list.impl.ResourceListQueryImpl;
30  import org.opencastproject.util.data.Option;
31  
32  /**
33   * Query for the jobs list.
34   *
35   * The following filters can be used:
36   * <ul>
37   * <li>hostname</li>
38   * <li>status</li>
39   * </ul>
40   */
41  public class JobsListQuery extends ResourceListQueryImpl {
42  
43    /** Prefix for the filter labels. */
44    private static final String FILTER_PREFIX = "FILTERS.JOBS";
45    /** Hostname filter name. */
46    public static final String FILTER_NAME_HOSTNAME = "hostname";
47    /** Hostname filter label. */
48    public static final String FILTER_LABEL_HOSTNAME = FILTER_PREFIX + ".HOSTNAME.LABEL";
49    /** NodeName filter name. */
50    public static final String FILTER_NAME_NODE_NAME = "nodeName";
51    /** NodeName filter label. */
52    public static final String FILTER_LABEL_NODE_NAME = FILTER_PREFIX + ".NODE_NAME.LABEL";
53    /** Status filter name. */
54    public static final String FILTER_NAME_STATUS = "status";
55    /** Status filter label. */
56    public static final String FILTER_LABEL_STATUS = FILTER_PREFIX + ".STATUS.LABEL";
57  
58    /** Default constructor. */
59    public JobsListQuery() {
60      super();
61      availableFilters.add(createHostnameFilter(Option.<String> none()));
62      availableFilters.add(createNodeNameFilter(Option.<String> none()));
63      availableFilters.add(createStatusFilter(Option.<String> none()));
64    }
65  
66    /**
67     * Copy constructor for the base class {@code ResourceListQuery}.
68     *
69     * @param query copy values from the given query
70     */
71    public JobsListQuery(ResourceListQuery query) {
72      this();
73      availableFilters.addAll(query.getAvailableFilters());
74  
75      for (ResourceListFilter filter : query.getFilters())
76        addFilter(filter);
77  
78      sortBy = query.getSortBy();
79      if (query.getOffset().isSome())
80        setOffset(query.getOffset().get());
81      if (query.getLimit().isSome())
82        setLimit(query.getLimit().get());
83    }
84  
85    /**
86     * Add a {@link ResourceListFilter} filter to the query with the given hostname.
87     *
88     * @param hostname the hostname to filter for
89     */
90    public void withHostname(String hostname) {
91      addFilter(createHostnameFilter(Option.option(hostname)));
92    }
93  
94    /**
95     * Add a {@link ResourceListFilter} filter to the query with the given node name.
96     *
97     * @param nodeName the node name to filter for
98     */
99    public void withNodeName(String nodeName) {
100     addFilter(createNodeNameFilter(Option.option(nodeName)));
101   }
102 
103   /**
104    * Add a {@link ResourceListFilter} filter to the query with the given status.
105    *
106    * @param status the status to filter for
107    */
108   public void withStatus(String status) {
109     addFilter(createStatusFilter(Option.option(status)));
110   }
111 
112   /**
113    * Add a {@link ResourceListFilter} filter to the query with the given free text.
114    *
115    * @param freeText the free text to filter for
116    */
117   public void withFreeText(String freeText) {
118     addFilter(createFreeTextFilter(Option.option(freeText)));
119   }
120 
121   /**
122    * Returns an {@link Option} containing the hostname used to filter if set.
123    * {@link Option#none()} otherwise.
124    *
125    * @return an {@link Option} containing the hostname or none.
126    */
127   public Option<String> getHostname() {
128     return getFilterValue(FILTER_NAME_HOSTNAME);
129   }
130 
131   /**
132    * Returns an {@link Option} containing the node name used to filter if set.
133    * {@link Option#none()} otherwise.
134    *
135    * @return an {@link Option} containing the node name or none.
136    */
137   public Option<String> getNodeName() {
138     return getFilterValue(FILTER_NAME_NODE_NAME);
139   }
140 
141   /**
142    * Returns an {@link Option} containing the status used to filter if set.
143    * {@link Option#none()} otherwise.
144    *
145    * @return an {@link Option} containing the status or none.
146    */
147   public Option<String> getStatus() {
148     return getFilterValue(FILTER_NAME_STATUS);
149   }
150 
151   /**
152    * Returns an {@link Option} containing the free text used to filter if set.
153    * {@link Option#none()} otherwise.
154    *
155    * @return an {@link Option} containing the free text or none.
156    */
157   public Option<String> getFreeText() {
158     return getFilterValue(ResourceListFilter.FREETEXT);
159   }
160 
161   /**
162    * Create a new {@link ResourceListFilter} based on a hostname.
163    *
164    * @param value the hostname to filter on wrapped in an {@link Option} or {@link Option#none()}
165    * @return a new {@link ResourceListFilter} for a hostname based query
166    */
167   public static <String> ResourceListFilter<String> createHostnameFilter(Option<String> value) {
168     return FiltersUtils.generateFilter(
169             value,
170             FILTER_NAME_HOSTNAME,
171             FILTER_LABEL_HOSTNAME,
172             ResourceListFilter.SourceType.SELECT,
173             Option.some(ServersListProvider.LIST_HOSTNAME));
174   }
175 
176   /**
177    * Create a new {@link ResourceListFilter} based on a nodeName.
178    *
179    * @param value the nodeName to filter on wrapped in an {@link Option} or {@link Option#none()}
180    * @return a new {@link ResourceListFilter} for a nodeName based query
181    */
182   public static ResourceListFilter<String> createNodeNameFilter(Option<String> value) {
183     return FiltersUtils.generateFilter(
184             value,
185             FILTER_NAME_NODE_NAME,
186             FILTER_LABEL_NODE_NAME,
187             ResourceListFilter.SourceType.SELECT,
188             Option.some(ServersListProvider.LIST_NODE_NAME));
189   }
190 
191   /**
192    * Create a new {@link ResourceListFilter} based on a status.
193    *
194    * @param value the status to filter on wrapped in an {@link Option} or {@link Option#none()}
195    * @return a new {@link ResourceListFilter} for a status based query
196    */
197   public static <String> ResourceListFilter<String> createStatusFilter(Option<String> value) {
198     return FiltersUtils.generateFilter(
199             value,
200             FILTER_NAME_STATUS,
201             FILTER_LABEL_STATUS,
202             ResourceListFilter.SourceType.SELECT,
203             Option.some(JobsListProvider.LIST_STATUS));
204   }
205 
206   /**
207    * Create a new {@link ResourceListFilter} based on a free text.
208    *
209    * @param value the free text to filter on wrapped in an {@link Option} or {@link Option#none()}
210    * @return a new {@link ResourceListFilter} for a free text based query
211    */
212   public static <String> ResourceListFilter<String> createFreeTextFilter(Option<String> value) {
213     return FiltersUtils.generateFilter(
214             value,
215             ResourceListFilter.FREETEXT,
216             ResourceListFilter.FREETEXT,
217             ResourceListFilter.SourceType.FREETEXT,
218             null);
219   }
