83    @Before
84    public void setUp() throws URISyntaxException, NotFoundException, IOException, ListProviderException {
85      TreeMap<String, String> collection = new TreeMap<String, String>();
86      collection.put("Entry 1", "Value 1");
87      collection.put("Entry 2", "Value 2");
88      collection.put("Entry 3", "Value 3");
89  
90      BundleContext bundleContext = EasyMock.createNiceMock(BundleContext.class);
91      EasyMock.replay(bundleContext);
92  
93      listProvidersService = EasyMock.createMock(ListProvidersService.class);
94      EasyMock.expect(
95              listProvidersService.getList(EasyMock.anyString(), EasyMock.anyObject(ResourceListQueryImpl.class),
96                      EasyMock.anyBoolean())).andReturn(collection).anyTimes();
97      EasyMock.expect(
98              listProvidersService.isTranslatable(EasyMock.anyString()))
99              .andThrow(new ListProviderException("not implemented")).anyTimes();
100     EasyMock.expect(
101             listProvidersService.getDefault(EasyMock.anyString()))
102             .andThrow(new ListProviderException("not implemented")).anyTimes();
103     EasyMock.replay(listProvidersService);
104 
105     Properties props = new Properties();
106     InputStream in = getClass().getResourceAsStream("/catalog-adapter/event.properties");
107     props.load(in);
108     in.close();
109 
110     eventProperties = PropertiesUtil.toDictionary(props);
111 
112     mediaPackageElementFlavor = new MediaPackageElementFlavor(FLAVOR_STRING.split("/")[0], FLAVOR_STRING.split("/")[1]);
113 
114     URI eventDublincoreURI = getClass().getResource("/catalog-adapter/event-dublincore.xml").toURI();
115 
116     workspace = EasyMock.createMock(Workspace.class);
117     EasyMock.expect(workspace.read(eventDublincoreURI))
118             .andAnswer(() -> new FileInputStream(new File(eventDublincoreURI)));
119     EasyMock.replay(workspace);
120 
121     Catalog eventCatalog = EasyMock.createMock(Catalog.class);
122     EasyMock.expect(eventCatalog.getURI()).andReturn(eventDublincoreURI).anyTimes();
123     EasyMock.replay(eventCatalog);
124     Catalog[] catalogs = { eventCatalog };
125     mediapackage = EasyMock.createMock(MediaPackage.class);
126     EasyMock.expect(mediapackage.getCatalogs(mediaPackageElementFlavor)).andReturn(catalogs).anyTimes();
127     EasyMock.replay(mediapackage);
128 
129     dictionary = new Hashtable<String, String>();
130 
131     dictionary.put(CONF_ORGANIZATION_KEY, ORGANIZATION_STRING);
132     dictionary.put(CONF_FLAVOR_KEY, FLAVOR_STRING);
133     dictionary.put(CONF_TITLE_KEY, TITLE_STRING);
134 
135     dictionary.put(MetadataField.CONFIG_PROPERTY_PREFIX + ".title." + MetadataField.CONFIG_INPUT_ID_KEY, title);
136     dictionary.put(MetadataField.CONFIG_PROPERTY_PREFIX + ".title." + MetadataField.CONFIG_LABEL_KEY, label);
137     dictionary.put(MetadataField.CONFIG_PROPERTY_PREFIX + ".title." + MetadataField.CONFIG_TYPE_KEY, type);
138     dictionary.put(MetadataField.CONFIG_PROPERTY_PREFIX + ".title." + MetadataField.CONFIG_READ_ONLY_KEY, readOnly);
139     dictionary.put(MetadataField.CONFIG_PROPERTY_PREFIX + ".title." + MetadataField.CONFIG_REQUIRED_KEY, required);
140     dictionary.put(MetadataField.CONFIG_PROPERTY_PREFIX + ".title." + MetadataField.CONFIG_LIST_PROVIDER_KEY,
141             listProvider);
142     dictionary.put(MetadataField.CONFIG_PROPERTY_PREFIX + ".title." + MetadataField.CONFIG_COLLECTION_ID_KEY,
143             collectionID);
144   }

146   @Test
147   public void testGetFields() throws Exception {
148     String eventJson = IOUtils.toString(getClass().getResource("/catalog-adapter/event.json"));
149 
150     ConfigurableEventDCCatalogUIAdapter configurationDublinCoreCatalogUIAdapter = new ConfigurableEventDCCatalogUIAdapter();
151     configurationDublinCoreCatalogUIAdapter.setListProvidersService(listProvidersService);
152     configurationDublinCoreCatalogUIAdapter.setWorkspace(workspace);
153     configurationDublinCoreCatalogUIAdapter.updated(eventProperties);
154 
155     DublinCoreMetadataCollection abstractMetadata = configurationDublinCoreCatalogUIAdapter.getFields(mediapackage);
156     assertThat(eventJson, SameJSONAs.sameJSONAs(RestUtils.getJsonString(MetadataJson.collectionToJson(abstractMetadata,
157             true)))
158             .allowingAnyArrayOrdering());
159   }
