254   /**
255    * {@inheritDoc}
256    */
257   @Override
258   public SmilObject getElementOrNull(String elementId) {
259     if (getId().equals(elementId)) {
260       return this;
261     }
262     for (SmilMediaParam media : getParams()) {
263       SmilObject element = ((SmilMediaParamImpl) media).getElementOrNull(elementId);
264       if (element != null) {
265         return element;
266       }
267     }
268     return null;
269   }
270 
271   /**
272    * {@inheritDoc}
273    */
274   @Override
275   public SmilObject removeElement(String elementId) {
276     SmilObject child = null;
277     for (SmilObject element : params) {
278       if (element.getId().equals(elementId)) {
279         child = element;
280         break;
281       } else {
282         SmilObject removed = ((SmilObjectImpl) element).removeElement(elementId);
283         if (removed != null) {
284           return removed;
285         }
286       }
287     }
288     if (child != null) {
289       params.remove(child);
290     }
291     return child;
292   }
293 
294   /**
295    * Convert time unit to milliseconds.
296    *
297    * @param timeUnit to convert
298    * @return time unit in milliseconds
299    * @throws SmilException if time unit format can't parsed
300    */
301   protected long convertTimeToMS(String timeUnit) throws SmilException {
302     TimeUnit unit;
303     long time = -1;
304     if (timeUnit.endsWith("ms")) {
305       unit = TimeUnit.MILLISECONDS;
306       time = Long.parseLong(timeUnit.replace("ms", "").trim());
307     } else if (timeUnit.endsWith("s")) {
308       unit = TimeUnit.SECONDS;
309       time = (long) Double.parseDouble(timeUnit.replace("s", "").trim());
310     } else {
311       // TODO: parse other formats
312       throw new SmilException("failed parsing time unit");
313     }
314 
315     return unit.toMillis(time);
316   }

323     for (SmilObject child : getParams()) {
324       elements.add(child);
325       ((SmilObjectImpl) child).putAllChilds(elements);
326     }
327   }
