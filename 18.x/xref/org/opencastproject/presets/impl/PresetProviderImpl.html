22  package org.opencastproject.presets.impl;
23  
24  import org.opencastproject.presets.api.PresetProvider;
25  import org.opencastproject.security.api.SecurityService;
26  import org.opencastproject.series.api.SeriesService;
27  import org.opencastproject.util.NotFoundException;
28  
29  import org.apache.commons.lang3.StringUtils;
30  import org.osgi.service.component.annotations.Component;
31  import org.osgi.service.component.annotations.Reference;
32  import org.slf4j.Logger;
33  import org.slf4j.LoggerFactory;
34  
35  /**
36   * A service that provides presets.
37   */
38  @Component(
39      immediate = true,
40      service = PresetProvider.class,
41      property = {
42          "service.description=Provides Series and Organization Presets"
43      }
44  )
45  public class PresetProviderImpl implements PresetProvider {
46  
47    private static final Logger logger = LoggerFactory.getLogger(PresetProvider.class);
48  
49    private SeriesService seriesService;
50    /** The security service to get the current user's organization */
51    private SecurityService securityService;
52  
53    @Reference
54    public void setSeriesService(SeriesService seriesService) {
55      this.seriesService = seriesService;
56    }
57  
58    @Reference
59    public void setSecurityService(SecurityService securityService) {
60      this.securityService = securityService;
61    }
62  
63    @Override
64    public String getProperty(String seriesID, String propertyName) throws NotFoundException {
65      String propertyValue = "";
66      if (StringUtils.trimToNull(seriesID) != null) {
67        try {
68          propertyValue = seriesService.getSeriesProperty(seriesID, propertyName);
69        } catch (NotFoundException e) {
70          logger.debug("The property {} was not found in the series {} so we will try to find it in the organization.",
71                  propertyName, seriesID);
72        } catch (Exception e) {
73          logger.warn(
74              "Unable to get the property {} from the series {} so we will try to find it in the "
75                  + "organization. The exception was ",
76              propertyName, seriesID, e);
77        }
78      }
79      if (StringUtils.isBlank(propertyValue)) {
80        propertyValue = securityService.getOrganization().getProperties().get(propertyName);
81        if (StringUtils.isBlank(propertyValue)) {
82          throw new NotFoundException("Unable to find the property in either the series or organization");
83        }
84      }
85      return propertyValue;
86    }
87  
88  }
