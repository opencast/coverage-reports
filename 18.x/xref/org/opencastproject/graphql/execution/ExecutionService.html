56  @Component(
57      service = { ExecutionService.class }
58  )
59  public class ExecutionService {
60  
61    private static final Logger logger = LoggerFactory.getLogger(ExecutionService.class);
62  
63    private final SecurityService securityService;
64    private final SchemaService schemaService;
65    private final BundleContext bundleContext;
66  
67    private final Map<String, GraphQL> organizationGraphQL = new ConcurrentHashMap<>();
68  
69    public @interface ExecutionConfiguration {
70      int execution_max_query_complexity() default 1000;
71  
72      int execution_max_query_depth() default 25;
73  
74    }
75  
76    private ExecutionConfiguration config;
77  
78    @Activate
79    public ExecutionService(
80        @Reference SchemaService schemaService,
81        @Reference SecurityService securityService,
82        BundleContext bundleContext,
83        ExecutionConfiguration config
84    ) {
85      this.schemaService = schemaService;
86      this.securityService = securityService;
87      this.bundleContext = bundleContext;
88  
89      updateConfiguration(config);
90    }
91  
92    @Modified
93    public void updateConfiguration(ExecutionConfiguration config) {
94      if (config.execution_max_query_complexity() <= 0) {
95        throw new IllegalArgumentException("execution_max_query_complexity must be greater than 0");
96      }
97      if (config.execution_max_query_depth() <= 0) {
98        throw new IllegalArgumentException("execution_max_query_depth must be greater than 0");
99      }
100     this.config = config;
101   }
102 
103   public ExecutionResult execute(
104       String query,
105       String operationName,
106       Map<String, Object> variables,
107       Map<String, Object> extensions) {
108 
109     Map<?, Object> context = new HashMap<>();
110 
111     return execute(ExecutionInput.newExecutionInput()
112         .query(query)
113         .operationName(operationName)
114         .variables((variables != null) ? variables : Collections.emptyMap())
115         .extensions((extensions != null) ? extensions : Collections.emptyMap())
116         .graphQLContext(context)
117         .build());
118   }
119 
120   public ExecutionResult execute(ExecutionInput executionInput) {
121     try {
122       var context = OpencastContextManager.initiateContext(bundleContext);
123 
124       context.setOrganization(securityService.getOrganization());
125       context.setUser(securityService.getUser());
126 
127       executionInput.getGraphQLContext().put(OpencastContextManager.CONTEXT, context);
128 
129       var graphQL = getGraphQL(securityService.getOrganization().getId());
130 
131       if (graphQL == null) {
132         return ExecutionErrorResult.newExecutionResult()
133             .addError(new GraphQLNotFoundException("No GraphQL schema found for organization `"
134                 + securityService.getOrganization().getId() + "`")
135             ).build();
136       }
137 
138       return graphQL.execute(executionInput);
139     } finally {
140       OpencastContextManager.clearContext();
141     }
142   }
143 
144   private GraphQL getGraphQL(String organizationId) {
145     GraphQL graphQL = organizationGraphQL.get(organizationId);
146     GraphQLSchema schema = schemaService.get(organizationId);
147 
148     List<Instrumentation> chainedList = new ArrayList<>(
149         List.of(new MaxQueryDepthInstrumentation(config.execution_max_query_depth()),
150             new MaxQueryComplexityInstrumentation(config.execution_max_query_complexity())));
151 
152     if (logger.isTraceEnabled()) {
153       logger.trace("Enabling tracing instrumentation for organization `{}`", organizationId);
154       chainedList.add(new TracingInstrumentation());
155     }
156 
157     if (schema == null) {
158       return null;
159     }
160 
161     if (graphQL == null || !schema.equals(graphQL.getGraphQLSchema())) {
162       var exceptionHandler = new OpencastDataFetcherExceptionHandler();
163       graphQL = GraphQL.newGraphQL(schema)
164           .queryExecutionStrategy(new AsyncExecutionStrategy(exceptionHandler))
165           .mutationExecutionStrategy(new AsyncSerialExecutionStrategy(exceptionHandler))
166           .preparsedDocumentProvider(new QueryCache())
167           .executionIdProvider(new OrganizationExecutionIdProvider(organizationId))
168           .defaultDataFetcherExceptionHandler(exceptionHandler)
169           .instrumentation(new ChainedInstrumentation(chainedList))
170           .build();
171       organizationGraphQL.put(organizationId, graphQL);
172     }
173 
174     return graphQL;
175   }
