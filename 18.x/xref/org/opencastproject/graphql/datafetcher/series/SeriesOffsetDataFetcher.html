22  package org.opencastproject.graphql.datafetcher.series;
23  
24  import org.opencastproject.elasticsearch.api.SearchIndexException;
25  import org.opencastproject.elasticsearch.api.SearchResult;
26  import org.opencastproject.elasticsearch.index.ElasticsearchIndex;
27  import org.opencastproject.elasticsearch.index.objects.series.Series;
28  import org.opencastproject.elasticsearch.index.objects.series.SeriesSearchQuery;
29  import org.opencastproject.graphql.datafetcher.ElasticsearchDataFetcher;
30  import org.opencastproject.graphql.exception.GraphQLRuntimeException;
31  import org.opencastproject.graphql.exception.OpencastErrorType;
32  import org.opencastproject.graphql.execution.context.OpencastContext;
33  import org.opencastproject.graphql.series.GqlSeriesList;
34  import org.opencastproject.graphql.type.input.SeriesOrderByInput;
35  import org.opencastproject.security.api.Permissions;
36  import org.opencastproject.security.api.SecurityService;
37  import org.opencastproject.security.api.User;
38  
39  import java.util.Objects;
40  
41  import graphql.schema.DataFetchingEnvironment;
42  
43  public class SeriesOffsetDataFetcher extends ElasticsearchDataFetcher<GqlSeriesList> {
44  
45    private User user;
46    private boolean writeOnly;
47  
48    public SeriesOffsetDataFetcher withUser(User user) {
49      this.user = user;
50      return this;
51    }
52  
53    public SeriesOffsetDataFetcher writeOnly(boolean writeOnly) {
54      this.writeOnly = writeOnly;
55      return this;
56    }
57  
58    @Override
59    public GqlSeriesList get(OpencastContext opencastContext, DataFetchingEnvironment dataFetchingEnvironment) {
60      SecurityService securityService = opencastContext.getService(SecurityService.class);
61      ElasticsearchIndex searchIndex = opencastContext.getService(ElasticsearchIndex.class);
62  
63      SeriesSearchQuery seriesSearchQuery = new SeriesSearchQuery(
64          securityService.getOrganization().getId(),
65          Objects.requireNonNullElse(user, securityService.getUser())
66      );
67  
68      seriesSearchQuery = addPaginationParams(seriesSearchQuery, dataFetchingEnvironment);
69  
70      seriesSearchQuery = addSeriesOrderByParams(seriesSearchQuery, dataFetchingEnvironment);
71  
72      seriesSearchQuery = addQueryParams(seriesSearchQuery, dataFetchingEnvironment);
73  
74      if (writeOnly) {
75        seriesSearchQuery.withAction(Permissions.Action.WRITE);
76      }
77  
78      try {
79        SearchResult<Series> result = searchIndex.getByQuery(seriesSearchQuery);
80        return new GqlSeriesList(result);
81      } catch (SearchIndexException e) {
82        throw new GraphQLRuntimeException(OpencastErrorType.InternalError, e);
83      }
84    }
85  
86    public SeriesSearchQuery addSeriesOrderByParams(
87        SeriesSearchQuery seriesSearchQuery,
88        final DataFetchingEnvironment environment
89    ) {
90      SeriesOrderByInput seriesOrderBy = parseObjectParam("orderBy", SeriesOrderByInput.class, environment);
91      if (seriesOrderBy == null) {
92        return seriesSearchQuery;
93      }
94  
95      if (seriesOrderBy.getTitle() != null) {
96        seriesSearchQuery.sortByTitle(seriesOrderBy.getTitle().getOrder());
97      }
98      if (seriesOrderBy.getSubject() != null) {
99        seriesSearchQuery.sortBySubject(seriesOrderBy.getSubject().getOrder());
100     }
101     if (seriesOrderBy.getCreator() != null) {
102       seriesSearchQuery.sortByCreator(seriesOrderBy.getCreator().getOrder());
103     }
104     if (seriesOrderBy.getPublishers() != null) {
105       seriesSearchQuery.sortByPublishers(seriesOrderBy.getPublishers().getOrder());
106     }
107     if (seriesOrderBy.getContributors() != null) {
108       seriesSearchQuery.sortByContributors(seriesOrderBy.getContributors().getOrder());
109     }
110     if (seriesOrderBy.getDescription() != null) {
111       seriesSearchQuery.sortByDescription(seriesOrderBy.getDescription().getOrder());
112     }
113     if (seriesOrderBy.getLanguage() != null) {
114       seriesSearchQuery.sortByLanguage(seriesOrderBy.getLanguage().getOrder());
115     }
116     if (seriesOrderBy.getRightHolder() != null) {
117       seriesSearchQuery.sortByRightsHolder(seriesOrderBy.getRightHolder().getOrder());
118     }
119     if (seriesOrderBy.getLicense() != null) {
120       seriesSearchQuery.sortByLicense(seriesOrderBy.getLicense().getOrder());
121     }
122     if (seriesOrderBy.getCreated() != null) {
123       seriesSearchQuery.sortByCreatedDateTime(seriesOrderBy.getCreated().getOrder());
124     }
125 
126     return seriesSearchQuery;
127   }
128 
129 }
