87    protected static String dataResolutionToInfluxGrouping(DataResolution dataResolution) {
88      switch (dataResolution) {
89        case HOURLY:
90          return " GROUP BY time(1h)";
91        case DAILY:
92          return " GROUP BY time(1d)";
93        case WEEKLY:
94          return " GROUP BY time(1w, -3d)";  // -3d because otherwise, influx starts weeks on Thursdays
95        case MONTHLY:
96          return ""; // not available in influx -> we have to do multiple queries with different periods
97        case YEARLY:
98          return ""; // not available in influx -> we have to do multiple queries with different periods
99        default:
100         throw new IllegalArgumentException("unmapped DataResolution: " + dataResolution.name());
101     }
102   }
103 
104   protected static List<Tuple<Instant, Instant>> getPeriods(
105       final Instant from,
106       final Instant to,
107       DataResolution resolution,
108       ZoneId zoneId
109   ) {
110     switch (resolution) {
111       case MONTHLY:
112         return getMonthPeriods(getMonths(from, to, zoneId), from, to);
113       case YEARLY:
114         return getYearPeriods(getYears(from, to, zoneId), from, to);
115       case DAILY: // Will be handled by influx grouping. No need to divide into periods.
116       case WEEKLY: // Will be handled by influx grouping. No need to divide into periods.
117       default:
118         return Collections.singletonList(new Tuple<>(from, to));
119     }
120   }
121 
122   private static List<Tuple<Instant, Instant>> getMonthPeriods(List<YearMonth> months, Instant from, Instant to) {
123     final List<Tuple<Instant, Instant>> result = new ArrayList<>();
124     for (YearMonth month : months) {
125       final Instant start = month.atDay(1).atStartOfDay().toInstant(ZoneOffset.UTC);
126       final Instant end = month.atEndOfMonth().atTime(23, 59, 59, 1_000_000_000 - 1).toInstant(ZoneOffset.UTC);
127       result.add(new Tuple<>(Ordering.natural().max(start, from), Ordering.natural().min(end, to)));
128     }
129     return result;
130   }
131 
132   private static List<Tuple<Instant, Instant>> getYearPeriods(List<Year> years, Instant from, Instant to) {
133     final List<Tuple<Instant, Instant>> result = new ArrayList<>();
134     for (Year year : years) {
135       final Instant start = year.atDay(1).atStartOfDay().toInstant(ZoneOffset.UTC);
136       final Instant end = year.atMonthDay(MonthDay.of(12, 31))
137           .atTime(23, 59, 59, 1_000_000_000 - 1).toInstant(ZoneOffset.UTC);
138       result.add(new Tuple<>(Ordering.natural().max(start, from), Ordering.natural().min(end, to)));
139     }
140     return result;
141   }
142 
143   private static List<YearMonth> getMonths(final Instant from, final Instant to, ZoneId zoneId) {
144     LocalDateTime localStart = LocalDateTime.ofInstant(from, zoneId);
145     final LocalDateTime localEnd = LocalDateTime.ofInstant(to, zoneId);
146     final List<YearMonth> months = new ArrayList<>();
147     while (!localStart.isAfter(localEnd)) {
148       months.add(YearMonth.from(localStart));
149       localStart = localStart.plusMonths(1).withDayOfMonth(1).withHour(0).withMinute(0).withSecond(0).withNano(0);
150     }
151     return months;
152   }
153 
154   private static List<Year> getYears(final Instant from, final Instant to, ZoneId zoneId) {
155     LocalDateTime localStart = LocalDateTime.ofInstant(from, zoneId);
156     final LocalDateTime localEnd = LocalDateTime.ofInstant(to, zoneId);
157     final List<Year> years = new ArrayList<>();
158     while (!localStart.isAfter(localEnd)) {
159       years.add(Year.from(localStart));
160       localStart = localStart.plusYears(1).withMonth(1)
161           .withDayOfMonth(1).withHour(0).withMinute(0).withSecond(0).withNano(0);
162     }
163     return years;
164   }

171   @Override
172   public boolean equals(Object o) {
173     if (!(o instanceof InfluxStatisticsProvider)) {
174       return false;
175     }
176     final StatisticsProvider other = (StatisticsProvider) o;
177     return this.getId().equals(other.getId());
178   }
