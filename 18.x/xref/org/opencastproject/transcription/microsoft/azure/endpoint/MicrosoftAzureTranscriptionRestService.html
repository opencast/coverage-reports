22  package org.opencastproject.transcription.microsoft.azure.endpoint;
23  
24  import org.opencastproject.job.api.JobProducer;
25  import org.opencastproject.rest.AbstractJobProducerEndpoint;
26  import org.opencastproject.serviceregistry.api.ServiceRegistry;
27  import org.opencastproject.transcription.microsoft.azure.MicrosoftAzureTranscriptionService;
28  import org.opencastproject.util.doc.rest.RestService;
29  
30  import org.osgi.service.component.ComponentContext;
31  import org.osgi.service.component.annotations.Activate;
32  import org.osgi.service.component.annotations.Component;
33  import org.osgi.service.component.annotations.Reference;
34  import org.osgi.service.jaxrs.whiteboard.propertytypes.JaxrsResource;
35  import org.slf4j.Logger;
36  import org.slf4j.LoggerFactory;
37  
38  import javax.ws.rs.Path;
39  
40  @Path("/transcripts/azure")
41  @RestService(
42      name = "MicrosoftAzureTranscriptionRestService",
43      title = "Transcription Service REST Endpoint (uses Microsoft Azure Speech services)",
44      abstractText = "Uses external service to generate transcriptions of recordings.",
45      notes = {
46          "All paths above are relative to the REST endpoint base (something like http://your.server/transcripts)",
47      }
48  )
49  @Component(
50      immediate = true,
51      service = MicrosoftAzureTranscriptionRestService.class,
52      property = {
53          "service.description=Microsoft Azure Transcription REST Endpoint",
54          "opencast.service.type=org.opencastproject.transcription.microsoft.azure",
55          "opencast.service.path=/transcripts/azure",
56          "opencast.service.jobproducer=true"
57      }
58  )
59  @JaxrsResource
60  public class MicrosoftAzureTranscriptionRestService extends AbstractJobProducerEndpoint {
61  
62    /**
63     * The logger
64     */
65    private static final Logger logger = LoggerFactory.getLogger(MicrosoftAzureTranscriptionRestService.class);
66  
67    /**
68     * The transcription service
69     */
70    protected MicrosoftAzureTranscriptionService service;
71  
72    /**
73     * The service registry
74     */
75    protected ServiceRegistry serviceRegistry = null;
76  
77    @Activate
78    public void activate(ComponentContext cc) {
79      logger.debug("activate()");
80    }
81  
82    @Reference
83    public void setTranscriptionService(MicrosoftAzureTranscriptionService service) {
84      this.service = service;
85    }
86  
87    @Reference
88    public void setServiceRegistry(ServiceRegistry service) {
89      this.serviceRegistry = service;
90    }
91  
92  
93    @Override
94    public JobProducer getService() {
95      return service;
96    }
97  
98    @Override
99    public ServiceRegistry getServiceRegistry() {
100     return serviceRegistry;
101   }
102 
103 }
