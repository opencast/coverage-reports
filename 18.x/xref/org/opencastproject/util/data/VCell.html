23  package org.opencastproject.util.data;
24  
25  import static org.opencastproject.util.EqualsUtil.ne;
26  import static org.opencastproject.util.RequireUtil.notNull;
27  import static org.opencastproject.util.data.Cells.fcell;
28  import static org.opencastproject.util.data.Tuple.tuple;
29  
30  /**
31   * Value cell, a mutable data container.
32   *
33   * Cells provide a pattern that reverses the listener pattern. Instead of propagating changes of a value to registered
34   * listeners via callbacks, the dependent object just holds a reference to a cell and pulls the value when needed.
35   *
36   * Cells must not contain null!
37   *
38   * @param <A>
39   *          the value type
40   */
41  public final class VCell<A> extends Cell<A> {
42    private volatile A a;
43    private int change;
44    private final boolean stable;
45  
46    private final Object lock = new Object();
47  
48    public VCell(A a, boolean stable) {
49      this.a = notNull(a, "a");
50      this.stable = stable;
51      change = 1;
52    }
53  
54    /** Constructor function. */
55    public static <A> VCell<A> cell(A a) {
56      return new VCell<A>(a, true);
57    }
58  
59    /** Create a cell containing none. */
60    public static <A> VCell<Option<A>> ocell() {
61      return cell(Option.<A> none());
62    }
63  
64    /** Get the cell's value. */
65    @Override
66    public A get() {
67      return a;
68    }
69  
70    @Override
71    protected Tuple<A, Object> change() {
72      synchronized (lock) {
73        if (!stable)
74          change += 1;
75        return tuple(a, (Object) change);
76      }
77    }
78  
79    /** Set the cell's value. */
80    public void set(A a) {
81      synchronized (lock) {
82        if (ne(a, this.a)) {
83          this.a = a;
84          change += 1;
85        }
86      }
87    }
88  
89    public <B> Cell<B> lift(Function<A, B> f) {
90      return fcell(this, f);
91    }
92  }
