237   @Override
238   public WorkflowOperationResult start(final WorkflowInstance workflowInstance, final JobContext context)
239       throws WorkflowOperationException {
240 
241     ConfiguredTagsAndFlavors tagsAndFlavors = getTagsAndFlavors(workflowInstance,
242         Configuration.many, Configuration.many, Configuration.many, Configuration.none);
243     final MediaPackage mediaPackage = workflowInstance.getMediaPackage();
244     final WorkflowOperationInstance operation = workflowInstance.getCurrentOperation();
245     final List<MediaPackageElementFlavor> configuredSourceFlavors = tagsAndFlavors.getSrcFlavors();
246     final List<String> configuredSourceTags = tagsAndFlavors.getSrcTags();
247     final List<String> configuredTargetTags = tagsAndFlavors.getTargetTags();
248     final boolean noSuffix = Boolean.parseBoolean(trimToEmpty(operation.getConfiguration(NO_SUFFIX)));
249     final String startDateString = trimToEmpty(operation.getConfiguration(SET_START_DATE));
250     final String seriesId = trimToEmpty(operation.getConfiguration(SET_SERIES_ID));
251     final String title = trimToEmpty(operation.getConfiguration(SET_TITLE));
252     final int numberOfEvents = Integer.parseInt(operation.getConfiguration(NUMBER_PROPERTY));
253     final String configuredPropertyNamespaces = trimToEmpty(operation.getConfiguration(PROPERTY_NAMESPACES_PROPERTY));
254     int maxNumberOfEvents = MAX_NUMBER_DEFAULT;
255 
256     if (operation.getConfiguration(MAX_NUMBER_PROPERTY) != null) {
257       maxNumberOfEvents = Integer.parseInt(operation.getConfiguration(MAX_NUMBER_PROPERTY));
258     }
259 
260     if (numberOfEvents > maxNumberOfEvents) {
261       throw new WorkflowOperationException("Number of events to create exceeds the maximum of "
262           + maxNumberOfEvents + ". Aborting.");
263     }
264 
265     SeriesInformation series = null;
266     AccessControlList seriesAccessControl = null;
267     if (!seriesId.isEmpty() && !seriesId.startsWith("${") && !seriesId.endsWith("}")) {
268       try {
269         final DublinCoreCatalog dc = seriesService.getSeries(seriesId);
270         series = new SeriesInformation(seriesId, dc, dc.get(DublinCore.PROPERTY_TITLE).get(0).getValue());
271         seriesAccessControl = seriesService.getSeriesAccessControl(seriesId);
272       } catch (SeriesException e) {
273         throw new WorkflowOperationException(e);
274       } catch (NotFoundException e) {
275         throw new WorkflowOperationException("couldn't find series for ID \"" + seriesId + "\"");
276       } catch (UnauthorizedException e) {
277         throw new WorkflowOperationException("not allowed to access series \"" + seriesId + "\"");
278       }
279     }
280 
281     logger.info("Creating {} new media packages from media package with id {}.", numberOfEvents,
282         mediaPackage.getIdentifier());
283 
284     final String[] propertyNamespaces = split(configuredPropertyNamespaces, ",");
285     final String copyNumberPrefix = trimToEmpty(operation.getConfiguration(COPY_NUMBER_PREFIX_PROPERTY));
286 
287     final SimpleElementSelector elementSelector = new SimpleElementSelector();
288     for (MediaPackageElementFlavor flavor : configuredSourceFlavors) {
289       elementSelector.addFlavor(flavor);
290     }
291 
292     final List<String> removeTags = new ArrayList<>();
293     final List<String> addTags = new ArrayList<>();
294     final List<String> overrideTags = new ArrayList<>();
295 
296     for (String tag : configuredTargetTags) {
297       if (tag.startsWith(MINUS)) {
298         removeTags.add(tag);
299       } else if (tag.startsWith(PLUS)) {
300         addTags.add(tag);
301       } else {
302         overrideTags.add(tag);
303       }
304     }
305 
306     for (String tag : configuredSourceTags) {
307       elementSelector.addTag(tag);
308     }
309 
310     // Filter elements to copy based on input tags and input flavors
311     final Collection<MediaPackageElement> elements = elementSelector.select(mediaPackage, false);
312     final Collection<Publication> internalPublications = new HashSet<>();
313 
314     final List<String> seriesAclTags = new ArrayList<>();
315     for (MediaPackageElement e : mediaPackage.getElements()) {
316       if (e instanceof Publication) {
317         if (InternalPublicationChannel.CHANNEL_ID.equals(((Publication) e).getChannel())) {
318           internalPublications.add((Publication) e);
319         }
320         elements.remove(e); // don't duplicate publications
321       }
322       if (MediaPackageElements.EPISODE.equals(e.getFlavor())) {
323         // Remove episode DC since we will add a new one (with changed title)
324         elements.remove(e);
325       }
326       // The series DC changes
327       if (series != null && MediaPackageElements.SERIES.equals(e.getFlavor())) {
328         // Remove episode DC since we will add a new one
329         elements.remove(e);
330       }
331       if (series != null && MediaPackageElements.XACML_POLICY_SERIES.equals(e.getFlavor())) {
332         seriesAclTags.addAll(Arrays.asList(e.getTags()));
333         elements.remove(e);
334       }
335     }
336 
337     final MediaPackageElement[] originalEpisodeDc = mediaPackage.getElementsByFlavor(MediaPackageElements.EPISODE);
338     if (originalEpisodeDc.length != 1) {
339       throw new WorkflowOperationException("Media package " + mediaPackage.getIdentifier() + " has "
340           + originalEpisodeDc.length + " episode dublin cores while it is expected to have exactly 1. Aborting.");
341     }
342 
343     String mpIds = "";
344     String sep = "";
345     Map<String, String> properties = new HashMap<>();
346     for (int i = 0; i < numberOfEvents; i++) {
347       final List<URI> temporaryFiles = new ArrayList<>();
348       MediaPackage newMp = null;
349       try {
350         String newMpId = workflowInstance.getConfiguration("newMpId");
351         if (newMpId == null) {
352           newMpId = UUID.randomUUID().toString();
353         }
354         // Clone the media package (without its elements)
355         String useTitle;
356         if (title.isEmpty() || (title.startsWith("${") && (title.endsWith("}")))) {
357           final DublinCoreCatalog dublinCore = DublinCoreUtil.loadEpisodeDublinCore(workspace, mediaPackage).get();
358           useTitle = dublinCore.getFirst(DublinCore.PROPERTY_TITLE);
359         } else {
360           useTitle = title;
361         }
362         if (!noSuffix) {
363           useTitle = String.format("%s (%s %d)", useTitle, copyNumberPrefix, i + 1);
364         }
365         Date mpDate;
366         if (!startDateString.isEmpty() || (!startDateString.startsWith("${") && (!startDateString.endsWith("}")))) {
367           try {
368             mpDate = ADMIN_UI_DATE_FORMAT.parse(startDateString);
369             logger.info("Setting StartDate to {}", mpDate);
370           } catch (ParseException ex) {
371             logger.warn("Could not parse: {} as date time", startDateString);
372             mpDate = mediaPackage.getDate();
373             logger.warn("Using original event date {} as default", mpDate);
374           }
375         } else {
376           mpDate = mediaPackage.getDate();
377           logger.warn("No date set, using original event date {} as default", mpDate);
378         }
379         newMp = copyMediaPackage(mediaPackage, series, newMpId, useTitle, mpDate);
380 
381         if (series != null) {
382           URI newSeriesURI = null;
383           String newSeriesId = UUID.randomUUID().toString();
384           try (InputStream seriesDCInputStream = IOUtils.toInputStream(series.dc.toXmlString(), "UTF-8")) {
385             newSeriesURI = workspace.put(newMpId, newSeriesId, "dublincore.xml", seriesDCInputStream);
386           }
387           MediaPackageElementBuilder elementBuilder = MediaPackageElementBuilderFactory.newInstance()
388                   .newElementBuilder();
389           MediaPackageElement newSeriesMpElement = elementBuilder.elementFromURI(newSeriesURI,
390                   Catalog.TYPE, MediaPackageElements.SERIES);
391           newSeriesMpElement.setIdentifier(newSeriesId);
392           newMp.add(newSeriesMpElement);
393 
394           if (seriesAccessControl != null) {
395             newMp = authorizationService.setAcl(newMp, AclScope.Series, seriesAccessControl).getA();
396             for (MediaPackageElement seriesAclMpe : newMp.getElementsByFlavor(MediaPackageElements.XACML_POLICY_SERIES)) {
397               for (final String tag : seriesAclTags) {
398                 seriesAclMpe.addTag(tag);
399               }
400             }
401           }
402         }
403 
404         // Create and add new episode dublin core with changed title
405         newMp = copyDublinCore(mediaPackage, originalEpisodeDc[0],
406               newMp, series, removeTags, addTags, overrideTags,
407               temporaryFiles, mpDate);
408 
409         // Clone regular elements
410         for (final MediaPackageElement e : elements) {
411           final MediaPackageElement element = (MediaPackageElement) e.clone();
412           updateTags(element, removeTags, addTags, overrideTags);
413           newMp.add(element);
414         }
415 
416         // Clone internal publications
417         for (final Publication originalPub : internalPublications) {
418          copyPublication(originalPub, mediaPackage, newMp, removeTags, addTags, overrideTags, temporaryFiles);
419         }
420 
421         assetManager.takeSnapshot(AssetManager.DEFAULT_OWNER, newMp);
422 
423         // Clone properties of media package
424         for (String namespace : propertyNamespaces) {
425           copyProperties(namespace, mediaPackage, newMp);
426         }
427 
428         // Store media package ID as workflow property
429         properties.put("duplicate_media_package_" + (i + 1) + "_id", newMp.getIdentifier().toString());
430         mpIds += sep + newMp.getIdentifier().toString();
431         sep = ", ";
432       } catch (IOException | MediaPackageException e) {
433         throw new WorkflowOperationException(e);
434       } finally {
435         cleanup(temporaryFiles, Optional.ofNullable(newMp));
436       }
437     }
438     properties.put("duplicate_media_package_ids", mpIds);
439     return createResult(mediaPackage, properties, Action.CONTINUE, 0);
440   }
441 
442   private void cleanup(List<URI> temporaryFiles, Optional<MediaPackage> newMp) {
443     // Remove temporary files of new media package
444     for (URI temporaryFile : temporaryFiles) {
445       try {
446         workspace.delete(temporaryFile);
447       } catch (NotFoundException e) {
448         logger.debug("{} could not be found in the workspace and hence, cannot be deleted.", temporaryFile);
449       } catch (IOException e) {
450         logger.warn("Failed to delete {} from workspace.", temporaryFile);
451       }
452     }
453     newMp.ifPresent(mp -> {
454       try {
455         workspace.cleanup(mp.getIdentifier());
456       } catch (IOException e) {
457         logger.warn("Failed to cleanup the workspace for media package {}", mp.getIdentifier());
458       }
459     });
460   }

462   private void updateTags(
463       MediaPackageElement element,
464       List<String> removeTags,
465       List<String> addTags,
466       List<String> overrideTags) {
467     element.setIdentifier(null);
468 
469     if (overrideTags.size() > 0) {
470       element.clearTags();
471       for (String overrideTag : overrideTags) {
472         element.addTag(overrideTag);
473       }
474     } else {
475       for (String removeTag : removeTags) {
476         element.removeTag(removeTag.substring(MINUS.length()));
477       }
478       for (String tag : addTags) {
479         element.addTag(tag.substring(PLUS.length()));
480       }
481     }
482   }

484   private MediaPackage copyMediaPackage(
485       final MediaPackage source,
486       final SeriesInformation series,
487       final String newMpId,
488       final String title,
489       final Date startDate
490       ) throws WorkflowOperationException {
491     // We are not using MediaPackage.clone() here, since it does "too much" for us (e.g. copies all the attachments)
492     MediaPackage destination;
493     try {
494       destination = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder().createNew(new IdImpl(newMpId));
495     } catch (MediaPackageException e) {
496       logger.error("Failed to create media package " + e.getLocalizedMessage());
497       throw new WorkflowOperationException(e);
498     }
499     logger.info("Created mediapackage {}", destination);
500     destination.setDate(source.getDate());
501     if (series != null) {
502       destination.setSeries(series.id);
503       destination.setSeriesTitle(series.title);
504     } else {
505       destination.setSeries(source.getSeries());
506       destination.setSeriesTitle(source.getSeriesTitle());
507     }
508     destination.setDuration(source.getDuration());
509     destination.setLanguage(source.getLanguage());
510     destination.setLicense(source.getLicense());
511     destination.setDate(startDate);
512     destination.setTitle(title);
513     return destination;
514   }

516   private void copyPublication(
517       Publication sourcePublication,
518       MediaPackage source,
519       MediaPackage destination,
520       List<String> removeTags,
521       List<String> addTags,
522       List<String> overrideTags,
523       List<URI> temporaryFiles) throws WorkflowOperationException {
524     final String newPublicationId = UUID.randomUUID().toString();
525     final Publication newPublication = PublicationImpl.publication(newPublicationId,
526         InternalPublicationChannel.CHANNEL_ID, null, null);
527 
528     // re-distribute elements of publication to internal publication channel
529     final Collection<MediaPackageElement> sourcePubElements = new HashSet<>();
530     sourcePubElements.addAll(Arrays.asList(sourcePublication.getAttachments()));
531     sourcePubElements.addAll(Arrays.asList(sourcePublication.getCatalogs()));
532     sourcePubElements.addAll(Arrays.asList(sourcePublication.getTracks()));
533     for (final MediaPackageElement e : sourcePubElements) {
534       try {
535         // We first have to copy the media package element into the workspace
536         final MediaPackageElement element = (MediaPackageElement) e.clone();
537         try (InputStream inputStream = workspace.read(element.getURI())) {
538           final URI tmpUri = workspace.put(destination.getIdentifier().toString(), element.getIdentifier(),
539               FilenameUtils.getName(element.getURI().toString()), inputStream);
540           temporaryFiles.add(tmpUri);
541           element.setIdentifier(null);
542           element.setURI(tmpUri);
543         }
544 
545         // Now we can distribute it to the new media package
546         destination.add(element); // Element has to be added before it can be distributed
547         final Job job = distributionService.distribute(InternalPublicationChannel.CHANNEL_ID, destination,
548             element.getIdentifier());
549         final MediaPackageElement distributedElement =
550             JobUtil.payloadAsMediaPackageElement(serviceRegistry).apply(job);
551         destination.remove(element);
552 
553         updateTags(distributedElement, removeTags, addTags, overrideTags);
554 
555         PublicationImpl.addElementToPublication(newPublication, distributedElement);
556       } catch (Exception exception) {
557         throw new WorkflowOperationException(exception);
558       }
559     }
560 
561     // Using an altered copy of the source publication's URI is a bit hacky,
562     // but it works without knowing the URI pattern...
563     String publicationUri = sourcePublication.getURI().toString();
564     publicationUri = publicationUri.replace(source.getIdentifier().toString(), destination.getIdentifier().toString());
565     publicationUri = publicationUri.replace(sourcePublication.getIdentifier(), newPublicationId);
566     newPublication.setURI(URI.create(publicationUri));
567     destination.add(newPublication);
568   }

570   private MediaPackage copyDublinCore(
571       final MediaPackage source,
572       final MediaPackageElement sourceDublinCore,
573       final MediaPackage destination,
574       final SeriesInformation series,
575       final List<String> removeTags,
576       final List<String> addTags,
577       final List<String> overrideTags,
578       final List<URI> temporaryFiles,
579       final Date creationDate
580   ) throws WorkflowOperationException {
581     final DublinCoreCatalog destinationDublinCore = DublinCoreUtil.loadEpisodeDublinCore(workspace, source).get();
582     destinationDublinCore.setIdentifier(null);
583     destinationDublinCore.setURI(sourceDublinCore.getURI());
584     destinationDublinCore.set(DublinCore.PROPERTY_CREATED, OpencastMetadataCodec.encodeDate(creationDate, Precision.Second));
585     destinationDublinCore.set(DublinCore.PROPERTY_TITLE, destination.getTitle());
586     if (StringUtils.isNotBlank(destinationDublinCore.getFirst(DublinCore.PROPERTY_TEMPORAL))) {
587       DublinCoreValue eventTime = EncodingSchemeUtils.encodePeriod(new DCMIPeriod(creationDate, creationDate), Precision.Second);
588       destinationDublinCore.set(DublinCore.PROPERTY_TEMPORAL, eventTime);
589     }
590     if (series != null) {
591       destinationDublinCore.set(DublinCore.PROPERTY_IS_PART_OF, series.id);
592     }
593     try (InputStream inputStream = IOUtils.toInputStream(destinationDublinCore.toXmlString(), "UTF-8")) {
594       final String elementId = UUID.randomUUID().toString();
595       final URI newUrl = workspace.put(destination.getIdentifier().toString(), elementId, "dublincore.xml",
596           inputStream);
597       temporaryFiles.add(newUrl);
598       final MediaPackageElement mpe = destination.add(newUrl, MediaPackageElement.Type.Catalog,
599           MediaPackageElements.EPISODE);
600       for (String tag : sourceDublinCore.getTags()) {
601         mpe.addTag(tag);
602       }
603       updateTags(mpe, removeTags, addTags, overrideTags);
604       mpe.setIdentifier(elementId);
605     } catch (IOException e) {
606       throw new WorkflowOperationException(e);
607     }
608     return destination;
609   }

611   private void copyProperties(String namespace, MediaPackage source, MediaPackage destination) {
612     final AQueryBuilder q = assetManager.createQuery();
613     final AResult properties = q.select(q.propertiesOf(namespace))
614         .where(q.mediaPackageId(source.getIdentifier().toString())).run();
615     if (properties.getRecords().stream().findFirst().isEmpty()) {
616       logger.info("No properties to copy for media package {}, namespace {}.", source.getIdentifier(), namespace);
617       return;
618     }
619     for (final Property p : properties.getRecords().stream().findFirst().get().getProperties()) {
620       final PropertyId newPropId = PropertyId.mk(destination.getIdentifier().toString(), namespace, p.getId()
621           .getName());
622       assetManager.setProperty(Property.mk(newPropId, p.getValue()));
623     }
624   }
