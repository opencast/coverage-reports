21  package org.opencastproject.security.urlsigning.service.impl;
22  
23  import static java.util.Objects.requireNonNull;
24  import static org.opencastproject.security.urlsigning.exception.UrlSigningException.urlNotSupported;
25  
26  import org.opencastproject.security.urlsigning.exception.UrlSigningException;
27  import org.opencastproject.security.urlsigning.provider.UrlSigningProvider;
28  import org.opencastproject.security.urlsigning.service.UrlSigningService;
29  import org.opencastproject.urlsigning.common.Policy;
30  
31  import org.joda.time.DateTime;
32  import org.joda.time.DateTimeConstants;
33  import org.joda.time.DateTimeZone;
34  import org.osgi.service.component.annotations.Component;
35  import org.osgi.service.component.annotations.Reference;
36  import org.osgi.service.component.annotations.ReferenceCardinality;
37  import org.osgi.service.component.annotations.ReferencePolicy;
38  import org.slf4j.Logger;
39  import org.slf4j.LoggerFactory;
40  
41  import java.util.List;
42  import java.util.concurrent.CopyOnWriteArrayList;
43  
44  @Component(
45      immediate = true,
46      service = { UrlSigningService.class }
47  )
48  public class UrlSigningServiceImpl implements UrlSigningService {
49  
50    /** The logging facility */
51    private static final Logger logger = LoggerFactory.getLogger(UrlSigningServiceImpl.class);
52  
53    /** List of registered signing providers */
54    private final List<UrlSigningProvider> signingProviders = new CopyOnWriteArrayList<>();
55  
56    /** OSGi callback for registering {@link UrlSigningProvider} */
57    @Reference(
58        cardinality = ReferenceCardinality.MULTIPLE,
59        policy = ReferencePolicy.DYNAMIC,
60        unbind = "unregisterSigningProvider"
61    )
62    void registerSigningProvider(final UrlSigningProvider provider) {
63      signingProviders.add(provider);
64      logger.info("{} registered", provider);
65    }
66  
67    /** OSGi callback for unregistering {@link UrlSigningProvider} */
68    void unregisterSigningProvider(final UrlSigningProvider provider) {
69      signingProviders.remove(provider);
70      logger.info("{} unregistered", provider);
71    }
72  
73    @Override
74    public boolean accepts(String baseUrl) {
75      for (final UrlSigningProvider provider : signingProviders) {
76        if (provider.accepts(baseUrl)) {
77          logger.debug("{} accepted to sign base URL '{}'", provider, baseUrl);
78          return true;
79        }
80      }
81  
82      logger.debug("No provider accepted to sign the URL '{}'", baseUrl);
83      return false;
84    }
85  
86    @Override
87    public String sign(final String baseUrl, final Long validUntilDuration, final Long validFromDuration,
88            final String ipAddr) throws UrlSigningException {
89      requireNonNull(validUntilDuration);
90      DateTime validUntil = new DateTime(DateTimeZone.UTC).plus(validUntilDuration * DateTimeConstants.MILLIS_PER_SECOND);
91      DateTime validFrom = validFromDuration == null ? null : new DateTime(DateTimeZone.UTC).plus(validFromDuration
92              * DateTimeConstants.MILLIS_PER_SECOND);
93      return sign(baseUrl, validUntil, validFrom, ipAddr);
94    }
95  
96    @Override
97    public String sign(final String baseUrl, final DateTime validUntil, final DateTime validFrom, final String ipAddr)
98            throws UrlSigningException {
99      requireNonNull(baseUrl);
100     requireNonNull(validUntil);
101 
102     final Policy policy = Policy.mkPolicyValidFromWithIP(baseUrl, validUntil, validFrom, ipAddr);
103 
104     for (final UrlSigningProvider provider : signingProviders) {
105       if (provider.accepts(baseUrl)) {
106         logger.debug("{} accepted to sign base URL '{}'", provider, baseUrl);
107         return provider.sign(policy);
108       }
109     }
110 
111     logger.warn("No signing provider accepted to sign URL '{}'", baseUrl);
112     throw urlNotSupported();
113   }
114 
115 }
