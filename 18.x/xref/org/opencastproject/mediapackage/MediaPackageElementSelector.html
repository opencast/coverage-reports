22  package org.opencastproject.mediapackage;
23  
24  import java.util.Collection;
25  import java.util.List;
26  
27  /**
28   * A <code>MedikaPackageElementSelector</code> is the way to set up rules for extracting elements from a media package
29   * dependent on their flavor.
30   */
31  public interface MediaPackageElementSelector<T extends MediaPackageElement> {
32  
33    /**
34     * Returns the media package elements that are matched by this selector.
35     *
36     * @param mediaPackage
37     *          the media package
38     * @param withTagsAndFlavors
39     *          define if the elements must match with flavors and tags, or just one of these parameters
40     * @return the selected elements
41     */
42    Collection<T> select(MediaPackage mediaPackage, boolean withTagsAndFlavors);
43  
44    /**
45     * Returns the elements that are matched by this selector from the given publication of the given media package.
46     * If the publication could not be found on the media package, returns an empty collection.
47     *
48     * @param mediaPackage
49     *          the media package
50     * @param publicationChannel
51     *          the publication to get elements from
52     * @param withTagsAndFlavors
53     *          define if the elements must match with flavors and tags, or just one of these parameters
54     * @return the selected elements
55     */
56    Collection<T> select(MediaPackage mediaPackage, String publicationChannel, boolean withTagsAndFlavors);
57  
58    /**
59     * Returns the media package elements that are matched by this selector.
60     *
61     * @param elements
62     *          the elements to select from
63     * @param withTagsAndFlavors
64     *          define if the elements must match with flavors and tags, or just one of these parameters
65     * @return the selected elements
66     */
67    Collection<T> select(List<MediaPackageElement> elements, boolean withTagsAndFlavors);
68  }
