64    /**
65     * Update a {@link DublinCoreCatalog} with the values contained within a {@link DublinCoreMetadataCollection}
66     *
67     * @param dc
68     *          The {@link DublinCoreCatalog} to update the values within.
69     * @param metadata
70     *          The {@link DublinCoreMetadataCollection} data definitions and values to update the catalog with.
71     */
72    public static void updateDublincoreCatalog(DublinCoreCatalog dc, DublinCoreMetadataCollection metadata) {
73      for (MetadataField field : metadata.getOutputFields().values()) {
74        if (field.isUpdated() && field.getValue() != null) {
75          final String namespace = field.getNamespace() == null ? DublinCore.TERMS_NS_URI : field.getNamespace();
76          final EName ename = new EName(namespace, field.getInputID());
77          if (field.getType() == MetadataField.Type.START_DATE) {
78            setStartDate(dc, field, ename);
79          } else if (field.getType() == MetadataField.Type.DURATION) {
80            // WARN: the duration change assumes the catalog's start date is already up to date.
81            setDuration(dc, field, ename);
82          } else if (field.getType() == Type.DATE) {
83            // Skip over metadata field tagged with key "created".
84            // DC created should only be modified by changing the start date, see MH-12250
85            if (! DublinCore.PROPERTY_CREATED.equals(ename))
86              setDate(dc, field, ename);
87          } else if (field.getType() == MetadataField.Type.MIXED_TEXT || field.getType() == Type.ITERABLE_TEXT) {
88            setIterableString(dc, field, ename);
89          } else {
90            if (field.isRequired() && StringUtils.isBlank(field.getValue().toString()))
91              throw new IllegalArgumentException(
92                      String.format(
93                              "The event metadata field with id '%s' and the metadata type '%s' is required and can not be empty!.",
94                              field.getInputID(), field.getType()));
95            dc.set(ename, field.getValue().toString());
96          }
97        } else if (field.getValue() == null && field.isRequired()) {
98          throw new IllegalArgumentException(String.format(
99                  "The event metadata field with id '%s' and the metadata type '%s' is required and can not be empty!.",
100                 field.getInputID(), field.getType()));
101       }
102     }
103   }

105   /**
106    * Set the value of an iterable string (each element as an separate entry)
107    *
108    * @param dc    The dublin core catalog to add the iterable string values to (or remove it if empty)
109    * @param field The {@link MetadataField} with the values to update.
110    * @param ename The {@link EName} of the property in the {@link DublinCoreCatalog} to update.
111    */
112   private static void setIterableString(DublinCoreCatalog dc, MetadataField field, final EName ename) {
113     if (field.getValue() != null) {
114       dc.remove(ename);
115       if (field.getValue() instanceof String) {
116         String valueString = (String) field.getValue();
117         dc.set(ename, valueString);
118       } else {
119         @SuppressWarnings("unchecked")
120         Iterable<String> valueIterable = (Iterable<String>) field.getValue();
121         for (String valueString : valueIterable) {
122           if (StringUtils.isNotBlank(valueString))
123             dc.add(ename, valueString);
124         }
125       }
126     }
127   }

129   private static Opt<DCMIPeriod> getPeriodFromCatalog(DublinCoreCatalog dc, EName ename) {
130     List<DublinCoreValue> periodStrings = dc.get(ename);
131     Opt<DCMIPeriod> p = Opt.<DCMIPeriod> none();
132     for (DublinCoreValue periodString : periodStrings) {
133       p = Opt.nul(EncodingSchemeUtils.decodePeriod(periodString.getValue()));
134     }
135     if (p.isNone()) {
136       // fall back to created date with zero duration
137       // opencast keep the event start date and created date in sync
138       // if the start date isn't set, we can grab the created value
139       DublinCoreValue createdDCValue = dc.getFirstVal(DublinCore.PROPERTY_CREATED);
140       if (createdDCValue != null) {
141         Date createdDate = EncodingSchemeUtils.decodeDate(createdDCValue.getValue());
142         p = Opt.nul(new DCMIPeriod(createdDate, createdDate));
143       }
144     }
145     return p;
146   }

148   /**
149    * @param period
150    *          The dublin core period for this event.
151    * @return The milliseconds between the start and end time for this event.
152    */
153   static Long getDuration(Opt<DCMIPeriod> period) {
154     Long duration = 0L;
155     if (period.isSome() && period.get().hasStart() && period.get().hasEnd()) {
156       return period.get().getEnd().getTime() - period.get().getStart().getTime();
157     }
158     return duration;
159   }

161   /**
162    * Sets a date object with the correct formatting into the dublin core catalog.
163    *
164    * @param dc
165    *          The dublin core catalog to insert the date.
166    * @param field
167    *          The field with the date value and pattern ready to format.
168    * @param ename
169    *          The unique id for the dublin core property.
170    */
171   private static void setDate(DublinCoreCatalog dc, MetadataField field, EName ename) {
172     if (field.getValue() instanceof Date && field.getPattern() == null) {
173       throw new IllegalArgumentException("There needs to be a pattern property set for " + field.getInputID() + ":"
174               + field.getOutputID() + ":" + field.getValue() + " metadata field to store and retrieve the result.");
175     }
176     if (field.getValue() instanceof Date) {
177       SimpleDateFormat sdf = new SimpleDateFormat(field.getPattern());
178       dc.set(ename, sdf.format((Date) field.getValue()));
179     } else {
180       dc.set(ename, field.getValue().toString());
181     }
182   }

184   /**
185    * Sets the start date in a dublin core catalog to the right value and keeps the start time and duration the same.
186    *
187    * @param dc
188    *          The dublin core catalog to adjust
189    * @param field
190    *          The metadata field that contains the start date.
191    * @param ename
192    *          The EName in the catalog to identify the property that has the dublin core period.
193    */
194   static void setStartDate(DublinCoreCatalog dc, MetadataField field, EName ename) {
195     if (field.getValue() == null
196             || (field.getValue() instanceof String && StringUtils.isBlank(field.getValue().toString()))) {
197       logger.debug("No value was set for metadata field with dublin core id '{}' and json id '{}'", field.getInputID(),
198               field.getOutputID());
199       return;
200     }
201     try {
202       // Get the current date
203       SimpleDateFormat dateFormat = MetadataField.getSimpleDateFormatter(field.getPattern());
204       Date startDate = dateFormat.parse((String) field.getValue());
205       // Get the current period
206       Opt<DCMIPeriod> period = getPeriodFromCatalog(dc, ename);
207       // Get the current duration
208       Long duration = getDuration(period);
209       // Get the current end date based on new date and duration.
210       DateTime endDate = new DateTime(startDate.getTime() + duration);
211       dc.set(ename, EncodingSchemeUtils.encodePeriod(new DCMIPeriod(startDate, endDate.toDate()),
212               Precision.Second));
213       // ensure that DC created is start date, see MH-12250
214       setDate(dc, field, DublinCore.PROPERTY_CREATED);
215     } catch (ParseException e) {
216       logger.error("Not able to parse date {} to update the dublin core because:", field.getValue(), e);
217     }
218   }

220   /**
221    * Gets the current hour, minute and second from a dublin core period if available.
222    *
223    * @param period
224    *          The current period from dublin core.
225    * @return A new DateTime with the current hour, minute and second.
226    */
227   private static DateTime getCurrentStartDateTime(Opt<DCMIPeriod> period) {
228     DateTime currentStartTime = new DateTime();
229     currentStartTime = currentStartTime.withZone(DateTimeZone.UTC);
230     currentStartTime = currentStartTime.withYear(2001);
231     currentStartTime = currentStartTime.withMonthOfYear(1);
232     currentStartTime = currentStartTime.withDayOfMonth(1);
233     currentStartTime = currentStartTime.withHourOfDay(0);
234     currentStartTime = currentStartTime.withMinuteOfHour(0);
235     currentStartTime = currentStartTime.withSecondOfMinute(0);
236 
237     if (period.isSome() && period.get().hasStart()) {
238       DateTime fromDC = new DateTime(period.get().getStart().getTime());
239       fromDC = fromDC.withZone(DateTimeZone.UTC);
240       currentStartTime = currentStartTime.withZone(DateTimeZone.UTC);
241       currentStartTime = currentStartTime.withYear(fromDC.getYear());
242       currentStartTime = currentStartTime.withMonthOfYear(fromDC.getMonthOfYear());
243       currentStartTime = currentStartTime.withDayOfMonth(fromDC.getDayOfMonth());
244       currentStartTime = currentStartTime.withHourOfDay(fromDC.getHourOfDay());
245       currentStartTime = currentStartTime.withMinuteOfHour(fromDC.getMinuteOfHour());
246       currentStartTime = currentStartTime.withSecondOfMinute(fromDC.getSecondOfMinute());
247     }
248     return currentStartTime;
249   }

251   /**
252    * Sets the duration in a dublin core catalog to the right value and keeps the start date and start time the same.
253    *
254    * @param dc
255    *          The dublin core catalog to adjust
256    * @param field
257    *          The metadata field that contains the duration.
258    * @param ename
259    *          The EName in the catalog to identify the property that has the dublin core period.
260    */
261   static void setDuration(DublinCoreCatalog dc, MetadataField field, EName ename) {
262     if (field.getValue() == null) {
263       logger.error("No value was set for metadata field with dublin core id '{}' and json id '{}'", field.getInputID(),
264               field.getOutputID());
265       return;
266     }
267 
268     // Get the current period
269     Opt<DCMIPeriod> period = getPeriodFromCatalog(dc, ename);
270     // Get the current duration
271     Long duration = 0L;
272     try {
273       duration = Long.parseLong(field.getValue().toString());
274     } catch (NumberFormatException e) {
275       logger.debug("Unable to parse the duration's value '{}' as a long value. Trying it as a period next.",
276               field.getValue());
277     }
278     if (duration < 1L) {
279       duration = getDuration(period);
280     }
281     // Get the current start date (WARN: this assumes any start time updates have already been performed)
282     DateTime startDateTime = getCurrentStartDateTime(period);
283     // Get the current end date based on new date and duration.
284     DateTime endDate = new DateTime(startDateTime.toDate().getTime() + duration);
285     dc.set(ename, EncodingSchemeUtils.encodePeriod(new DCMIPeriod(startDateTime.toDate(), endDate.toDate()),
286             Precision.Second));
287   }

289   public static Map<String, MetadataField> getDublinCoreProperties(Dictionary<String, ?> configProperties) {
290 
291     final Map<String,Map<String, String>> allProperties = new HashMap<>();
292 
293     for (String configObject : Collections.list(configProperties.keys())) {
294       String propertyName = getDublinCorePropertyName(configObject);
295       String propertyKey = getDublinCorePropertyKey(configObject);
296 
297       if (propertyName != null && propertyKey != null) {
298         Map<String,String> metadataFieldProperties = allProperties.computeIfAbsent(
299                 propertyName,
300                 key -> new HashMap<>());
301         metadataFieldProperties.put(propertyKey, configProperties.get(configObject).toString());
302       }
303     }
304 
305     final Map<String, MetadataField> metadataFieldsMap = new TreeMap<>();
306     for (Map<String, String> metadataFieldPropertiesMap : allProperties.values()) {
307       MetadataField metadataField = MetadataField.createMetadataField(metadataFieldPropertiesMap);
308       metadataFieldsMap.put(metadataField.getOutputID(), metadataField);
309     }
310 
311     return metadataFieldsMap;
312   }

314   private static boolean isDublinCoreProperty(String propertyKey) {
315     return !StringUtils.isBlank(propertyKey) && propertyKey.split("\\.").length == 3
316             && propertyKey.split("\\.")[0].equalsIgnoreCase(MetadataField.CONFIG_PROPERTY_PREFIX);
317   }
318 
319   static String getDublinCorePropertyName(String propertyKey) {
320     if (isDublinCoreProperty(propertyKey)) {
321       return propertyKey.split("\\.")[1];
322     }
323     return null;
324   }
325 
326   static String getDublinCorePropertyKey(String propertyKey) {
327     if (isDublinCoreProperty(propertyKey)) {
328       return propertyKey.split("\\.")[2];
329     }
330     return null;
331   }
