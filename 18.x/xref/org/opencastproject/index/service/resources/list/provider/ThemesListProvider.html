22  package org.opencastproject.index.service.resources.list.provider;
23  
24  import org.opencastproject.elasticsearch.api.SearchIndexException;
25  import org.opencastproject.elasticsearch.api.SearchResult;
26  import org.opencastproject.elasticsearch.api.SearchResultItem;
27  import org.opencastproject.elasticsearch.index.ElasticsearchIndex;
28  import org.opencastproject.elasticsearch.index.objects.theme.IndexTheme;
29  import org.opencastproject.elasticsearch.index.objects.theme.ThemeSearchQuery;
30  import org.opencastproject.list.api.ListProviderException;
31  import org.opencastproject.list.api.ResourceListProvider;
32  import org.opencastproject.list.api.ResourceListQuery;
33  import org.opencastproject.security.api.SecurityService;
34  import org.opencastproject.util.requests.SortCriterion.Order;
35  
36  import org.osgi.framework.BundleContext;
37  import org.osgi.service.component.annotations.Activate;
38  import org.osgi.service.component.annotations.Component;
39  import org.osgi.service.component.annotations.Reference;
40  import org.slf4j.Logger;
41  import org.slf4j.LoggerFactory;
42  
43  import java.util.HashMap;
44  import java.util.Map;
45  
46  @Component(
47      service = ResourceListProvider.class,
48      property = {
49          "service.description=Themes list provider",
50          "opencast.service.type=org.opencastproject.index.service.resources.list.provider.ThemesListProvider"
51      }
52  )
53  public class ThemesListProvider implements ResourceListProvider {
54  
55    private static final String PROVIDER_PREFIX = "THEMES";
56    public static final String NAME = PROVIDER_PREFIX + ".NAME";
57    public static final String DESCRIPTION = PROVIDER_PREFIX + ".DESCRIPTION";
58  
59    private static final String[] NAMES = { PROVIDER_PREFIX, NAME, DESCRIPTION };
60  
61    private static final Logger logger = LoggerFactory.getLogger(ThemesListProvider.class);
62  
63    private ElasticsearchIndex searchIndex;
64  
65    private SecurityService securityService;
66  
67    @Activate
68    protected void activate(BundleContext bundleContext) {
69      logger.info("Themes list provider activated!");
70    }
71  
72    /** OSGi callback for the search index. */
73    @Reference
74    public void setIndex(ElasticsearchIndex index) {
75      this.searchIndex = index;
76    }
77  
78    /** OSGi callback for the security service. */
79    @Reference
80    public void setSecurityService(SecurityService securityService) {
81      this.securityService = securityService;
82    }
83  
84    @Override
85    public String[] getListNames() {
86      return NAMES;
87    }
88  
89    @Override
90    public Map<String, String> getList(String listName, ResourceListQuery query)
91            throws ListProviderException {
92      Map<String, String> list = new HashMap<String, String>();
93  
94      if (NAME.equals(listName)) {
95        ThemeSearchQuery themeQuery = new ThemeSearchQuery(securityService.getOrganization().getId(),
96                securityService.getUser());
97        themeQuery.withOffset(query.getOffset().getOrElse(0));
98        int limit = query.getLimit().getOrElse(Integer.MAX_VALUE - themeQuery.getOffset());
99        themeQuery.withLimit(limit);
100       themeQuery.sortByName(Order.Ascending);
101       SearchResult<IndexTheme> results = null;
102       try {
103         results = searchIndex.getByQuery(themeQuery);
104       } catch (SearchIndexException e) {
105         logger.error("The admin UI Search Index was not able to get the themes", e);
106         throw new ListProviderException("No themes list for list name " + listName + " found!");
107       }
108 
109       for (SearchResultItem<IndexTheme> item : results.getItems()) {
110         IndexTheme theme = item.getSource();
111         list.put(Long.toString(theme.getIdentifier()), theme.getName());
112       }
113     }
114     else if (DESCRIPTION.equals(listName)) {
115       ThemeSearchQuery themeQuery = new ThemeSearchQuery(securityService.getOrganization().getId(),
116               securityService.getUser());
117       themeQuery.withOffset(query.getOffset().getOrElse(0));
118       int limit = query.getLimit().getOrElse(Integer.MAX_VALUE - themeQuery.getOffset());
119       themeQuery.withLimit(limit);
120       themeQuery.sortByName(Order.Ascending);
121       SearchResult<IndexTheme> results = null;
122       try {
123         results = searchIndex.getByQuery(themeQuery);
124       } catch (SearchIndexException e) {
125         logger.error("The admin UI Search Index was not able to get the themes", e);
126         throw new ListProviderException("No themes list for list name " + listName + " found!");
127       }
128 
129       for (SearchResultItem<IndexTheme> item : results.getItems()) {
130         IndexTheme theme = item.getSource();
131         if (theme.getDescription() == null) {
132           theme.setDescription("");
133         }
134         else {
135           theme.getDescription();
136         }
137         list.put(Long.toString(theme.getIdentifier()), theme.getDescription());
138       }
139     }
140 
141     return list;
142   }
143 
144   @Override
145   public boolean isTranslatable(String listName) {
146     return false;
147   }
148 
149   @Override
150   public String getDefault() {
151     return null;
152   }
153 }
