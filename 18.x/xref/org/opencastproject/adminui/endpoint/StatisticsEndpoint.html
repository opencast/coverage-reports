137   @GET
138   @Path("providers.json")
139   @Produces(MediaType.APPLICATION_JSON)
140   @RestQuery(name = "getprovidersbyresourcetype", description = "Returns the available statistics providers for an (optional) resource type", returnDescription = "The available statistics providers as JSON", restParameters = {
141     @RestParameter(name = "resourceType", description = "The resource type: either 'episode', 'series' or 'organization'", isRequired = false, type = STRING)},
142     responses = {
143       @RestResponse(description = "Returns the providers for the given resource type as JSON, or all, if the resource type is missing", responseCode = HttpServletResponse.SC_OK),
144       @RestResponse(description = "If the current user is not authorized to perform this action", responseCode = HttpServletResponse.SC_UNAUTHORIZED)
145     })
146   public Response getProviders(
147     @QueryParam("resourceType") final String resourceTypeStr) {
148     ResourceType resourceType;
149     try {
150       if (resourceTypeStr == null) {
151         resourceType = null;
152       } else {
153         resourceType = Enum.valueOf(ResourceType.class, resourceTypeStr.toUpperCase());
154       }
155     } catch (IllegalArgumentException e) {
156       return RestUtil.R.badRequest("invalid value for 'resourceType'");
157     }
158 
159     JSONArray result = new JSONArray();
160     statisticsService
161       .getProviders(resourceType)
162       .stream()
163       .map(this::providerToJson)
164       .forEach(result::add);
165     return Response.ok(result.toJSONString()).build();
166   }

168   private static String providerTypeString(StatisticsProvider provider) {
169     if (provider instanceof TimeSeriesProvider) {
170       return TIME_SERIES_PROVIDER_TYPE;
171     }
172     return "unknown";
173   }
174 
175   private JSONObject providerToJson(StatisticsProvider provider) {
176     final JSONObject providerObj = new JSONObject();
177     providerObj.put("providerId", provider.getId());
178     providerObj.put("providerType", providerTypeString(provider));
179     providerObj.put("title", provider.getTitle());
180     if (provider instanceof TimeSeriesProvider) {
181       providerObj.put("dataResolutions", resolutionsToJson(((TimeSeriesProvider) provider).getDataResolutions()));
182     }
183     providerObj.put("description", provider.getDescription());
184     return providerObj;
185   }
186 
187   private JSONArray resolutionsToJson(Set<DataResolution> resolutions) {
188     JSONArray result = new JSONArray();
189     for (DataResolution dataResolution : resolutions) {
190       result.add(dataResolutionToJson(dataResolution));
191     }
192     return result;
193   }
194 
195   private String dataResolutionToJson(DataResolution dataResolution) {
196     return dataResolution.toString().toLowerCase();
197   }

199   @POST
200   @Path("data.json")
201   @Produces(MediaType.APPLICATION_JSON)
202   @RestQuery(name = "getproviderdata", description = "Returns the statistical data for a specific provider and a specific resource", returnDescription = "The statistical data as JSON", restParameters = {
203     @RestParameter(name = "data", isRequired = true, description = "A list of statistical data requests, containing a provider id, from, to, the resource id and a resolution - all as JSON", type = RestParameter.Type.TEXT) },
204     responses = {
205       @RestResponse(description = "Returns the statistical data for the given resource type as JSON", responseCode = HttpServletResponse.SC_OK),
206       @RestResponse(description = "If the current user is not authorized to perform this action", responseCode = HttpServletResponse.SC_UNAUTHORIZED)
207     })
208   public Response getProviderData(@FormParam("data") String data) {
209     if (StringUtils.isBlank(data)) {
210       return RestUtil.R.badRequest("No data set");
211     }
212     Gson gson = new Gson();
213     RawProviderQuery[] dataJson;
214     try {
215       dataJson = gson.fromJson(data, RawProviderQuery[].class);
216     } catch (Exception e) {
217       logger.warn("Unable to parse data {}", data);
218       return RestUtil.R.badRequest("Unable to parse data");
219     }
220 
221     JSONArray result = new JSONArray();
222     try {
223       Arrays
224         .stream(dataJson)
225         .map(ProviderQuery::new)
226         .flatMap(q ->
227           statisticsService
228             .getProvider(q.getProviderId())
229             .map(Stream::of).orElseGet(Stream::empty)
230             .peek(p -> checkAccess(q.getResourceId(), p.getResourceType()))
231             .map(p -> timeSeriesToJson(
232               p.getId(),
233               statisticsService.getTimeSeriesData(p, q.getResourceId(), q.getFrom(), q.getTo(), q.getDataResolution(),
234                 ZoneId.systemDefault()))))
235         .forEach(result::add);
236     } catch (IllegalArgumentException e) {
237       return RestUtil.R.badRequest(e.getMessage());
238     }
239     return Response.ok(result.toJSONString()).build();
240   }

242   @GET
243   @Path("export.csv")
244   @Produces(MediaType.TEXT_PLAIN)
245   @RestQuery(name = "getcsvdata", description = "Returns the statistical data for a specific provider and a specific resource as CSV.", returnDescription = "The statistical data as CSV", restParameters = {
246     @RestParameter(name = "providerId", isRequired = true, description = "The provider id", type = RestParameter.Type.TEXT),
247     @RestParameter(name = "resourceId", isRequired = true, description = "The resource id", type = RestParameter.Type.TEXT),
248     @RestParameter(name = "from", isRequired = true, description = "The from date in iso 8601 UTC notation", type = RestParameter.Type.TEXT),
249     @RestParameter(name = "to", isRequired = true, description = "The to date in iso 8601 UTC notation", type = RestParameter.Type.TEXT),
250     @RestParameter(name = "dataResolution", isRequired = true, description = "The data resolution. Valid values are 'HOURLY', 'DAILY', 'WEEKLY', 'MONTHLY', and 'YEARLY'", type = RestParameter.Type.TEXT)},
251     responses = {
252       @RestResponse(description = "Returns the statistical data for the given resource type as csv", responseCode = HttpServletResponse.SC_OK),
253       @RestResponse(description = "If the current user is not authorized to perform this action", responseCode = HttpServletResponse.SC_UNAUTHORIZED)
254     })
255   public Response getCSVData(
256     @QueryParam("providerId") String providerId,
257     @QueryParam("resourceId") String resourceId,
258     @QueryParam("from") String fromStr,
259     @QueryParam("to") String toStr,
260     @QueryParam("dataResolution") String dataResolutionStr) {
261     try {
262       final ProviderQuery q = new ProviderQuery(providerId, fromStr, toStr, dataResolutionStr, resourceId);
263       final StatisticsProvider p = statisticsService
264         .getProvider(providerId).orElseThrow(() -> new IllegalArgumentException("Unknown provider: " + providerId));
265       checkAccess(q.getResourceId(), p.getResourceType());
266       final String csv = statisticsExportService.getCSV(p, q.getResourceId(), q.getFrom(), q.getTo(), q.getDataResolution(),
267         searchIndex, ZoneId.systemDefault());
268       return Response.ok().entity(csv).build();
269     } catch (IllegalArgumentException e) {
270       return RestUtil.R.badRequest(e.getMessage());
271     } catch (SearchIndexException e) {
272       return RestUtil.R.serverError();
273     } catch (NotFoundException e) {
274       return RestUtil.R.notFound(resourceId);
275     } catch (UnauthorizedException e) {
276       return Response.status(Response.Status.UNAUTHORIZED).build();
277     }
278   }

280   private JSONObject timeSeriesToJson(String providerId, TimeSeries timeSeriesData) {
281     final JSONObject result = new JSONObject();
282     result.put("providerId", providerId);
283     result.put("providerType", "timeSeries");
284     result.put("labels", timeSeriesData.getLabels());
285     result.put("values", timeSeriesData.getValues());
286     if (timeSeriesData.getTotal().isPresent()) {
287       result.put("total", timeSeriesData.getTotal().getAsDouble());
288     }
289     return result;
290   }

292   private void checkAccess(final String resourceId, final ResourceType resourceType) {
293     try {
294       switch (resourceType) {
295         case EPISODE:
296           checkMediapackageAccess(resourceId);
297           break;
298         case SERIES:
299           checkSeriesAccess(resourceId);
300           break;
301         case ORGANIZATION:
302           checkOrganizationAccess(resourceId);
303           break;
304         // Thanks CheckStyle, very sensible
305         default:
306           break;
307       }
308     } catch (UnauthorizedException | SearchIndexException e) {
309       chuck(e);
310     }
311   }
312 
313   private void checkMediapackageAccess(final String mpId) throws UnauthorizedException, SearchIndexException {
314     final Opt<Event> event = indexService.getEvent(mpId, searchIndex);
315     if (event.isNone()) {
316       // IndexService checks permissions and returns None if user is unauthorized
317       throw new UnauthorizedException(securityService.getUser(), "read");
318     }
319   }
320 
321   private void checkSeriesAccess(final String seriesId) throws UnauthorizedException, SearchIndexException {
322     final Optional<Series> series = searchIndex.getSeries(seriesId, securityService.getOrganization().getId(), securityService.getUser());
323     if (series.isEmpty()) {
324       // IndexService checks permissions and returns None if user is unauthorized
325       throw new UnauthorizedException(securityService.getUser(), "read");
326     }
327   }
328 
329   private void checkOrganizationAccess(final String orgId) throws UnauthorizedException {
330     final User currentUser = securityService.getUser();
331     final Organization currentOrg = securityService.getOrganization();
332     final String currentOrgAdminRole = currentOrg.getAdminRole();
333     final String currentOrgId = currentOrg.getId();
334 
335     final boolean userIsInOrg = currentOrgId.equals(orgId);
336 
337     boolean userIsAdmin = currentUser.hasRole(GLOBAL_ADMIN_ROLE)
338       || (currentUser.hasRole(currentOrgAdminRole) && userIsInOrg);
339 
340     boolean userIsAuthorized = currentUser.hasRole(STATISTICS_ORGANIZATION_UI_ROLE) && userIsInOrg;
341 
342     if (!userIsAdmin && !userIsAuthorized) {
343       throw new UnauthorizedException(currentUser, "read");
344     }
345   }
