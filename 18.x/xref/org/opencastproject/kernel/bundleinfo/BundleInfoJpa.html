49  @Entity(name = "BundleInfo")
50  @Table(name = "oc_bundleinfo", uniqueConstraints = { @UniqueConstraint(columnNames = { "host", "bundle_name",
51          "bundle_version" }) })
52  @NamedQueries({
53          @NamedQuery(name = "BundleInfo.findAll", query = "select a from BundleInfo a order by a.host, a.bundleSymbolicName"),
54          @NamedQuery(name = "BundleInfo.deleteAll", query = "delete from BundleInfo"),
55          @NamedQuery(name = "BundleInfo.deleteByHost", query = "delete from BundleInfo where host = :host"),
56          @NamedQuery(name = "BundleInfo.delete", query = "delete from BundleInfo where host = :host and bundleId = :bundleId") })
57  public class BundleInfoJpa {
58    @Id
59    @Column(name = "id")
60    @GeneratedValue
61    protected long id;
62  
63    @Column(name = "host", length = 128, nullable = false)
64    protected String host;
65  
66    @Column(name = "bundle_name", length = 128, nullable = false)
67    protected String bundleSymbolicName;
68  
69    @Column(name = "bundle_id", length = 128, nullable = false)
70    protected long bundleId;
71  
72    @Column(name = "bundle_version", length = 128, nullable = false)
73    protected String bundleVersion;
74  
75    @Column(name = "build_number", length = 128, nullable = true)
76    protected String buildNumber;
77  
78    @Column(name = "db_schema_version", length = 128, nullable = true)
79    protected String dbSchemaVersion;
80  
81    public static BundleInfoJpa create(BundleInfo a) {
82      final BundleInfoJpa dto = new BundleInfoJpa();
83      dto.host = a.getHost();
84      dto.bundleSymbolicName = a.getBundleSymbolicName();
85      dto.bundleId = a.getBundleId();
86      dto.bundleVersion = a.getBundleVersion();
87      for (String x : a.getBuildNumber())
88        dto.buildNumber = x;
89      return dto;
90    }
91  
92    public BundleInfo toBundleInfo() {
93      return bundleInfo(host, bundleSymbolicName, bundleId, bundleVersion, option(buildNumber), option(dbSchemaVersion));
94    }
95  
96    /** Find all in database. */
97    public static final Function<EntityManager, List<BundleInfoJpa>> findAll = namedQuery
98        .findAll("BundleInfo.findAll", BundleInfoJpa.class);
99  
100   /** Find all bundles whose symbolic names start with one of the given prefixes. */
101   public static Function<EntityManager, List<BundleInfoJpa>> findAll(final String... prefixes) {
102     return em -> {
103       final CriteriaBuilder cb = em.getCriteriaBuilder();
104       final CriteriaQuery<BundleInfoJpa> q = cb.createQuery(BundleInfoJpa.class);
105       final Root<BundleInfoJpa> r = q.from(BundleInfoJpa.class);
106       q.select(r);
107       final Expression<String> symbolicNamePath = r.get("bundleSymbolicName");
108       final Predicate[] likes = Stream.of(prefixes)
109           .map(prefix -> cb.like(symbolicNamePath, prefix + "%"))
110           .toArray(Predicate[]::new);
111       q.where(cb.or(likes));
112       q.orderBy(cb.asc(r.get("host")), cb.asc(symbolicNamePath));
113       return em.createQuery(q).getResultList();
114     };
115   }
116 
117   /** Delete all entities of BundleInfo from the database. */
118   public static final Function<EntityManager, Integer> deleteAll = namedQuery.update("BundleInfo.deleteAll");
119 
120   /** Delete all entities of BundleInfo of a certain host from the database. */
121   public static Function<EntityManager, Integer> deleteByHost(String host) {
122     return namedQuery.update("BundleInfo.deleteByHost", Pair.of("host", host));
123   }
124 
125   /** Delete a bundle info. */
126   public static Function<EntityManager, Integer> delete(final String host, final long bundleId) {
127     return namedQuery.update("BundleInfo.delete", Pair.of("host", host), Pair.of("bundleId", bundleId));
128   }
