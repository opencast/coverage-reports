99    @Override
100   @SuppressWarnings("unchecked")
101   public List<String> get(EName property, final String language) {
102     RequireUtil.notNull(property, "property");
103     RequireUtil.notNull(language, "language");
104     if (LANGUAGE_ANY.equals(language)) {
105       return (List<String>) CollectionUtils.collect(getValuesAsList(property), new Transformer() {
106         @Override
107         public Object transform(Object o) {
108           return ((CatalogEntry) o).getValue();
109         }
110       });
111     } else {
112       final List<String> values = new ArrayList<String>();
113       final boolean langUndef = LANGUAGE_UNDEFINED.equals(language);
114       CollectionUtils.forAllDo(getValuesAsList(property), new Closure() {
115         @Override
116         public void execute(Object o) {
117           CatalogEntry c = (CatalogEntry) o;
118           String lang = c.getAttribute(XML_LANG_ATTR);
119           if ((langUndef && lang == null) || (language.equals(lang)))
120             values.add(c.getValue());
121         }
122       });
123       return values;
124     }
125   }

127   @Override
128   public List<DublinCoreValue> get(EName property) {
129     RequireUtil.notNull(property, "property");
130     return getValuesAsList(property).stream()
131             .map(this::toDublinCoreValue)
132             .collect(Collectors.toList());
133   }

135   private DublinCoreValue toDublinCoreValue(CatalogEntry e) {
136     final String langRaw = e.getAttribute(XML_LANG_ATTR);
137     final String lang = langRaw != null ? langRaw : LANGUAGE_UNDEFINED;
138     final String typeRaw = e.getAttribute(XSI_TYPE_ATTR);
139     if (typeRaw != null) {
140       return DublinCoreValue.mk(e.getValue(), lang, toEName(typeRaw));
141     } else {
142       return DublinCoreValue.mk(e.getValue(), lang);
143     }
144   }

146   private final Function<CatalogEntry, DublinCoreValue> toDublinCoreValue = new Function<CatalogEntry, DublinCoreValue>() {
147     @Override
148     public DublinCoreValue apply(CatalogEntry e) {
149       return toDublinCoreValue(e);
150     }
151   };

153   @Override
154   public Map<EName, List<DublinCoreValue>> getValues() {
155     return mlist(data.values().iterator())
156             .foldl(new HashMap<EName, List<DublinCoreValue>>(),
157                     new Function2<HashMap<EName, List<DublinCoreValue>>, List<CatalogEntry>, HashMap<EName, List<DublinCoreValue>>>() {
158                       @Override
159                       public HashMap<EName, List<DublinCoreValue>> apply(HashMap<EName, List<DublinCoreValue>> map,
160                               List<CatalogEntry> entries) {
161                         if (entries.size() > 0) {
162                           final EName property = entries.get(0).getEName();
163                           map.put(property, mlist(entries).map(toDublinCoreValue).value());
164                         }
165                         return map;
166                       }
167                     });
168   }

198   private CatalogEntry getFirstCatalogEntry(EName property, String language) {
199     CatalogEntry entry = null;
200     switch (language) {
201       case LANGUAGE_UNDEFINED:
202         entry = getFirstLocalizedValue(property, null);
203         break;
204       case LANGUAGE_ANY:
205         for (CatalogEntry value : getValuesAsList(property)) {
206           entry = value;
207           // Prefer values without language information
208           if (!value.hasAttribute(XML_LANG_ATTR))
209             break;
210         }
211         break;
212       default:
213         entry = getFirstLocalizedValue(property, language);
214         break;
215     }
216     return entry;
217   }

220   public String getAsText(EName property, String language, String delimiter) {
221     RequireUtil.notNull(property, "property");
222     RequireUtil.notNull(language, "language");
223     RequireUtil.notNull(delimiter, "delimiter");
224     final List<CatalogEntry> values;
225     switch (language) {
226       case LANGUAGE_UNDEFINED:
227         values = getLocalizedValuesAsList(property, null);
228         break;
229       case LANGUAGE_ANY:
230         values = getValuesAsList(property);
231         break;
232       default:
233         values = getLocalizedValuesAsList(property, language);
234         break;
235     }
236     return values.size() > 0 ? $(values).mkString(delimiter) : null;
237   }

266   private boolean hasMultiplePropertyValues(EName property, String language) {
267     if (LANGUAGE_ANY.equals(language)) {
268       return getValuesAsList(property).size() > 1;
269     } else {
270       int counter = 0;
271       for (CatalogEntry entry : getValuesAsList(property)) {
272         if (equalLanguage(language, entry.getAttribute(XML_LANG_ATTR)))
273           counter++;
274         if (counter > 1)
275           return true;
276       }
277       return false;
278     }
279   }

294   private boolean hasPropertyValue(EName property, final String language) {
295     if (LANGUAGE_ANY.equals(language)) {
296       return getValuesAsList(property).size() > 0;
297     } else {
298       return CollectionUtils.find(getValuesAsList(property), new Predicate() {
299         @Override
300         public boolean evaluate(Object o) {
301           return equalLanguage(((CatalogEntry) o).getAttribute(XML_LANG_ATTR), language);
302         }
303       }) != null;
304     }
305   }

341   private void setValue(EName property, @Nullable String value, String language, @Nullable EName encodingScheme) {
342     if (value == null) {
343       // No value, remove the whole element
344       removeValue(property, language);
345     } else {
346       String lang = !LANGUAGE_UNDEFINED.equals(language) ? language : null;
347       removeLocalizedValues(property, lang);
348       add(property, value, language, encodingScheme);
349     }
350   }

378   void add(EName property, String value, String language, @Nullable EName encodingScheme) {
379     // Ignore empty rootTag element
380     if (LANGUAGE_UNDEFINED.equals(language) && !property.equals(rootTag)) {
381       if (encodingScheme == null) {
382         addElement(property, value);
383       } else {
384         addTypedElement(property, value, encodingScheme);
385       }
386     } else {
387       // Language defined
388       if (encodingScheme == null) {
389         addLocalizedElement(property, value, language);
390       } else {
391         addTypedLocalizedElement(property, value, language, encodingScheme);
392       }
393     }
394   }

409   private void removeValue(EName property, String language) {
410     switch (language) {
411       case LANGUAGE_ANY:
412         removeElement(property);
413         break;
414       case LANGUAGE_UNDEFINED:
415         removeLocalizedValues(property, null);
416         break;
417       default:
418         removeLocalizedValues(property, language);
419         break;
420     }
421   }

428   @Override
429   public Object clone() {
430     DublinCoreCatalog clone = new DublinCoreCatalog();
431     clone.setIdentifier(getIdentifier());
432     clone.setFlavor(getFlavor());
433     clone.setSize(getSize());
434     clone.setChecksum(getChecksum());
435     clone.bindings = bindings; // safe, since XmlNamespaceContext is immutable
436     clone.rootTag = rootTag;
437     for (Map.Entry<EName, List<CatalogEntry>> entry : data.entrySet()) {
438       EName elmName = entry.getKey();
439       EName elmNameCopy = new EName(elmName.getNamespaceURI(), elmName.getLocalName());
440       List<CatalogEntry> elmsCopy = new ArrayList<CatalogEntry>();
441       for (CatalogEntry catalogEntry : entry.getValue()) {
442         elmsCopy.add(new CatalogEntry(catalogEntry.getEName(), catalogEntry.getValue(), catalogEntry.getAttributes()));
443       }
444       clone.data.put(elmNameCopy, elmsCopy);
445     }
446     return clone;
447   }

454   boolean equalLanguage(String a, String b) {
455     return (a == null && eq(b, LANGUAGE_UNDEFINED)) || (b == null && eq(a, LANGUAGE_UNDEFINED)) || eq(a, LANGUAGE_ANY)
456             || eq(b, LANGUAGE_ANY) || (a != null && eq(a, b));
457   }
