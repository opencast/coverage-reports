60  /**
61   * The index endpoint allows the management of the elasticsearch index.
62   */
63  @Path("/index")
64  @RestService(name = "IndexEndpoint", title = "Index Endpoint",
65      abstractText = "Provides operations related to the index that serves both the Admin UI and the External API",
66      notes = {})
67  @Component(
68          immediate = true,
69          property = {
70                  "service.description=Index Endpoint",
71                  "opencast.service.type=org.opencastproject.elasticsearch.index.endpoint",
72                  "opencast.service.path=/index"
73          },
74          service = { IndexEndpoint.class }
75  )
76  @JaxrsResource
77  public class IndexEndpoint {

79    /** The logging facility */
80    private static final Logger logger = LoggerFactory.getLogger(IndexEndpoint.class);

114   @POST
115   @Path("clear")
116   @RestQuery(name = "clearIndex", description = "Clear the index",
117       returnDescription = "OK if index is cleared", responses = {
118       @RestResponse(description = "Index is cleared", responseCode = HttpServletResponse.SC_OK),
119       @RestResponse(description = "Unable to clear index",
120               responseCode = HttpServletResponse.SC_INTERNAL_SERVER_ERROR) })
121   public Response clearIndex() {
122     final SecurityContext securityContext = new SecurityContext(securityService, securityService.getOrganization(),
123         securityService.getUser());
124     return securityContext.runInContext(() -> {
125       try {
126         logger.info("Clear the index");
127         elasticsearchIndex.clear();
128         return R.ok();
129       } catch (Throwable t) {
130         logger.error("Clearing the index failed", t);
131         return R.serverError();
132       }
133     });
134   }

136   @POST
137   @Path("rebuild/{service}")
138   @RestQuery(name = "partiallyRebuildIndex",
139       description = "Repopulates the Index from an specific service",
140       returnDescription = "OK if repopulation has started", pathParameters = {
141       @RestParameter(name = "service", isRequired = true, description = "The service to recreate index from. "
142         + "The available services are: Themes, Series, Scheduler, AssetManager, Comments, Workflow and Search. "
143         + "The service order (see above) is very important except for Search! Make sure, you do not run index rebuild"
144         + "for more than one service at a time!",
145         type = RestParameter.Type.STRING) }, responses = {
146       @RestResponse(description = "OK if repopulation has started", responseCode = HttpServletResponse.SC_OK) })
147   public Response partiallyRebuildIndex(@PathParam("service") final String serviceStr) {
148     Service service = EnumUtils.getEnum(Service.class, serviceStr);
149     if (service == null) {
150       return R.badRequest("The given path param for service was invalid.");
151     }
152     IndexProducer indexProducer = indexRebuildService.getIndexProducer(service);
153 
154     final SecurityContext securityContext = new SecurityContext(securityService, securityService.getOrganization(),
155         securityService.getUser());
156     executor.execute(() -> securityContext.runInContext(() -> {
157       try {
158         logger.info("Starting to repopulate the index from service {}", service);
159         indexRebuildService.rebuildIndex(indexProducer, DataType.ALL);
160       } catch (Throwable t) {
161         logger.error("Repopulating the index failed", t);
162       }
163     }));
164     return R.ok();
165   }

167   @POST
168   @Path("rebuild/{service}/{type}")
169   @RestQuery(name = "partiallyRebuildIndexByPart",
170       description = "Repopulates the index for a specific service, but only for a specific type of data "
171         + "in order to save time. "
172         + "Only use this endpoint if you have been explicitly told to or know what you are doing, else you risk "
173         + "inconsistent data in the index!",
174       returnDescription = "OK if repopulation has started",
175       pathParameters = {
176           @RestParameter(
177               name = "service",
178               isRequired = true,
179               description = "The service to recreate index from. The available services are: AssetManager. ",
180               type = RestParameter.Type.STRING
181           ),
182           @RestParameter(
183               name = "type",
184               isRequired = true,
185               description = "The type of data to re-index. The available types are: ACL. ",
186               type = RestParameter.Type.STRING
187           )
188       },
189       responses = {
190         @RestResponse(
191             description = "OK if repopulation has started",
192             responseCode = HttpServletResponse.SC_OK
193         ),
194         @RestResponse(
195             description = "If the given parameters aren't one of the supported values",
196             responseCode = HttpServletResponse.SC_BAD_REQUEST
197         )
198       }
199   )
200   public Response partiallyRebuildIndexByType(
201       @PathParam("service") final String serviceStr,
202       @PathParam("type") final String dataTypeStr
203   ) {
204     Service service = EnumUtils.getEnum(Service.class, serviceStr);
205     if (service == null) {
206       return R.badRequest("The given path param for service was invalid.");
207     }
208     IndexProducer indexProducer = indexRebuildService.getIndexProducer(service);
209 
210     DataType dataType = EnumUtils.getEnum(DataType.class, dataTypeStr);
211     if (dataType == null || !indexProducer.dataTypeSupported(dataType)) {
212       return R.badRequest("The given path param for data type was invalid.");
213     }
214 
215     final SecurityContext securityContext = new SecurityContext(securityService, securityService.getOrganization(),
216             securityService.getUser());
217     executor.execute(() -> securityContext.runInContext(() -> {
218       try {
219         logger.info("Starting to repopulate the index from service {}", service);
220         indexRebuildService.rebuildIndex(indexProducer, dataType);
221       } catch (Throwable t) {
222         logger.error("Repopulating the index failed", t);
223       }
224     }));
225     return R.ok();
226   }

237     executor.execute(() -> securityContext.runInContext(() -> {
238       try {
239         logger.info("Starting to repopulate the index");

242         logger.error("Repopulating the index failed", t);
243       }
244     }));
245     return R.ok();
246   }

248   @POST
249   @Path("resume/{service}")
250   @RestQuery(name = "resumeIndexRebuild",
251           description = "Starts repopulating the Index from an specific service and will then continue with the rest "
252                   + "of the services that come afterwards",
253           returnDescription = "OK if repopulation has started", pathParameters = {
254           @RestParameter(name = "service", isRequired = true, description = "The service to start recreating the index "
255                   + "from. "
256                   + "The available services are: Themes, Series, Scheduler, AssetManager, Comments and Workflow. "
257                   + "All services that come after the specified service in the order above will also run.",
258                   type = RestParameter.Type.STRING) }, responses = {
259           @RestResponse(description = "OK if repopulation has started", responseCode = HttpServletResponse.SC_OK) })
260   public Response resumeIndexRebuild(@PathParam("service") final String serviceStr) {
261     Service service = EnumUtils.getEnum(Service.class, serviceStr);
262     if (service == null) {
263       return R.badRequest("The given path param for service was invalid.");
264     }
265 
266     final SecurityContext securityContext = new SecurityContext(securityService, securityService.getOrganization(),
267             securityService.getUser());
268     executor.execute(() -> securityContext.runInContext(() -> {
269       try {
270         logger.info("Resume repopulating the index from service {}", service);
271         indexRebuildService.resumeIndexRebuild(service);
272       } catch (Throwable t) {
273         logger.error("Repopulating the index failed", t);
274       }
275     }));
276     return R.ok();
277   }

279   @GET
280   @Path("rebuild/states.json")
281   @Produces(MediaType.APPLICATION_JSON)
282   @RestQuery(name = "getrebuildstates", description = "Returns the index rebuild service"
283           + "repopulation states", returnDescription = "The repopulation states of the index rebuild services",
284           responses = {
285           @RestResponse(description = "Returns the repopulation states of the index rebuild services",
286           responseCode = HttpServletResponse.SC_OK),
287     })
288   public Response getRebuildStates() {
289     Map<String, String> states = indexRebuildService.getRebuildStates();
290     JSONArray statesAsJson = new JSONArray();
291     for (Map.Entry<String, String> entry : states.entrySet()) {
292       JSONObject data = new JSONObject();
293       data.put("type", entry.getKey());
294       data.put("state", entry.getValue());
295       data.put("executionOrder", IndexRebuildService.Service.valueOf(entry.getKey()).ordinal());
296       statesAsJson.add(data);
297     }
298     JSONObject service = new JSONObject();
299     service.put("service", statesAsJson);
300     JSONObject services = new JSONObject();
301     services.put("services", service);
302     return Response.ok(services.toJSONString()).build();
303   }
