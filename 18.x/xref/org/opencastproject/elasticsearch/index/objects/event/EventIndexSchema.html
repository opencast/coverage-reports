25  import org.opencastproject.elasticsearch.impl.IndexSchema;
26  
27  /**
28   * Interface defining the mapping between data and field names in the search index.
29   */
30  public interface EventIndexSchema extends IndexSchema {
31  
32    /** The unique identifier */
33    String UID = "uid";
34  
35    /** The organization */
36    String ORGANIZATION = "organization";
37  
38    /** The recording object */
39    String OBJECT = "object";
40  
41    /** The event series identifier */
42    String SERIES_ID = "series_id";
43  
44    /** The event series name */
45    String SERIES_NAME = "series_name";
46  
47    /** The key in the input documents representing the prefix to extended metadata */
48    String EXTENDED_METADATA_PREFIX = "extended_metadata_";
49  
50    /** The access policy */
51    String ACCESS_POLICY = "access_policy";
52  
53    /** The key in the input documents representing the prefix to an access control entry */
54    String ACL_PERMISSION_PREFIX = "acl_permission_";
55  
56    /** The name of the managed ACL used by the event (if set) */
57    String MANAGED_ACL = "managed_acl";
58  
59    /** The has comments field name */
60    String HAS_COMMENTS = "has_comments";
61  
62    /** The has open comments field name */
63    String HAS_OPEN_COMMENTS = "has_open_comments";
64  
65    /** The comments */
66    String COMMENTS = "comments";
67  
68    /** The event has open comment that it needs cutting */
69    String NEEDS_CUTTING = "needs_cutting";
70  
71    /** The publications */
72    String PUBLICATION = "publication";
73  
74    /** The is published field name */
75    String IS_PUBLISHED = "is_published";
76  
77    /** The event status */
78    String EVENT_STATUS = "event_status";
79  
80    /*
81     * Dublincore fields
82     */
83  
84    /** The event description */
85    String DESCRIPTION = "description";
86  
87    /** The event location */
88    String LOCATION = "location";
89  
90    /** The event language */
91    String LANGUAGE = "language";
92  
93    /** The recording title */
94    String TITLE = "title";
95  
96    /** The recording start date */
97    String START_DATE = "start_date";
98  
99    /** The recording end date */
100   String END_DATE = "end_date";
101 
102   /** The recording duration */
103   String DURATION = "duration";
104 
105   /** The contributors */
106   String CONTRIBUTOR = "contributor";
107 
108   /** The contributors */
109   String PRESENTER = "presenter";
110 
111   /** The subject */
112   String SUBJECT = "subject";
113 
114   /** The event source */
115   String SOURCE = "source";
116 
117   /** The creation date */
118   String CREATED = "created";
119 
120   /** The creator */
121   String CREATOR = "creator";
122 
123   /** The publisher */
124   String PUBLISHER = "publisher";
125 
126   /** The license */
127   String LICENSE = "license";
128 
129   /** The rights */
130   String RIGHTS = "rights";
131 
132   /*
133    * Workflow specific fields
134    */
135 
136   /** The workflow state */
137   String WORKFLOW_STATE = "workflow_state";
138 
139   /** The workflow id */
140   String WORKFLOW_ID = "workflow_id";
141 
142   /** The workflow definition id */
143   String WORKFLOW_DEFINITION_ID = "workflow_definition_id";
144 
145   /*
146    * Scheduler specific fields
147    */
148 
149   /** The recording status */
150   String RECORDING_STATUS = "recording_status";
151 
152   /*
153    * Technical Metadata
154    */
155   /** Id of the agent that will record. */
156   String AGENT_ID = "agent_id";
157 
158   /** The ending time of the recording. */
159   String TECHNICAL_END = "technical_end";
160 
161   /** The start time of the recording. */
162   String TECHNICAL_START = "technical_start";
163 
164   /** The technical presenters of the recording. */
165   String TECHNICAL_PRESENTERS = "technical_presenters";
166 
167   /*
168    * Archive specific fields
169    */
170 
171   /** The archive version */
172   String ARCHIVE_VERSION = "archive_version";
173 
174 }
