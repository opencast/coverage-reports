52  /** Contains the actual harvesting logic.  */
53  final class Harvest {
54    /** Private constructor for utility class with static methods */
55    private Harvest() { }
56  
57    /**
58     * A duration to allow some leeway for certain database operations to be slow.
59     *
60     * The correct usage of the harvesting API depends on the correct adjustment of the {@code since}
61     * parameter between different requests. If `since` is increased too much, some event/series
62     * modifications could be missed. One fact which makes this harder is that the modified date of
63     * events/series can be significantly lower/before the time those changes are written to the
64     * database or index. That's because the "now" timestamp is created in Java and written to the
65     * database afterwards. Unfortunately, we are not just talking about milliseconds, but up to many
66     * seconds. This can have different reasons, but since Opencast can be distributed, network plays
67     * a big role here.
68     *
69     * But to not completely give up on an incremental harvesting API, we define an arbitrary "buffer
70     * period". Writes that take longer (i.e. the time between `new Date()` and the serialization in
71     * the DB/index) than this buffer could lead to missed updates with this harvesting API.
72     */
73    private static final long TIME_BUFFER_SIZE = 3 * 60 * 1000;
74  
75    private static final Logger logger = LoggerFactory.getLogger(Harvest.class);
76  
77    static Jsons.Obj harvest(
78        int preferredAmount,
79        Date since,
80        SearchService searchService,
81        SeriesService seriesService,
82        AuthorizationService authorizationService,
83        SecurityService securityService,
84        PlaylistService playlistService,
85        Workspace workspace
86    ) throws UnauthorizedException, SeriesException {
87      final var org = securityService.getOrganization().getId();
88  
89      var user = securityService.getUser();
90      var orgAdminRole = securityService.getOrganization().getAdminRole();
91      var isAdmin = user.hasRole(SecurityConstants.GLOBAL_ADMIN_ROLE) || user.hasRole(orgAdminRole);
92      if (!isAdmin) {
93        throw new UnauthorizedException(user, "Only (org-) admins can access the Tobira harvest API");
94      }
95  
96      // ===== Retrieve information about events, series, and playlists =============================
97      //
98      // In this step, we always request `preferredAmount + 1` in order to figure out the values for
99      // `hasMore` and `includesItemsUntil` more precisely.
100 
101     // Retrieve episodes from index.
102     final SearchSourceBuilder q = new SearchSourceBuilder().query(
103             QueryBuilders.boolQuery()
104                 .must(QueryBuilders.termQuery(SearchResult.ORG, org))
105                 .must(QueryBuilders.rangeQuery(SearchResult.MODIFIED_DATE).gte(since.getTime()))
106                 .must(QueryBuilders.termQuery(SearchResult.TYPE, SearchService.IndexEntryType.Episode)))
107         .sort(SearchResult.MODIFIED_DATE, SortOrder.ASC)
108         .size(preferredAmount + 1);
109     final SearchResultList results = searchService.search(q);
110     final var hasMoreEvents = results.getHits().size() >= preferredAmount + 1;
111 
112     logger.debug("Retrieved {} events from the index during harvest", results.getHits().size());
113 
114     // Start tracking the timestamp upper limit. It starts with "now" but gets decreased whenever we
115     // query items that are limited by `preferredAmount`. We can use this timestamp to restrict
116     // queries below, to avoid loading useless items: any item modified after this timestamp will be
117     // requested in the next harvest request anyway.
118     var includesItemsUntilRaw = new Date();
119     if (hasMoreEvents) {
120       includesItemsUntilRaw = results.getHits().get(results.getHits().size() - 1).getModifiedDate();
121     }
122 
123 
124     // Retrieve series from DB.
125     final var rawSeries = seriesService.getAllForAdministrativeRead(
126         since,
127         Optional.of(includesItemsUntilRaw),
128         preferredAmount + 1
129     );
130     final var hasMoreSeriesInRange = rawSeries.size() >= preferredAmount + 1;
131     logger.debug("Retrieved {} series from the database during harvest", rawSeries.size());
132 
133     if (hasMoreSeriesInRange) {
134       final var lastSeriesUpdated = rawSeries.get(rawSeries.size() - 1).getModifiedDate();
135       if (lastSeriesUpdated.before(includesItemsUntilRaw)) {
136         includesItemsUntilRaw = lastSeriesUpdated;
137       }
138     }
139 
140 
141     // Retrieve playlists from DB.
142     final var rawPlaylists = playlistService.getAllForAdministrativeRead(
143         since,
144         includesItemsUntilRaw,
145         preferredAmount + 1
146     );
147     final var hasMorePlaylistsInRange = rawPlaylists.size() >= preferredAmount + 1;
148     logger.debug("Retrieved {} playlists from the database during harvest", rawPlaylists.size());
149 
150     if (hasMorePlaylistsInRange) {
151       final var lastPlaylistUpdated = rawPlaylists.get(rawPlaylists.size() - 1).getUpdated();
152       if (lastPlaylistUpdated.before(includesItemsUntilRaw)) {
153         includesItemsUntilRaw = lastPlaylistUpdated;
154       }
155     }
156 
157 
158     // ===== Convert all items into the JSON output representation ================================
159 
160     // We limit to `preferredAmount` here again, because we fetched `preferredAmount + 1` above.
161     final var eventItems = results.getHits().stream()
162         .limit(preferredAmount)
163         .map(event -> {
164           try {
165             return new Item(event, authorizationService, workspace);
166           } catch (Exception e) {
167             var id = event == null ? null : event.getId();
168             logger.error("Error reading event '{}' (skipping...)", id, e);
169             return null;
170           }
171         })
172         .filter(item -> item != null);
173 
174     final var seriesItems = rawSeries.stream()
175         .limit(preferredAmount)
176         .map(series -> {
177           try {
178             return new Item(series);
179           } catch (Exception e) {
180             var id = series == null ? null : series.getId();
181             logger.error("Error reading series '{}' (skipping...)", id, e);
182             return null;
183           }
184         })
185         .filter(item -> item != null);
186 
187     final var playlistItems = rawPlaylists.stream()
188         .limit(preferredAmount)
189         .map(playlist -> {
190           try {
191             return new Item(playlist);
192           } catch (Exception e) {
193             var id = playlist == null ? null : playlist.getId();
194             logger.error("Error reading playlist '{}' (skipping...)", id, e);
195             return null;
196           }
197         })
198         .filter(item -> item != null);
199 
200 
201     // Combine events, series, and playlists into one combined list and sort it. We filter out all
202     // items that were modified after `includesItemsUntilRaw` as those are transferred in the next
203     // request anyway. So this is just about response size savings.
204     //
205     // The sorting is, again, not for correctness, because consumers of this API need to be
206     // able to deal with that. However, sorting this here will result in fewer temporary objects
207     // or invalid states in the consumer.
208     Date finalIncludesItemsUntilRaw = includesItemsUntilRaw; // copy for lambda
209     final var items = Stream.of(eventItems, seriesItems, playlistItems)
210         .flatMap(Function.identity())
211         .filter(item -> !item.getModifiedDate().after(finalIncludesItemsUntilRaw))
212         .collect(Collectors.toCollection(ArrayList::new));
213     items.sort(Comparator.comparing(item -> item.getModifiedDate()));
214 
215 
216     // Obtain information to allow Tobira to plan the next harvesting request.
217     final var hasMore = hasMoreEvents || hasMoreSeriesInRange || hasMorePlaylistsInRange;
218 
219     // The `includesItemsUntil` we return has to be at least `TIME_BUFFER_SIZE` in the past. See
220     // the constant's documentation for more information on that.
221     final var includesItemsUntil = Math.min(
222         includesItemsUntilRaw.getTime(),
223         new Date().getTime() - TIME_BUFFER_SIZE
224     );
225 
226 
227     // Assembly full response.
228     final var outItems = items.stream()
229         .map(item -> item.getJson())
230         .collect(Collectors.toCollection(ArrayList::new));
231     final var json = Jsons.obj(
232         Jsons.p("includesItemsUntil", includesItemsUntil),
233         Jsons.p("hasMore", hasMore),
234         Jsons.p("items", Jsons.arr(outItems))
235     );
236     logger.debug(
237         "Returning {} items from harvesting (hasMore = {}, includesItemsUntil = {})",
238         items.size(),
239         hasMore,
240         new Date(includesItemsUntil)
241     );
242 
243     return json;
244   }
