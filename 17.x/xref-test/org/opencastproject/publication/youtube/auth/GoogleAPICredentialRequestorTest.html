22  package org.opencastproject.publication.youtube.auth;
23  
24  import static org.easymock.EasyMock.anyObject;
25  import static org.easymock.EasyMock.createMock;
26  import static org.easymock.EasyMock.expect;
27  import static org.easymock.EasyMock.replay;
28  
29  import org.opencastproject.publication.youtube.UnitTestUtils;
30  
31  import com.google.api.client.googleapis.auth.oauth2.GoogleCredential;
32  import com.google.api.client.util.store.DataStore;
33  
34  import org.json.simple.parser.ParseException;
35  import org.junit.Rule;
36  import org.junit.Test;
37  import org.junit.rules.TemporaryFolder;
38  
39  import java.io.File;
40  import java.io.IOException;
41  
42  public class GoogleAPICredentialRequestorTest {
43  
44    @Rule
45    public TemporaryFolder testFolder = new TemporaryFolder();
46  
47    @Test(expected = IllegalArgumentException.class)
48    public void testInvalidArgs() throws IOException, ParseException {
49      GoogleAPICredentialRequestor.main(new String[0]);
50    }
51  
52    @Test(expected = IllegalArgumentException.class)
53    public void testFileDoesNotExist() throws IOException, ParseException {
54      GoogleAPICredentialRequestor.main(new String[] {
55          "/this/path/does/exist",
56          "credentialDataStore",
57          "dataStoreDirectory"
58      });
59    }
60  
61    @Test
62    public void testClientSecrets() throws IOException, ParseException {
63      final String clientId = "clientId";
64      final String credentialDataStore = "credentialDataStore";
65      final String dataStoreDirectory = "dataStoreDirectory";
66      final File clientSecrets = UnitTestUtils.getMockClientSecretsFile(clientId,
67          testFolder.newFile("client-secrets-youtube-v3.json"));
68      try {
69        final OAuth2CredentialFactory credentialFactory = createMock(OAuth2CredentialFactory.class);
70        expect(credentialFactory.getDataStore(credentialDataStore, dataStoreDirectory))
71          .andReturn(createMock(DataStore.class)).once();
72        expect(credentialFactory.getGoogleCredential(anyObject(DataStore.class), anyObject(ClientCredentials.class)))
73          .andReturn(new GoogleCredential()).once();
74        replay(credentialFactory);
75        GoogleAPICredentialRequestor.setCredentialFactory(credentialFactory);
76        GoogleAPICredentialRequestor
77          .main(new String[] {clientSecrets.getAbsolutePath(), credentialDataStore, dataStoreDirectory});
78      } finally {
79        if (clientSecrets != null) {
80          clientSecrets.delete();
81        }
82      }
83    }
84  }
