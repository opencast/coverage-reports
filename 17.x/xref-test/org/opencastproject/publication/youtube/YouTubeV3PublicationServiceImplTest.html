61    private YouTubeV3PublicationServiceImpl service;
62    private YouTubeAPIVersion3Service youTubeService;
63    private OrganizationDirectoryService orgDirectory;
64    private SecurityService security;
65    private ServiceRegistry registry;
66    private UserDirectoryService userDirectoryService;
67    private Workspace workspace;
68  
69    @Rule
70    public TemporaryFolder testFolder = new TemporaryFolder();
71  
72    @Before
73    public void before() throws Exception {
74      youTubeService = createMock(YouTubeAPIVersion3Service.class);
75      youTubeService.initialize(anyObject(ClientCredentials.class));
76      expectLastCall();
77      orgDirectory = createMock(OrganizationDirectoryService.class);
78      security = createMock(SecurityService.class);
79      registry = createMock(ServiceRegistry.class);
80      List<HostRegistration> hosts = new LinkedList<HostRegistration>();
81      HostRegistration host = new HostRegistrationInMemory("localhost", "localhost", "AllInOne", 1.0F, 1, 1024L);
82      hosts.add(host);
83      expect(registry.getHostRegistrations()).andReturn(hosts).anyTimes();
84      userDirectoryService = createMock(UserDirectoryService.class);
85      workspace = createMock(Workspace.class);
86      //
87      service = new YouTubeV3PublicationServiceImpl(youTubeService);
88      service.setOrganizationDirectoryService(orgDirectory);
89      service.setSecurityService(security);
90      service.setServiceRegistry(registry);
91      service.setUserDirectoryService(userDirectoryService);
92      service.setWorkspace(workspace);
93    }
94  
95    @Test
96    public void testPublishNewPlaylist() throws Exception {
97      final File baseDir = new File(this.getClass().getResource("/mediapackage").toURI());
98      final String xml = FileUtils.readFileToString(new File(baseDir, "manifest.xml"));
99      final MediaPackage mediaPackage = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder()
100         .loadFromXml(xml);
101     //
102     expect(youTubeService.getMyPlaylistByTitle(mediaPackage.getTitle())).andReturn(null).once();
103     expect(youTubeService.createPlaylist(mediaPackage.getSeriesTitle(), null, mediaPackage.getSeries()))
104       .andReturn(new Playlist()).once();
105     expect(youTubeService.addVideoToMyChannel(anyObject(VideoUpload.class))).andReturn(new Video()).once();
106     expect(youTubeService.addPlaylistItem(anyObject(String.class), anyObject(String.class)))
107       .andReturn(new PlaylistItem()).once();
108 
109     expect(registry.createJob(anyObject(String.class), anyObject(String.class), anyObject(List.class),
110           anyObject(Float.class))).andReturn(new JobImpl()).once();
111     replay(youTubeService, orgDirectory, security, registry, userDirectoryService, workspace);
112     service.updated(getServiceProperties());
113     service.publish(mediaPackage, mediaPackage.getTracks()[0]);
114   }
115 
116   private Dictionary getServiceProperties() throws IOException {
117     final Properties p = new Properties();
118     YouTubeUtils.put(p, YouTubeKey.credentialDatastore, "credentialDatastore");
119     YouTubeUtils.put(p, YouTubeKey.scopes, "foo");
120     final String absolutePath = UnitTestUtils.getMockClientSecretsFile("clientId",
121         testFolder.newFile("client-secrets-youtube-v3.json")).getAbsolutePath();
122     YouTubeUtils.put(p, YouTubeKey.clientSecretsV3, absolutePath);
123     YouTubeUtils.put(p, YouTubeKey.dataStore, "dataStore");
124     YouTubeUtils.put(p, YouTubeKey.keywords, "foo");
125     YouTubeUtils.put(p, YouTubeKey.defaultPlaylist, "foo");
126     YouTubeUtils.put(p, YouTubeKey.makeVideosPrivate, "true");
127     // maxFieldLength is optional so we skip
128     return p;
129   }
