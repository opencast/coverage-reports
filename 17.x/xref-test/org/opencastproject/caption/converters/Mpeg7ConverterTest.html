23  package org.opencastproject.caption.converters;
24  
25  import org.opencastproject.caption.api.Caption;
26  import org.opencastproject.caption.api.CaptionConverterException;
27  import org.opencastproject.caption.api.IllegalTimeFormatException;
28  import org.opencastproject.caption.api.Time;
29  import org.opencastproject.caption.impl.TimeImpl;
30  
31  import org.apache.commons.io.IOUtils;
32  import org.junit.After;
33  import org.junit.Assert;
34  import org.junit.Before;
35  import org.junit.Test;
36  
37  import java.io.ByteArrayInputStream;
38  import java.io.ByteArrayOutputStream;
39  import java.io.IOException;
40  import java.io.InputStream;
41  import java.util.List;
42  
43  /**
44   * Test class for Mpeg7 format.
45   *
46   */
47  public class Mpeg7ConverterTest {
48  
49    // SubRip converter
50    private Mpeg7CaptionConverter converter;
51    // mpeg7 sample
52    private InputStream inputStream;
53    // output stream
54    private ByteArrayOutputStream outputStream;
55    // expected second segment start time
56    private Time time;
57    // expected output
58    private static final String EXPECTED_LAST_CAPTION = "ENJOYS GETTING QUESTIONS";
59    // Captions language
60    private static final String LANGUAGE = "en";
61    // Sample file
62    private static final String FILE = "/sample.mpeg7.xml";
63  
64    @Before
65    public void setUp() throws IOException, IllegalTimeFormatException {
66      converter = new Mpeg7CaptionConverter();
67      inputStream = Mpeg7ConverterTest.class.getResourceAsStream(FILE);
68      outputStream = new ByteArrayOutputStream();
69      time = new TimeImpl(0, 0, 5, 89);
70    }
71  
72    @Test
73    public void testImportAndExport() {
74      try {
75        // Test import from example file
76        List<Caption> collection = testImport(inputStream);
77  
78        IOUtils.closeQuietly(inputStream);
79  
80        converter.exportCaption(outputStream, collection, LANGUAGE);
81  
82        inputStream = new ByteArrayInputStream(outputStream.toByteArray());
83  
84        testImport(inputStream);
85  
86      } catch (Exception e) {
87        e.printStackTrace();
88        Assert.fail(e.getMessage());
89      }
90    }
91  
92    public List<Caption> testImport(InputStream inputStream) {
93      List<Caption> collection = null;
94  
95      try {
96        collection = converter.importCaption(inputStream, LANGUAGE);
97  
98        int nbCaption = collection.size();
99  
100       // Check size
101       Assert.assertEquals(25, nbCaption);
102 
103       // Check the last caption value
104       Assert.assertEquals(EXPECTED_LAST_CAPTION, collection.get(nbCaption - 1).getCaption()[0]);
105 
106       // Check start time from second segment
107       Assert.assertEquals(0, collection.get(1).getStartTime().compareTo(time));
108 
109     } catch (CaptionConverterException e) {
110       e.printStackTrace();
111       Assert.fail(e.getMessage());
112     }
113 
114     return collection;
115   }
116 
117   @After
118   public void tear() throws IOException {
119     IOUtils.closeQuietly(inputStream);
120     IOUtils.closeQuietly(outputStream);
121   }
122 
123 }
