22  package org.opencastproject.transcription.microsoft.azure;
23  
24  import org.apache.commons.lang3.StringUtils;
25  import org.junit.Assert;
26  import org.junit.Before;
27  import org.junit.Test;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import java.io.File;
32  import java.io.IOException;
33  import java.net.URISyntaxException;
34  import java.net.URL;
35  import java.util.Map;
36  
37  public class MicrosoftAzureStorageClientTest {
38  
39    private static final Logger logger = LoggerFactory.getLogger(MicrosoftAzureStorageClientTest.class);
40  
41    /**
42     * Provide the Microsoft Azure storage account name as value here or
43     * put the value into AZURE_STORAGE_ACCOUNT_NAME environment variable.
44     */
45    private String azureStorageAccountName;
46    /**
47     * Provide the Microsoft Azure account access key as value here or
48     * put the value into AZURE_ACCOUNT_ACCESS_KEY environment variable.
49     */
50    private String azureAccountAccessKey;
51  
52    private String azureStorageContainerName;
53    private boolean enabled;   // will be set in setUp according the values of
54                               // azureStorageAccountName and azureAccountAccessKey
55  
56    private MicrosoftAzureStorageClient azureStorageClient;
57  
58    @Before
59    public void setUp() throws MicrosoftAzureStorageClientException {
60      azureStorageAccountName = System.getenv().getOrDefault("AZURE_STORAGE_ACCOUNT_NAME", "");
61      azureAccountAccessKey = System.getenv().getOrDefault("AZURE_ACCOUNT_ACCESS_KEY", "");
62      azureStorageContainerName = System.getenv().getOrDefault("AZURE_STORAGE_CONTAINER_NAME", "opencast-transcriptions");
63      enabled = StringUtils.isNotBlank(azureStorageAccountName)
64          && StringUtils.isNotBlank(azureAccountAccessKey);
65      if (!enabled) {
66        logger.debug("Skip tests.");
67        return;
68      }
69      azureStorageClient = new MicrosoftAzureStorageClient(new MicrosoftAzureAuthorization(azureStorageAccountName,
70          azureAccountAccessKey));
71      logger.debug("Run tests on Azure storage account {}", azureStorageAccountName);
72    }
73  
74    @Test
75    public void containerExists()
76            throws MicrosoftAzureStorageClientException, IOException, MicrosoftAzureNotAllowedException {
77      if (!enabled) {
78        return;
79      }
80      boolean containerExists = azureStorageClient.containerExists(azureStorageContainerName);
81      Assert.assertTrue(containerExists);
82    }
83  
84    @Test
85    public void getContainerProperties()
86            throws MicrosoftAzureStorageClientException, IOException, MicrosoftAzureNotFoundException,
87            MicrosoftAzureNotAllowedException {
88      if (!enabled) {
89        return;
90      }
91      Map<String, String> containerProps = azureStorageClient.getContainerProperties(azureStorageContainerName);
92      Assert.assertNotNull(containerProps);
93      Assert.assertFalse(containerProps.isEmpty());
94      logger.debug("Container properties: {}", containerProps);
95    }
96  
97    @Test
98    public void createContainer()
99            throws MicrosoftAzureStorageClientException, IOException, MicrosoftAzureNotAllowedException {
100     if (!enabled) {
101       return;
102     }
103     azureStorageClient.createContainer("opencast-transcriptions2");
104   }
105 
106   @Test
107   public void uploadFile()
108           throws MicrosoftAzureStorageClientException, MicrosoftAzureNotAllowedException, IOException,
109           URISyntaxException {
110     if (!enabled) {
111       return;
112     }
113     URL testFileUrl = MicrosoftAzureStorageClientTest.class.getResource("/test.txt");
114     File testFile = new File(testFileUrl.toURI());
115     String blobUrl = azureStorageClient.uploadFile(testFile, azureStorageContainerName, null, "test.txt");
116     Assert.assertTrue("Azure storage blob URL should end with /" + azureStorageContainerName + "/test.txt",
117         StringUtils.endsWithIgnoreCase(blobUrl, "/" + azureStorageContainerName + "/test.txt"));
118   }
119 
120   @Test
121   public void uploadFile2()
122           throws MicrosoftAzureStorageClientException, MicrosoftAzureNotAllowedException, IOException,
123           URISyntaxException {
124     if (!enabled) {
125       return;
126     }
127     URL testFileUrl = MicrosoftAzureStorageClientTest.class.getResource("/test.ogg");
128     File testFile = new File(testFileUrl.toURI());
129     String blobUrl = azureStorageClient.uploadFile(testFile, azureStorageContainerName, null, "test.ogg");
130     Assert.assertTrue("Azure storage blob URL should end with /" + azureStorageContainerName + "/test.txt",
131         StringUtils.endsWithIgnoreCase(blobUrl, "/" + azureStorageContainerName + "/test.ogg"));
132   }
133 
134   @Test
135   public void deleteFile()
136           throws MicrosoftAzureStorageClientException, MicrosoftAzureNotAllowedException, IOException,
137           URISyntaxException {
138     if (!enabled) {
139       return;
140     }
141     URL testFileUrl = MicrosoftAzureStorageClientTest.class.getResource("/test.txt");
142     File testFile = new File(testFileUrl.toURI());
143     String blobUrl = azureStorageClient.uploadFile(testFile, azureStorageContainerName, null, "test.txt");
144     Assert.assertNotNull(blobUrl);
145     azureStorageClient.deleteFile(new URL(blobUrl));
146   }
147 }
