22  package org.opencastproject.authorization.xacml.manager.util;
23  
24  import static org.junit.Assert.assertEquals;
25  import static org.junit.Assert.assertFalse;
26  import static org.junit.Assert.assertTrue;
27  
28  import org.opencastproject.authorization.xacml.manager.api.ManagedAcl;
29  import org.opencastproject.authorization.xacml.manager.impl.ManagedAclImpl;
30  import org.opencastproject.security.api.AccessControlEntry;
31  import org.opencastproject.security.api.AccessControlList;
32  
33  import org.codehaus.jettison.json.JSONObject;
34  import org.junit.Test;
35  
36  /**
37   * Unit tests for {@link AccessInformationUtil}
38   */
39  public class AccessInformationUtilTest {
40  
41    private static final String ORGANISATION_1_ID = "org-1";
42    private static final String MANAGED_ACL_1_NAME = "Managed ACL 1";
43  
44    private static final String ROLE_ADMIN = "ROLE_ADMIN";
45    private static final String ROLE_STUDENT = "ROLE_STUDENT";
46  
47    private static final String ACTION_READ = "ACTION_READ";
48    private static final String ACTION_WRITE = "ACTION_WRITE";
49  
50    private static final AccessControlEntry ACE_ROLE_ADMIN_ALLOW_ACTION_READ = new AccessControlEntry(ROLE_ADMIN,
51            ACTION_READ, true);
52    private static final AccessControlEntry ACE_ROLE_ADMIN_ALLOW_ACTION_WRITE = new AccessControlEntry(ROLE_ADMIN,
53            ACTION_WRITE, true);
54    private static final AccessControlEntry ACE_ROLE_STUDENT_ALLOW_ACTION_READ = new AccessControlEntry(ROLE_STUDENT,
55            ACTION_READ, true);
56    private static final AccessControlEntry ACE_ROLE_STUDENT_DISALLOW_ACTION_WRITE = new AccessControlEntry(ROLE_STUDENT,
57            ACTION_WRITE, false);
58  
59    /**
60     * Test method for {@link AccessInformationUtil#serializeManagedAcl(ManagedAcl)}
61     */
62    @Test
63    public void testSerializeManagedAcl() throws Exception {
64      AccessControlList acl = new AccessControlList();
65      acl.getEntries().add(ACE_ROLE_ADMIN_ALLOW_ACTION_READ);
66  
67      ManagedAcl manAcl = new ManagedAclImpl(1L, MANAGED_ACL_1_NAME, ORGANISATION_1_ID, acl);
68  
69      JSONObject aclJson = AccessInformationUtil.serializeManagedAcl(manAcl);
70      assertEquals(1L, aclJson.getLong("id"));
71      assertEquals(MANAGED_ACL_1_NAME, aclJson.getString("name"));
72    }
73  
74    /**
75     * Test method for {@link AccessInformationUtil#serializeManagedAcl(ManagedAcl)}
76     */
77    @Test(expected = IllegalArgumentException.class)
78    public void testSerializeManagedAclWithNull() throws Exception {
79      AccessInformationUtil.serializeManagedAcl(null);
80    }
81  
82    /**
83     * Test method for {@link AccessInformationUtil#serializePrivilegesByRole(AccessControlList)}
84     */
85    @Test
86    public void testSerializePrivilegesByRole() throws Exception {
87      AccessControlList acl = new AccessControlList();
88      acl.getEntries().add(ACE_ROLE_ADMIN_ALLOW_ACTION_READ);
89      acl.getEntries().add(ACE_ROLE_ADMIN_ALLOW_ACTION_WRITE);
90      acl.getEntries().add(ACE_ROLE_STUDENT_ALLOW_ACTION_READ);
91      acl.getEntries().add(ACE_ROLE_STUDENT_DISALLOW_ACTION_WRITE);
92  
93      JSONObject privilegesByRole = AccessInformationUtil.serializePrivilegesByRole(acl);
94      assertTrue(privilegesByRole.getJSONObject(ROLE_ADMIN).getBoolean(ACTION_READ));
95      assertTrue(privilegesByRole.getJSONObject(ROLE_ADMIN).getBoolean(ACTION_WRITE));
96      assertTrue(privilegesByRole.getJSONObject(ROLE_STUDENT).getBoolean(ACTION_READ));
97      assertFalse(privilegesByRole.getJSONObject(ROLE_STUDENT).getBoolean(ACTION_WRITE));
98    }
99  
100   /**
101    * Test method for {@link AccessInformationUtil#serializePrivilegesByRole(AccessControlList)}
102    */
103   @Test(expected = IllegalArgumentException.class)
104   public void testSerializePrivilegesByRoleWithNull() throws Exception {
105     AccessInformationUtil.serializePrivilegesByRole(null);
106   }
107 
108 }
