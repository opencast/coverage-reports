155   public Track makeTrack(URI uri, String logicalName) throws JAXBException, IOException, SAXException {
156     String xml = "<oc:track xmlns:oc=\"http://mediapackage.opencastproject.org\" type=\"presentation/source\"><oc:tags/><oc:url>"
157             + uri + "</oc:url><oc:duration>1</oc:duration><oc:logicalname>" + logicalName
158             + "</oc:logicalname></oc:track>";
159     InputStream inputStream = new ByteArrayInputStream(xml.getBytes(StandardCharsets.UTF_8));
160     TrackImpl track;
161     try {
162       track = unmarshaller.unmarshal(XmlSafeParser.parse(inputStream), TrackImpl.class).getValue();
163     } finally {
164       IoSupport.closeQuietly(inputStream);
165     }
166     return track;
167   }

169   protected HashMap<File, File> hlsmoveFiles(List<File> files) throws Exception, IOException {
170     // rename all the segments to match encoding profile
171     HashMap<File, File> outputs = new HashMap<File, File>();
172     for (File file : files) {
173       File destFile = File.createTempFile(FilenameUtils.getBaseName(file.getName()),
174               "." + FilenameUtils.getExtension(file.getName()),
175               file.getAbsoluteFile().getParentFile());
176       FileUtils.copyFile(file, destFile);
177       outputs.put(file, destFile);
178     }
179     return outputs;
180   }

226   @Test
227   public void testFileReferences() throws Exception {
228     Set<String> variants = AdaptivePlaylist.getVariants(master2);
229     assertEquals(variants.size(), 2);
230     Set<String> segments = AdaptivePlaylist.getVariants(variantFile);
231     assertEquals(segments.size(), 1);
232     Set<String> allsegments = AdaptivePlaylist.getReferencedFiles(master2, true);
233     assertEquals(allsegments.size(), 2);
234     Set<String> allFiles = AdaptivePlaylist.getReferencedFiles(master2, false);
235     assertEquals(allFiles.size(), 4);
236     List<File> files = new ArrayList<File>();
237     files.add(master2);
238     files.add(variantFile);
239     files.add(variantFile1);
240     files.add(srcFile);
241     files.add(srcFile1);
242     HashMap<File, File> movedFiles = hlsmoveFiles(files);
243     HashMap<String, String> nameMap = new HashMap<String, String>();
244     movedFiles.forEach((k, v) -> {
245       nameMap.put(k.getName(), v.getName());
246     });
247     for (File f : movedFiles.values()) {
248       if (AdaptivePlaylist.isPlaylist(f)) {
249         File destFile = File.createTempFile(f.getName(), FilenameUtils.getExtension(f.getName()),
250                 f.getAbsoluteFile().getParentFile());
251         Set<String> oldref = AdaptivePlaylist.getVariants(f);
252         AdaptivePlaylist.hlsRewriteFileReference(f, destFile, nameMap);
253         variants = AdaptivePlaylist.getVariants(destFile);
254         assertTrue(variants.size() > 0);
255         // the new name is used in the playlist
256         assertTrue(variants.contains(nameMap.get(oldref.toArray()[0])));
257       }
258     }
259     for (File f : movedFiles.values()) {
260       FileUtil.delete(f);
261     }
262   }
