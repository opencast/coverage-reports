53    /** The user and role directory */

66    @Before
67    public void setUp() throws Exception {
68      org = new DefaultOrganization();
69      userName = "sampleUser";
70  
71      role1 = new JaxbRole("role1", org);
72      role2 = new JaxbRole("role2", org);
73      role3 = new JaxbRole("role3", org);
74  
75      JaxbUser user1 = new JaxbUser(userName, "matterhorn", org, role1, role2);
76      user1.setManageable(true);
77      User user2 = new JaxbUser(userName, "secret", "test", org, role2, role3);
78      User user3 = new JaxbUser("userSample", "test", org, role2, role3);
79  
80      List<User> users = new ArrayList<User>();
81      users.add(user1);
82  
83      UserProvider provider1 = EasyMock.createNiceMock(UserProvider.class);
84      EasyMock.expect(provider1.getOrganization()).andReturn(org.getId()).anyTimes();
85      EasyMock.expect(provider1.loadUser((String) EasyMock.anyObject())).andReturn(user1).anyTimes();
86      EasyMock.expect(provider1.findUsers("%mple%", 0, 0)).andReturn(users.iterator()).once();
87      EasyMock.expect(provider1.findUsers("%mple%", 0, 0)).andReturn(users.iterator()).once();
88      EasyMock.expect(provider1.getUsers()).andReturn(users.iterator()).once();
89      EasyMock.expect(provider1.getName()).andReturn("test").once();
90  
91      List<User> users2 = new ArrayList<User>();
92      users2.add(user3);
93  
94      UserProvider provider2 = EasyMock.createNiceMock(UserProvider.class);
95      EasyMock.expect(provider2.getOrganization()).andReturn(org.getId()).anyTimes();
96      EasyMock.expect(provider2.loadUser((String) EasyMock.anyObject())).andReturn(user2).anyTimes();
97      EasyMock.expect(provider2.findUsers("%mple%", 0, 0)).andReturn(users2.iterator()).once();
98      EasyMock.expect(provider2.findUsers("%mple%", 0, 0)).andReturn(users2.iterator()).once();
99      EasyMock.expect(provider2.getUsers()).andReturn(users2.iterator()).once();
100     EasyMock.expect(provider2.getName()).andReturn("matterhorn").once();
101 
102     List<Role> roles1 = new ArrayList<Role>();
103     roles1.add(new JaxbRole("ROLE_ASTRO_2011", org));
104     roles1.add(new JaxbRole("ROLE_ASTRO_2012", org));
105 
106     List<Role> rolesForUser1 = new ArrayList<Role>();
107     rolesForUser1.add(new JaxbRole("ROLE_ASTRO_2012", org));
108 
109     List<Role> findRoles1 = new ArrayList<Role>();
110     findRoles1.add(new JaxbRole("ROLE_ASTRO_2012", org));
111 
112     RoleProvider roleProvider1 = EasyMock.createNiceMock(RoleProvider.class);
113     EasyMock.expect(roleProvider1.getOrganization()).andReturn(org.getId()).anyTimes();
114     EasyMock.expect(roleProvider1.getRolesForUser((String) EasyMock.anyObject())).andReturn(rolesForUser1).anyTimes();
115     EasyMock.expect(roleProvider1.findRoles("%", Role.Target.ALL, 0, 0)).andReturn(roles1.iterator()).anyTimes();
116     EasyMock.expect(roleProvider1.findRoles("%2012%", Role.Target.ALL, 0, 0)).andReturn(findRoles1.iterator()).once();
117     EasyMock.expect(roleProvider1.findRoles("%2012%", Role.Target.ALL, 0, 0)).andReturn(findRoles1.iterator()).once();
118 
119     List<Role> roles2 = new ArrayList<Role>();
120     roles2.add(new JaxbRole("ROLE_MATH_2011", org));
121     roles2.add(new JaxbRole("ROLE_MATH_2012", org));
122 
123     List<Role> rolesForUser2 = new ArrayList<Role>();
124     rolesForUser2.add(new JaxbRole("ROLE_MATH_2012", org));
125 
126     List<Role> findRoles2 = new ArrayList<Role>();
127     findRoles2.add(new JaxbRole("ROLE_MATH_2012", org));
128 
129     RoleProvider roleProvider2 = EasyMock.createNiceMock(RoleProvider.class);
130     EasyMock.expect(roleProvider2.getOrganization()).andReturn(org.getId()).anyTimes();
131     EasyMock.expect(roleProvider2.getRolesForUser((String) EasyMock.anyObject())).andReturn(rolesForUser2).anyTimes();
132     EasyMock.expect(roleProvider2.findRoles("%", Role.Target.ALL, 0, 0)).andReturn(roles2.iterator()).anyTimes();
133     EasyMock.expect(roleProvider2.findRoles("%2012%", Role.Target.ALL, 0, 0)).andReturn(findRoles2.iterator()).once();
134     EasyMock.expect(roleProvider2.findRoles("%2012%", Role.Target.ALL, 0, 0)).andReturn(findRoles2.iterator()).once();
135 
136     RoleProvider otherOrgRoleProvider = EasyMock.createNiceMock(RoleProvider.class);
137     EasyMock.expect(otherOrgRoleProvider.getOrganization()).andReturn("otherOrg").anyTimes();
138     EasyMock.expect(otherOrgRoleProvider.getRolesForUser((String) EasyMock.anyObject())).andReturn(rolesForUser2)
139             .anyTimes();
140 
141     EasyMock.expect(otherOrgRoleProvider.findRoles("%", Role.Target.ALL, 0, 0))
142         .andReturn(roles2.iterator())
143         .anyTimes();
144     EasyMock.expect(otherOrgRoleProvider.findRoles("%2012%", Role.Target.ALL, 0, 0))
145         .andReturn(new ArrayList<Role>().iterator())
146         .anyTimes();
147 
148     SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
149     EasyMock.expect(securityService.getOrganization()).andReturn(org).anyTimes();
150 
151     EasyMock.replay(provider1, provider2, roleProvider1, roleProvider2, otherOrgRoleProvider, securityService);
152 
153     directory = new UserAndRoleDirectoryServiceImpl();
154     directory.activate(null);
155     directory.setSecurityService(securityService);
156     directory.addUserProvider(provider1);
157     directory.addUserProvider(provider2);
158     directory.addRoleProvider(roleProvider1);
159     directory.addRoleProvider(roleProvider2);
160     directory.addRoleProvider(otherOrgRoleProvider);
161   }

163   @Test
164   public void testUserMerge() throws Exception {
165     User mergedUser = directory.loadUser(userName);
166     Set<Role> roles = mergedUser.getRoles();
167     assertTrue(roles.contains(role1));
168     assertTrue(roles.contains(role2));
169     assertTrue(roles.contains(role3));
170     assertNotNull(mergedUser.getPassword());
171     assertEquals(org.getId(), mergedUser.getOrganization().getId());
172     assertEquals(userName, mergedUser.getUsername());
173     assertEquals("matterhorn,test", mergedUser.getProvider());
174     assertTrue(mergedUser.isManageable());
175     assertTrue(((JaxbUser) mergedUser).isManageable());
176   }

192   @Test
193   @SuppressWarnings("unchecked")
194   public void testFindUsers() {
195     List<User> users = IteratorUtils.toList(directory.findUsers("%mple%", 0, 0));
196     Assert.assertEquals(2, users.size());
197     Assert.assertTrue(userName.equals(users.get(0).getUsername()) || userName.equals(users.get(1).getUsername()));
198     Assert.assertTrue("userSample".equals(users.get(0).getUsername())
199             || "userSample".equals(users.get(1).getUsername()));
200 
201     // Test limit and offset
202     users = IteratorUtils.toList(directory.findUsers("%mple%", 1, 1));
203     Assert.assertEquals(1, users.size());
204   }

206   @Test
207   @SuppressWarnings("unchecked")
208   public void testFindRoles() {
209     List<Role> roles = directory.findRoles("%2012%", Role.Target.ALL, 0, 0);
210     Assert.assertEquals(2, roles.size());
211     Assert.assertTrue("ROLE_MATH_2012".equals(roles.get(0).getName())
212             || "ROLE_MATH_2012".equals(roles.get(1).getName()));
213     Assert.assertTrue("ROLE_ASTRO_2012".equals(roles.get(0).getName())
214             || "ROLE_ASTRO_2012".equals(roles.get(1).getName()));
215 
216     // Test limit and offset
217     roles = directory.findRoles("%2012%", Role.Target.ALL, 1, 1);
218     Assert.assertEquals(1, roles.size());
219     Assert.assertTrue("ROLE_ASTRO_2012".equals(roles.get(0).getName())
220             || "ROLE_MATH_2012".equals(roles.get(0).getName()));
221   }
