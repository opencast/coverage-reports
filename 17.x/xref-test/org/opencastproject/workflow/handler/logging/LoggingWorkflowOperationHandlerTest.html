22  package org.opencastproject.workflow.handler.logging;
23  
24  import org.opencastproject.mediapackage.MediaPackage;
25  import org.opencastproject.mediapackage.MediaPackageBuilder;
26  import org.opencastproject.mediapackage.MediaPackageBuilderImpl;
27  import org.opencastproject.workflow.api.WorkflowInstance;
28  import org.opencastproject.workflow.api.WorkflowOperationInstance;
29  
30  import org.apache.commons.io.FileUtils;
31  import org.apache.commons.io.filefilter.TrueFileFilter;
32  import org.easymock.EasyMock;
33  import org.junit.Assert;
34  import org.junit.Before;
35  import org.junit.Rule;
36  import org.junit.Test;
37  import org.junit.rules.TemporaryFolder;
38  
39  import java.io.File;
40  import java.util.Collections;
41  import java.util.List;
42  
43  public class LoggingWorkflowOperationHandlerTest {
44  
45    private LoggingWorkflowOperationHandler operation = new LoggingWorkflowOperationHandler();
46    private WorkflowInstance workflow;
47    private WorkflowOperationInstance instance;
48  
49    @Rule
50    public TemporaryFolder testFolder = new TemporaryFolder();
51  
52    @Before
53    public void setUp() throws Exception {
54      MediaPackageBuilder builder = new MediaPackageBuilderImpl();
55      MediaPackage mediaPackage = builder.createNew();
56  
57      instance = EasyMock.createNiceMock(WorkflowOperationInstance.class);
58      EasyMock.expect(instance.getId()).andReturn(2L).anyTimes();
59  
60      workflow = EasyMock.createNiceMock(WorkflowInstance.class);
61      EasyMock.expect(workflow.getMediaPackage()).andReturn(mediaPackage).anyTimes();
62      EasyMock.expect(workflow.getId()).andReturn(1L).anyTimes();
63      EasyMock.expect(workflow.getCurrentOperation()).andReturn(instance).anyTimes();
64      EasyMock.expect(workflow.getOperations()).andReturn(List.of(instance)).anyTimes();
65      EasyMock.expect(workflow.getConfigurations()).andReturn(Collections.emptyMap()).anyTimes();
66    }
67  
68    @Test
69    public void testSimpleLog() throws Exception {
70      EasyMock.replay(workflow, instance);
71      operation.start(workflow, null);
72    }
73  
74    @Test
75    public void testLogToFile() throws Exception {
76      File tempFolder = testFolder.newFolder("logging.test");
77      EasyMock.expect(instance.getConfiguration("directory")).andReturn(tempFolder.getAbsolutePath()).anyTimes();
78      EasyMock.expect(instance.getConfiguration("mediapackage-xml")).andReturn("true").anyTimes();
79      EasyMock.expect(instance.getConfiguration("workflowinstance-xml")).andReturn("true").anyTimes();
80      EasyMock.replay(workflow, instance);
81      operation.start(workflow, null);
82      Assert.assertEquals(3, FileUtils.listFilesAndDirs(tempFolder, TrueFileFilter.TRUE, TrueFileFilter.TRUE).size());
83    }
84  
85  }
