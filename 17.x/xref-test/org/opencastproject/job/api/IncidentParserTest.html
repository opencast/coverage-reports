22  package org.opencastproject.job.api;
23  
24  import static org.junit.Assert.assertEquals;
25  import static org.junit.Assert.assertThat;
26  import static org.opencastproject.util.IoSupport.loadFileFromClassPathAsString;
27  import static org.opencastproject.util.data.Tuple.tuple;
28  import static org.xmlmatchers.XmlMatchers.similarTo;
29  import static org.xmlmatchers.transform.XmlConverters.the;
30  
31  import org.opencastproject.job.api.Incident.Severity;
32  
33  import org.apache.commons.io.IOUtils;
34  import org.junit.Test;
35  import org.slf4j.Logger;
36  import org.slf4j.LoggerFactory;
37  
38  import java.util.Arrays;
39  import java.util.Collections;
40  import java.util.Date;
41  
42  public class IncidentParserTest {
43  
44    private static final Logger logger = LoggerFactory.getLogger(IncidentParserTest.class);
45  
46    @Test
47    public void testSerializationOfJaxbIncident() throws Exception {
48      final Incident incident = new IncidentImpl(1, 2, "service", "localhost", new Date(0), Severity.FAILURE, "code",
49              Arrays.asList(tuple("title", "content"), tuple("Another title", "...and even more content")),
50              Collections.singletonMap("key", "value"));
51      final String marshaled = IncidentParser.I.toXml(new JaxbIncident(incident));
52      logger.info(marshaled);
53      assertThat(the(marshaled),
54              similarTo(the(loadFileFromClassPathAsString("/org/opencastproject/job/api/expected-incident-1.xml").get())));
55      final Incident unmarshaled = IncidentParser.I.parseIncidentFromXml(IOUtils.toInputStream(marshaled)).toIncident();
56      assertEquals(incident, unmarshaled);
57    }
58  
59    @Test
60    public void testSerializationOfJaxbIncidentTree() throws Exception {
61      final IncidentTree tree = new IncidentTreeImpl(
62              Arrays.asList(incident(1), incident(2)),
63              Collections.singletonList(new IncidentTreeImpl(
64                      Collections.singletonList(incident(3)),
65                      Collections.singletonList(new IncidentTreeImpl(
66                              Arrays.asList(incident(4), incident(5)),
67                              Collections.emptyList())))));
68      final String marshaled = IncidentParser.I.toXml(new JaxbIncidentTree(tree));
69      logger.info(marshaled);
70      assertThat(the(marshaled),
71              similarTo(the(loadFileFromClassPathAsString("/org/opencastproject/job/api/expected-incident-tree-1.xml")
72                      .get())));
73      final IncidentTree unmarshaled = IncidentParser.I.parseIncidentTreeFromXml(IOUtils.toInputStream(marshaled))
74              .toIncidentTree();
75      assertEquals(tree, unmarshaled);
76    }
77  
78    public Incident incident(long id) {
79      return new IncidentImpl(id, 2, "service", "localhost", new Date(id), Severity.FAILURE, "code",
80              Collections.singletonList(tuple("title", "content")), Collections.singletonMap("key", "value"));
81    }
82  }
