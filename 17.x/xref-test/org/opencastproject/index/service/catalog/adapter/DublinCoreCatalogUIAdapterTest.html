373   @Test
374   public void testGetDuration() {
375     int startHour = 19;
376     int startMinute = 35;
377     int startSecond = 19;
378     int endHour = 20;
379     int endMinute = 48;
380     int endSecond = 23;
381 
382     // Test a none() period
383     Opt<DCMIPeriod> emptyPeriod = Opt.none();
384     Long result = DublinCoreMetadataUtil.getDuration(emptyPeriod);
385     assertEquals(new Long(0L), result);
386 
387     DateTime periodStart = new DateTime(2014, 11, 4, startHour, startMinute, startSecond, DateTimeZone.UTC);
388     DateTime periodEnd = new DateTime(2014, 11, 4, endHour, endMinute, endSecond, DateTimeZone.UTC);
389 
390     // Test a period that is missing the start date
391     DCMIPeriod withoutStartPeriod = new DCMIPeriod(null, periodEnd.toDate());
392     result = DublinCoreMetadataUtil.getDuration(Opt.some(withoutStartPeriod));
393     assertEquals(new Long(0L), result);
394 
395     // Test a period with a start but no end.
396     DCMIPeriod withoutEndPeriod = new DCMIPeriod(periodStart.toDate(), null);
397     result = DublinCoreMetadataUtil.getDuration(Opt.some(withoutEndPeriod));
398     assertEquals(new Long(0L), result);
399 
400     // Test a period with both a start and an end.
401     DCMIPeriod standardPeriod = new DCMIPeriod(periodStart.toDate(), periodEnd.toDate());
402     result = DublinCoreMetadataUtil.getDuration(Opt.some(standardPeriod));
403     assertEquals(new Long(4384000L), result);
404   }
