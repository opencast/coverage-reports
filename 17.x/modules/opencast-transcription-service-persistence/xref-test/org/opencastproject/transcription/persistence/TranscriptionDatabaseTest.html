72    @Test
73    public void testStoreJobControl() throws Exception {
74      long dt1 = System.currentTimeMillis();
75      database.storeJobControl(MP_ID, TRACK_ID, JOB_ID, STATUS, TRACK_DURATION, DATE_EXPECTED, PROVIDER);
76      long dt2 = System.currentTimeMillis();
77  
78      TranscriptionJobControl j = database.findByJob(JOB_ID);
79      Assert.assertNotNull(j);
80      Assert.assertEquals(MP_ID, j.getMediaPackageId());
81      Assert.assertEquals(TRACK_ID, j.getTrackId());
82      Assert.assertEquals(STATUS, j.getStatus());
83      Assert.assertEquals(TRACK_DURATION, j.getTrackDuration());
84      long created = j.getDateCreated().getTime();
85      Assert.assertTrue(dt1 <= created && created <= dt2);
86    }

190   @Test
191   public void testUpdateJobControlToTranscriptionComplete() throws Exception {
192     database.storeJobControl(MP_ID, TRACK_ID, JOB_ID, STATUS, TRACK_DURATION, DATE_EXPECTED, PROVIDER);
193 
194     TranscriptionJobControl j = database.findByJob(JOB_ID);
195     Assert.assertNotNull(j);
196     Assert.assertEquals(MP_ID, j.getMediaPackageId());
197     Assert.assertEquals(TRACK_ID, j.getTrackId());
198     Assert.assertEquals(STATUS, j.getStatus());
199     Assert.assertNull(j.getDateCompleted());
200 
201     long dt1 = System.currentTimeMillis();
202     database.updateJobControl(JOB_ID, TranscriptionJobControl.Status.TranscriptionComplete.name());
203     long dt2 = System.currentTimeMillis();
204 
205     j = database.findByJob(JOB_ID);
206     Assert.assertNotNull(j);
207     Assert.assertEquals(MP_ID, j.getMediaPackageId());
208     Assert.assertEquals(TRACK_ID, j.getTrackId());
209     Assert.assertEquals(TranscriptionJobControl.Status.TranscriptionComplete.name(), j.getStatus());
210     long completed = j.getDateCompleted().getTime();
211     Assert.assertTrue(dt1 <= completed && completed <= dt2);
212   }
