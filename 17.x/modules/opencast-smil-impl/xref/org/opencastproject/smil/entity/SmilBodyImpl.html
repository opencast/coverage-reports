22  package org.opencastproject.smil.entity;
23  
24  import org.opencastproject.smil.api.SmilException;
25  import org.opencastproject.smil.entity.api.SmilBody;
26  import org.opencastproject.smil.entity.api.SmilObject;
27  import org.opencastproject.smil.entity.media.SmilMediaObjectImpl;
28  import org.opencastproject.smil.entity.media.api.SmilMediaObject;
29  import org.opencastproject.smil.entity.media.container.SmilMediaContainerImpl;
30  import org.opencastproject.smil.entity.media.container.api.SmilMediaContainer;
31  
32  import java.util.Collections;
33  import java.util.LinkedList;
34  import java.util.List;
35  
36  import javax.xml.bind.annotation.XmlElementRef;
37  import javax.xml.bind.annotation.XmlRootElement;
38  
39  /**
40   * {@link SmilBody} implementation.
41   */
42  @XmlRootElement(name = "body")
43  public class SmilBodyImpl extends SmilObjectImpl implements SmilBody {
44  
45    /**
46     * Media Elements
47     */
48    private List<SmilMediaObject> mediaElements = new LinkedList<SmilMediaObject>();
49  
50    /**
51     * {@inheritDoc}
52     */
53    @Override
54    public List<SmilMediaObject> getMediaElements() {
55      return Collections.unmodifiableList(mediaElements);
56    }
57  
58    /**
59     * Returns SMIL media elements.
60     *
61     * @return the media elements
62     */
63    @XmlElementRef(type = SmilMediaObjectImpl.class)
64    protected List<SmilMediaObject> getMediaObjects() {
65      return mediaElements;
66    }
67  
68    /**
69     * Set SMIL media elements.
70     *
71     * @param mediaElements the mediae lements to set
72     */
73    protected void setMediaObjects(List<SmilMediaObject> mediaElements) {
74      this.mediaElements = mediaElements;
75    }
76  
77    /**
78     * {@inheritDoc}
79     */
80    @Override
81    protected String getIdPrefix() {
82      return "b";
83    }
84  
85    /**
86     * Add given {@link SmilMediaObject} to the element with given Id.
87     *
88     * @param mediaObject to add
89     * @param parentId where to add new media
90     * @throws SmilException if there is no element inside with given Id
91     */
92    public void addMediaElement(SmilMediaObject mediaObject, String parentId) throws SmilException {
93      if (getId().equals(parentId)) {
94        mediaElements.add(mediaObject);
95      } else {
96        for (SmilMediaObject element : mediaElements) {
97          if (element.isContainer() && (element.getId().equals(parentId)
98                  || ((SmilMediaContainer) element).isParentOf(parentId))) {
99            ((SmilMediaContainerImpl) element).addMediaObject(mediaObject, parentId);
100           return;
101         }
102       }
103       throw new SmilException("There is no element with id " + parentId);
104     }
105   }
106 
107   /**
108    * {@inheritDoc}
109    */
110   @Override
111   public SmilObject removeElement(String elementId) {
112     SmilMediaObject child = null;
113     for (SmilMediaObject element : mediaElements) {
114       if (element.getId().equals(elementId)) {
115         child = element;
116         break;
117       } else {
118         SmilObject removed = ((SmilObjectImpl) element).removeElement(elementId);
119         if (removed != null) {
120           return removed;
121         }
122       }
123     }
124     if (child != null) {
125       mediaElements.remove(child);
126     }
127     return child;
128   }
129 
130   /**
131    * Remove all media Elements inside.
132    */
133   public void clear() {
134     mediaElements.clear();
135   }
136 
137   /**
138    * {@inheritDoc}
139    */
140   @Override
141   public SmilObject getElementOrNull(String elementId) {
142     if (getId().equals(elementId)) {
143       return this;
144     }
145     for (SmilMediaObject media : getMediaElements()) {
146       SmilObject element = ((SmilMediaObjectImpl) media).getElementOrNull(elementId);
147       if (element != null) {
148         return element;
149       }
150     }
151     return null;
152   }
153 
154   /**
155    * {@inheritDoc}
156    */
157   @Override
158   public void putAllChilds(List<SmilObject> elements) {
159     for (SmilObject child : getMediaElements()) {
160       elements.add(child);
161       ((SmilObjectImpl) child).putAllChilds(elements);
162     }
163   }
164 }
