21  package org.opencastproject.security.urlsigning;
22  
23  import static org.junit.Assert.assertEquals;
24  
25  import org.junit.Test;
26  
27  import java.net.URI;
28  import java.net.URISyntaxException;
29  
30  public class WowzaResourceStrategyImplTest {
31    @Test
32    public void testGetRTMPSource() throws URISyntaxException {
33      /**
34       * With the explicit Wowza delimiter.
35       */
36      // With explicit seperation between application and stream.
37      URI noPathDelimiter = new URI("rtmp://hostname.com/matterhorn-engage/_definst_/mp4:myvideo.mov");
38      String result = WowzaResourceStrategyImpl.getRTMPResource(noPathDelimiter);
39      assertEquals("mp4:myvideo.mov", result);
40  
41      // With path, delimiter, no extension
42      URI pathDelimiterNoExtension = new URI(
43          "rtmp://hostname.com/matterhorn-engage/_definst_/mp4:path/to/video/myvideo");
44      result = WowzaResourceStrategyImpl.getRTMPResource(pathDelimiterNoExtension);
45      assertEquals("mp4:path/to/video/myvideo", result);
46  
47      // With path, delimiter, extension
48      URI pathDelimiterExtension = new URI(
49          "rtmp://hostname.com/matterhorn-engage/_definst_/mp4:path/to/video/myvideo.mov");
50      result = WowzaResourceStrategyImpl.getRTMPResource(pathDelimiterExtension);
51      assertEquals("mp4:path/to/video/myvideo.mov", result);
52  
53      // With path, delimiter, extension and query string
54      URI pathDelimiterExtensionQueryString = new URI(
55          "rtmp://hostname.com/matterhorn-engage/_definst_/mp4:path/to/video/myvideo.mov?param1=value1&param2=value2");
56      result = WowzaResourceStrategyImpl.getRTMPResource(pathDelimiterExtensionQueryString);
57      assertEquals("mp4:path/to/video/myvideo.mov?param1=value1&param2=value2", result);
58  
59      /**
60       * With the format mp4: and no Wowza delimiter.
61       */
62      // No Path, extension and format.
63      URI noPathNoDelimiterFormat = new URI("rtmp://hostname.com/matterhorn-engage/mp4:myvideo.mov");
64      result = WowzaResourceStrategyImpl.getRTMPResource(noPathNoDelimiterFormat);
65      assertEquals("mp4:myvideo.mov", result);
66  
67      // With path, no delimiter, no extension and format
68      URI noPathNoDelimiterNoExtensionFormat = new URI(
69          "rtmp://hostname.com/matterhorn-engage/mp4:path/to/video/myvideo");
70      result = WowzaResourceStrategyImpl.getRTMPResource(noPathNoDelimiterNoExtensionFormat);
71      assertEquals("mp4:path/to/video/myvideo", result);
72  
73      // With path, no delimiter, extension and format
74      URI pathNoDelimiterExtensionFormat = new URI(
75          "rtmp://hostname.com/matterhorn-engage/mp4:path/to/video/myvideo.mov");
76      result = WowzaResourceStrategyImpl.getRTMPResource(pathNoDelimiterExtensionFormat);
77      assertEquals("mp4:path/to/video/myvideo.mov", result);
78  
79      // With path, no delimiter, extension, format and query string
80      URI pathNoDelimiterExtensionFormatQueryString = new URI(
81          "rtmp://hostname.com/matterhorn-engage/mp4:path/to/video/myvideo.mov?param1=value1&param2=value2");
82      result = WowzaResourceStrategyImpl.getRTMPResource(pathNoDelimiterExtensionFormatQueryString);
83      assertEquals("mp4:path/to/video/myvideo.mov?param1=value1&param2=value2", result);
84  
85      // FLV with extension
86      URI flvWithExtension = new URI(
87          "rtmp://hostname.com/matterhorn-engage/flv:path/to/video/myvideo.flv?param1=value1&param2=value2");
88      result = WowzaResourceStrategyImpl.getRTMPResource(flvWithExtension);
89      assertEquals("flv:path/to/video/myvideo.flv?param1=value1&param2=value2", result);
90  
91      // FLV without extension
92      URI flvWithoutExtension = new URI(
93          "rtmp://hostname.com/matterhorn-engage/flv:path/to/video/myvideo?param1=value1&param2=value2");
94      result = WowzaResourceStrategyImpl.getRTMPResource(flvWithoutExtension);
95      assertEquals("flv:path/to/video/myvideo?param1=value1&param2=value2", result);
96  
97      // Without format
98      URI withoutFormat = new URI(
99          "rtmp://hostname.com/matterhorn-engage/path/to/video/myvideo.mp4?param1=value1&param2=value2");
100     result = WowzaResourceStrategyImpl.getRTMPResource(withoutFormat);
101     assertEquals("path/to/video/myvideo.mp4?param1=value1&param2=value2", result);
102   }
103 }
