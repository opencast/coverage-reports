55  /**
56   * Take look in specified catalog for specified term, if the value matches the specified value add the target-tags
57   */
58  @Component(
59      immediate = true,
60      service = WorkflowOperationHandler.class,
61      property = {
62          "service.description=Tag by Dublin Core Term Workflow Handler",
63          "workflow.operation=tag-by-dcterm"
64      }
65  )
66  public class TagByDublinCoreTermWOH extends ResumableWorkflowOperationHandlerBase {
67  
68    private static final Logger logger = LoggerFactory.getLogger(TagByDublinCoreTermWOH.class);
69    private static final String PLUS = "+";
70    private static final String MINUS = "-";
71  
72    /** Name of the configuration option that provides the source flavors we are looking for */
73    public static final String SOURCE_FLAVORS_PROPERTY = "source-flavors";
74  
75    /** Name of the configuration option that provides the target tags we are looking for */
76    public static final String TARGET_TAGS_PROPERTY = "target-tags";
77  
78    /** Name of the configuration option that provides the catalog to examine */
79    public static final String DCCATALOG_PROPERTY = "dccatalog";
80  
81    /** Name of the configuration option that provides Dublin Core term/element */
82    public static final String DCTERM_PROPERTY = "dcterm";
83  
84    /** Name of the configuration option that provides term's default value if not present */
85    public static final String DEFAULT_VALUE_PROPERTY = "default-value";
86  
87    /** Name of the configuration option that provides value to match */
88    public static final String MATCH_VALUE_PROPERTY = "match-value";
89  
90    /** Name of the configuration option that provides the copy boolean we are looking for */
91    public static final String COPY_PROPERTY = "copy";
92  
93    /** The local workspace */
94    private Workspace workspace = null;
95  
96    /**
97     * Callback for declarative services configuration that will introduce us to the local workspace service.
98     * Implementation assumes that the reference is configured as being static.
99     *
100    * @param workspace
101    *          an instance of the workspace
102    */
103   @Reference
104   public void setWorkspace(Workspace workspace) {
105     this.workspace = workspace;
106   }
107 
108   /**
109    * {@inheritDoc}
110    *
111    * @see org.opencastproject.workflow.api.WorkflowOperationHandler#start(org.opencastproject.workflow.api.WorkflowInstance,
112    *      JobContext)
113    */
114   @Override
115   public WorkflowOperationResult start(WorkflowInstance workflowInstance, JobContext context)
116           throws WorkflowOperationException {
117 
118     MediaPackage mediaPackage = workflowInstance.getMediaPackage();
119     WorkflowOperationInstance currentOperation = workflowInstance.getCurrentOperation();
120 
121     ConfiguredTagsAndFlavors tagsAndFlavors = getTagsAndFlavors(workflowInstance,
122         Configuration.many, Configuration.many, Configuration.many, Configuration.many);
123     List<MediaPackageElementFlavor> configuredSourceFlavors = tagsAndFlavors.getSrcFlavors();
124     List<String> configuredSourceTags = tagsAndFlavors.getSrcTags();
125     String configuredCatalog = StringUtils.trimToEmpty(currentOperation.getConfiguration(DCCATALOG_PROPERTY));
126     String configuredDCTerm = StringUtils.trimToEmpty(currentOperation.getConfiguration(DCTERM_PROPERTY));
127     String configuredDefaultValue = StringUtils.trimToNull(currentOperation.getConfiguration(DEFAULT_VALUE_PROPERTY));
128     String configuredMatchValue = StringUtils.trimToEmpty(currentOperation.getConfiguration(MATCH_VALUE_PROPERTY));
129     List<MediaPackageElementFlavor> configuredTargetFlavor = tagsAndFlavors.getTargetFlavors();
130     List<String> configuredTargetTags = tagsAndFlavors.getTargetTags();
131     boolean copy = BooleanUtils.toBoolean(currentOperation.getConfiguration(COPY_PROPERTY));
132 
133     SimpleElementSelector elementSelector = new SimpleElementSelector();
134     for (MediaPackageElementFlavor flavor : configuredSourceFlavors) {
135       elementSelector.addFlavor(flavor);
136     }
137     for (String tag : configuredSourceTags) {
138       elementSelector.addTag(tag);
139     }
140 
141     List<String> removeTags = new ArrayList<>();
142     List<String> addTags = new ArrayList<>();
143     List<String> overrideTags = new ArrayList<>();
144 
145     for (String tag : configuredTargetTags) {
146       if (tag.startsWith(MINUS)) {
147         removeTags.add(tag);
148       } else if (tag.startsWith(PLUS)) {
149         addTags.add(tag);
150       } else {
151         overrideTags.add(tag);
152       }
153     }
154 
155     // Find Catalog
156     Catalog[] catalogs = mediaPackage.getCatalogs(new MediaPackageElementFlavor("dublincore", StringUtils.lowerCase(configuredCatalog)));
157 
158     if (catalogs != null && catalogs.length > 0) {
159       Boolean foundValue = false;
160       EName dcterm = new EName(TERMS_NS_URI, configuredDCTerm);
161 
162       // Find DCTerm
163       for (Catalog catalog : catalogs) {
164         DublinCoreCatalog dc = DublinCoreUtil.loadDublinCore(workspace, catalog);
165         // Match Value
166         List<DublinCoreValue> values = dc.get(dcterm);
167         if (values.isEmpty()) {
168           // Use default
169           if (configuredDefaultValue != null) {
170             foundValue = configuredDefaultValue.equals(configuredMatchValue);
171           }
172         } else {
173           foundValue = values.contains(DublinCoreValue.mk(configuredMatchValue));
174         }
175       }
176 
177       if (foundValue) {
178         if (copy) {
179           logger.info("Retagging mediapackage elements as a copy");
180         } else {
181           logger.info("Retagging mediapackage elements");
182         }
183 
184         Collection<MediaPackageElement> elements = elementSelector.select(mediaPackage, false);
185         for (MediaPackageElement e : elements) {
186           MediaPackageElement element = e;
187 
188           if (copy) {
189             element = (MediaPackageElement) e.clone();
190             element.setIdentifier(null);
191             element.setURI(e.getURI()); // use the same URI as the original
192           }
193           if (!configuredTargetFlavor.isEmpty()) {
194             element.setFlavor(configuredTargetFlavor.get(0));
195           }
196 
197           if (overrideTags.size() > 0) {
198             element.clearTags();
199             for (String tag : overrideTags) {
200               element.addTag(tag);
201             }
202           } else {
203             for (String tag : removeTags) {
204               element.removeTag(tag.substring(MINUS.length()));
205             }
206             for (String tag : addTags) {
207               element.addTag(tag.substring(PLUS.length()));
208             }
209           }
210 
211           if (copy) {
212             mediaPackage.addDerived(element, e);
213           }
214         }
215       } // if foundValue
216     } // if catalogs
217 
218     return createResult(mediaPackage, Action.CONTINUE);
219   }
