51  /**
52   * Workflow operation handler for analyzing tracks and set control variables.
53   */
54  @Component(
55      immediate = true,
56      service = WorkflowOperationHandler.class,
57      property = {
58          "service.description=Analyze Tracks Operation Handler",
59          "workflow.operation=analyze-tracks"
60      }
61  )
62  public class AnalyzeTracksWorkflowOperationHandler extends AbstractWorkflowOperationHandler {
63  
64    /** Configuration key for the "flavor" of the tracks to use as a source input */
65    static final String OPT_SOURCE_FLAVOR = "source-flavor";
66  
67    /** Configuration key for video aspect ratio to check */
68    static final String OPT_VIDEO_ASPECT = "aspect-ratio";
69  
70    /** Configuration key to define behavior if no track matches */
71    static final String OPT_FAIL_NO_TRACK = "fail-no-track";
72  
73    /** The logging facility */
74    private static final Logger logger = LoggerFactory
75            .getLogger(AnalyzeTracksWorkflowOperationHandler.class);
76  
77    @Override
78    public WorkflowOperationResult start(WorkflowInstance workflowInstance, JobContext context)
79            throws WorkflowOperationException {
80  
81      logger.info("Running analyze-tracks workflow operation on workflow {}", workflowInstance.getId());
82      final MediaPackage mediaPackage = workflowInstance.getMediaPackage();
83      Map<String, String> properties = new HashMap<>();
84  
85      ConfiguredTagsAndFlavors tagsAndFlavors = getTagsAndFlavors(workflowInstance, Configuration.none, Configuration.one, Configuration.none, Configuration.none);
86      final MediaPackageElementFlavor singleSourceFlavor = tagsAndFlavors.getSingleSrcFlavor();
87      TrackSelector trackSelector = new TrackSelector();
88      trackSelector.addFlavor(singleSourceFlavor);
89      final Collection<Track> tracks = trackSelector.select(mediaPackage, false);
90      if (tracks.size() <= 0) {
91        if (BooleanUtils.toBoolean(getConfig(workflowInstance, OPT_FAIL_NO_TRACK, "false"))) {
92          throw new WorkflowOperationException("No matching tracks for flavor " + singleSourceFlavor.toString());
93        }
94        logger.info("No tracks with specified flavors ({}) to analyse.", singleSourceFlavor.toString());
95        return createResult(mediaPackage, properties, Action.CONTINUE, 0);
96      }
97  
98      List<Fraction> aspectRatios = getAspectRatio(getConfig(workflowInstance, OPT_VIDEO_ASPECT, ""));
99  
100     for (Track track : tracks) {
101       final String varName = toVariableName(track.getFlavor());
102       properties.put(varName + "_media", "true");
103       properties.put(varName + "_video", Boolean.toString(track.hasVideo()));
104       properties.put(varName + "_audio", Boolean.toString(track.hasAudio()));
105       properties.put(varName + "_subtitle", Boolean.toString(track.hasSubtitle()));
106 
107       // Check resolution
108       if (track.hasVideo()) {
109         for (VideoStream video: ((TrackImpl) track).getVideo()) {
110           // Set resolution variables
111           properties.put(varName + "_resolution_x", video.getFrameWidth().toString());
112           properties.put(varName + "_resolution_y", video.getFrameHeight().toString());
113           Fraction trackAspect = Fraction.getReducedFraction(video.getFrameWidth(), video.getFrameHeight());
114           properties.put(varName + "_aspect", trackAspect.toString());
115           properties.put(varName + "_framerate", video.getFrameRate().toString());
116 
117           // Check if we should fall back to nearest defined aspect ratio
118           if (!aspectRatios.isEmpty()) {
119             trackAspect = getNearestAspectRatio(trackAspect, aspectRatios);
120             properties.put(varName + "_aspect_snap", trackAspect.toString());
121           }
122         }
123       }
124     }
125     logger.info("Finished analyze-tracks workflow operation adding the properties: {}", properties);
126     return createResult(mediaPackage, properties, Action.CONTINUE, 0);
127   }
128 
129   /**
130    * Get nearest aspect ratio from list
131    *
132    * @param videoAspect
133    *        Aspect ratio of video to check
134    * @param aspects
135    *        List of aspect ratios to snap to.
136    * @return Nearest aspect ratio
137    */
138   Fraction getNearestAspectRatio(final Fraction videoAspect, final List<Fraction> aspects) {
139     Fraction nearestAspect = aspects.get(0);
140     for (Fraction aspect: aspects) {
141       if (videoAspect.subtract(nearestAspect).abs().compareTo(videoAspect.subtract(aspect).abs()) > 0) {
142         nearestAspect = aspect;
143       }
144     }
145     return nearestAspect;
146   }
147 
148   /**
149    * Get aspect ratios to check from configuration string.
150    *
151    * @param aspectConfig
152    *        Configuration string
153    * @return List of aspect rations to check
154    */
155   List<Fraction> getAspectRatio(String aspectConfig) {
156     List<Fraction> aspectRatios = new ArrayList<>();
157     for (String aspect: aspectConfig.split(" *, *")) {
158       if (StringUtils.isNotBlank(aspect)) {
159         aspectRatios.add(Fraction.getFraction(aspect).reduce());
160       }
161     }
162     return aspectRatios;
163   }
164 
165   /** Create a name for a workflow variable from a flavor */
166   private static String toVariableName(final MediaPackageElementFlavor flavor) {
167     return flavor.getType() + "_" + flavor.getSubtype();
168   }
169 }
