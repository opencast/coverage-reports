194   /**
195    * {@inheritDoc}
196    *
197    * @see org.opencastproject.workflow.api.WorkflowOperationHandler#start(org.opencastproject.workflow.api.WorkflowInstance,
198    *      JobContext)
199    */
200   @Override
201   public WorkflowOperationResult start(final WorkflowInstance workflowInstance, JobContext context)
202           throws WorkflowOperationException {
203     logger.debug("Running series workflow operation");
204 
205     MediaPackage mediaPackage = workflowInstance.getMediaPackage();
206 
207     Opt<String> optSeries = getOptConfig(workflowInstance.getCurrentOperation(), SERIES_PROPERTY);
208     Opt<String> optAttachFlavors = getOptConfig(workflowInstance.getCurrentOperation(), ATTACH_PROPERTY);
209     Boolean applyAcl = getOptConfig(workflowInstance.getCurrentOperation(), APPLY_ACL_PROPERTY).map(toBoolean)
210             .getOr(false);
211     Opt<String> optCopyMetadata = getOptConfig(workflowInstance.getCurrentOperation(), COPY_METADATA_PROPERTY);
212     String defaultNamespace = getOptConfig(workflowInstance.getCurrentOperation(), DEFAULT_NS_PROPERTY)
213             .getOr(DublinCore.TERMS_NS_URI);
214     logger.debug("Using default namespace: '{}'", defaultNamespace);
215 
216     if (optSeries.isSome() && !optSeries.get().equals(mediaPackage.getSeries())) {
217       logger.info("Changing series id from '{}' to '{}'", StringUtils.trimToEmpty(mediaPackage.getSeries()),
218               optSeries.get());
219       mediaPackage.setSeries(optSeries.get());
220     }
221 
222     String seriesId = mediaPackage.getSeries();
223     if (seriesId == null) {
224       logger.info("No series set, skip operation");
225       return createResult(mediaPackage, Action.SKIP);
226     }
227 
228     DublinCoreCatalog series;
229     try {
230       series = seriesService.getSeries(seriesId);
231     } catch (NotFoundException e) {
232       logger.info("No series with the identifier '{}' found, skip operation", seriesId);
233       return createResult(mediaPackage, Action.SKIP);
234     } catch (UnauthorizedException e) {
235       logger.warn("Not authorized to get series with identifier '{}' found, skip operation", seriesId);
236       return createResult(mediaPackage, Action.SKIP);
237     } catch (SeriesException e) {
238       logger.error("Unable to get series with identifier '{}', skip operation:", seriesId, e);
239       throw new WorkflowOperationException(e);
240     }
241 
242     mediaPackage.setSeriesTitle(series.getFirst(DublinCore.PROPERTY_TITLE));
243 
244     // Process extra metadata
245     HashSet<EName> extraMetadata = new HashSet<>();
246     if (optCopyMetadata.isSome()) {
247       for (String strEName : optCopyMetadata.get().split(",+\\s*"))
248         try {
249           if (!strEName.isEmpty()) {
250             extraMetadata.add(EName.fromString(strEName, defaultNamespace));
251           }
252         } catch (IllegalArgumentException iae) {
253           logger.warn("Ignoring incorrect dublincore metadata property: '{}'", strEName);
254         }
255     }
256 
257     // Update the episode catalog
258     for (Catalog episodeCatalog : mediaPackage.getCatalogs(MediaPackageElements.EPISODE)) {
259       DublinCoreCatalog episodeDublinCore = DublinCoreUtil.loadDublinCore(workspace, episodeCatalog);
260       // Make sure the MP catalog has bindings defined
261       episodeDublinCore.addBindings(
262               XmlNamespaceContext.mk(XmlNamespaceBinding.mk(DublinCore.TERMS_NS_PREFIX, DublinCore.TERMS_NS_URI)));
263       episodeDublinCore.addBindings(XmlNamespaceContext
264               .mk(XmlNamespaceBinding.mk(DublinCore.ELEMENTS_1_1_NS_PREFIX, DublinCore.ELEMENTS_1_1_NS_URI)));
265       episodeDublinCore.addBindings(XmlNamespaceContext
266               .mk(XmlNamespaceBinding.mk(DublinCores.OC_PROPERTY_NS_PREFIX, DublinCores.OC_PROPERTY_NS_URI)));
267       episodeDublinCore.set(DublinCore.PROPERTY_IS_PART_OF, seriesId);
268       for (EName property : extraMetadata) {
269         if (!episodeDublinCore.hasValue(property) && series.hasValue(property)) {
270           episodeDublinCore.set(property, series.get(property));
271         }
272       }
273       try (InputStream in = IOUtils.toInputStream(episodeDublinCore.toXmlString(), "UTF-8")) {
274         String filename = FilenameUtils.getName(episodeCatalog.getURI().toString());
275         URI uri = workspace.put(mediaPackage.getIdentifier().toString(), episodeCatalog.getIdentifier(), filename, in);
276         episodeCatalog.setURI(uri);
277         // setting the URI to a new source so the checksum will most like be invalid
278         episodeCatalog.setChecksum(null);
279       } catch (Exception e) {
280         logger.error("Unable to update episode catalog isPartOf field", e);
281         throw new WorkflowOperationException(e);
282       }
283     }
284 
285     // Attach series catalogs
286     if (optAttachFlavors.isSome()) {
287       // Remove existing series catalogs
288       AbstractMediaPackageElementSelector<Catalog> catalogSelector = new CatalogSelector();
289       String[] seriesFlavors = StringUtils.split(optAttachFlavors.get(), ",");
290       for (String flavor : seriesFlavors) {
291         if ("*".equals(flavor)) {
292           catalogSelector.addFlavor("*/*");
293         } else {
294           catalogSelector.addFlavor(flavor);
295         }
296       }
297       for (Catalog c : catalogSelector.select(mediaPackage, false)) {
298         if (MediaPackageElements.SERIES.equals(c.getFlavor()) || "series".equals(c.getFlavor().getSubtype())) {
299           mediaPackage.remove(c);
300         }
301       }
302 
303       List<SeriesCatalogUIAdapter> adapters = getSeriesCatalogUIAdapters();
304       for (String flavorString : seriesFlavors) {
305         MediaPackageElementFlavor flavor;
306         if ("*".equals(flavorString)) {
307           flavor = MediaPackageElementFlavor.parseFlavor("*/*");
308         } else {
309           flavor = MediaPackageElementFlavor.parseFlavor(flavorString);
310         }
311         for (SeriesCatalogUIAdapter a : adapters) {
312           MediaPackageElementFlavor adapterFlavor = MediaPackageElementFlavor.parseFlavor(a.getFlavor().toString());
313           if (flavor.matches(adapterFlavor)) {
314             if (MediaPackageElements.SERIES.eq(a.getFlavor().toString())) {
315               addDublinCoreCatalog(series, MediaPackageElements.SERIES, mediaPackage);
316             } else {
317               try {
318                 Opt<byte[]> seriesElementData = seriesService.getSeriesElementData(seriesId, adapterFlavor.getType());
319                 if (seriesElementData.isSome()) {
320                   DublinCoreCatalog catalog = DublinCores.read(new ByteArrayInputStream(seriesElementData.get()));
321                   addDublinCoreCatalog(catalog, adapterFlavor, mediaPackage);
322                 } else {
323                   logger.warn("No extended series catalog found for flavor '{}' and series '{}', skip adding catalog",
324                           adapterFlavor.getType(), seriesId);
325                 }
326               } catch (SeriesException e) {
327                 logger.error("Unable to load extended series metadata for flavor {}", adapterFlavor.getType());
328                 throw new WorkflowOperationException(e);
329               }
330             }
331           }
332         }
333       }
334     }
335 
336     if (applyAcl) {
337       try {
338         AccessControlList acl = seriesService.getSeriesAccessControl(seriesId);
339         if (acl != null)
340           authorizationService.setAcl(mediaPackage, AclScope.Series, acl);
341       } catch (Exception e) {
342         logger.error("Unable to update series ACL", e);
343         throw new WorkflowOperationException(e);
344       }
345     }
346     return createResult(mediaPackage, Action.CONTINUE);
347   }

349   /**
350    * @param organization
351    *          The organization to filter the results with.
352    * @return A {@link List} of {@link SeriesCatalogUIAdapter} that provide the metadata to the front end.
353    */
354   private List<SeriesCatalogUIAdapter> getSeriesCatalogUIAdapters() {
355     String organization = securityService.getOrganization().getId();
356     return Stream.$(seriesCatalogUIAdapters).filter(seriesOrganizationFilter._2(organization)).toList();
357   }
358 
359   private MediaPackage addDublinCoreCatalog(DublinCoreCatalog catalog, MediaPackageElementFlavor flavor,
360           MediaPackage mediaPackage) throws WorkflowOperationException {
361     try (InputStream in = IOUtils.toInputStream(catalog.toXmlString(), "UTF-8")) {
362       String elementId = UUID.randomUUID().toString();
363       URI catalogUrl = workspace.put(mediaPackage.getIdentifier().toString(), elementId, "dublincore.xml", in);
364       logger.info("Adding catalog with flavor {} to mediapackage {}", flavor, mediaPackage);
365       MediaPackageElement mpe = mediaPackage.add(catalogUrl, MediaPackageElement.Type.Catalog, flavor);
366       mpe.setIdentifier(elementId);
367       mpe.setChecksum(Checksum.create(ChecksumType.DEFAULT_TYPE, workspace.get(catalogUrl)));
368       return mediaPackage;
369     } catch (IOException | NotFoundException e) {
370       throw new WorkflowOperationException(e);
371     }
372   }

374   private static final Fn2<SeriesCatalogUIAdapter, String, Boolean> seriesOrganizationFilter = new Fn2<SeriesCatalogUIAdapter, String, Boolean>() {
375     @Override
376     public Boolean apply(SeriesCatalogUIAdapter catalogUIAdapter, String organization) {
377       return catalogUIAdapter.handlesOrganization(organization);
378     }
379   };
