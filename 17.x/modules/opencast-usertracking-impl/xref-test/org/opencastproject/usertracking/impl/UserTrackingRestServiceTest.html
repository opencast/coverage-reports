36  import org.easymock.EasyMock;
37  import org.junit.Before;
38  import org.osgi.framework.BundleContext;
39  import org.osgi.service.component.ComponentContext;
40  
41  import java.util.Dictionary;
42  
43  import javax.servlet.http.HttpServletRequest;
44  import javax.servlet.http.HttpSession;
45  
46  public class UserTrackingRestServiceTest {
47  
48    public static final String MOCK_SESSION_ID = "mock session id";
49    public static final String REMOTE_IP = "127.0.0.1";
50    public static final String PROXY_IP = "127.0.0.2";
51    public static final String MOCK_USER = "Mock User";
52  
53    private UserTrackingRestService service;
54  
55    @Before
56    public void setUp() throws UserTrackingException {
57      SecurityService security = EasyMock.createMock(SecurityService.class);
58      EasyMock.expect(security.getUser())
59              .andReturn(
60                      new JaxbUser(MOCK_USER, "test", new DefaultOrganization(), new JaxbRole("ROLE_USER",
61                              new DefaultOrganization()))).anyTimes();
62  
63      BundleContext bc = EasyMock.createMock(BundleContext.class);
64      EasyMock.expect(bc.getProperty(OpencastConstants.SERVER_URL_PROPERTY)).andReturn("http://www.example.org:8080")
65              .anyTimes();
66  
67      @SuppressWarnings("rawtypes")
68      Dictionary dict = EasyMock.createMock(Dictionary.class);
69      EasyMock.expect(dict.get(RestConstants.SERVICE_PATH_PROPERTY)).andReturn("/usertracking").anyTimes();
70  
71      ComponentContext context = EasyMock.createMock(ComponentContext.class);
72      EasyMock.expect(context.getBundleContext()).andReturn(bc).anyTimes();
73      EasyMock.expect(context.getProperties()).andReturn(dict).anyTimes();
74  
75      UserActionImpl ua = EasyMock.createMock(UserActionImpl.class);
76      EasyMock.expect(ua.getId()).andReturn(4L).anyTimes();
77  
78      UserTrackingService usertracking = EasyMock.createMock(UserTrackingService.class);
79      EasyMock.expect(usertracking.addUserFootprint(EasyMock.isA(UserAction.class), EasyMock.isA(UserSession.class))).andReturn(ua).anyTimes();
80  
81      EasyMock.replay(security, bc, dict, context, ua, usertracking);
82  
83      service = new UserTrackingRestService();
84      service.setSecurityService(security);
85      service.setService(usertracking);
86      service.activate(context);
87    }
88  
89    private HttpServletRequest getMockHttpSession() {
90      HttpSession session = EasyMock.createMock(HttpSession.class);
91      EasyMock.expect(session.getId()).andReturn(MOCK_SESSION_ID).anyTimes();
92      HttpServletRequest request = EasyMock.createMock(HttpServletRequest.class);
93      EasyMock.expect(request.getSession()).andReturn(session).anyTimes();
94      EasyMock.expect(request.getHeader("X-FORWARDED-FOR")).andReturn(null).anyTimes();
95      EasyMock.expect(request.getRemoteAddr()).andReturn(REMOTE_IP).anyTimes();
96      EasyMock.replay(session, request);
97      return request;
98    }
