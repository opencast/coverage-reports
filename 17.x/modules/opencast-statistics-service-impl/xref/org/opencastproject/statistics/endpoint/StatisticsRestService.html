121   @GET
122   @Produces(MediaType.APPLICATION_JSON)
123   @Path("writeDuration")
124   @RestQuery(
125       name = "writeDuration",
126       description = "Writes a duration",
127       returnDescription = "",
128       responses = {
129           @RestResponse(responseCode = SC_NO_CONTENT, description = "Writing worked")
130       }
131   )
132   public Response writeDuration(
133       @QueryParam("organizationId") final String organizationId,
134       @QueryParam("measurementName") final String measurementName,
135       @QueryParam("retentionPolicy") final String retentionPolicy,
136       @QueryParam("organizationIdResourceName") final String organizationIdResourceName,
137       @QueryParam("fieldName") final String fieldName,
138       @QueryParam("temporalResolution") final String temporalResolution,
139       @QueryParam("duration") final String duration) {
140     try {
141       statisticsService
142               .writeDuration(organizationId, measurementName, retentionPolicy, organizationIdResourceName, fieldName,
143                       TimeUnit.valueOf(temporalResolution), Duration.parse(duration));
144       return Response.ok(Response.Status.NO_CONTENT).build();
145     } catch (Exception e) {
146       logger.error("Could not retrieve providers: {}", e.getMessage());
147       throw new WebApplicationException(Response.Status.INTERNAL_SERVER_ERROR);
148     }
149   }
150 
151   @GET
152   @Produces(MediaType.APPLICATION_JSON)
153   @Path("providers.json")
154   @RestQuery(
155       name = "getAllAsJson",
156       description = "Returns active providers",
157       returnDescription = "Returns the active providers JSON document",
158       responses = {
159           @RestResponse(responseCode = SC_OK, description = "The active providers.")
160       }
161   )
162   public Response getProviders() {
163     try {
164       return Response.ok(providersToJson(statisticsService.getProviders()).toJSONString()).build();
165     } catch (Exception e) {
166       logger.error("Could not retrieve providers: {}", e.getMessage());
167       throw new WebApplicationException(Response.Status.INTERNAL_SERVER_ERROR);
168     }
169   }

171   @GET
172   @Produces(MediaType.APPLICATION_JSON)
173   @Path("timeseries/{providerId:.+}.json")
174   @RestQuery(
175       name = "getTimeSeriesAsJson",
176       description = "Returns the time series data for the given providerId",
177       returnDescription = "Returns the time series data JSON document",
178       pathParameters = {
179           @RestParameter(name = "providerId", isRequired = true, description = "The provider identifier", type = STRING)
180       },
181       restParameters = {
182           @RestParameter(
183               name = "resourceId",
184               description = "The id of the resource to get the data for. E.g. episode id.",
185               isRequired = true,
186               type = STRING
187           ),
188           @RestParameter(
189               name = "from",
190               description = "Start of the time series as ISO 8601 UTC date string",
191               isRequired = true,
192               type = STRING
193           ),
194           @RestParameter(
195               name = "to",
196               description = "End of the time series as ISO 8601 UTC date string",
197               isRequired = true,
198               type = STRING
199           ),
200           @RestParameter(
201               name = "resolution",
202               description = "Data aggregation level. Must be one of 'daily', 'weekly', 'monthly', 'yearly'",
203               isRequired = true,
204               type = STRING
205           ),
206           @RestParameter(
207               name = "zoneId",
208               description = "The time zone id to use for calculations",
209               isRequired = true,
210               type = STRING
211           ),
212       },
213       responses = {
214           @RestResponse(responseCode = SC_OK, description = "The time series data.")
215       }
216   )
217   public Response getTimeSeriesData(
218       @PathParam("providerId") final String providerId,
219       @QueryParam("resourceId") final String resourceId,
220       @QueryParam("from") final String from,
221       @QueryParam("to") final String to,
222       @QueryParam("resolution") final String resolution,
223       @QueryParam("zoneId") final String zoneId
224   ) {
225     try {
226       final TimeSeries timeSeries = statisticsService.getTimeSeriesData(
227           statisticsService.getProvider(providerId).orElseThrow(NotFoundException::new),
228           resourceId,
229           Instant.parse(from),
230           Instant.parse(to),
231           DataResolution.fromString(resolution),
232           ZoneId.of(zoneId)
233       );
234       return Response.ok(timeSeriesToJson(timeSeries).toJSONString()).build();
235     } catch (Exception e) {
236       logger.error("Could not retrieve time series data for provider {}: {}", providerId, e.getMessage());
237       throw new WebApplicationException(Response.Status.INTERNAL_SERVER_ERROR);
238     }
239   }

241   @SuppressWarnings("unchecked")
242   private static JSONObject timeSeriesToJson(final TimeSeries timeSeries) {
243     final JSONObject json = new JSONObject();
244     final JSONArray jsonLabels = new JSONArray();
245     final JSONArray jsonValues = new JSONArray();
246     for (int i = 0; i < timeSeries.getValues().size(); i++) {
247       jsonLabels.add(timeSeries.getLabels().get(i));
248       jsonValues.add(timeSeries.getValues().get(i));
249     }
250     json.put("labels", jsonLabels);
251     json.put("values", jsonValues);
252     if (timeSeries.getTotal().isPresent()) {
253       json.put("total", timeSeries.getTotal().getAsDouble());
254     }
255     return json;
256   }
257 
258   @SuppressWarnings("unchecked")
259   private static JSONArray providersToJson(final Set<StatisticsProvider> providers) {
260     final JSONArray json = new JSONArray();
261     for (StatisticsProvider provider : providers) {
262       final JSONObject jsonProvider = new JSONObject();
263       final JSONArray jsonResolutions = new JSONArray();
264       jsonProvider.put("id", provider.getId());
265       jsonProvider.put("title", provider.getTitle());
266       jsonProvider.put("description", provider.getDescription());
267       jsonProvider.put("resourceType", provider.getResourceType().name());
268       if (provider instanceof TimeSeriesProvider) {
269         for (DataResolution resolution : ((TimeSeriesProvider) provider).getDataResolutions()) {
270           jsonResolutions.add(resolution.name());
271         }
272         jsonProvider.put("dataResolutions", jsonResolutions);
273       }
274       json.add(jsonProvider);
275     }
276     return json;
277   }
