48  public class CaptureAgentsEndpointTest {
49    private static final RestServiceTestEnv rt = testEnvForClasses(TestCaptureAgentsEndpoint.class);
50    private final JSONParser parser = new JSONParser();
51  
52    private JSONObject getCaptureAgent(String name, JSONArray captureAgents) {
53      JSONObject result = null;
54      for (Object agent : captureAgents) {
55        JSONObject agentJson = (JSONObject) agent;
56        if (!StringUtils.isBlank(agentJson.get("Name").toString())
57                && agentJson.get("Name").toString().equalsIgnoreCase(name)) {
58          result = agentJson;
59        }
60      }
61      return result;
62    }
63  
64    /**
65     * Check the results from the capture agent endpoint.
66     *
67     * @param expected
68     *          The expected JSONObject
69     * @param actual
70     *          The actual JSONObject
71     * @param checkInputs
72     *          Whether to check the capture agent's inputs ]
73     */
74    private void checkCaptureAgentResults(String expected, String actual, boolean checkInputs) throws ParseException {
75      JSONObject expectedJson = (JSONObject) parser.parse(expected);
76      JSONObject resultJson = (JSONObject) parser.parse(actual);
77      assertEquals("The expected total should match the result total", expectedJson.get("total"), resultJson.get("total"));
78      assertEquals("The expected offset should match the result offset", expectedJson.get("offset"),
79              resultJson.get("offset"));
80      assertEquals("The expected count should match the result count", expectedJson.get("count"), resultJson.get("count"));
81      assertEquals("The expected limit should match the result limit", expectedJson.get("limit"), resultJson.get("limit"));
82      checkCaptureAgentResults((JSONArray) expectedJson.get("results"), (JSONArray) resultJson.get("results"),
83              checkInputs);
84    }
85  
86    /**
87     * Check an individual agent's actual json against an expected json
88     *
89     * @param expectedJson
90     *          The expected json array
91     * @param resultJson
92     *          The actual json array
93     * @param checkInputs
94     *          Whether to check the agent's inputs
95     */
96    private void checkCaptureAgentResults(JSONArray expectedJson, JSONArray resultJson, boolean checkInputs) {
97      for (Object captureAgent : resultJson) {
98        JSONObject captureAgentJson = (JSONObject) captureAgent;
99        String name = captureAgentJson.get("Name").toString();
100       if (!StringUtils.isBlank(name)) {
101         JSONObject expected = getCaptureAgent(name, expectedJson);
102         assertEquals(expected.get("Name"), captureAgentJson.get("Name"));
103         assertEquals(expected.get("Status"), captureAgentJson.get("Status"));
104         assertEquals(expected.get("Update"), captureAgentJson.get("Update"));
105 
106         if (checkInputs) {
107           // Check inputs
108           JSONObject expectedInputs = (JSONObject) ((JSONArray) expected.get("inputs")).get(0);
109           JSONObject actualInputs = (JSONObject) ((JSONArray) captureAgentJson.get("inputs")).get(0);
110           assertEquals(expectedInputs.get("id"), actualInputs.get("id"));
111           assertEquals(expectedInputs.get("value"), actualInputs.get("value"));
112         }
113       } else {
114         fail("Capture agents needs to have a name as it is a unique id.");
115       }
116     }
117   }

119   @Test
120   public void testGetAllCaptureAgents() throws Exception {
121     String expectedWithInputs = IOUtils.toString(CaptureAgentsEndpointTest.class.getResource("/capture_agents.json"),
122       "utf-8");
123     String expectedWithoutInputs = IOUtils.toString(CaptureAgentsEndpointTest.class
124             .getResource("/capture_agents_noinputs.json"), "utf-8");
125 
126     String result = given().queryParam("inputs", true).expect().statusCode(HttpStatus.SC_OK)
127             .contentType(ContentType.JSON).body("total", equalTo(4)).body("offset", equalTo(0))
128             .body("limit", equalTo(0)).body("results", hasSize(4)).when().get(rt.host("/agents.json")).asString();
129 
130     checkCaptureAgentResults(expectedWithInputs, result, true);
131 
132     result = given().expect().statusCode(HttpStatus.SC_OK).contentType(ContentType.JSON).body("total", equalTo(4))
133             .body("offset", equalTo(0)).body("limit", equalTo(0)).body("results", hasSize(4)).when()
134             .get(rt.host("/agents.json")).asString();
135 
136     checkCaptureAgentResults(expectedWithoutInputs, result, false);
137   }

139   @Test
140   public void testGetAllWithParams() throws ParseException, IOException {
141     int limit = 0;
142     int offset = 2;
143     int total = 4;
144 
145     given().queryParam("limit", limit).queryParam("offset", offset).expect().statusCode(HttpStatus.SC_OK)
146             .contentType(ContentType.JSON).body("total", equalTo(total)).body("offset", equalTo(offset))
147             .body("limit", equalTo(limit)).body("results", hasSize(total - offset)).when()
148             .get(rt.host("/agents.json"));
149 
150     offset = 0;
151     limit = 2;
152 
153     given().queryParam("limit", limit).queryParam("offset", offset).expect().statusCode(HttpStatus.SC_OK)
154             .contentType(ContentType.JSON).body("total", equalTo(total)).body("offset", equalTo(offset))
155             .body("limit", equalTo(limit)).body("results", hasSize(limit)).when()
156             .get(rt.host("/agents.json"));
157 
158     offset = 2;
159     limit = 2;
160 
161     given().queryParam("limit", limit).queryParam("offset", offset).expect().statusCode(HttpStatus.SC_OK)
162             .contentType(ContentType.JSON).body("total", equalTo(total)).body("offset", equalTo(offset))
163             .body("limit", equalTo(limit)).body("results", hasSize(limit)).when()
164             .get(rt.host("/agents.json"));
165   }

167   @Test
168   public void testSortAgentsByLastUpdatedTimestamp() throws ParseException, IOException {
169     int total = 4;
170 
171     given().queryParam("sort", "updated:ASC").expect().statusCode(HttpStatus.SC_OK)
172             .contentType(ContentType.JSON).body("total", equalTo(total))
173             .body("results", hasSize(total))
174             .body("results[0].Update", equalTo("2014-05-26T15:37:02Z"))
175             .body("results[1].Update", equalTo("2016-05-26T07:07:07Z"))
176             .body("results[2].Update", equalTo("2016-06-09T06:00:00Z"))
177             .body("results[3].Update", equalTo("2016-06-09T18:00:00Z"))
178             .when().get(rt.host("/agents.json"));
179 
180     given().queryParam("sort", "updated:DESC").expect().statusCode(HttpStatus.SC_OK)
181             .contentType(ContentType.JSON).body("total", equalTo(total))
182             .body("results", hasSize(total))
183             .body("results[3].Update", equalTo("2014-05-26T15:37:02Z"))
184             .body("results[2].Update", equalTo("2016-05-26T07:07:07Z"))
185             .body("results[1].Update", equalTo("2016-06-09T06:00:00Z"))
186             .body("results[0].Update", equalTo("2016-06-09T18:00:00Z"))
187             .when().get(rt.host("/agents.json"));
188   }
