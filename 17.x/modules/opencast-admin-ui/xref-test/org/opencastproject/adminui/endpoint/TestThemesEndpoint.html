149   private SearchResult<IndexTheme> createThemeCaptureResult(
150           final Capture<ThemeSearchQuery> myCapture) {
151     SearchResultImpl<IndexTheme> searchResults = new SearchResultImpl<IndexTheme>(
152             myCapture.getValue(), 0, 0);
153     if (myCapture.hasCaptured()) {
154       if (myCapture.getValue().getIdentifiers().length == 1 && myCapture.getValue().getIdentifiers()[0] == theme1Id) {
155         SearchResultItem<IndexTheme> searchResultItem = getThemeSearchResultItem(
156                 theme1Id, "theme-1-name");
157         searchResults.addResultItem(searchResultItem);
158       } else if (myCapture.getValue().getIdentifiers().length == 0) {
159         SearchResultItem<IndexTheme> searchResultItem1 = getThemeSearchResultItem(
160                 theme1Id, "theme-1-name");
161         searchResults.addResultItem(searchResultItem1);
162         SearchResultItem<IndexTheme> searchResultItem2 = getThemeSearchResultItem(
163                 theme2Id, "theme-2-name");
164         searchResults.addResultItem(searchResultItem2);
165         SearchResultItem<IndexTheme> searchResultItem3 = getThemeSearchResultItem(
166                 theme3Id, "theme-3-name");
167         searchResults.addResultItem(searchResultItem3);
168       }
169     }
170     return searchResults;
171   }

173   @SuppressWarnings("unchecked")
174   private SearchResultItem<IndexTheme> getThemeSearchResultItem(Long id,
175           String name) {
176     IndexTheme theme = new IndexTheme(
177             id, defaultOrg.getId());
178     theme.setCreationDate(creationDate);
179     theme.setName(name);
180     theme.setCreator("Test User");
181     theme.setBumperFile("uuid1");
182     theme.setWatermarkFile("uuid2");
183     SearchResultItem<IndexTheme> searchResultItem = EasyMock
184             .createMock(SearchResultItem.class);
185     EasyMock.expect(searchResultItem.getSource()).andReturn(theme);
186     EasyMock.expect(searchResultItem.compareTo(EasyMock.anyObject(SearchResultItem.class))).andReturn(1);
187     EasyMock.replay(searchResultItem);
188     return searchResultItem;
189   }
190 
191   private SearchResult<Series> createSeriesCaptureResult(Capture<SeriesSearchQuery> myCapture) {
192     SearchResultImpl<Series> searchResults = new SearchResultImpl<Series>(myCapture.getValue(), 0, 0);
193     if (myCapture.hasCaptured()) {
194       SearchResultItem<Series> searchResultItem1 = getSeriesSearchResultItem("Series1Id", "Series 1 Title");
195       searchResults.addResultItem(searchResultItem1);
196       SearchResultItem<Series> searchResultItem2 = getSeriesSearchResultItem("Series2Id", "Series 2 Title");
197       searchResults.addResultItem(searchResultItem2);
198       SearchResultItem<Series> searchResultItem3 = getSeriesSearchResultItem("Series3Id", "Series 3 Title");
199       searchResults.addResultItem(searchResultItem3);
200     }
201     return searchResults;
202   }
203 
204   private SearchResultItem<Series> getSeriesSearchResultItem(String seriesId, String title) {
205     Series series = new Series(seriesId, defaultOrg.getId());
206     series.setTitle(title);
207     SearchResultItem<Series> searchResultItem = EasyMock.createNiceMock(SearchResultItem.class);
208     EasyMock.expect(searchResultItem.getSource()).andReturn(series);
209     EasyMock.expect(searchResultItem.compareTo(EasyMock.anyObject(SearchResultItem.class))).andReturn(1);
210     EasyMock.replay(searchResultItem);
211     return searchResultItem;
212   }
