22  package org.opencastproject.adminui.endpoint;
23  
24  import org.opencastproject.mediapackage.MediaPackageBuilderImpl;
25  import org.opencastproject.serviceregistry.api.HostRegistration;
26  import org.opencastproject.serviceregistry.api.JaxbHostRegistration;
27  import org.opencastproject.serviceregistry.api.JaxbServiceRegistration;
28  import org.opencastproject.serviceregistry.api.JaxbServiceStatistics;
29  import org.opencastproject.serviceregistry.api.ServiceRegistry;
30  import org.opencastproject.serviceregistry.api.ServiceState;
31  import org.opencastproject.serviceregistry.api.ServiceStatistics;
32  
33  import org.easymock.EasyMock;
34  import org.junit.Ignore;
35  
36  import java.util.ArrayList;
37  import java.util.List;
38  
39  import javax.ws.rs.Path;
40  
41  @Path("/")
42  @Ignore
43  public class TestServicesEndpoint extends ServicesEndpoint {
44  
45    private static final String HOST1_NAME = "host1";
46    private static final String HOST2_NAME = "host2";
47    private static final String HOST3_NAME = "host3";
48    private static final String HOST4_NAME = "host4";
49  
50    private ServiceRegistry serviceRegistry;
51    private MediaPackageBuilderImpl mpBuilder;
52  
53    public TestServicesEndpoint() throws Exception {
54      mpBuilder = new MediaPackageBuilderImpl();
55      this.serviceRegistry = EasyMock.createNiceMock(ServiceRegistry.class);
56  
57      List<HostRegistration> hosts = new ArrayList<HostRegistration>();
58      hosts.add(new JaxbHostRegistration(HOST1_NAME, "1.1.1.1", "node1", 100000, 8, 8, true, false));
59      hosts.add(new JaxbHostRegistration(HOST2_NAME, "1.1.1.2", "node2", 400000, 4, 8, true, true));
60      hosts.add(new JaxbHostRegistration(HOST3_NAME, "1.1.1.3", "node3", 200000, 2, 8, false, false));
61      hosts.add(new JaxbHostRegistration(HOST4_NAME, "1.1.1.4", "node4", 500000, 6, 8, true, true));
62  
63      JaxbServiceRegistration service1 = new JaxbServiceRegistration("service1", HOST1_NAME, "");
64      JaxbServiceRegistration service2 = new JaxbServiceRegistration("service2", HOST1_NAME, "");
65      JaxbServiceRegistration service3 = new JaxbServiceRegistration("service3", HOST2_NAME, "");
66      JaxbServiceRegistration service4 = new JaxbServiceRegistration("service4", HOST3_NAME, "");
67      JaxbServiceRegistration service5 = new JaxbServiceRegistration("service5", HOST2_NAME, "");
68      JaxbServiceRegistration service6 = new JaxbServiceRegistration("service6", HOST4_NAME, "");
69      service2.setServiceState(ServiceState.ERROR);
70      service4.setServiceState(ServiceState.WARNING);
71  
72      List<ServiceStatistics> statistics = new ArrayList<ServiceStatistics>();
73      statistics.add(new JaxbServiceStatistics(service1,      0,      0, 0, 0,  0));
74      statistics.add(new JaxbServiceStatistics(service2, 123000, 456000, 0, 3,  5));
75      statistics.add(new JaxbServiceStatistics(service3,  30000,  10000, 2, 5, 20));
76      statistics.add(new JaxbServiceStatistics(service4,  10000,  30000, 1, 1,  0));
77      statistics.add(new JaxbServiceStatistics(service5,      0,      0, 0, 0, 10));
78      statistics.add(new JaxbServiceStatistics(service6,      0,  60000, 0, 1,  0));
79  
80      EasyMock.expect(serviceRegistry.getHostRegistrations()).andReturn(hosts).anyTimes();
81      EasyMock.expect(serviceRegistry.getServiceStatistics()).andReturn(statistics).anyTimes();
82  
83      EasyMock.replay(serviceRegistry);
84  
85      this.setServiceRegistry(serviceRegistry);
86      this.activate();
87    }
88  
89  }
