39  import java.io.InputStream;

42  import uk.co.datumedge.hamcrest.json.SameJSONAs;
43  
44  public class UserSettingsTest {
45    @Test
46    public void toJsonInputEmptyExpectedEmptySettingsAndSignatures() throws Exception {
47      InputStream stream = SeriesEndpointTest.class.getResourceAsStream("/user_settings_test_empty.json");
48      InputStreamReader reader = new InputStreamReader(stream);
49      JSONObject expected = (JSONObject) new JSONParser().parse(reader);
50  
51      UserSettings userSetting = new UserSettings();
52      assertThat(expected.toJSONString(), SameJSONAs.sameJSONAs(userSetting.toJson().toJson()));
53    }
54  
55    @Test
56    public void toJsonInputSettingAndSignatureExpectedAllInJson() throws Exception {
57      InputStream stream = SeriesEndpointTest.class.getResourceAsStream("/user_settings_test_example.json");
58      InputStreamReader reader = new InputStreamReader(stream);
59      JSONObject expected = (JSONObject) new JSONParser().parse(reader);
60  
61      User creator = EasyMock.createMock(User.class);
62      EasyMock.expect(creator.getName()).andReturn("Users Name").anyTimes();
63      EasyMock.expect(creator.getUsername()).andReturn("username12").anyTimes();
64      EasyMock.expect(creator.getEmail()).andReturn("adam@fake.com").anyTimes();
65      EasyMock.replay(creator);
66      EmailAddress sender = new EmailAddress("adam@fake.com", "Other Name");
67      Option<EmailAddress> replyTo = none();
68      DateTime dateTime = new DateTime(1401465634101L);
69      dateTime.toDateTime(DateTimeZone.UTC);
70  
71      UserSetting userSetting = new UserSetting(98, "Test Key", "Test Value");
72      UserSettings userSettings = new UserSettings();
73      userSettings.setTotal(1);
74      userSettings.addUserSetting(userSetting);
75      assertThat(expected.toJSONString(), SameJSONAs.sameJSONAs(userSettings.toJson().toJson()));
76    }
77  }
