69      SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);

71      EasyMock.expect(securityService.getOrganization()).andReturn(new DefaultOrganization()).anyTimes();

73      EasyMock.replay(securityService);

75      mp1 = MediaPackageSupport.loadFromClassPath("/mp1.xml");

78      Workspace workspace = EasyMock.createNiceMock(Workspace.class);
79      EasyMock.expect(workspace.read(uri("series-dublincore.xml")))
80              .andAnswer(() -> getClass().getResourceAsStream("/series-dublincore.xml")).anyTimes();
81      EasyMock.expect(workspace.read(uri("episode-dublincore.xml")))
82              .andAnswer(() -> getClass().getResourceAsStream("/episode-dublincore.xml")).anyTimes();
83      EasyMock.expect(workspace.read(uri("mpeg7.xml")))
84              .andAnswer(() -> getClass().getResourceAsStream("/mpeg7.xml")).anyTimes();
85      EasyMock.expect(workspace.read(uri("series-xacml.xml")))
86              .andAnswer(() -> getClass().getResourceAsStream("/series-xacml.xml")).anyTimes();
87      EasyMock.replay(workspace);

89      oaiPmhDatabase = new OaiPmhDatabaseImpl();
90      oaiPmhDatabase.setEntityManagerFactory(newEntityManagerFactory(OaiPmhDatabaseImpl.PERSISTENCE_UNIT_NAME));
91      oaiPmhDatabase.setDBSessionFactory(getDbSessionFactory());
92      oaiPmhDatabase.setSecurityService(securityService);
93      oaiPmhDatabase.setWorkspace(workspace);
94      oaiPmhDatabase.activate(null);
95    }

109   @Test
110   public void testDeleting() throws Exception {
111     oaiPmhDatabase.store(mp1, REPOSITORY_ID_1);
112 
113     boolean failed = false;
114     try {
115       oaiPmhDatabase.delete(mp1.getIdentifier().toString(), "abc");
116     } catch (NotFoundException e) {
117       failed = true;
118     }
119     Assert.assertTrue(failed);
120 
121     oaiPmhDatabase.delete(mp1.getIdentifier().toString(), REPOSITORY_ID_1);
122 
123     SearchResult search = oaiPmhDatabase.search(query().mediaPackageId(mp1).build());
124     Assert.assertEquals(1, search.size());
125     Assert.assertTrue(search.getItems().get(0).isDeleted());
126   }

128   @Test
129   public void testRetrieving() throws Exception {
130     oaiPmhDatabase.store(mp1, REPOSITORY_ID_1);
131 
132     SearchResult search = oaiPmhDatabase.search(query().mediaPackageId(mp1).build());
133     Assert.assertEquals(1, search.size());
134     SearchResultItem searchResultItem = search.getItems().get(0);
135     Assert.assertEquals(REPOSITORY_ID_1, searchResultItem.getRepository());
136     Assert.assertEquals(mp1.getIdentifier().toString(), searchResultItem.getId());
137     Assert.assertEquals(mp1, searchResultItem.getMediaPackage());
138     Assert.assertFalse(searchResultItem.isDeleted());
139     Assert.assertEquals(DefaultOrganization.DEFAULT_ORGANIZATION_ID, searchResultItem.getOrganization());
140     Assert.assertNotNull(searchResultItem.getModificationDate());
141     Date modificationDate = searchResultItem.getModificationDate();
142     Assert.assertEquals(2, searchResultItem.getElements().size());
143     for (SearchResultElementItem catalogItem : searchResultItem.getElements()) {
144       Assert.assertNotNull(catalogItem.getFlavor());
145       Assert.assertNotNull(catalogItem.getXml());
146       Assert.assertEquals("dublincore/episode".equals(catalogItem.getFlavor().toLowerCase()),
147               catalogItem.isEpisodeDublinCore());
148       Assert.assertEquals("dublincore/series".equals(catalogItem.getFlavor().toLowerCase()),
149               catalogItem.isSeriesDublinCore());
150     }
151 
152     Date dateBeforeStoring = new Date();
153 
154     oaiPmhDatabase.store(mp1, REPOSITORY_ID_2);
155 
156     search = oaiPmhDatabase.search(query().mediaPackageId(mp1).build());
157     Assert.assertEquals(2, search.size());
158 
159     search = oaiPmhDatabase.search(query().mediaPackageId(mp1).repositoryId(REPOSITORY_ID_2).build());
160     Assert.assertEquals(1, search.size());
161     searchResultItem = search.getItems().get(0);
162     Assert.assertEquals(REPOSITORY_ID_2, searchResultItem.getRepository());
163     Assert.assertEquals(mp1.getIdentifier().toString(), searchResultItem.getId());
164     Assert.assertEquals(mp1, searchResultItem.getMediaPackage());
165     Assert.assertFalse(searchResultItem.isDeleted());
166     Assert.assertEquals(DefaultOrganization.DEFAULT_ORGANIZATION_ID, searchResultItem.getOrganization());
167     Assert.assertTrue(searchResultItem.getModificationDate() != null);
168     Assert.assertTrue(searchResultItem.getModificationDate().after(modificationDate));
169     Assert.assertEquals(2, searchResultItem.getElements().size());
170     for (SearchResultElementItem catalogItem : searchResultItem.getElements()) {
171       Assert.assertNotNull(catalogItem.getFlavor());
172       Assert.assertNotNull(catalogItem.getXml());
173       Assert.assertEquals("dublincore/episode".equals(catalogItem.getFlavor().toLowerCase()),
174               catalogItem.isEpisodeDublinCore());
175       Assert.assertEquals("dublincore/series".equals(catalogItem.getFlavor().toLowerCase()),
176               catalogItem.isSeriesDublinCore());
177     }
178 
179     search = oaiPmhDatabase.search(query().mediaPackageId(mp1).repositoryId(REPOSITORY_ID_1).build());
180     Assert.assertEquals(1, search.size());
181 
182     search = oaiPmhDatabase.search(query().mediaPackageId(mp1).repositoryId(REPOSITORY_ID_2).build());
183     Assert.assertEquals(1, search.size());
184 
185     search = oaiPmhDatabase
186             .search(query().mediaPackageId(mp1).repositoryId(REPOSITORY_ID_2).modifiedAfter(new Date()).build());
187     Assert.assertEquals(0, search.size());
188 
189     search = oaiPmhDatabase
190             .search(query().mediaPackageId(mp1).repositoryId(REPOSITORY_ID_2).modifiedAfter(dateBeforeStoring).build());
191     Assert.assertEquals(1, search.size());
192 
193     search = oaiPmhDatabase.search(query().modifiedAfter(dateBeforeStoring).build());
194     Assert.assertEquals(1, search.size());
195 
196     Date dateBeforeDeletion = new Date();
197 
198     oaiPmhDatabase.delete(mp2.getIdentifier().toString(), REPOSITORY_ID_2);
199 
200     Thread.sleep(10);
201 
202     search = oaiPmhDatabase.search(query().modifiedAfter(new Date()).build());
203     Assert.assertEquals(0, search.size());
204 
205     search = oaiPmhDatabase.search(query().modifiedAfter(dateBeforeDeletion).build());
206     Assert.assertEquals(1, search.size());
207   }

209   @Test
210   public void testLimitOffset() throws Exception {
211     oaiPmhDatabase.store(mp1, REPOSITORY_ID_1);
212     MediaPackage mp2 = (MediaPackage) mp1.clone();
213     mp2.setIdentifier(IdImpl.fromUUID());
214     oaiPmhDatabase.store(mp2, REPOSITORY_ID_2);
215     SearchResult search = oaiPmhDatabase.search(query().limit(2).build());
216     Assert.assertEquals(2, search.size());
217 
218     search = oaiPmhDatabase.search(query().limit(1).build());
219     Assert.assertEquals(1, search.size());
220     Assert.assertEquals(mp1.getIdentifier().toString(), search.getItems().get(0).getId());
221 
222     search = oaiPmhDatabase.search(query().offset(1).build());
223     Assert.assertEquals(1, search.size());
224     Assert.assertEquals(mp2.getIdentifier().toString(), search.getItems().get(0).getId());
225   }

227   @Test
228   public void testRemovalOfOrphanedElements() throws Exception {
229     oaiPmhDatabase.store(mp1, REPOSITORY_ID_1);
230     int count = 0;
231     for (SearchResultItem searchResultItem : oaiPmhDatabase
232             .search(queryRepo(REPOSITORY_ID_1).mediaPackageId(mp1).build()).getItems()) {
233       count++;
234       Assert.assertNotNull(searchResultItem.getMediaPackage());
235       Assert.assertNotNull(searchResultItem.getMediaPackage().getElementById("catalog-1"));
236     }
237     Assert.assertEquals(1, count);
238 
239     mp1.removeElementById("catalog-1");
240     oaiPmhDatabase.store(mp1, REPOSITORY_ID_1);
241     count = 0;
242     for (SearchResultItem searchResultItem : oaiPmhDatabase
243             .search(queryRepo(REPOSITORY_ID_1).mediaPackageId(mp1).build()).getItems()) {
244       count++;
245       Assert.assertNotNull(searchResultItem.getMediaPackage());
246       Assert.assertNull(searchResultItem.getMediaPackage().getElementById("catalog-1"));
247     }
248     Assert.assertEquals(1, count);
249   }

251   @Test
252   public void testModifyElement() throws Exception {
253     oaiPmhDatabase.store(mp1, REPOSITORY_ID_1);
254 
255     MediaPackageElement catalog1 = mp1.getElementById("catalog-1");
256     catalog1.setURI(URI.create("foo.xml"));
257     catalog1.setChecksum(null);
258     oaiPmhDatabase.store(mp1, REPOSITORY_ID_1);
259 
260     int count = 0;
261     for (SearchResultItem searchResultItem : oaiPmhDatabase
262             .search(queryRepo(REPOSITORY_ID_1).mediaPackageId(mp1).build()).getItems()) {
263       count++;
264       Assert.assertNotNull(searchResultItem.getMediaPackage());
265       MediaPackageElement mpe = searchResultItem.getMediaPackage().getElementById("catalog-1");
266       Assert.assertNotNull(mpe);
267       Assert.assertEquals("foo.xml", mpe.getURI().toString());
268     }
269     Assert.assertEquals(1, count);
270   }
