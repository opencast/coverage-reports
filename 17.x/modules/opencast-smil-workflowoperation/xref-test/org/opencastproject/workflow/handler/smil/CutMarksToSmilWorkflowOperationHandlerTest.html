53    private static final String SOURCE_MEDIA_FLAVORS = "source-media-flavors";
54    private static final String SOURCE_MEDIA_FLAVORS_KEY = "source/presenter";
55    private static final String SOURCE_JSON_FLAVOR = "source-json-flavor";
56    private static final String SOURCE_JSON_FLAVOR_KEY = "cut-marks/json";
57    private static final String TARGET_SMIL_FLAVOR = "target-smil-flavor";
58    private static final String TARGET_SMIL_FLAVOR_KEY = "a/b";

60    private static final String CORRECT_JSON_PATH = "/cutmarkstosmil_correct_1.json";
61    private static final String FAULTY_JSON_PATH = "/cutmarkstosmil_faulty_1.json";
62    private static final String SOME_TEST_VIDEO_PATH = "/testvideo_320x180.mp4";
63    private static final String SOME_RESULT_SMIL_PATH = "/smil.smil";
64  
65    private CutMarksToSmilWorkflowOperationHandler handler;
66  
67    private Workspace workspace;
68    private WorkflowInstance workflow;
69    private WorkflowOperationInstance instance;

71    @Before
72    public void setUp() throws Exception {
73      handler = new CutMarksToSmilWorkflowOperationHandler();
74  
75      // Create mocks
76      instance = EasyMock.createNiceMock(WorkflowOperationInstance.class);
77      EasyMock.expect(instance.getConfiguration(TARGET_SMIL_FLAVOR)).andReturn(TARGET_SMIL_FLAVOR_KEY).anyTimes();
78  
79      workspace = EasyMock.createNiceMock(Workspace.class);
80  
81      workflow = EasyMock.createNiceMock(WorkflowInstance.class);
82      EasyMock.expect(workflow.getCurrentOperation()).andReturn(instance).anyTimes();
83  
84      handler.setWorkspace(workspace);
85  
86      smilServiceSetup();
87    }
88  
89    /**
90     * Must be called by every test method for test-specific setup
91     * @param jsonPaths Paths to json files that will be added to the source flavor
92     * @param trackFlavors Track flavor will be created if it does not exist yet, also a track will be added to the flavor
93     * @throws Exception
94     */
95    private void specificSetUp(String[] jsonPaths, String[] trackFlavors) throws Exception {
96      MediaPackage mediaPackage = new MediaPackageBuilderImpl().createNew();
97      mediaPackage.setIdentifier(new IdImpl("123-456"));
98  
99      // Add JSON to mp
100     for (String path : jsonPaths) {
101       URI newURI = getClass().getResource(path).toURI();
102       // To avoid NullPointerEx when grabbing Absolute Track Path
103       EasyMock.expect(workspace.get(newURI)).andReturn(new File(newURI)).anyTimes();
104       mediaPackage.add(newURI, MediaPackageElement.Type.Catalog,
105               MediaPackageElementFlavor.parseFlavor(SOURCE_JSON_FLAVOR_KEY));
106     }
107 
108     // Add track with flavor to mp
109     for (String flavor : trackFlavors) {
110       TrackImpl track = new TrackImpl();
111       track.setFlavor(MediaPackageElementFlavor.parseFlavor(flavor));
112       // Make sure track contains a video
113       VideoStreamImpl videoStream = new VideoStreamImpl("test1");
114       track.setVideo(Collections.singletonList(videoStream));
115       URI trackURI = getClass().getResource(SOME_TEST_VIDEO_PATH).toURI();   // Absolute URI
116       track.setURI(trackURI);
117       track.setDuration(10000L);
118       mediaPackage.add(track);
119     }
120 
121     EasyMock.expect(workflow.getMediaPackage()).andReturn(mediaPackage).anyTimes();
122     EasyMock.replay(workspace, workflow);
123     handler.setWorkspace(workspace);
124 
125     EasyMock.expect(instance.getConfiguration(SOURCE_MEDIA_FLAVORS))
126         .andReturn(String.join(",", trackFlavors))
127         .anyTimes();
128     EasyMock.expect(instance.getConfiguration(SOURCE_JSON_FLAVOR)).andReturn(SOURCE_JSON_FLAVOR_KEY).anyTimes();
129     EasyMock.replay(instance);
130   }
131 
132   /**
133    * Initializes the smil service used for creating the final smil
134    * @throws Exception
135    */
136   private void smilServiceSetup() throws Exception {
137     final URI mpSmilURI = getClass().getResource(SOME_RESULT_SMIL_PATH).toURI();
138 
139     SmilMediaContainer objectPar = EasyMock.createNiceMock(SmilMediaContainer.class);
140     EasyMock.expect(objectPar.getId()).andReturn("test").anyTimes();
141     EasyMock.replay(objectPar);
142 
143     Smil smil = EasyMock.createNiceMock(Smil.class);
144     EasyMock.expect(smil.toXML()).andReturn(mpSmilURI.toString()).anyTimes();
145     EasyMock.replay(smil);
146 
147     SmilResponse response = EasyMock.createNiceMock(SmilResponse.class);
148     EasyMock.expect(response.getSmil()).andReturn(smil).anyTimes();
149     EasyMock.expect(response.getEntity()).andReturn(objectPar).anyTimes();
150     EasyMock.replay(response);
151 
152     SmilService smilService = EasyMock.createNiceMock(SmilService.class);
153     EasyMock.expect(smilService.createNewSmil(EasyMock.anyObject(MediaPackage.class))).andReturn(response).anyTimes();
154     EasyMock.expect(smilService.addParallel(smil)).andReturn(response).anyTimes();
155     EasyMock.expect(smilService.addClips(EasyMock.anyObject(Smil.class), EasyMock.anyString(),
156         EasyMock.anyObject(Track[].class), EasyMock.anyLong(), EasyMock.anyLong())).andReturn(response).anyTimes();
157     EasyMock.expect(workspace.put(EasyMock.anyString(), EasyMock.anyString(), EasyMock.anyString(),
158         EasyMock.anyObject(InputStream.class))).andReturn(mpSmilURI);
159     EasyMock.replay(smilService);
160     handler.setSmilService(smilService);
161   }

163   @Test
164   public void testComplete() throws Exception {
165     String[] jsonPaths = new String[] {CORRECT_JSON_PATH};
166     String[] trackFlavors = new String[] {SOURCE_MEDIA_FLAVORS_KEY};
167     specificSetUp(jsonPaths, trackFlavors);
168 
169     WorkflowOperationResult result = handler.start(workflow, null);
170     Assert.assertEquals(WorkflowOperationResult.Action.CONTINUE, result.getAction());
171 
172     MediaPackage mpNew = result.getMediaPackage();
173     Catalog[] catalogs = mpNew.getCatalogs(MediaPackageElementFlavor.parseFlavor(TARGET_SMIL_FLAVOR_KEY));
174     Assert.assertEquals(1, catalogs.length);
175   }
