106     replay(currentJob);

117     replay(organizationDirectoryService);

122         "admin", 4.0f, 2, 4194304);

156         .andReturn(basicResponse.getHeaders("WWW-Authenticate"))
157         .anyTimes();
158     nonceResponse = EasyMock.createMock(CloseableHttpResponse.class);

161         .anyTimes();
162     expect(nonceResponse.getHeaders("WWW-Authenticate")).andReturn(new Header[] {}).anyTimes();
163     replay(okResponse, digestResponse, nonceResponse);
