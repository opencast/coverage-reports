36  public class BundleInfoDbTest {
37    @Test
38    public void testPersistence() {
39      final BundleInfoDb db = db();
40      final BundleInfo info = bundleInfo("localhost", "bundle", 1L, "1.4.0", some("1345"), some("9012"));
41      db.store(info);
42      assertEquals("db contains an element", 1, db.getBundles().size());
43      for (final BundleInfo a : db.getBundles()) {
44        run(BundleInfo.class, new BundleInfo() {
45          @Override
46          public String getHost() {
47            assertEquals("host persisted", info.getHost(), a.getHost());
48            return null;
49          }
50  
51          @Override
52          public String getBundleSymbolicName() {
53            assertEquals("bundle symbolic name persisted", info.getBundleSymbolicName(), a.getBundleSymbolicName());
54            return null;
55          }
56  
57          @Override
58          public long getBundleId() {
59            assertEquals("bundle id persisted", info.getBundleId(), a.getBundleId());
60            return 0L;
61          }
62  
63          @Override
64          public String getBundleVersion() {
65            assertEquals("bundle version persisted", info.getBundleVersion(), a.getBundleVersion());
66            return null;
67          }
68  
69          @Override
70          public Option<String> getBuildNumber() {
71            assertEquals("build number persisted", info.getBuildNumber(), a.getBuildNumber());
72            return null;
73          }
74  
75          @Override
76          public BundleVersion getVersion() {
77            assertEquals("bundle version persisted", info.getVersion().getBundleVersion(), a.getVersion()
78                    .getBundleVersion());
79            assertEquals("build number persisted", info.getVersion().getBuildNumber(), a.getVersion().getBuildNumber());
80            return null;
81          }
82        });
83      }
84      db.clearAll();
85      assertEquals("db is empty", 0, db.getBundles().size());
86      //
87      db.store(bundleInfo("localhost", "bundle", 2L, "1.4.1", Option.<String> none()));
88      assertEquals("no build number", Option.<String> none(), db.getBundles().get(0).getBuildNumber());
89    }

91    @Test(expected = BundleInfoDbException.class)
92    public void testContstraints() {
93      final BundleInfoDb db = db();
94      db.store(bundleInfo("localhost", "bundle-1", 1L, "1.4.0", none("")));
95      // insert violates constraints
96      db.store(bundleInfo("localhost", "bundle-1", 5L, "1.4.0", some("ae41b09")));
97    }

99    @Test
100   public void testPrefixQuery() {
101     final BundleInfoDb db = db();
102     db.store(bundleInfo("localhost", "matterhorn-1", 1L, "1.4.0", none("")));
103     db.store(bundleInfo("localhost", "matterhorn-2", 2L, "1.4.0", none("")));
104     db.store(bundleInfo("localhost", "matterhorn-3", 3L, "1.4.0", none("")));
105     db.store(bundleInfo("localhost", "eth-1", 4L, "1.4.0", none("")));
106     db.store(bundleInfo("localhost", "opencast-1", 5L, "1.4.0", none("")));
107     db.store(bundleInfo("localhost", "opencast-2", 6L, "1.4.0", none("")));
108     assertEquals(6, db.getBundles().size());
109     assertEquals(3, db.getBundles("matterhorn").size());
110     assertEquals(1, db.getBundles("eth").size());
111     assertEquals(2, db.getBundles("opencast").size());
112     assertEquals(0, db.getBundles("UNKNOWN").size());
113     assertEquals(4, db.getBundles("matterhorn", "eth").size());
114   }

116   @Test
117   public void testDeleteOne() {
118     final BundleInfoDb db = db();
119     db.store(bundleInfo("localhost", "matterhorn-1", 1L, "1.4.0", none("")));
120     db.store(bundleInfo("localhost", "matterhorn-2", 2L, "1.4.0", none("")));
121     db.store(bundleInfo("localhost", "matterhorn-3", 3L, "1.4.0", none("")));
122     assertEquals(3, db.getBundles().size());
123     db.delete("localhost", db.getBundles().get(0).getBundleId());
124     assertEquals(2, db.getBundles().size());
125   }

127   @Test
128   public void testDeleteByHost() {
129     final BundleInfoDb db = db();
130     db.store(bundleInfo("localhost", "matterhorn-1", 1L, "1.4.0", none("")));
131     db.store(bundleInfo("remote", "matterhorn-1", 1L, "1.4.0", none("")));
132     assertEquals("db size", 2, db.getBundles().size());
133     db.clear("localhost");
134     assertEquals("db size", 1, db.getBundles().size());
135     db.clear("remote");
136     assertEquals("db size", 0, db.getBundles().size());
137   }

139   private BundleInfoDb db() {
140     final DBSession db = DBTestEnv.newDBSession(OsgiBundleInfoDb.PERSISTENCE_UNIT);
141     return new AbstractBundleInfoDb() {
142       @Override
143       protected DBSession getDBSession() {
144         return db;
145       }
146     };
147   }
