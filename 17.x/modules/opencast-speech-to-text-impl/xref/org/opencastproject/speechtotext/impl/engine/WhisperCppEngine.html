199   @Activate
200   @Modified
201   public void activate(ComponentContext cc) {
202     logger.debug("Activated/Modified WhisperC++ engine service class");
203     whispercppExecutable = StringUtils.defaultIfBlank(
204         (String) cc.getProperties().get(WHISPERCPP_EXECUTABLE_PATH_CONFIG_KEY), WHISPERCPP_EXECUTABLE_DEFAULT_PATH);
205     logger.debug("Set WhisperC++ path to {}", whispercppExecutable);
206 
207     whispercppModel = StringUtils.defaultIfBlank(
208         (String) cc.getProperties().get(WHISPERCPP_MODEL_CONFIG_KEY), WHISPERCPP_MODEL_DEFAULT);
209     logger.debug("WhisperC++ Language model set to {}", whispercppModel);
210 
211     whispercppBeamSize = OsgiUtil.getOptCfgAsInt(cc.getProperties(), WHISPERCPP_BEAM_SIZE_CONFIG_KEY);
212     if (whispercppBeamSize.isSome()) {
213       logger.debug("WhisperC++ beam size set to {}", whispercppBeamSize);
214     }
215 
216     whispercppMaxLength = OsgiUtil.getOptCfgAsInt(cc.getProperties(), WHISPERCPP_MAX_LENGTH_CONFIG_KEY);
217     if (whispercppMaxLength.isSome()) {
218       logger.debug("WhisperC++ maximum segment length set to {}", whispercppMaxLength);
219     }
220 
221     whispercppThreads = OsgiUtil.getOptCfgAsInt(cc.getProperties(), WHISPERCPP_THREADS_CONFIG_KEY);
222     if (whispercppThreads.isSome()) {
223       logger.debug("WhisperC++ number of threads set to {}", whispercppThreads);
224     }
225 
226     whispercppProcessors = OsgiUtil.getOptCfgAsInt(cc.getProperties(), WHISPERCPP_PROCESSORS_CONFIG_KEY);
227     if (whispercppProcessors.isSome()) {
228       logger.debug("WhisperC++ number of processors set to {}", whispercppProcessors);
229     }
230 
231     whispercppMaxContext = OsgiUtil.getOptCfgAsInt(cc.getProperties(), WHISPERCPP_MAX_CONTEXT_CONFIG_KEY);
232     if (whispercppMaxContext.isSome()) {
233       logger.debug("WhisperC++ max context set to {}", whispercppMaxContext);
234     }
235 
236     whispercppSplitOnWord = OsgiUtil.getOptCfgAsBoolean(cc.getProperties(), WHISPERCPP_SPLIT_ON_WORD_CONFIG_KEY);
237     if (whispercppSplitOnWord.isSome()) {
238       logger.debug("WhisperC++ split on word set to {}", whispercppSplitOnWord);
239     }
240 
241     whispercppBestOf = OsgiUtil.getOptCfgAsInt(cc.getProperties(), WHISPERCPP_BEST_OF_CONFIG_KEY);
242     if (whispercppBestOf.isSome()) {
243       logger.debug("WhisperC++ best of set to {}", whispercppBestOf);
244     }
245 
246     whispercppWordThreshold = OsgiUtil.getOptCfg(cc.getProperties(), WHISPERCPP_WORD_THRESHOLD_CONFIG_KEY).bind(
247         Strings.toDouble);
248     if (whispercppWordThreshold.isSome()) {
249       logger.debug("WhisperC++ word threshold set to {}", whispercppWordThreshold);
250     }
251 
252     whispercppEntropyThreshold = OsgiUtil.getOptCfg(cc.getProperties(), WHISPERCPP_ENTROPY_THRESHOLD_CONFIG_KEY).bind(
253         Strings.toDouble);
254     if (whispercppEntropyThreshold.isSome()) {
255       logger.debug("WhisperC++ entropy threshold set to {}", whispercppEntropyThreshold);
256     }
257 
258     whispercppLogProbThreshold = OsgiUtil.getOptCfg(cc.getProperties(), WHISPERCPP_LOG_PROB_THRESHOLD_CONFIG_KEY).bind(
259         Strings.toDouble);
260     if (whispercppLogProbThreshold.isSome()) {
261       logger.debug("WhisperC++ log prob threshold set to {}", whispercppLogProbThreshold);
262     }
263 
264     whispercppDiarization = OsgiUtil.getOptCfgAsBoolean(cc.getProperties(), WHISPERCPP_DIARIZATION_CONFIG_KEY);
265     if (whispercppDiarization.isSome()) {
266       logger.debug("WhisperC++ diarization set to {}", whispercppDiarization);
267     }
268 
269     whispercppTinyDiarization = OsgiUtil.getOptCfgAsBoolean(cc.getProperties(), WHISPERCPP_TINY_DIARIZATION_CONFIG_KEY);
270     if (whispercppTinyDiarization.isSome()) {
271       logger.debug("WhisperC++ tiny diarization set to {}", whispercppTinyDiarization);
272     }
273

275     if (whispercppNoFallback.isSome()) {
276       logger.debug("WhisperC++ no fallback set to {}", whispercppNoFallback);
277     }
278 
279     whispercppArgs = Objects.toString(cc.getProperties().get(WHISPERCPP_ARGS_CONFIG_KEY), "").trim().split("\\s+");
280     logger.debug("Additional args for WhisperC++: {}", (Object) whispercppArgs);
281 
282     autoEncode = BooleanUtils.toBoolean(Objects.toString(
283         cc.getProperties().get(AUTO_ENCODING_CONFIG_KEY),
284         AUTO_ENCODING_DEFAULT.toString()));
285     logger.debug("Automatically convert input media: {}", autoEncode);

287     ffmpegBinary = Objects.toString(cc.getBundleContext().getProperty(FFMPEG_BINARY_CONFIG_KEY), DEFAULT_FFMPEG_BINARY);
288     logger.debug("ffmpeg binary set to {}", ffmpegBinary);
289 
290     logger.debug("Finished activating/updating speech-to-text service");
291   }

293   /**
294    * {@inheritDoc}
295    *
296    * @see org.opencastproject.speechtotext.api.SpeechToTextEngine#generateSubtitlesFile(File, File, String, Boolean)
297    */
298   @Override
299   public Result generateSubtitlesFile(File mediaFile, File workingDirectory, String language, Boolean translate)
300           throws SpeechToTextEngineException {
301 
302     var whisperInput = mediaFile.getAbsolutePath();
303     if (autoEncode) {
304       whisperInput = FilenameUtils.concat(workingDirectory.getAbsolutePath(), UUID.randomUUID() + ".wav");
305       var ffmpegCommand = List.of(
306           ffmpegBinary,
307           "-i", mediaFile.getAbsolutePath(),
308           "-ar", "16000",
309           "-ac", "1",
310           "-c:a", "pcm_s16le",
311           whisperInput);
312       try {
313         execCommand(ffmpegCommand);
314       } catch (IOException e) {
315         throw new SpeechToTextEngineException("Failed to convert audio file", e);
316       } catch (InterruptedException e) {
317         throw new RuntimeException(e);
318       }
319     }
320 
321     if (!whisperInput.toLowerCase().endsWith(".wav")) {
322       throw new SpeechToTextEngineException("WhisperC++ currently doesn't support any media extension other than wav");
323     }
324 
325     String outputName = FilenameUtils.getBaseName(mediaFile.getAbsolutePath());
326 
327     List<String> command = new ArrayList<>(List.of(
328         whispercppExecutable,
329         whisperInput,
330         "--model", whispercppModel,
331         "-ovtt",
332         "-oj",
333         "--output-file", FilenameUtils.concat(workingDirectory.getAbsolutePath(), outputName)));
334     if (whispercppBeamSize.isSome()) {
335       command.add("-bs");
336       command.add(Integer.toString(whispercppBeamSize.get()));
337     }
338     if (whispercppMaxLength.isSome()) {
339       command.add("-ml");
340       command.add(Integer.toString(whispercppMaxLength.get()));
341     }
342     if (whispercppThreads.isSome()) {
343       command.add("-t");
344       command.add(Integer.toString(whispercppThreads.get()));
345     }
346     if (whispercppProcessors.isSome()) {
347       command.add("-p");
348       command.add(Integer.toString(whispercppProcessors.get()));
349     }
350     if (whispercppMaxContext.isSome()) {
351       command.add("-mc");
352       command.add(Integer.toString(whispercppMaxContext.get()));
353     }
354     if (whispercppSplitOnWord.isSome() && whispercppSplitOnWord.get()) {
355       command.add("-sow");
356     }
357     if (whispercppBestOf.isSome()) {
358       command.add("-bo");
359       command.add(Integer.toString(whispercppBestOf.get()));
360     }
361     if (whispercppWordThreshold.isSome()) {
362       command.add("-wt");
363       command.add(String.format(Locale.US, "%f", whispercppWordThreshold.get()));
364     }
365     if (whispercppEntropyThreshold.isSome()) {
366       command.add("-et");
367       command.add(String.format(Locale.US, "%f", whispercppEntropyThreshold.get()));
368     }
369     if (whispercppLogProbThreshold.isSome()) {
370       command.add("-lpt");
371       command.add(String.format(Locale.US, "%f", whispercppLogProbThreshold.get()));
372     }
373     if (whispercppDiarization.isSome() && whispercppDiarization.get()) {
374       command.add("-di");
375     }
376     if (whispercppTinyDiarization.isSome() && whispercppTinyDiarization.get()) {
377       command.add("-tdrz");
378     }
379     if (whispercppNoFallback.isSome() && whispercppNoFallback.get()) {
380       command.add("-nf");
381     }
382 
383     String subtitleLanguage;
384 
385     // set language of the source audio if known
386     if (!language.isBlank()) {
387       // Convert ISO3 language code to ISO2 if possible, as WhisperC++ expects ISO2 codes.
388       // If the conversion is not possible, retain the original language code.
389       logger.info("Found language '{}'", language);
390       language = LangCodeUtil.iso3ToIso2(language, language);
391       logger.info("Using language code '{}' for transcription process", language);
392       command.add("--language");
393       command.add(language);
394     } else {
395       logger.debug("Auto-detecting language");
396       command.add("--language");
397       command.add("auto");
398     }
399 
400     if (translate) {
401       command.add("--translate");
402       logger.info("Translation enabled");
403       subtitleLanguage = "en";
404     } else {
405       subtitleLanguage = language;
406     }
407 
408     command.addAll(Arrays.asList(whispercppArgs));
409 
410     logger.info("Executing WhisperC++'s transcription command: {}", command);
411 
412     File vtt;
413 
414     try {
415       execCommand(command);
416 
417       vtt = new File(workingDirectory, outputName + ".vtt");
418       if (!vtt.isFile()) {
419         throw new SpeechToTextEngineException("WhisperC++ produced no output");
420       }
421       logger.info("Subtitles file generated successfully: {}", vtt);
422     } catch (Exception e) {
423       logger.info("Transcription failed closing WhisperC++ transcription process for: {}", whisperInput);
424       throw new SpeechToTextEngineException(e);
425     }
426 
427     // Detect language if not set
428     if (subtitleLanguage.isBlank()) {
429       JSONParser jsonParser = new JSONParser();
430       File json = new File(workingDirectory, outputName + ".json");
431       try {
432         FileReader reader = new FileReader(json);
433         Object obj = jsonParser.parse(reader);
434         JSONObject jsonObject = (JSONObject) obj;
435         JSONObject result = (JSONObject) jsonObject.get("result");
436         subtitleLanguage = (String) result.get("language");
437         // convert language name to iso3 if necessary or take default
438         subtitleLanguage = LangCodeUtil.getIso2FromLang(subtitleLanguage, subtitleLanguage);
439         logger.info("Language detected by WhisperC++: {}", subtitleLanguage);
440       } catch (Exception e) {
441         logger.info("Error reading WhisperC++ JSON file for: {}", mediaFile);
442         throw new SpeechToTextEngineException(e);
443       } finally {
444         FileUtils.deleteQuietly(json);
445       }
446     }
447 
448     return new Result(subtitleLanguage, vtt);
449   }

451   private void execCommand(List<String> command) throws IOException, InterruptedException, SpeechToTextEngineException {
452     logger.info("Executing command: {}", command);
453     Process process = null;
454 
455     try {
456       ProcessBuilder processBuilder = new ProcessBuilder(command);
457       processBuilder.redirectErrorStream(true);
458       processBuilder.redirectInput(ProcessBuilder.Redirect.PIPE)
459           .redirectError(ProcessBuilder.Redirect.PIPE)
460           .redirectOutput(ProcessBuilder.Redirect.PIPE);
461       process = processBuilder.start();
462 
463       try (BufferedReader in = new BufferedReader(new InputStreamReader(process.getInputStream()))) {
464         String line;
465         while ((line = in.readLine()) != null) { // consume process output
466           logger.debug(line);
467         }
468       }
469 
470       // wait until the task is finished
471       int exitCode = process.waitFor();
472       logger.info("Process finished with exit code {}", exitCode);
473 
474       if (exitCode != 0) {
475         var error = "";
476         try (var errorStream = process.getInputStream()) {
477           error = "\n Output:\n" + IOUtils.toString(errorStream, StandardCharsets.UTF_8);
478         }
479         throw new SpeechToTextEngineException(
480             String.format("Process exited abnormally with status %d (command: %s) %s", exitCode, command, error));
481       }
482     } finally {
483       IoSupport.closeQuietly(process);
484     }
485   }
