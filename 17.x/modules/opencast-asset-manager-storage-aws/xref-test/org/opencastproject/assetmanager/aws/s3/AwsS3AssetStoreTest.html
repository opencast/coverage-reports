282   @Test
283   public void testTagging() throws Exception {
284     Capture<SetObjectTaggingRequest> tags = Capture.newInstance();
285     EasyMock.expect(s3Client.setObjectTagging(EasyMock.capture(tags))).andReturn(null).anyTimes();
286     //Fake mimetypes to trigger, or not trigger tagging
287     String[] mimetypes = { "audio", "image", "video" };
288     StoragePath path;
289     String objectKey;
290     Optional<MimeType> type;
291     Source source;
292     for (String mimetype : mimetypes) {
293       path = new StoragePath(ORG_ID, MP_ID, new VersionImpl(1L), mimetype);
294       type = Optional.of(MimeType.mimeType(mimetype, "fake"));
295       objectKey = KEY_VERSION_1 + mimetype + ".xml";
296       source = Source.mk(uri, null, type);
297       //We need to do this every time we upload something.
298       setupUpload(objectKey);
299       EasyMock.replay(s3Client, s3Transfer);
300       store.put(path, source);
301       EasyMock.resetToNice(s3Client, s3Transfer);
302       EasyMock.expect(s3Client.setObjectTagging(EasyMock.capture(tags))).andReturn(null).anyTimes();
303 
304       //Check with AWS that the tags are applied correctly
305       Assert.assertTrue(tags.hasCaptured());
306       SetObjectTaggingRequest req = tags.getValue();
307       Assert.assertEquals(BUCKET_NAME, req.getBucketName());
308       Assert.assertEquals(objectKey, req.getKey());
309       Assert.assertEquals("Freezable", req.getTagging().getTagSet().get(0).getKey());
310       Assert.assertEquals("true", req.getTagging().getTagSet().get(0).getValue());
311 
312       tags.reset();
313     }
314     path = new StoragePath(ORG_ID, MP_ID, new VersionImpl(1L), "fake");
315     type = Optional.of(MimeType.mimeType("non-freezable", "fake"));
316     objectKey = KEY_VERSION_1 + "fake.xml";
317     source = Source.mk(uri, null, type);
318     setupUpload(objectKey);
319     EasyMock.replay(s3Client, s3Transfer);
320     store.put(path, source);
321     //Check with AWS that the tags are not applied since this isn't in the allowlist
322     Assert.assertFalse(tags.hasCaptured());
323   }

327     GetObjectTaggingResult gotr = EasyMock.createStrictMock(GetObjectTaggingResult.class);
328     List<Tag> tags = List.of(new Tag("Freezable", "true"));
329     EasyMock.expect(gotr.getTagSet()).andReturn(tags).once();

332 
333     EasyMock.expect(metadata.getStorageClass()).andReturn(StorageClass.Standard.toString()).times(4);
334     EasyMock.expect(metadata.getStorageClass()).andReturn(StorageClass.Glacier.toString()).once();
335     EasyMock.expect(metadata.getStorageClass()).andReturn(StorageClass.Standard.toString()).once();
336     EasyMock.expect(metadata.getStorageClass()).andReturn(StorageClass.Glacier.toString()).once();
337     EasyMock.expect(metadata.getOngoingRestore()).andReturn(false);
338     EasyMock.expect(metadata.getRestoreExpirationTime()).andReturn(new Date()).once();
339     EasyMock.expect(metadata.getStorageClass()).andReturn(StorageClass.Standard.toString()).once();
340     EasyMock.expect(metadata.getStorageClass()).andReturn(StorageClass.DeepArchive.toString()).once();
341     EasyMock.expect(s3Client.getObjectTagging(EasyMock.anyObject(GetObjectTaggingRequest.class)))
342         .andReturn(gotr).once();
343     //FIXME: Mock appropriate results, not that they're checked.
344     EasyMock.expect(s3Client.copyObject(EasyMock.anyObject(CopyObjectRequest.class))).andReturn(null).anyTimes();
345     EasyMock.replay(gotr, metadata);
346 
347     setupUpload(OBJECT_KEY_1, metadata);
348     EasyMock.replay(s3Client, s3Transfer);
349 
350     StoragePath path = new StoragePath(ORG_ID, MP_ID, new VersionImpl(1L), ASSET_ID);
351     Source source = Source.mk(uri);
352     store.put(path, source);
353 
354     //This asset doesn't exist
355     Assert.assertEquals("NONE",
356         store.getAssetStorageClass(new StoragePath(ORG_ID, MP_ID, new VersionImpl(0L), ASSET_ID)));
357     //This does, test the default
358     Assert.assertEquals(OBJECT_KEY_1, store.getAssetObjectKey(path));
359     Assert.assertEquals(StorageClass.Standard.toString(), store.getAssetStorageClass(path));
360 
361     //Set the storage class to the same thing -> should be a noop, and not change the class
362     store.modifyAssetStorageClass(path, StorageClass.Standard.toString());
363     Assert.assertEquals(StorageClass.Standard.toString(), store.getAssetStorageClass(path));
364 
365     //change the storage type to glacier or deep archive
366     store.modifyAssetStorageClass(path, StorageClass.Glacier.toString());
367     Assert.assertEquals(StorageClass.Glacier.toString(), store.getAssetStorageClass(path));
368 
369     //We're in Glacier, which means we have to restore before we can move classes
370     store.modifyAssetStorageClass(path, StorageClass.ReducedRedundancy.toString());
371     Assert.assertEquals(StorageClass.Glacier.toString(), store.getAssetStorageClass(path));
372     //'Restore' the object.  NB: The mock here is skipping a whole bunch of logic
373     store.initiateRestoreAsset(path, Integer.MAX_VALUE);
374     //Asset is now in regular S3, sooo...
375     store.modifyAssetStorageClass(path, StorageClass.DeepArchive.toString());
376     Assert.assertEquals(StorageClass.DeepArchive.toString(), store.getAssetStorageClass(path));
377 
378     EasyMock.verify();
379   }

508     store.put(path, Source.mk(uri));
509     // Store another version of same asset
510     StoragePath path2 = new StoragePath(ORG_ID, MP_ID, new VersionImpl(2L), ASSET_ID);
511     store.put(path2, Source.mk(uri));
512 
513     // Check if mappings were saved to db

523     Assert.assertEquals(MP_ID, mapping2.getMediaPackageId());
524     Assert.assertEquals(2L, mapping2.getVersion().longValue());
525 
526     // Delete version 1 only
527     DeletionSelector versionSelector = DeletionSelector.delete(ORG_ID, MP_ID, new VersionImpl(1L));
528     Assert.assertTrue(store.delete(versionSelector));
529 
530     // Check if version1 was deleted but version 2 is still there
531     Assert.assertNull(database.findMapping(path));
