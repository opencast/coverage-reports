93    @Test
94    public void testDeleteByValue() throws Exception {
95      final String[] mp = createAndAddMediaPackagesSimple(3, 1, 1);
96      am.setProperty(p.agent.mk(mp[0], "agent-1"));
97      am.setProperty(p.legacyId.mk(mp[0], "id"));
98      am.setProperty(p.agent.mk(mp[1], "agent-2"));
99      am.setProperty(p.agent.mk(mp[2], "agent-2"));
100     assertTotals(3, 3, 4);
101     assertEquals(0, q.delete(OWNER, p.agent.target()).where(p.agent.eq("unknown-agent")).run());
102     assertTotals(3, 3, 4);
103     assertEquals(2, q.delete(OWNER, p.agent.target()).where(p.agent.eq("agent-2")).run());
104     assertEquals(1, q.delete(OWNER, p.legacyId.target()).where(p.agent.eq("agent-1")).run());
105     assertTotals(3, 3, 1);
106     final RichAResult r = enrich(q.select(q.properties()).where(q.mediaPackageId(mp[0])).run());
107     assertEquals("Media package " + mp[0] + " should still have the agent property",
108                  p.agent.name(), r.getProperties().stream().findFirst().get().getId().getFqn());
109     assertEquals("The media package itself should be found", 1, r.getSize());
110   }

114     final String[] mp1 = createAndAddMediaPackagesSimple(3, 1, 1, Opt.some("123"));

126                   .list(Q_SNAPSHOT.mediaPackageId))));

128       assertEquals(
129           "Two properties should be deleted",
130           2,

132       );
133     }
134     assertTotals(6, 6, 1);
135     assertEquals(Value.mk("agent-5"),
136         enrich(q.select(p.agent.target()).run()).getProperties().stream().findFirst().get().getValue());
137   }

202   @Test
203   public void testDeleteAllByMixedPredicates() throws Exception {
204     final Snapshot[] mpSeries1 = createAndAddMediaPackages(3, 2, 2, Opt.some("series-1"));
205     final Snapshot[] mpSeries2 = createAndAddMediaPackages(3, 1, 1, Opt.some("series-2"));
206     final Snapshot[] mpNoSeries = createAndAddMediaPackages(3, 2, 2, Opt.<String>none());
207     // set some properties
208     am.setProperty(Properties.mkProperty(p.agent, mpSeries1[0], "agent-1"));
209     am.setProperty(Properties.mkProperty(p.count, mpSeries1[0], 2L));
210     //
211     am.setProperty(Properties.mkProperty(p.agent, mpNoSeries[1], "agent-2"));
212     //
213     am.setProperty(Properties.mkProperty(p.agent, mpNoSeries[2], "agent-1"));
214     am.setProperty(Properties.mkProperty(p.legacyId, mpNoSeries[2], "id"));
215     //
216     am.setProperty(Properties.mkProperty(p.agent, mpSeries2[0], "agent-4"));
217     //
218     assertTotals(15, 15, 6);
219     assertEquals("All properties of mpSeries1[0] (2) should be deleted",
220                  2, q.delete(OWNER, q.properties())
221                          .where(q.organizationId(DefaultOrganization.DEFAULT_ORGANIZATION_ID)
222                                         .and(p.agent.eq("agent-1")).and(q.seriesId().exists()))
223                          .name("AND")
224                          .run());
225     assertTotals(15, 15, 4);
226     assertEquals("All properties of mpNoSeries[2] (2) and mpNoSeries[1] (1) should be deleted",
227                  3, q.delete(OWNER, q.properties())
228                          .where(q.organizationId(DefaultOrganization.DEFAULT_ORGANIZATION_ID)
229                                         .and(p.agent.eq("agent-1").or(p.agent.eq("agent-2"))))
230                          .name("OR")
231                          .run());
232     assertTotals(15, 15, 1);
233   }

236   public void testDeleteOfNamespaceByNonExistingProperty() throws Exception {
237     final String[] mp = createAndAddMediaPackagesSimple(2, 1, 1);
238     am.setProperty(p.agent.mk(mp[0], "agent"));
239     am.setProperty(p.approved.mk(mp[0], true));
240     am.setProperty(p.count.mk(mp[0], 1L));
241     am.setProperty(p2.agent.mk(mp[0], "agent"));
242     //
243     am.setProperty(p.agent.mk(mp[1], "agent"));
244     am.setProperty(p.approved.mk(mp[1], true));
245     am.setProperty(p.legacyId.mk(mp[1], "id"));
246     //
247 //    if (true) return;
248     assertEquals(
249         "All properties of media package " + mp[1] + "should be deleted",
250         3,
251         q.delete(OWNER, q.propertiesOf(p.namespace()))
252             .where(q.organizationId(DefaultOrganization.DEFAULT_ORGANIZATION_ID)
253                 .and(p.agent.eq("agent"))
254                 .and(p.count.notExists()))
255             .name("by non existing property")
256             .run()
257     );
258     {
259       final RichAResult r = enrich(q.select(q.properties()).where(q.mediaPackageId(mp[1])).run());
260       assertEquals(0, r.countProperties());
261     }
262     {
263       final RichAResult r = enrich(q.select(q.properties()).where(q.mediaPackageId(mp[0])).run());
264       assertEquals(
265           "No properties should have been deleted from the other media package since it has the 'count' property set",
266           4,
267           r.countProperties()
268       );
269     }

272   @Test
273   public void testRemoveProperties() throws Exception {
274     final String[] mp = createAndAddMediaPackagesSimple(2, 1, 1);
275     am.setProperty(p.agent.mk(mp[0], "agent"));
276     am.setProperty(p.approved.mk(mp[0], true));
277     am.setProperty(p.count.mk(mp[0], 1L));
278     am.setProperty(p2.agent.mk(mp[0], "agent"));
279     //
280     am.setProperty(p.agent.mk(mp[1], "agent"));
281     am.setProperty(p.approved.mk(mp[1], true));
282     am.setProperty(p.legacyId.mk(mp[1], "id"));
283     assertEquals(0L, Properties.removeProperties(am, OWNER, "unknown_org", mp[0], p.namespace()));
284     assertEquals(0L, Properties.removeProperties(
285         am, OWNER, DefaultOrganization.DEFAULT_ORGANIZATION_ID, "unknown-mp-id", p.namespace()
286     ));
287     assertEquals(0L, Properties.removeProperties(
288         am, OWNER, DefaultOrganization.DEFAULT_ORGANIZATION_ID, mp[0], "unknown-namespace"
289     ));
290     assertEquals(3L, Properties.removeProperties(
291         am, OWNER, DefaultOrganization.DEFAULT_ORGANIZATION_ID, mp[0], p.namespace()
292     ));
293     assertEquals(1L, enrich(q.select(q.properties()).where(q.mediaPackageId(mp[0])).run()).countProperties());
294     assertEquals(1L, Properties.removeProperties(
295         am, OWNER, DefaultOrganization.DEFAULT_ORGANIZATION_ID, mp[0], p2.namespace()
296     ));
297     assertEquals(3L, Properties.removeProperties(
298         am, OWNER, DefaultOrganization.DEFAULT_ORGANIZATION_ID, mp[1], p.namespace()
299     ));
