227   /**
228    * Create a number of media packages with one catalog each and add it to the
229    * AssetManager. Return the media package IDs as an array.
230    * <p>
231    * Please note that each media package creates two assets in the store--the
232    * catalog and the manifest--but only one asset in the database which is the
233    * catalog. The manifest is represented in the snapshot table, not the asset
234    * table.
235    *
236    * @param amount
237    *         the amount of media packages to create
238    * @param minVersions
239    *         the minimum amount of versions to create per media package
240    * @param maxVersions
241    *         the maximum amount of versions to create per media package
242    * @param seriesId
243    *         an optional series ID
244    */
245   protected String[] createAndAddMediaPackagesSimple(
246       int amount,
247       final int minVersions,
248       final int maxVersions,
249       final Opt<String> seriesId
250   ) {
251     return Arrays.stream(createAndAddMediaPackages(amount, minVersions, maxVersions, seriesId))
252         .map(s -> s.getMediaPackage().getIdentifier().toString())
253         .collect(Collectors.toSet())
254         .toArray(new String[]{});
255   }
256 
257   /**
258    * Like {@link #createAndAddMediaPackagesSimple(int, int, int, Opt)} but without series ID.
259    */
260   protected String[] createAndAddMediaPackagesSimple(int amount, final int minVersions, final int maxVersions) {
261     return createAndAddMediaPackagesSimple(amount, minVersions, maxVersions, Opt.<String>none());
262   }

264   /**
265    * Continuous versions.
266    *
267    * @see #createAndAddMediaPackages(int, int, int, boolean, Opt)
268    */
269   protected Snapshot[] createAndAddMediaPackages(
270           int amount, final int minVersions, final int maxVersions, final Opt<String> seriesId) {
271     return createAndAddMediaPackages(amount, minVersions, maxVersions, true, seriesId);
272   }
273 
274   /**
275    * @param continuousVersions true if version numbers should be increased continuously, false if there should be
276    *          discontinuities
277    * @see #createAndAddMediaPackagesSimple(int, int, int, Opt)
278    */
279   protected Snapshot[] createAndAddMediaPackages(
280           int amount,
281           final int minVersions, final int maxVersions,
282           final boolean continuousVersions,
283           final Opt<String> seriesId) {
284     logger.info("Create {} media packages with {} to {} snapshots each", amount, minVersions, maxVersions);
285     final Stream<Snapshot> inserts = Stream.cont(inc()).take(amount).bind(new FnX<Integer, Iterable<Snapshot>>() {
286       @Override public Iterable<Snapshot> applyX(final Integer mpCount) throws Exception {
287         final MediaPackage mp = mkMediaPackage(mkCatalog());
288         for (String sid : seriesId) {
289           mp.setSeries(sid);
290         }
291         final int versions = (int) (Math.random() * ((double) maxVersions - minVersions) + minVersions);
292         final String mpId = mp.getIdentifier().toString();
293         logger.debug("Going to take {} snapshot/s of media package {}", versions, mpId);
294         return Stream.cont(inc()).take(versions).map(new Fn<Integer, Snapshot>() {
295           @Override public Snapshot apply(Integer versionCount) {
296             if (!continuousVersions) {
297               // insert a gap into the version claim
298               am.getDatabase().claimVersion(mp.getIdentifier().toString());
299             }
300             logger.debug("Taking snapshot {} of media package {}", versionCount + 1, mpId);
301             return am.takeSnapshot(OWNER, mp);
302           }
303         });
304       }
305     });
306     return Collections.toArray(Snapshot.class, inserts.toList());
307   }

340   /**
341    * Create a test asset store.
342    */
343   protected AssetStore mkAssetStore(String storeType) {
344     return new AssetStore() {
345       private Set<StoragePath> store = new HashSet<>();
346 
347       private void logSize() {
348         logger.debug("Store contains {} asset(s)", store.size());
349       }
350 
351       @Override public void put(StoragePath path, Source source) throws AssetStoreException {
352         store.add(path);
353         logSize();
354       }
355 
356       @Override public boolean copy(StoragePath from, StoragePath to) throws AssetStoreException {
357         if (store.contains(from)) {
358           store.add(to);
359           logSize();
360           return true;
361         } else {
362           return false;
363         }
364       }
365 
366       @Override public Optional<InputStream> get(StoragePath path) throws AssetStoreException {
367         return IoSupport.openClassPathResource("/dublincore-a.xml").isSome()
368             ? Optional.of(IoSupport.openClassPathResource("/dublincore-a.xml").get())
369             : Optional.empty();
370       }
371 
372       @Override public boolean contains(StoragePath path) throws AssetStoreException {
373         return store.contains(path);
374       }
375 
376       @Override public boolean delete(DeletionSelector sel) throws AssetStoreException {
377         logger.info("Delete from asset store " + sel);
378         final Set<StoragePath> newStore = new HashSet<>();
379         boolean deleted = false;
380         for (StoragePath s : store) {
381           if (!(sel.getOrganizationId().equals(s.getOrganizationId())
382               && sel.getMediaPackageId().equals(s.getMediaPackageId())
383               && (sel.getVersion().isPresent() ? sel.getVersion().get().equals(s.getVersion()) : true))) {
384             newStore.add(s);
385           } else {
386             deleted = true;
387           }
388         }
389         store = newStore;
390         logSize();
391         return deleted;
392       }
393 
394       @Override public Option<Long> getTotalSpace() {
395         return Option.none();
396       }
397 
398       @Override public Option<Long> getUsableSpace() {
399         return Option.none();
400       }
401 
402       @Override public Option<Long> getUsedSpace() {
403         return Option.some((long) store.size());
404       }
405 
406       @Override public String getStoreType() {
407         return storeType;
408       }
409     };
410   }
