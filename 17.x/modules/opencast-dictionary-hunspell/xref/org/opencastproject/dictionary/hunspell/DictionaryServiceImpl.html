47  /**
48   * This dictionary service implementation passes the input text
49   * to the hunspell spell checker and returns its results.
50   */
51  @Component(
52      immediate = true,
53      service = DictionaryService.class,
54      property = {
55          "service.description=Dictionary Service"
56      }
57  )
58  public class DictionaryServiceImpl implements DictionaryService {
59  
60    /** The logging facility */
61    private static final Logger logger =
62        LoggerFactory.getLogger(DictionaryServiceImpl.class);
63  
64    public static final String HUNSPELL_BINARY_CONFIG_KEY =
65        "org.opencastproject.dictionary.hunspell.binary";
66  
67    public static final String HUNSPELL_COMMAND_CONFIG_KEY =
68        "org.opencastproject.dictionary.hunspell.command";
69  
70    /* The hunspell binary to execute */
71    private String binary = "hunspell";
72  
73    /* The regular command line options for filtering */
74    private String command = " -i utf-8 -d de_DE,en_GB,en_US -G";
75  
76    public void setBinary(String b) {
77      binary = b;
78    }
79  
80    public String getBinary() {
81      return binary;
82    }
83  
84    public void setCommand(String c) {
85      command = c;
86    }
87  
88    public String getCommand() {
89      return command;
90    }

92    /**
93     * OSGi callback on component activation.
94     *
95     * @param  ctx  the bundle context
96     */
97    @Activate
98    void activate(BundleContext ctx) throws UnsupportedEncodingException {
99      Dictionary<String, String> properties = new Hashtable<String, String>();
100     properties.put(ARTIFACT, "dictionary");
101     ctx.registerService(ReadinessIndicator.class.getName(),
102         new ReadinessIndicator(), properties);
103 
104     /* Get hunspell binary from config file */
105     String binary = ctx.getProperty(HUNSPELL_BINARY_CONFIG_KEY);
106     if (binary != null) {
107       logger.info("Setting hunspell binary to '{}'", binary);
108       this.binary = binary;
109     }
110 
111     /* Get hunspell command line options from config file */
112     String command = ctx.getProperty(HUNSPELL_COMMAND_CONFIG_KEY);
113     if (command != null) {
114       logger.info("Setting hunspell command line options to '{}'", command);
115       this.command = command;
116     }
117   }

120   /**
121    * Run hunspell with text as input.
122    **/
123   public LinkedList<String> runHunspell(String text) throws Throwable {
124 
125     // create a new list of arguments for our process
126     String commandLine = binary + ' ' + command;
127     String[] commandList = commandLine.split("\\s+");
128 
129 
130     InputStream  stdout = null;
131     InputStream  stderr = null;
132     OutputStream stdin  = null;
133     Process p = null;
134     BufferedReader bufr = null;
135     LinkedList<String> words = new LinkedList<String>();
136 
137     logger.info("Executing hunspell command '{}'", StringUtils.join(commandList, " "));
138     p = new ProcessBuilder(commandList).start();
139     stderr = p.getErrorStream();
140     stdout = p.getInputStream();
141     stdin  = p.getOutputStream();
142 
143     /* Pipe text through hunspell for filtering */
144     stdin.write(text.getBytes("UTF-8"));
145     stdin.flush();
146     stdin.close();
147 
148     /* Get output of hunspell */
149     String line;
150     bufr = new BufferedReader(new InputStreamReader(stdout, "UTF-8"));
151     while ((line = bufr.readLine()) != null) {
152       words.add(line);
153     }
154     bufr.close();
155 
156     /* Get error messages */
157     bufr = new BufferedReader(new InputStreamReader(stderr));
158     while ((line = bufr.readLine()) != null) {
159       logger.warn(line);
160     }
161     bufr.close();
162 
163     if (p.waitFor() != 0) {
164       logger.error("Hunspell reported an error (Missing dictionaries?)");
165       throw new IllegalStateException("Hunspell returned error code");
166     }
167 
168     return words;
169   }

172   /**
173    * Filter the text according to the rules defined by the dictionary
174    * implementation used. This implementation will just let the whole text pass
175    * through.
176    *
177    * @return filtered text
178    **/
179   @Override
180   public Textual cleanUpText(String text) {
181 
182     LinkedList<String> words = null;
183 
184     try {
185       words = runHunspell(text);
186     } catch (Throwable t) {
187       logger.error("Error executing hunspell");
188       logger.error(t.getMessage(), t);
189       return null;
190     }
191 
192 
193     String result = StringUtils.join(words, " ");
194     if ("".equals(result)) {
195       return null;
196     }
197     return new TextualImpl(result);
198   }
