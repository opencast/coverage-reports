22  package org.opencastproject.security.api;
23  
24  /**
25   * Common security constant definitions.
26   */
27  public interface SecurityConstants {
28  
29    /** Header name for the desired organization */
30    String ORGANIZATION_HEADER = "X-Opencast-Matterhorn-Organization";
31  
32    /** Header name for the desired user */
33    String USER_HEADER = "X-Opencast-Matterhorn-User";
34  
35    /** Header name for running an operation as a desired user. Same as X-Opencast-Matterhorn-User. */
36    String RUN_AS_USER_HEADER = "X-RUN-AS-USER";
37  
38    /** Header name for the desired role */
39    String ROLES_HEADER = "X-Opencast-Matterhorn-Roles";
40  
41    /** Header name for running an operation with a desired role. Same as X-Opencast-Matterhorn-Roles. */
42    String RUN_WITH_ROLES = "X-RUN-WITH-ROLES";
43  
44    /** Name of the Opencast admin role */
45    String GLOBAL_ADMIN_ROLE = "ROLE_ADMIN";
46  
47    /** Name of the Opencast capture agent role */
48    String GLOBAL_CAPTURE_AGENT_ROLE = "ROLE_CAPTURE_AGENT";
49  
50    /** Name of the Opencast global sudo role */
51    String GLOBAL_SUDO_ROLE = "ROLE_SUDO";
52  
53    /** Name of the Opencast anonymous role */
54    String GLOBAL_ANONYMOUS_USERNAME = "anonymous";
55  
56    /** The roles associated with the Opencast system account */
57    String[] GLOBAL_SYSTEM_ROLES = new String[] { GLOBAL_ADMIN_ROLE, GLOBAL_SUDO_ROLE };
58  
59    /** The roles associated with the Opencast capture agent account */
60    String[] GLOBAL_CAPTURE_AGENT_ROLES = new String[] { GLOBAL_CAPTURE_AGENT_ROLE };
61  
62    /** The administrator user configuration option */
63    String GLOBAL_ADMIN_USER_PROPERTY = "org.opencastproject.security.admin.user";
64  
65    /** The prefix for episode role ids */
66    String EPISODE_ROLE_ID_PREFIX = "ROLE_EPISODE";
67  
68  }
