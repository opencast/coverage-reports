35    @Test
36    public void testInvalidConstructorArguments() {
37  
38      String[][] inputArguments = new String[][] {
39        new String[] { null, null },
40        new String[] { "valid", null },
41        new String[] { null, "valid" },
42        new String[] { "", "" },
43        new String[] { "valid", "" },
44        new String[] { "", "valid" },
45        new String[] { " \t \n \r ", " \t \n \r " },
46        new String[] { " \t \n \r ", "" },
47        new String[] { "", " \t \n \r " },
48        new String[] { "valid", " \t \n \r " },
49        new String[] { " \t \n \r ", "valid" },
50        new String[] { "invalid/type", "valid_subtype" },
51        new String[] { "valid_type", "invalid/subtype" }
52      };
53  
54      for (String[] arguments : inputArguments) {
55        try {
56          new MediaPackageElementFlavor(arguments[0], arguments[1]);
57        } catch (Throwable t) {
58          assertEquals(format("Catched unexpected exception. The message was: %s", t.getMessage()),
59                  IllegalArgumentException.class, t.getClass());
60          continue;
61        }
62        fail(format("Flavor constructor did not throw an expected exception with invalid arguments \"%s\" and \"%s\"",
63                arguments[0], arguments[1]));
64      }
65    }

67    @Test
68    public void testEquals() {
69      assertEquals("Leading or trailing whitespace in the flavor type should be trimmed",
70              flavor(" \r \n \t type \t \n \r ", "subtype"), flavor("type", "subtype"));
71      assertEquals("Leading or trailing whitespace in the flavor subtype should be ignored",
72              flavor("type", " \r \n \t subtype \t \n \r "), flavor("type", "subtype"));
73      assertEquals("Leading or trailing whitespace in both the flavor type and subtype should be trimmed",
74              flavor(" \r \n \t type \t \n \r ", " \r \n \t subtype \t \n \r "), flavor("type", "subtype"));
75      assertEquals("Flavor type should be case-insensitive", flavor("TYPE", "subtype"), flavor("type", "subtype"));
76      assertEquals("Flavor type should be case-insensitive", flavor("type", "SUBTYPE"), flavor("type", "subtype"));
77      assertEquals("Flavor parts should be case-insensitive", flavor("TYPE", "SUBTYPE"), flavor("type", "subtype"));
78    }

80    @Test
81    public void testMatches() throws Exception {
82      assertFalse("null should not match any flavor", flavor("a", "b").matches(null));
83      assertTrue("Equal flavors should match", flavor("a", "b").matches(flavor("a", "b")));
84      assertFalse("Unequal flavors should not match", flavor("a", "b").matches(flavor("b", "a")));
85      assertTrue("Wildcard type did not match", flavor("*", "b").matches(flavor("a", "b")));
86      assertTrue("Match with wildcard type should be commutative",
87              flavor("a", "b").matches(flavor("*", "b")) && flavor("*", "b").matches(flavor("a", "b")));
88      assertTrue("Wildcard subtype did not match", flavor("a", "*").matches(flavor("a", "b")));
89      assertTrue("Match with wildcard subtype should be commutative",
90              flavor("a", "b").matches(flavor("a", "*")) && flavor("a", "*").matches(flavor("a", "b")));
91      assertTrue("Wildcard in both type and subtype did not match", flavor("*", "*").matches(flavor("a", "b")));
92      assertTrue("Match with wildcards in both type and subtype should be commutative",
93              flavor("*", "*").matches(flavor("a", "b")) && flavor("a", "b").matches(flavor("*", "*")));
94      assertTrue("Wildcard matches", flavor("*", "*").matches(flavor("*", "b")));
95      assertTrue("Match between wildcards should be commutative",
96              flavor("*", "*").matches(flavor("*", "b")) && flavor("*", "b").matches(flavor("*", "*")));
97      assertTrue("Wildcard matches", flavor("*", "*").matches(flavor("a", "*")));
98      assertTrue("Match between wildcards should be commutative",
99              flavor("*", "*").matches(flavor("a", "*")) && flavor("a", "*").matches(flavor("*", "*")));
100     assertTrue("Wildcard matches", flavor("*", "*").matches(flavor("*", "*")));
101   }

103   @Test
104   public void testParse() {
105 
106     String[] invalidFlavors = new String[] {
107             null,
108             "",
109             "/",
110             "valid/",
111             "/valid",
112             " \r \n \t /",
113             " \r \n \t /valid",
114             "/ \r \n \t ",
115             "valid/ \r \n \t ",
116             " \r \n \t / \r \n \t ",
117             "too/many/slashes",
118             "/ \r \n \t too \r / \r many   / \t \r \n slashes \t"
119     };
120 
121     for (String invalid : invalidFlavors) {
122       try {
123         MediaPackageElementFlavor.parseFlavor(invalid);
124       } catch (Throwable t) {
125         assertEquals(format("Catched unexpected exception. The message was: %s", t.getMessage()),
126                 IllegalArgumentException.class, t.getClass());
127         continue;
128       }
129       fail(format("Invalid flavor should fail but was parsed without errors: \"%s\"", invalid));
130     }
131   }
