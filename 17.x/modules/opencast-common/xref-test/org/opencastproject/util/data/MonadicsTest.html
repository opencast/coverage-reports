175   @Test
176   public void testLazyMap() {
177     final boolean[] applied = {false};
178     IteratorMonadic<Integer> im = mlazy(asList(1, 2, 3, 4, 5))
179             .map(new Function<Integer, Integer>() {
180               @Override
181               public Integer apply(Integer i) {
182                 applied[0] = true;
183                 return i * i;
184               }
185             })
186             .map(new Function<Integer, Integer>() {
187               @Override
188               public Integer apply(Integer i) {
189                 applied[0] = true;
190                 return i + i;
191               }
192             });
193     assertFalse(applied[0]);
194     im.value();
195     assertFalse(applied[0]);
196     List<Integer> eval = im.eval();
197     assertTrue(applied[0]);
198     assertArrayEquals(new Integer[]{2, 8, 18, 32, 50}, eval.toArray(new Integer[]{}));
199     // test empty input
200     assertTrue(Collections.toList(
201             mlazy(Collections.<Integer>nil()).map(Functions.<Integer>identity()).value()
202     ).isEmpty());
203   }

406     {
407       List<Tuple<String, Integer>> r = mlist(new String[]{"a", "b", "c"}).zip(list(1, 2, 3, 4, 5)).value();
408       assertEquals(3, r.size());
409       assertEquals(tuple("b", 2), r.get(1));
410     }
411     {
412       List<Tuple<String, Integer>> r = mlist(new String[]{"a", "b", "c"}).zip(Collections.<Integer>nil()).value();

414     }
415     {
416       List<Tuple<String, Integer>> r = mlist(list("a", "b", "c")).zip(list(1, 2)).value();
417       assertEquals(2, r.size());
418       assertEquals(tuple("b", 2), r.get(1));
419     }
420     {
421       List<Tuple<String, Integer>> r = mlist(list("a", "b", "c").iterator()).zip(list(1, 2)).value();
422       assertEquals(2, r.size());
423       assertEquals(tuple("b", 2), r.get(1));
424     }
425     {
426       List<Tuple<String, Integer>> r = mlist(Collections.<String>nil().iterator()).zip(list(1, 2)).value();
427       assertEquals(0, r.size());

444   private static <A> Function<A, Iterator<A>> twice() {
445     return new Function<A, Iterator<A>>() {
446       @Override
447       public Iterator<A> apply(A a) {
448         return iterator(a, a);
449       }
450     };
451   }
