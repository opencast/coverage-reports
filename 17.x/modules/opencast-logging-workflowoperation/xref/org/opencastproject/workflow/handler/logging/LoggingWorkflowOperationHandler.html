49  /**
50   * The <code>LoggingWorkflowOperationHandlerTest</code> will log the current state of a workflow instance and its media
51   * package at a given point of a workflow.
52   */
53  @Component(
54      property = {
55          "service.description=Logging Workflow Operation Handler",
56          "workflow.operation=log"
57      },
58      immediate = true,
59      service = WorkflowOperationHandler.class
60  )
61  public class LoggingWorkflowOperationHandler extends AbstractWorkflowOperationHandler {
62  
63    /** The logging facility */
64    private static final Logger logger = LoggerFactory.getLogger(LoggingWorkflowOperationHandler.class);
65  
66    /**
67     * {@inheritDoc}
68     *
69     * @see org.opencastproject.workflow.api.WorkflowOperationHandler#start(
70     *      org.opencastproject.workflow.api.WorkflowInstance, JobContext)
71     */
72    @Override
73    public WorkflowOperationResult start(WorkflowInstance workflowInstance, JobContext context)
74            throws WorkflowOperationException {
75      WorkflowOperationInstance operation = workflowInstance.getCurrentOperation();
76  
77      String directoryPath = StringUtils.trimToNull(operation.getConfiguration("directory"));
78      File directory = null;
79      if (directoryPath != null) {
80        directory = new File(directoryPath);
81        try {
82          FileUtils.forceMkdir(directory);
83        } catch (IOException e) {
84          throw new WorkflowOperationException(String.format("Failed to create output directory '%s'", directoryPath), e);
85        }
86      }
87  
88  
89      boolean logMediaPackageJSON = BooleanUtils.toBoolean(operation.getConfiguration("mediapackage-json"));
90      boolean logMediaPackageXML = BooleanUtils.toBoolean(operation.getConfiguration("mediapackage-xml"));
91      boolean logWorkflowInstance = BooleanUtils.toBoolean(operation.getConfiguration("workflowinstance-xml"));
92  
93      if (!(logMediaPackageJSON || logMediaPackageXML || logWorkflowInstance)) {
94        logMediaPackageJSON = true;
95      }
96  
97      MediaPackage mediaPackage = workflowInstance.getMediaPackage();
98  
99      // Log media package as JSON
100     if (logMediaPackageJSON) {
101       String filename = String.format("workflow-%d-%d-mediapackage-%s.json", workflowInstance.getId(),
102                                       operation.getId(), mediaPackage.getIdentifier());
103       saveOrLog(MediaPackageParser.getAsJSON(mediaPackage), directory, filename);
104     }
105 
106     // Log media package as XML
107     if (logMediaPackageXML) {
108       String filename = String.format("workflow-%d-%d-mediapackage-%s.xml", workflowInstance.getId(),
109               operation.getId(), mediaPackage.getIdentifier());
110       saveOrLog(MediaPackageParser.getAsXml(mediaPackage), directory, filename);
111     }
112 
113     // Log workflow instance as XML
114     if (logWorkflowInstance) {
115       String filename = String.format("workflow-%d-%d.xml", workflowInstance.getId(), operation.getId());
116       try {
117         saveOrLog(XmlWorkflowParser.toXml(workflowInstance), directory, filename);
118       } catch (WorkflowParsingException e) {
119         throw new WorkflowOperationException(e);
120       }
121     }
122 
123     // Continue with unmodified media package
124     return createResult(mediaPackage, Action.CONTINUE);
125   }
126 
127   /**
128    * Log data to directory if set, otherwise use the system logger.
129    *
130    * @param data
131    *          Data to log
132    * @param directory
133    *          Directory to log to. If set to null, the default logger is used.
134    * @param filename
135    *          Filename to write the data to. Used by internal logger as well.
136    * @throws WorkflowOperationException
137    *          In case the data could not be written
138    */
139   private void saveOrLog(final String data, final File directory, final String filename)
140           throws WorkflowOperationException {
141     // Write to directory if set
142     if (directory != null) {
143       File file = new File(directory, filename);
144       try {
145         logger.info("Logging current workflow state to to {}", file.getAbsolutePath());
146         FileUtils.writeStringToFile(file, data, Charset.defaultCharset());
147       } catch (IOException e) {
148         throw new WorkflowOperationException(e);
149       }
150     } else {
151       // â€¦otherwise, just use the logger
152       logger.info("({}):\n{}", filename, data);
153     }
154   }
