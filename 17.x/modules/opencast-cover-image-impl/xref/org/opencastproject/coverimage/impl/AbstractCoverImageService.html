110   @Override
111   protected String process(Job job) throws Exception {
112 
113     List<String> arguments = job.getArguments();
114     String xml = arguments.get(0);
115     String xsl = arguments.get(1);
116     int width = Integer.valueOf(arguments.get(2));
117     int height = Integer.valueOf(arguments.get(3));
118     String posterImage = arguments.get(4);
119     String targetFlavor = arguments.get(5);
120 
121     Operation op = null;
122     op = Operation.valueOf(job.getOperation());
123     switch (op) {
124       case Generate:
125         Attachment result = generateCoverImageInternal(job, xml, xsl, width, height, posterImage, targetFlavor);
126         return MediaPackageElementParser.getAsXml(result);
127       default:
128         throw new IllegalStateException("Don't know how to handle operation '" + job.getOperation() + "'");
129     }
130   }
131 
132   @Override
133   public Job generateCoverImage(String xml, String xsl, String width, String height, String posterImageUri,
134           String targetFlavor) throws CoverImageException {
135 
136     // Null values are not passed to the arguments list
137     if (posterImageUri == null) {
138       posterImageUri = "";
139     }
140 
141     try {
142       return serviceRegistry.createJob(JOB_TYPE, Operation.Generate.toString(),
143           Arrays.asList(xml, xsl, width, height, posterImageUri, targetFlavor));
144     } catch (ServiceRegistryException e) {
145       throw new CoverImageException("Unable to create a job", e);
146     }
147   }
148 
149   protected Attachment generateCoverImageInternal(Job job, String xml, String xsl, int width, int height,
150       String posterImage, String targetFlavor) throws CoverImageException {
151 
152     URI result;
153     File tempSvg = null;
154     File tempPng = null;
155 
156     try (StringReader xmlReader =  new StringReader(xml)) {
157       Document xslDoc = parseXsl(xsl);
158 
159       // Create temp SVG file for transformation result
160       tempSvg = createTempFile(job, ".svg");
161       Result svg = new StreamResult(tempSvg);
162 
163       // Load Metadata (from resources)
164       InputSource xmlSource = new InputSource(xmlReader);
165 
166       // Transform XML metadata with stylesheet to SVG
167       transformSvg(svg, xmlSource, xslDoc, width, height, posterImage);
168 
169       // Rasterize SVG to PNG
170       tempPng = createTempFile(job, ".png");
171       rasterizeSvg(tempSvg, tempPng);
172 
173       try (FileInputStream in = new FileInputStream(tempPng)) {
174         result = workspace.putInCollection(COVERIMAGE_WORKSPACE_COLLECTION, job.getId() + "_coverimage.png", in);
175         log.debug("Put the cover image into the workspace ({})", result);
176       } catch (IOException e) {
177         log.warn("Error while putting resulting image into workspace collection '{}'",
178                 COVERIMAGE_WORKSPACE_COLLECTION, e);
179         throw new CoverImageException("Error while putting resulting image into workspace collection", e);
180       }
181     } finally {
182       FileUtils.deleteQuietly(tempSvg);
183       FileUtils.deleteQuietly(tempPng);
184       log.debug("Removed temporary files");
185     }
186 
187     return (Attachment) MediaPackageElementBuilderFactory.newInstance().newElementBuilder()
188         .elementFromURI(result, Type.Attachment, MediaPackageElementFlavor.parseFlavor(targetFlavor));
189   }
190 
191   protected static Document parseXsl(String xsl) throws CoverImageException {
192     if (StringUtils.isBlank(xsl)) {
193       throw new IllegalArgumentException("XSL string must not be empty");
194     }
195 
196     DocumentBuilderFactory dbFactory = XmlSafeParser.newDocumentBuilderFactory();
197 
198     dbFactory.setNamespaceAware(true);
199     Document xslDoc;
200     try {
201       log.debug("Parse given XSL to a org.w3c.dom.Document object");
202       DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
203       xslDoc = dBuilder.parse(new InputSource(new ByteArrayInputStream(xsl.getBytes("utf-8"))));
204     } catch (ParserConfigurationException e) {
205       // this should never happen...
206       throw new CoverImageException("The XSLT parser has serious configuration errors", e);
207     } catch (SAXException e) {
208       log.warn("Error while parsing the XSLT stylesheet: {}", e.getMessage());
209       throw new CoverImageException("Error while parsing the XSLT stylesheet", e);
210     } catch (IOException e) {
211       log.warn("Error while reading the XSLT stylesheet: {}", e.getMessage());
212       throw new CoverImageException("Error while reading the XSLT stylesheet", e);
213     }
214     return xslDoc;
215   }
216 
217   protected static void transformSvg(Result svg, InputSource xmlSource, Document xslDoc, int width, int height,
218       String posterImage) throws TransformerFactoryConfigurationError, CoverImageException {
219     if (svg == null || xmlSource == null || xslDoc == null) {
220       throw new IllegalArgumentException("Neither svg nor xmlSource nor xslDoc must be null");
221     }
222 
223     Thread thread = Thread.currentThread();
224     ClassLoader loader = thread.getContextClassLoader();
225     thread.setContextClassLoader(AbstractCoverImageService.class.getClassLoader());
226     try {
227       Transformer transformer;
228       // CHECKSTYLE:OFF
229       // Xalan Transformer can't be configured safely
230       // xslDoc is an already parsed Document
231       transformer = TransformerFactory
232           .newInstance("com.sun.org.apache.xalan.internal.xsltc.trax.TransformerFactoryImpl", null)
233           .newTransformer(new DOMSource(xslDoc));
234       // CHECKSTYLE:ON
235 
236       transformer.setParameter("width", width);
237       transformer.setParameter("height", height);
238       if (isNotBlank(posterImage)) {
239         transformer.setParameter("posterimage", posterImage);
240       }
241 
242       log.debug("Transform XML source to SVG");
243       // Xalan Transformer can't be configured safely
244       // Preparsing the XML with a safe parser
245       transformer.transform(new DOMSource(XmlSafeParser.parse(xmlSource)), svg);
246     } catch (TransformerConfigurationException e) {
247       // this should never happen...
248       throw new CoverImageException("The XSL transformer factory has serious configuration errors", e);
249     } catch (TransformerException | IOException | SAXException e) {
250       log.warn("Error while transforming SVG to image: {}", e.getMessage());
251       throw new CoverImageException("Error while transforming SVG to image", e);
252     } finally {
253       thread.setContextClassLoader(loader);
254     }
255   }

257   protected File createTempFile(Job job, String suffix) throws CoverImageException {
258     File tempFile;
259     try {
260       tempFile = File.createTempFile(COVERIMAGE_WORKSPACE_COLLECTION, Long.toString(job.getId()) + "_" + suffix);
261       log.debug("Created temporary file {}", tempFile);
262     } catch (IOException e) {
263       log.warn("Error creating temporary file:", e);
264       throw new CoverImageException("Error creating temporary file", e);
265     }
266     return tempFile;
267   }
268 
269   protected static void rasterizeSvg(File svgSource, File pngResult) throws CoverImageException {
270     SVGConverter converter = new SVGConverter();
271     converter.setDestinationType(DestinationType.PNG);
272     converter.setDst(pngResult);
273     converter.setSources(new String[] { svgSource.getAbsolutePath() });
274     try {
275       log.debug("Start converting SVG to PNG");
276       converter.execute();
277     } catch (SVGConverterException e) {
278       log.warn("Error while converting the SVG to a PNG: {}", e.getMessage());
279       throw new CoverImageException("Error while converting the SVG to a PNG", e);
280     }
281   }
