53  /**
54   * REST endpoint for {@link CoverImageService}
55   */
56  @Path("/cover-image")
57  @RestService(
58      name = "coverimage",
59      title = "Cover Image Service",
60      abstractText = "This endpoint triggers generation of cover images",
61      notes = {
62          "All paths above are relative to the REST endpoint base (something like http://your.server/files)",
63          "If the service is down or not working it will return a status 503, this means the the underlying service is "
64              + "not working and is either restarting or has failed",
65          "A status code 500 means a general failure has occurred which is not recoverable and was not anticipated. In "
66              + "other words, there is a bug! You should file an error report with your server logs from the time when "
67              + "the error occurred: <a href=\"https://github.com/opencast/opencast/issues\">Opencast Issue Tracker</a>"
68      }
69  )
70  @Component(
71      immediate = true,
72      service = CoverImageEndpoint.class,
73      property = {
74          "service.description=Cover Image Service REST Endpoint",
75          "opencast.service.type=org.opencastproject.coverimage",
76          "opencast.service.path=/cover-image",
77          "opencast.service.jobproducer=true"
78      }
79  )
80  @JaxrsResource
81  public class CoverImageEndpoint extends AbstractJobProducerEndpoint {
82  
83    /** Reference to the service registry service */
84    private ServiceRegistry serviceRegistry;
85  
86    /** Reference to the cover image service */
87    private CoverImageService coverImageService;
88  
89    /** The logging facility */
90    private static final Logger logger = LoggerFactory.getLogger(CoverImageEndpoint.class);
91  
92    @Override
93    public JobProducer getService() {
94      if (coverImageService instanceof JobProducer) {
95        return (JobProducer) coverImageService;
96      } else {
97        return null;
98      }
99    }
100 
101   @Override
102   public ServiceRegistry getServiceRegistry() {
103     return serviceRegistry;
104   }
105 
106   @POST
107   @Path("generate")
108   @Produces(MediaType.TEXT_XML)
109   @RestQuery(
110       name = "generate",
111       description = "Generates a cover image based on the given metadata",
112       restParameters = {
113           @RestParameter(description = "Metadata XML", isRequired = false, name = "xml", type = Type.TEXT),
114           @RestParameter(description = "XSLT stylesheet", isRequired = true, name = "xsl", type = Type.TEXT),
115           @RestParameter(
116               description = "Width of the cover image",
117               isRequired = true,
118               name = "width", type = Type.INTEGER, defaultValue = "1600"
119           ),
120           @RestParameter(
121               description = "Height of the cover image",
122               isRequired = true,
123               name = "height",
124               type = Type.INTEGER,
125               defaultValue = "900"
126           ),
127           @RestParameter(
128               description = "URI of poster image",
129               isRequired = false,
130               name = "posterimage",
131               type = Type.STRING
132           ),
133           @RestParameter(
134               description = "Flavor of target cover image",
135               isRequired = true,
136               name = "targetflavor",
137               type = Type.STRING,
138               defaultValue = "image/cover"
139           )
140       },
141       responses = {
142           @RestResponse(
143               description = "Results in an xml document containing the job for the cover image generation task",
144               responseCode = HttpServletResponse.SC_OK
145           ),
146           @RestResponse(
147               description = "If required parameters aren't set or not valid",
148               responseCode = HttpServletResponse.SC_BAD_REQUEST
149           )
150       },
151       returnDescription = ""
152   )
153   public Response generateCoverImage(
154       @FormParam("xml") String xml,
155       @FormParam("xsl") String xsl,
156       @FormParam("width") String width,
157       @FormParam("height") String height,
158       @FormParam("posterimage") String posterFlavor,
159       @FormParam("targetflavor") String targetFlavor
160   ) {
161     try {
162       Job job = coverImageService.generateCoverImage(xml, xsl, width, height, posterFlavor, targetFlavor);
163       return Response.ok().entity(new JaxbJob(job)).build();
164     } catch (CoverImageException e) {
165       logger.warn("Error while creating cover image job via REST endpoint: {}", e.getMessage());
166       return Response.status(Response.Status.INTERNAL_SERVER_ERROR).build();
167     }
168   }
169 
170   /**
171    * OSGi callback to set the a reference to the service registry.
172    *
173    * @param serviceRegistry
174    *          the service registry
175    */
176   @Reference
177   protected void setServiceRegistry(ServiceRegistry serviceRegistry) {
178     this.serviceRegistry = serviceRegistry;
179   }
180 
181   /**
182    * OSGi callback to set the a reference to the cover image service.
183    *
184    * @param coverImageService
185    */
186   @Reference
187   protected void setCoverImageService(CoverImageService coverImageService) {
188     this.coverImageService = coverImageService;
189   }
190 
191   /**
192    * Callback from OSGi that is called when this service is activated.
193    *
194    * @param cc
195    *          OSGi component context
196    */
197   @Activate
198   protected void activate(ComponentContext cc) {
199     logger.info("Cover Image REST Endpoint started");
200   }

202 }
