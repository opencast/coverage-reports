251   @Test
252   public void testWorkflowWithSecurityPolicy() throws Exception {
253 
254     // Create an ACL for the authorization service to return
255     AccessControlList acl = new AccessControlList();
256     acl.getEntries().add(new AccessControlEntry("ROLE_INSTRUCTOR", Permissions.Action.READ.toString(), true));
257     acl.getEntries().add(new AccessControlEntry("ROLE_INSTRUCTOR", Permissions.Action.WRITE.toString(), true));
258 
259     // Mock up an authorization service that always returns "true" for hasPermission()
260     AuthorizationService authzService = EasyMock.createNiceMock(AuthorizationService.class);
261     EasyMock.expect(authzService.getActiveAcl((MediaPackage) EasyMock.anyObject()))
262             .andReturn(Tuple.tuple(acl, AclScope.Series)).anyTimes();
263     EasyMock.expect(authzService.hasPermission((MediaPackage) EasyMock.anyObject(), (String) EasyMock.anyObject()))
264             .andReturn(true).anyTimes();
265     EasyMock.replay(authzService);
266     service.setAuthorizationService(authzService);
267 
268     // Create the workflow and its dependent object graph
269     WorkflowDefinitionImpl def = new WorkflowDefinitionImpl();
270     def.add(new WorkflowOperationDefinitionImpl("op1", "op1", null, true));
271     MediaPackage mp = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder().createNew();
272 
273     // As an instructor, create a workflow. We don't care if it passes or fails. We just care about access to it.
274     userResponder.setResponse(instructor1);
275     WorkflowInstance workflow = service.start(def, mp);
276     service.suspend(workflow.getId());
277 
278     // Ensure that this instructor can access the workflow
279     try {
280       service.getWorkflowById(workflow.getId());
281     } catch (Exception e) {
282       fail(e.getMessage());
283     }
284 
285     // Ensure the organization admin can access that workflow
286     userResponder.setResponse(DEFAULT_ORG_ADMIN);
287     try {
288       service.getWorkflowById(workflow.getId());
289     } catch (Exception e) {
290       fail(e.getMessage());
291     }
292 
293     // Ensure the global admin can access that workflow
294     userResponder.setResponse(globalAdmin);
295     try {
296       service.getWorkflowById(workflow.getId());
297     } catch (Exception e) {
298       fail(e.getMessage());
299     }
300 
301     // Ensure the other instructor from this organization can also see the workflow, since this is specified in the
302     // security policy
303     userResponder.setResponse(instructor2);
304     try {
305       service.getWorkflowById(workflow.getId());
306     } catch (Exception e) {
307       fail(e.getMessage());
308     }
309 
310     // TODO change to answer show in episode or series how to do it. Cool stuff
311 
312     // Ensure the instructor from a different org can not see the workflow, even though they share the same role
313     organizationResponder.setResponse(otherOrganization);
314     userResponder.setResponse(instructorFromDifferentOrg);
315     try {
316       service.getWorkflowById(workflow.getId());
317       fail();
318     } catch (Exception e) {
319       // expected
320     }
321   }

323   @Test
324   public void testWorkflowWithoutSecurityPolicy() throws Exception {
325     // Mock up an authorization service that always returns "false" for hasPermission()
326     AuthorizationService authzService = EasyMock.createNiceMock(AuthorizationService.class);
327     EasyMock.expect(authzService.getActiveAcl((MediaPackage) EasyMock.anyObject()))
328             .andReturn(Tuple.tuple(new AccessControlList(), AclScope.Series)).anyTimes();
329     EasyMock.expect(authzService.hasPermission((MediaPackage) EasyMock.anyObject(), (String) EasyMock.anyObject()))
330             .andReturn(false).anyTimes();
331     EasyMock.replay(authzService);
332     service.setAuthorizationService(authzService);
333 
334     // Create the workflow and its dependent object graph
335     WorkflowDefinitionImpl def = new WorkflowDefinitionImpl();
336     def.add(new WorkflowOperationDefinitionImpl("op1", "op1", null, true));
337     MediaPackage mp = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder().createNew();
338 
339     // As an instructor, create a workflow
340     userResponder.setResponse(instructor1);
341     WorkflowInstance workflow = service.start(def, mp);
342     service.suspend(workflow.getId());
343 
344     // Ensure that this instructor can access the workflow
345     try {
346       service.getWorkflowById(workflow.getId());
347     } catch (Exception e) {
348       fail(e.getMessage());
349     }
350 
351     // Ensure the organization admin can access that workflow
352     userResponder.setResponse(DEFAULT_ORG_ADMIN);
353     try {
354       service.getWorkflowById(workflow.getId());
355     } catch (Exception e) {
356       fail(e.getMessage());
357     }
358 
359     // Ensure the global admin can access that workflow
360     userResponder.setResponse(globalAdmin);
361     try {
362       service.getWorkflowById(workflow.getId());
363     } catch (Exception e) {
364       fail(e.getMessage());
365     }
366 
367     // Ensure the other instructor can not see the workflow, since there is no security policy granting access
368     userResponder.setResponse(instructor2);
369     try {
370       service.getWorkflowById(workflow.getId());
371       fail();
372     } catch (UnauthorizedException e) {
373       // expected
374     }
375 
376     // Ensure the instructor from a different org can not see the workflow, even though they share a role
377     organizationResponder.setResponse(otherOrganization);
378     userResponder.setResponse(instructorFromDifferentOrg);
379     try {
380       service.getWorkflowById(workflow.getId());
381       fail();
382     } catch (Exception e) {
383       // expected
384     }
