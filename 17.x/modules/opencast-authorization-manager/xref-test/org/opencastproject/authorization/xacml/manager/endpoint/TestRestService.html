108     EasyMock.expect(seriesSearchResult.getItems()).andReturn(new SearchResultItem[] {}).anyTimes();

156   private static AclDb newAclPersistence() {
157     JpaAclDb db = new JpaAclDb();

161     return db;

164   private static AuthorizationService newAuthorizationService() {
165     AccessControlList acl = new AccessControlList();
166     Attachment attachment = new AttachmentImpl();
167     MediaPackage mediapackage;
168     mediapackage = new MediaPackageBuilderImpl().createNew();
169     AuthorizationService authorizationService = EasyMock.createNiceMock(AuthorizationService.class);
170     EasyMock.expect(authorizationService.getActiveAcl((MediaPackage) EasyMock.anyObject()))
171             .andReturn(Tuple.tuple(acl, AclScope.Series)).anyTimes();
172     try {
173       EasyMock.expect(authorizationService.setAcl(
174               EasyMock.anyObject(MediaPackage.class),
175               EasyMock.anyObject(AclScope.class),
176               EasyMock.anyObject(AccessControlList.class)))
177               .andReturn(Tuple.tuple(mediapackage, attachment));
178     } catch (MediaPackageException e) {
179       throw new RuntimeException(e);
180     }
181     EasyMock.replay(authorizationService);
182 
183     return authorizationService;
184   }

186   private static AssetManager newAssetManager() {
187     Snapshot snapshot = EasyMock.createNiceMock(Snapshot.class);
188     EasyMock.expect(snapshot.getMediaPackage()).andReturn(new MediaPackageBuilderImpl().createNew()).anyTimes();
189     ARecord record = EasyMock.createNiceMock(ARecord.class);
190     LinkedHashSet<ARecord> records = new LinkedHashSet<>();
191     records.add(record);
192     EasyMock.expect(record.getSnapshot()).andReturn(Optional.of(snapshot)).anyTimes();
193 
194     AResult result = EasyMock.createNiceMock(AResult.class);
195     EasyMock.expect(result.getRecords()).andReturn(records).anyTimes();
196 
197     ASelectQuery select = EasyMock.createNiceMock(ASelectQuery.class);
198     EasyMock.expect(select.where(EasyMock.anyObject(Predicate.class))).andReturn(select).anyTimes();
199     EasyMock.expect(select.run()).andReturn(result).anyTimes();
200 
201     Predicate predicate = EasyMock.createNiceMock(Predicate.class);
202     EasyMock.expect(predicate.and(EasyMock.anyObject(Predicate.class))).andReturn(predicate).anyTimes();
203 
204     AQueryBuilder query = EasyMock.createNiceMock(AQueryBuilder.class);
205 
206     VersionField version = EasyMock.createNiceMock(VersionField.class);
207 
208     EasyMock.expect(query.version()).andReturn(version).anyTimes();
209     EasyMock.expect(query.mediaPackageId(EasyMock.anyString())).andReturn(predicate).anyTimes();
210     EasyMock.expect(query.select(EasyMock.anyObject(Target.class))).andReturn(select).anyTimes();
211 
212     AssetManager assetManager = EasyMock.createNiceMock(AssetManager.class);
213     EasyMock.expect(assetManager.getMediaPackage(EasyMock.anyString())).andReturn(Optional.of(
214         new MediaPackageBuilderImpl().createNew())).anyTimes();
215     EasyMock.expect(assetManager.createQuery()).andReturn(query).anyTimes();
216     EasyMock.replay(assetManager, version, query, predicate, select, result, record, snapshot);
217     return assetManager;
218   }
