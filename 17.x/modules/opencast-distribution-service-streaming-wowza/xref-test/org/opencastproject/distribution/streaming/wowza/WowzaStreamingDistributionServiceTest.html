258   @Test
259   public void testIncorrectStreamingURLs() throws Exception {
260     setUpDefault();
261 
262     final String[] inputStreamingUrls = new String[] {
263         "incorrect url",
264         "noschema.myserver.com/my/path/to/server",
265         "http://withhttp.example.com/path",
266         "https://withhttps.testing.com/this/is/a/path",
267         "rtmp://withrtmp.test.ext/another/path",
268         "rtmps://withrtmps.anothertest.test/path/to/server",
269         "other://withotherschema.test/mypath"
270     };
271 
272     final String[] outputStreamingUrls = new String[] {
273         null,
274         "http://noschema.myserver.com:10/my/path/to/server",
275         "http://withhttp.example.com:10/path",
276         "https://withhttps.testing.com:10/this/is/a/path",
277         null,
278         null,
279         null
280     };
281 
282     map.put(defaultPortProperty, port);
283     streamingService.activate(bundleContext, map);
284 
285     // Test service instance
286     for (int i = 0; i < inputStreamingUrls.length; i++) {
287 
288       map.put(defaultUrlProperty, inputStreamingUrls[i]);
289 
290       if (outputStreamingUrls[i] == null) {
291         try {
292           streamingService.modified(bundleContext, map);
293           fail("Should fail on invalid streaming URL");
294         } catch (ComponentException e) {
295           // expected
296         }
297       } else {
298         streamingService.activate(bundleContext, map);
299         assertEquals(new URI(outputStreamingUrls[i]), streamingService.streamingUrls.get(defaultTenant));
300       }
301     }
302   }
