111   @Override
112   public Map<String, String> getList(String listName, ResourceListQuery query)
113           throws ListProviderException {
114     SeriesSearchQuery seriesQuery = toSearchQuery(query);
115     Map<String, String> result = new HashMap<>();
116     if (TITLE.equals(listName)) {
117       seriesQuery.sortByTitle(SortCriterion.Order.Ascending);
118       for (String title : searchIndex.getTermsForField(SeriesIndexSchema.TITLE, Series.DOCUMENT_TYPE)) {
119         result.put(title, title);
120       }
121     } else if (CONTRIBUTORS.equals(listName)) {
122       seriesQuery.sortByContributors(SortCriterion.Order.Ascending);
123       for (String contributor : searchIndex.getTermsForField(SeriesIndexSchema.CONTRIBUTORS, Series.DOCUMENT_TYPE)) {
124         result.put(contributor, contributor);
125       }
126     } else if (ORGANIZERS.equals(listName)) {
127       seriesQuery.sortByOrganizers(SortCriterion.Order.Ascending);
128       for (String organizer : searchIndex.getTermsForField(SeriesIndexSchema.ORGANIZERS, Series.DOCUMENT_TYPE)) {
129         result.put(organizer, organizer);
130       }
131     } else {
132       try {
133         seriesQuery.sortByTitle(SortCriterion.Order.Ascending);
134         seriesQuery.sortByCreatedDateTime(SortCriterion.Order.Descending);
135         seriesQuery.sortByOrganizers(SortCriterion.Order.Ascending);
136         if (SERIES_WRITE_ONLY.equals(listName)) {
137           seriesQuery.withAction(Permissions.Action.WRITE);
138         }
139         SearchResult<Series> searchResult = searchIndex.getByQuery(seriesQuery);
140         Calendar calendar = Calendar.getInstance();
141         //We might have duplicate series names, so let's count them and see
142         Map<String, Long> duplicates = Arrays.stream(searchResult.getItems())
143             .map(series -> series.getSource().getTitle())
144             .collect(Collectors.groupingBy(Function.identity(), Collectors.counting()));
145         for (SearchResultItem<Series> item : searchResult.getItems()) {
146           Series s = item.getSource();
147           if (TITLE_EXTENDED.equals(listName)) {
148             Date created = s.getCreatedDateTime();
149             List<String> organizers = s.getOrganizers();
150             StringBuilder sb = new StringBuilder(s.getTitle());
151             if (created != null || (organizers != null && !organizers.isEmpty())) {
152               List<String> extendedTitleData = new ArrayList<>();
153               if (created != null) {
154                 calendar.setTime(created);
155                 extendedTitleData.add(Integer.toString(calendar.get(Calendar.YEAR)));
156               }
157               if (organizers != null && !organizers.isEmpty())
158                 extendedTitleData.addAll(organizers);
159               sb.append(" (").append(StringUtils.join(extendedTitleData, ", ")).append(")");
160             }
161             result.put(s.getIdentifier(), sb.toString());
162           } else if (PROVIDER_PREFIX.equals(listName)) {
163             String newSeriesName = s.getTitle();
164             if (duplicates.get(newSeriesName) > 1L) {
165               //If a series name is repeated, will add the first 7 characters of the series ID to the display name on the
166               //admin-ui
167               if (s.getIdentifier().length() > 8) {
168                 newSeriesName += " " + "(ID: " + s.getIdentifier().substring(0, 8) + "...)";
169               } else {
170                 newSeriesName += " " + "(ID: " + s.getIdentifier() + ")";
171               }
172               logger.trace(String.format("Repeated series title \"%s\" found, changing to \"%s\" for admin-ui display",
173                   s.getTitle(), newSeriesName));
174             }
175             result.put(s.getIdentifier(), newSeriesName);
176           } else {
177             result.put(s.getIdentifier(), s.getTitle());
178           }
179         }
180       } catch (SearchIndexException e) {
181         logger.warn("Unable to query series.", e);
182       }
183     }
184     return result;
185   }

197   /**
198    * Creates a series search query from resource list query.
199    *
200    * @param query a resource list query
201    * @return a series search query
202    */
203   protected SeriesSearchQuery toSearchQuery(ResourceListQuery query) {
204     SeriesSearchQuery seriesQuery = new SeriesSearchQuery(securityService.getOrganization().getId(), securityService.getUser());
205     if (query.getLimit().isSome()) {
206       seriesQuery.withLimit(query.getLimit().get());
207     }
208     if (query.getOffset().isSome()) {
209       seriesQuery.withOffset(query.getOffset().get());
210     }
211     if (query instanceof SeriesListQuery) {
212       if (((SeriesListQuery) query).getReadPermission().isSome()
213           || ((SeriesListQuery) query).getWritePermission().isSome()) {
214         seriesQuery.withoutActions();
215         if (((SeriesListQuery) query).getReadPermission().getOrElse(true)) {
216           seriesQuery.withAction(Permissions.Action.READ);
217         }
218         if (((SeriesListQuery) query).getWritePermission().getOrElse(false)) {
219           seriesQuery.withAction(Permissions.Action.WRITE);
220         }
221       }
222       for (ResourceListFilter filter : query.getFilters()) {
223         if (filter.getValue().isNone()) {
224           continue;
225         } else if (SeriesListQuery.FILTER_CREATIONDATE_NAME.equals(filter.getName())) {
226           Tuple<Date, Date> creationDate = (Tuple<Date, Date>) filter.getValue().get();
227           if (creationDate.getA() != null) {
228             seriesQuery.withCreatedFrom(creationDate.getA());
229           }
230           if (creationDate.getB() != null) {
231             seriesQuery.withCreatedTo(creationDate.getB());
232           }
233         } else if (SeriesListQuery.FILTER_CREATOR_NAME.equals(filter.getName())) {
234           seriesQuery.withCreator((String)filter.getValue().get());
235         } else if (SeriesListQuery.FILTER_CONTRIBUTORS_NAME.equals(filter.getName())) {
236           seriesQuery.withContributor((String)filter.getValue().get());
237         } else if (SeriesListQuery.FILTER_LANGUAGE_NAME.equals(filter.getName())) {
238           seriesQuery.withLanguage((String)filter.getValue().get());
239         } else if (SeriesListQuery.FILTER_LICENSE_NAME.equals(filter.getName())) {
240           seriesQuery.withLicense((String)filter.getValue().get());
241         } else if (SeriesListQuery.FILTER_ORGANIZERS_NAME.equals(filter.getName())) {
242           seriesQuery.withOrganizer((String)filter.getValue().get());
243         } else if (SeriesListQuery.FILTER_SUBJECT_NAME.equals(filter.getName())) {
244           seriesQuery.withSubject((String)filter.getValue().get());
245         } else if (SeriesListQuery.FILTER_TEXT_NAME.equals(filter.getName())) {
246           seriesQuery.withText((String)filter.getValue().get());
247         } else if (SeriesListQuery.FILTER_TITLE_NAME.equals(filter.getName())) {
248           seriesQuery.withTitle((String)filter.getValue().get());
249         }
250       }
251     }
252     return seriesQuery;
253   }
