36  /**
37   * Query for the series list.
38   *
39   * The following filters can be used:
40   * <ul>
41   * <li>contributors</li>
42   * <li>subject</li>
43   * <li>language</li>
44   * <li>creator</li>
45   * <li>license</li>
46   * <li>access policy</li>
47   * <li>creation date</li>
48   * </ul>
49   */
50  public class SeriesListQuery extends ResourceListQueryImpl {
51  
52    public static final String FILTER_ACL_NAME = "managedAcl";
53    private static final String FILTER_ACL_LABEL = "FILTERS.SERIES.ACCESS_POLICY.LABEL";
54  
55    public static final String FILTER_ACL_PERMISSION_READ_NAME = "acl_permission_read";

57

87    public SeriesListQuery() {
88      super();
89      this.availableFilters.add(createCreationDateFilter(Option.<Tuple<Date, Date>> none()));
90      this.availableFilters.add(createOrganizersFilter(Option.none()));
91      this.availableFilters.add(createContributorsFilter(Option.none()));
92    }
93  
94    /**
95     * Add a {@link ResourceListFilter} filter to the query with the given acl
96     *
97     * @param acl
98     *          the acl to filter for
99     */
100   public void withAccessPolicy(String acl) {
101     this.addFilter(createAccessPolicyFilter(Option.option(acl)));
102   }
103 
104   /**
105    * Returns an {@link Option} containing the acl used to filter if set
106    *
107    * @return an {@link Option} containing the acl or none.
108    */
109   public Option<String> getAccessPolicy() {
110     return this.getFilterValue(FILTER_ACL_NAME);
111   }
112 
113   public void withoutPermissions() {
114     ResourceListFilter filter = this.getFilter(FILTER_ACL_PERMISSION_READ_NAME);
115     if (filter != null) {
116       this.removeFilter(filter);
117     }
118     filter = this.getFilter(FILTER_ACL_PERMISSION_WRITE_NAME);
119     if (filter != null) {
120       this.removeFilter(filter);
121     }
122   }
123 
124   public void withReadPermission(boolean value) {
125     this.addFilter(createReadPermissionFilter(Option.option(value)));
126   }
127 
128   public void withWritePermission(boolean value) {
129     this.addFilter(createWritePermissionFilter(Option.option(value)));
130   }
131

134   }

292   /**
293    * Create a new {@link ResourceListFilter} based on access policy
294    *
295    * @param acl
296    *          the acl to filter on wrapped in an {@link Option} or {@link Option#none()}
297    * @return a new {@link ResourceListFilter} for access policy based query
298    */
299   public static ResourceListFilter<String> createAccessPolicyFilter(Option<String> acl) {
300     return FiltersUtils.generateFilter(acl, FILTER_ACL_NAME, FILTER_ACL_LABEL, SourceType.SELECT,
301             Option.some(AclListProvider.NAME));
302   }

304   public static ResourceListFilter<Boolean> createReadPermissionFilter(Option<Boolean> value) {

306         SourceType.SELECT, Option.none());
307   }
