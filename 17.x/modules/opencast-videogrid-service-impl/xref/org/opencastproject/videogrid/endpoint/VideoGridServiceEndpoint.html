63  /**
64   * A service endpoint to expose the {@link VideoGridService} via REST.
65   */
66  @Path("/videogrid")
67  @RestService(
68      name = "videogrid",
69      title = "VideoGrid Service",
70      abstractText = "The Video Grid Service creates a section of the final video grid file for the "
71          + "Video Grid Workflow Operation Handler.",
72      notes = {
73          "Only offers one service",
74          "Only meant to be used with the VideoGridWorkflowOperationHandler"
75      }
76  )
77  @Component(
78      immediate = true,
79      service = VideoGridServiceEndpoint.class,
80      property = {
81          "service.description=VideoGrid Service REST Endpoint",
82          "opencast.service.type=org.opencastproject.videogrid",
83          "opencast.service.path=/videogrid",
84          "opencast.service.jobproducer=true"
85      }
86  )
87  @JaxrsResource
88  public class VideoGridServiceEndpoint extends AbstractJobProducerEndpoint {
89  
90    /** The logger */
91    private static final Logger logger = LoggerFactory.getLogger(VideoGridServiceEndpoint.class);
92  
93    /** The videogrid service */
94    private VideoGridService videoGridService;
95  
96    /** The service registry */
97    private ServiceRegistry serviceRegistry = null;
98  
99    /**
100    * Callback from the OSGi declarative services to set the service registry.
101    *
102    * @param serviceRegistry
103    *          the service registry
104    */
105   @Reference
106   protected void setServiceRegistry(ServiceRegistry serviceRegistry) {
107     this.serviceRegistry = serviceRegistry;
108   }
109 
110   /**
111    * Sets the videoGridService
112    *
113    * @param videoGridService
114    *          the videoGridService
115    */
116   @Reference
117   public void setVideoGridService(VideoGridService videoGridService) {
118     this.videoGridService = videoGridService;
119   }
120 
121   @POST
122   @Produces(MediaType.TEXT_XML)
123   @Path("videogrid")
124   @RestQuery(
125       name = "videogrid",
126       description = "Create a section of the final video grid",
127       restParameters = {
128           @RestParameter(
129               name = "command",
130               isRequired = true,
131               type = STRING,
132               description = "An ffmpeg command as a list. File paths are replaced by track identifiers."
133           ),
134           @RestParameter(
135               name = "tracks",
136               isRequired = true,
137               type = RestParameter.Type.TEXT,
138               description = "The source tracks to concat as XML"
139           )
140       },
141       responses = {
142           @RestResponse(description = "Video created successfully", responseCode = HttpServletResponse.SC_OK),
143           @RestResponse(description = "Invalid data", responseCode = HttpServletResponse.SC_BAD_REQUEST),
144           @RestResponse(description = "Internal error", responseCode = HttpServletResponse.SC_INTERNAL_SERVER_ERROR)
145       },
146       returnDescription = "Returns the path to the generated video for the grid"
147   )
148   public Response createPartialTrack(
149           @FormParam("command") String commandString,
150           @FormParam("sourceTracks") String tracksXml) throws MediaPackageException, EncoderException {
151     Gson gson = new Gson();
152     try {
153       List<String> command = gson.fromJson(commandString, new TypeToken<List<String>>() { }.getType());
154       List<? extends MediaPackageElement> tracks = MediaPackageElementParser.getArrayFromXml(tracksXml);
155       logger.debug("Start videogrid");
156       Job job = videoGridService.createPartialTrack(command, tracks.toArray(new Track[tracks.size()]));
157       return Response.ok(new JaxbJob(job)).build();
158     } catch (JsonSyntaxException | NullPointerException e) {
159       logger.debug("Invalid data passed to REST endpoint:\ncommand: {}", commandString);
160       return Response.status(Response.Status.BAD_REQUEST).build();
161     } catch (VideoGridServiceException e) {
162       logger.error("Error generating videos {}", commandString, e);
163       return Response.serverError().build();
164     }
165   }
166 
167   /**
168    * {@inheritDoc}
169    *
170    * @see org.opencastproject.rest.AbstractJobProducerEndpoint#getService()
171    */
172   @Override
173   public JobProducer getService() {
174     if (videoGridService instanceof JobProducer) {
175       logger.debug("get videogrid service");
176       return (JobProducer) videoGridService;
177     }
178     return null;
179   }

188     return serviceRegistry;
189   }
