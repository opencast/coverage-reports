21  package org.opencastproject.caption.converters;
22  
23  import org.opencastproject.caption.api.Caption;
24  import org.opencastproject.caption.impl.CaptionImpl;
25  import org.opencastproject.caption.impl.TimeImpl;
26  
27  import org.apache.commons.io.IOUtils;
28  import org.junit.After;
29  import org.junit.Assert;
30  import org.junit.Before;
31  import org.junit.Test;
32  
33  import java.io.ByteArrayOutputStream;
34  import java.io.IOException;
35  import java.util.ArrayList;
36  import java.util.List;
37  
38  /**
39   * Test class for WebVTT format.
40   *
41   */
42  public class WebVttCaptionConveterTest {
43  
44    private WebVttCaptionConverter format;
45    private ByteArrayOutputStream outputStream;
46  
47    private static final String CAPTION_LINE = "This is caption testing line ";
48    // Expected output
49    private String expectedOutput = "WEBVTT\n\n00:00:49.520 --> 00:00:52.961\n" + CAPTION_LINE
50            + "1.\n\n00:00:54.123 --> 00:00:56.456\n" + CAPTION_LINE + "2.\n\n";
51  
52    @Before
53    public void setUp() throws IOException {
54      format = new WebVttCaptionConverter();
55      outputStream = new ByteArrayOutputStream();
56    }
57  
58    @After
59    public void tearDown() throws IOException {
60      IOUtils.closeQuietly(outputStream);
61    }
62  
63    @Test
64    public void testExport() throws Exception {
65      List<Caption> captionList = new ArrayList<Caption>();
66      String[] captionLines1 = new String[1];
67      captionLines1[0] = CAPTION_LINE + "1.";
68      captionList.add(new CaptionImpl(new TimeImpl(0, 0, 49, 520), new TimeImpl(0, 0, 52, 961), captionLines1));
69      String[] captionLines2 = new String[1];
70      captionLines2[0] = CAPTION_LINE + "2.";
71      captionList.add(new CaptionImpl(new TimeImpl(0, 0, 54, 123), new TimeImpl(0, 0, 56, 456), captionLines2));
72  
73      format.exportCaption(outputStream, captionList, null);
74      Assert.assertTrue(outputStream.toString("UTF-8").equals(expectedOutput));
75    }
76  }
