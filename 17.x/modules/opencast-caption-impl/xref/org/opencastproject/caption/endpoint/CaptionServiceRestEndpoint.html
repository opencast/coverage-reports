148   /**
149    * Convert captions in catalog from one format to another.
150    *
151    * @param inputType
152    *          input format
153    * @param outputType
154    *          output format
155    * @param catalogAsXml
156    *          catalog containing captions
157    * @param lang
158    *          caption language
159    * @return a Response containing receipt of for conversion
160    */
161   @POST
162   @Path("convert")
163   @Produces(MediaType.TEXT_XML)
164   @RestQuery(name = "convert", description = "Convert captions from one format to another.", restParameters = {
165           @RestParameter(description = "Captions to be converted.", isRequired = true, name = "captions", type = RestParameter.Type.TEXT),
166           @RestParameter(description = "Caption input format (for example: dfxp, subrip,...).", isRequired = false, defaultValue = "dfxp", name = "input", type = RestParameter.Type.STRING),
167           @RestParameter(description = "Caption output format (for example: dfxp, subrip,...).", isRequired = false, defaultValue = "subrip", name = "output", type = RestParameter.Type.STRING),
168           @RestParameter(description = "Caption language (for those formats that store such information).", isRequired = false, defaultValue = "en", name = "language", type = RestParameter.Type.STRING) }, responses = { @RestResponse(description = "OK, Conversion successfully completed.", responseCode = HttpServletResponse.SC_OK) }, returnDescription = "The converted captions file")
169   public Response convert(@FormParam("input") String inputType, @FormParam("output") String outputType,
170           @FormParam("captions") String catalogAsXml, @FormParam("language") String lang) {
171     MediaPackageElement element;
172     try {
173       element = MediaPackageElementParser.getFromXml(catalogAsXml);
174       if (!Catalog.TYPE.equals(element.getElementType()))
175         return Response.status(Response.Status.BAD_REQUEST).entity("Captions must be of type catalog.").build();
176     } catch (Exception e) {
177       logger.info("Unable to parse serialized captions");
178       return Response.status(Response.Status.BAD_REQUEST).build();
179     }
180 
181     try {
182       Job job;
183       if (StringUtils.isNotBlank(lang)) {
184         job = service.convert((Catalog) element, inputType, outputType, lang);
185       } else {
186         job = service.convert((Catalog) element, inputType, outputType);
187       }
188       return Response.ok().entity(new JaxbJob(job)).build();
189     } catch (Exception e) {
190       logger.error("Unable to convert captions: {}", e.getMessage());
191       return Response.status(Response.Status.INTERNAL_SERVER_ERROR).build();
192     }
193   }

195   /**
196    * Parses captions in catalog for language information.
197    *
198    * @param inputType
199    *          caption format
200    * @param catalogAsXml
201    *          catalog containing captions
202    * @return a Response containing XML with language information
203    */
204   @POST
205   @Path("languages")
206   @Produces(MediaType.TEXT_XML)
207   @RestQuery(name = "languages", description = "Get information about languages in caption catalog (if such information is available).", restParameters = {
208           @RestParameter(description = "Captions to be examined.", isRequired = true, name = "captions", type = RestParameter.Type.TEXT),
209           @RestParameter(description = "Caption input format (for example: dfxp, subrip,...).", isRequired = false, defaultValue = "dfxp", name = "input", type = RestParameter.Type.STRING) }, responses = { @RestResponse(description = "OK, information was extracted and retrieved", responseCode = HttpServletResponse.SC_OK) }, returnDescription = "Returned information about languages present in captions.")
210   public Response languages(@FormParam("input") String inputType, @FormParam("captions") String catalogAsXml) {
211     try {
212       MediaPackageElement element = MediaPackageElementParser.getFromXml(catalogAsXml);
213       if (!Catalog.TYPE.equals(element.getElementType())) {
214         return Response.status(Response.Status.BAD_REQUEST).entity("Captions must be of type catalog").build();
215       }
216 
217       String[] languageArray = service.getLanguageList((Catalog) element, inputType);
218 
219       // build response
220       DocumentBuilder docBuilder = XmlSafeParser.newDocumentBuilderFactory().newDocumentBuilder();
221       Document doc = docBuilder.newDocument();
222       Element root = doc.createElement("languages");
223       root.setAttribute("type", inputType);
224       root.setAttribute("url", element.getURI().toString());
225       for (String lang : languageArray) {
226         Element language = doc.createElement("language");
227         language.appendChild(doc.createTextNode(lang));
228         root.appendChild(language);
229       }
230 
231       DOMSource domSource = new DOMSource(root);
232       StringWriter writer = new StringWriter();
233       StreamResult result = new StreamResult(writer);
234       Transformer transformer = XmlSafeParser.newTransformerFactory().newTransformer();
235       transformer.transform(domSource, result);
236 
237       return Response.status(Response.Status.OK).entity(writer.toString()).build();
238     } catch (Exception e) {
239       logger.error("Unable to parse captions: {}", e.getMessage());
240       return Response.status(Response.Status.INTERNAL_SERVER_ERROR).build();
241     }
242   }
