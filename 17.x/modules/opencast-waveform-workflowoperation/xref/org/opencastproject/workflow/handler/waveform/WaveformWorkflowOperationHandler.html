58  /**
59   * Workflow operation for the waveform service.
60   */
61  @Component(
62      immediate = true,
63      name = "org.opencastproject.workflow.handler.videoeditor.WaveformWorkflowOperationHandler",
64      service = WorkflowOperationHandler.class,
65      property = {
66          "service.description=Waveform Workflow Operation Handler",
67          "workflow.operation=waveform"
68      }
69  )
70  public class WaveformWorkflowOperationHandler extends AbstractWorkflowOperationHandler {
71    private static final Logger logger = LoggerFactory.getLogger(WaveformWorkflowOperationHandler.class);
72  
73    /** Source flavor configuration property name. */
74    private static final String SOURCE_FLAVOR_PROPERTY = "source-flavor";
75  
76    /** Source tags configuration property name. */
77    private static final String SOURCE_TAGS_PROPERTY = "source-tags";
78  
79    /** Default value of pixel per minute configuration. */
80    private static final int DEFAULT_PIXELS_PER_MINUTE = 200;
81  
82    /** Default value of minimum width configuration. */
83    private static final int DEFAULT_MIN_WIDTH = 5000;
84  
85    /** Default value of maximum width configuration. */
86    private static final int DEFAULT_MAX_WIDTH = 20000;
87  
88    /** Default value of height configuration. */
89    private static final int DEFAULT_HEIGHT = 500;
90  
91    /** Pixel per minute of waveform image width configuration property name. */
92    private static final String PIXELS_PER_MINUTE_PROPERTY = "pixels-per-minute";
93  
94    /** Minimun width of waveform image configuration property name. */
95    private static final String MIN_WIDTH_PROPERTY = "min-width";
96  
97    /** Maximum width of waveform image configuration property name. */
98    private static final String MAX_WIDTH_PROPERTY = "max-width";
99  
100   /** Height of waveform image configuration property name. */
101   private static final String HEIGHT_PROPERTY = "height";
102 
103   /** Color of waveform image configuration property name. */
104   private static final String COLOR_PROPERTY = "color";
105 
106   /** The waveform service. */
107   private WaveformService waveformService = null;
108 
109   /** The workspace service. */
110   private Workspace workspace = null;
111 
112   @Override
113   public void activate(ComponentContext cc) {
114     super.activate(cc);
115     logger.info("Registering waveform workflow operation handler");
116   }
117 
118   /**
119    * {@inheritDoc}
120    *
121    * @see
122    * org.opencastproject.workflow.api.WorkflowOperationHandler#start(org.opencastproject.workflow.api.WorkflowInstance,
123    * org.opencastproject.job.api.JobContext)
124    */
125   @Override
126   public WorkflowOperationResult start(
127       WorkflowInstance workflowInstance, JobContext context
128   ) throws WorkflowOperationException {
129 
130     MediaPackage mediaPackage = workflowInstance.getMediaPackage();
131     logger.info("Start waveform workflow operation for mediapackage {}", mediaPackage);
132 
133     ConfiguredTagsAndFlavors tagsAndFlavors = getTagsAndFlavors(workflowInstance,
134         Configuration.many, Configuration.many, Configuration.many, Configuration.one);
135 
136     List<MediaPackageElementFlavor> sourceFlavorList = tagsAndFlavors.getSrcFlavors();
137     List<String> sourceTagList = tagsAndFlavors.getSrcTags();
138     if (sourceFlavorList.isEmpty() && sourceTagList.isEmpty()) {
139       throw new WorkflowOperationException(
140           String.format("Required property %s or %s not set", SOURCE_FLAVOR_PROPERTY, SOURCE_TAGS_PROPERTY));
141     }
142 
143     MediaPackageElementFlavor targetFlavor = tagsAndFlavors.getSingleTargetFlavor();
144     List<String> targetTagList = tagsAndFlavors.getTargetTags();
145 
146 
147     int pixelsPerMinute = NumberUtils.toInt(
148         StringUtils.trimToNull(workflowInstance.getCurrentOperation().getConfiguration(PIXELS_PER_MINUTE_PROPERTY)),
149         DEFAULT_PIXELS_PER_MINUTE
150     );
151 
152     int minWidth = NumberUtils.toInt(
153         StringUtils.trimToNull(workflowInstance.getCurrentOperation().getConfiguration(MIN_WIDTH_PROPERTY)),
154         DEFAULT_MIN_WIDTH
155     );
156 
157     int maxWidth = NumberUtils.toInt(
158         StringUtils.trimToNull(workflowInstance.getCurrentOperation().getConfiguration(MAX_WIDTH_PROPERTY)),
159         DEFAULT_MAX_WIDTH
160     );
161 
162     int height = NumberUtils.toInt(
163         StringUtils.trimToNull(workflowInstance.getCurrentOperation().getConfiguration(HEIGHT_PROPERTY)),
164         DEFAULT_HEIGHT
165     );
166 
167     String color = StringUtils.trimToNull(workflowInstance.getCurrentOperation().getConfiguration(COLOR_PROPERTY));
168 
169     try {
170       TrackSelector trackSelector = new TrackSelector();
171       for (MediaPackageElementFlavor flavor : sourceFlavorList) {
172         trackSelector.addFlavor(flavor);
173       }
174       for (String tag : sourceTagList) {
175         trackSelector.addTag(tag);
176       }
177       Collection<Track> sourceTracks = trackSelector.select(mediaPackage, false);
178       if (sourceTracks.isEmpty()) {
179         logger.info("No tracks found in mediapackage {} with specified {} = {}", mediaPackage, SOURCE_FLAVOR_PROPERTY,
180                 sourceFlavorList);
181         return createResult(mediaPackage, WorkflowOperationResult.Action.SKIP);
182       }
183 
184       List<Job> waveformJobs = new ArrayList<>(sourceTracks.size());
185       for (Track sourceTrack : sourceTracks) {
186         // Skip over track with no audio stream
187         if (!sourceTrack.hasAudio()) {
188           logger.info("Skipping waveform extraction of track {} since it has no audio", sourceTrack.getIdentifier());
189           continue;
190         }
191         try {
192           // generate waveform
193           logger.info("Creating waveform extraction job for track '{}' in mediapackage '{}'",
194               sourceTrack.getIdentifier(), mediaPackage);
195 
196           Job waveformJob = waveformService.createWaveformImage(sourceTrack, pixelsPerMinute, minWidth, maxWidth,
197               height, color);
198           waveformJobs.add(waveformJob);
199         } catch (MediaPackageException | WaveformServiceException e) {
200           logger.error("Creating waveform extraction job for track '{}' in media package '{}' failed",
201               sourceTrack.getIdentifier(), mediaPackage, e);
202         }
203       }
204 
205       logger.debug("Waiting for waveform jobs for media package {}", mediaPackage);
206       if (!waitForStatus(waveformJobs.toArray(new Job[waveformJobs.size()])).isSuccess()) {
207         throw new WorkflowOperationException(
208             String.format("Waveform extraction jobs for media package '%s' have not completed successfully",
209                 mediaPackage.getIdentifier()));
210       }
211 
212       // copy waveform attachments into workspace and add them to the media package
213       for (Job job : waveformJobs) {
214         String jobPayload = job.getPayload();
215         if (StringUtils.isEmpty(jobPayload)) {
216           continue;
217         }
218         MediaPackageElement waveformMpe = null;
219         try {
220           waveformMpe = MediaPackageElementParser.getFromXml(jobPayload);
221           URI newURI = workspace.moveTo(waveformMpe.getURI(),
222               mediaPackage.getIdentifier().toString(),
223               waveformMpe.getIdentifier(),
224               "waveform.png");
225           waveformMpe.setURI(newURI);
226         } catch (MediaPackageException ex) {
227           // unexpected job payload
228           throw new WorkflowOperationException("Can't parse waveform attachment from job " + job.getId());
229         } catch (NotFoundException ex) {
230           throw new WorkflowOperationException("Waveform image file '" + waveformMpe.getURI() + "' not found", ex);
231         } catch (IOException ex) {
232           throw new WorkflowOperationException("Can't get workflow image file '"
233               + waveformMpe.getURI() + "' from workspace");
234         }
235 
236         // set the waveform attachment flavor and add it to the media package
237         if ("*".equals(targetFlavor.getType())) {
238           targetFlavor = new MediaPackageElementFlavor(waveformMpe.getFlavor().getType(), targetFlavor.getSubtype());
239         }
240         if ("*".equals(targetFlavor.getSubtype())) {
241           targetFlavor = new MediaPackageElementFlavor(targetFlavor.getType(), waveformMpe.getFlavor().getSubtype());
242         }
243         waveformMpe.setFlavor(targetFlavor);
244         for (String tag : targetTagList) {
245           waveformMpe.addTag(tag);
246         }
247         mediaPackage.add(waveformMpe);
248       }
249 
250       logger.info("Waveform workflow operation for mediapackage {} completed", mediaPackage);
251       return createResult(mediaPackage, WorkflowOperationResult.Action.CONTINUE);
252 
253     } finally {
254       try {
255         workspace.cleanup(mediaPackage.getIdentifier(), true);
256       } catch (IOException e) {
257         throw new WorkflowOperationException(e);
258       }
259     }
260   }
