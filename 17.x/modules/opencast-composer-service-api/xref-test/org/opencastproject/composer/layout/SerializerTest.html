22  package org.opencastproject.composer.layout;
23  
24  import static org.junit.Assert.assertEquals;
25  import static org.opencastproject.composer.layout.Offset.offset;
26  
27  import org.opencastproject.util.JsonObj;
28  
29  import org.junit.Test;
30  
31  public class SerializerTest {
32    @Test
33    public void testSerialize() throws Exception {
34      final Dimension d = new Dimension(10, 30);
35      assertEquals(d, Serializer.dimension(JsonObj.jsonObj(Serializer.json(d).toJson())));
36      final Anchor a = new Anchor(0.134, 0.982);
37      assertEquals(a, Serializer.anchor(JsonObj.jsonObj(Serializer.json(a).toJson())));
38      final AnchorOffset ao = new AnchorOffset(a, a, offset(10, 20));
39      assertEquals(ao, Serializer.anchorOffset(JsonObj.jsonObj(Serializer.json(ao).toJson())));
40      final TwoShapeLayout tsl = new TwoShapeLayout(new Dimension(1200, 980),
41                                                    new Layout(new Dimension(300, 200), new Offset(0, 0)),
42                                                    new Layout(new Dimension(900, 780), new Offset(300, 200)));
43      final String tslSer = Serializer.json(tsl).toJson();
44      assertEquals(tsl, Serializer.twoShapeLayout(JsonObj.jsonObj(tslSer)));
45      final HorizontalCoverageLayoutSpec hcls = new HorizontalCoverageLayoutSpec(
46              new AnchorOffset(Anchors.TOP_LEFT, Anchors.BOTTOM_RIGHT, new Offset(10, 12)),
47              0.34);
48      assertEquals(hcls, Serializer.horizontalCoverageLayoutSpec(JsonObj.jsonObj(Serializer.json(hcls).toJson())));
49    }
50  
51    @Test(expected = RuntimeException.class)
52    public void testAccessFailure() {
53      JsonObj.jsonObj("{\"x\": [1, 2, 3]}").get(String.class, "x");
54    }
55  }
