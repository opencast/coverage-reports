103   // Versioning the Tobira API:
104   //
105   // Since both Tobira and this API are changing over time, we need some mechanism for ensuring they
106   // are compatible. We don't want to enforce a 1:1 thing, where a particular Tobira needs one
107   // exact API as that makes the update process harder (especially once this module is included in
108   // the community version). So instead we have some semver-like versioning here. Increase the
109   // minor version for backwards-compatible changes and the major version for breaking changes.
110   //
111   // Note that we cannot use the Opencast version as some institutions might want to patch their
112   // Opencast to include a newer Tobira module.
113   //
114   // So what's a breaking change and what not? For starters, the harvesting still needs to work with
115   // all Tobira versions that it worked with previously. Since Tobira ignores unknown fields,
116   // adding new JSON fields is a non-breaking change. You should also consider whether Tobira needs
117   // to resynchronize, i.e. to get new data.
118   private static final int VERSION_MAJOR = 1;
119   private static final int VERSION_MINOR = 6;
120   private static final String VERSION = VERSION_MAJOR + "." + VERSION_MINOR;
121

208   @GET
209   @Path("/harvest")
210   @Produces(APPLICATION_JSON)
211   @RestQuery(
212       name = "harvest",
213       description = "Harvesting API to get incremental updates about series and events.",
214       restParameters = {
215           @RestParameter(
216               name = "preferredAmount",
217               isRequired = true,
218               description = "A preferred number of items the request should return. This is "
219                   + "merely a rough guideline and the API might return more or fewer items than "
220                   + "this parameter. You cannot rely on an exact number of returned items! "
221                   + "In practice this API usually returns between 0 and twice this parameter "
222                   + "number of items.",
223               type = Type.INTEGER
224           ),
225           @RestParameter(
226               name = "since",
227               isRequired = true,
228               description = "Only return items that changed after or at this timestamp. "
229                   + "Specified in milliseconds since 1970-01-01T00:00:00Z.",
230               type = Type.INTEGER
231           ),
232       },
233       responses = {
234           @RestResponse(description = "Event and Series Data", responseCode = HttpServletResponse.SC_OK)
235       },
236       returnDescription = "Event and Series Data changed after the given timestamp"
237   )
238   public Response harvest(
239       @QueryParam("preferredAmount") Integer preferredAmount,
240       @QueryParam("since") Long since
241   ) {
242     // Parameter error handling
243     if (since == null) {
244       return badRequest("Required parameter 'since' not specified");
245     }
246     if (preferredAmount == null) {
247       return badRequest("Required parameter 'preferredAmount' not specified");
248     }
249     if (since < 0) {
250       return badRequest("Parameter 'since' < 0, but it has to be positive or 0");
251     }
252     if (preferredAmount <= 0) {
253       return badRequest("Parameter 'preferredAmount' <= 0, but it has to be positive");
254     }
255 
256     logger.debug("Request to '/harvest' with preferredAmount={} since={}", preferredAmount, since);
257 
258     try {
259       var json = Harvest.harvest(
260           preferredAmount,
261           new Date(since),
262           searchService, seriesService, authorizationService, securityService, playlistService, workspace);
263 
264       // TODO: encoding
265       return Response.ok(json.toJson()).build();
266     } catch (Exception e) {
267       logger.error("Unexpected exception in tobira/harvest", e);
268       return Response.serverError().build();
269     }
270   }
271 
272   private static Response badRequest(String msg) {
273     logger.warn("Bad request to tobira/harvest: {}", msg);
274     return Response.status(BAD_REQUEST).entity(msg).build();
275   }

277   /* Since CXF doesn't seem to like accepting multiple types on a single endpoint, this is what Tobira currently sends:
278     curl http://localhost/tobira/stats \
279         -H "Authorization: Basic YWRtaW46b3BlbmNhc3Q=" \
280         -H "Content-Type: application/json" \
281         -d '{
282         "num_realms": 3641,
283         "num_blocks": 5544,
284         "version": {
285       "identifier": "v3.4",
286           "build_time_utc": "Fri, 27 Jun 2025 09:13:53 +0000",
287           "git_commit_hash": "d421b168cc6a004dd008f4b8bc0de4070cd99c2c",
288           "git_was_dirty": true,
289           "target": "aarch64-apple-darwin"
290     },
291         "config": {
292       "download_button_shown": true,
293           "auth_source": "tobira-session",
294           "login_credentials_handler": "login-callback",
295           "session_endpoint_handler": "none",
296           "login_link_overridden": false,
297           "logout_link_overridden": false,
298           "uses_pre_auth": true
299     }
300   }' \
301     -v
302   */
303   @POST
304   @Path("/stats")
305   @Consumes(APPLICATION_JSON)
306   @RestQuery(
307       name = "stats",
308       description = "Accepts a json blob of statistical data about Tobira.  To test this properly see the code.",
309       responses = {
310           @RestResponse(description = "Stats parsed", responseCode = HttpServletResponse.SC_ACCEPTED)
311       },
312       returnDescription = "No data returned, just a 204 on success"
313   )
314   public Response acceptStats(@Context HttpServletRequest request) {
315     try (InputStream is = request.getInputStream()) {
316       String body = IOUtils.toString(is, request.getCharacterEncoding());
317       cachedStats = gson.fromJson(body, JsonElement.class).getAsJsonObject();
318     } catch (IOException e) {
319       logger.error("Error reading request body:", e);
320       return badRequest("Error reading response body");
321     } catch (IllegalStateException e) {
322       return Response.notAcceptable(null).build();
323     }
324     cachedStats.addProperty("updated", sdf.format(Calendar.getInstance().getTime()));
325 
326     db.execTx(em -> {
327       AdopterRegistrationExtra existing = em.find(AdopterRegistrationExtra.class, "tobira");
328       if (null != existing) {
329         existing.setData(gson.toJson(cachedStats));
330         em.merge(existing);
331       } else {
332         existing = new AdopterRegistrationExtra("tobira", gson.toJson(cachedStats));
333         em.persist(existing);
334       }
335     });
336 
337     return Response.noContent().build();
338   }
