59  public class AnalyzeAudioWorkflowOperationHandlerTest {
60  
61    private AnalyzeAudioWorkflowOperationHandler operationHandler;
62    private MediaPackage mp;
63    private URI uriMP;
64    private WorkflowInstance instance;
65    private WorkflowOperationInstance operationInstance = new WorkflowOperationInstance();
66  
67    @Before
68    public void setUp() throws Exception {
69      MediaPackageBuilder builder = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder();
70      uriMP = AnalyzeAudioWorkflowOperationHandler.class.getResource("/sox_mediapackage.xml").toURI();
71      mp = builder.loadFromXml(uriMP.toURL().openStream());
72  
73      URI soxTrackUri = AnalyzeAudioWorkflowOperationHandler.class.getResource("/sox-track.xml").toURI();
74      URI soxEncodeUri = AnalyzeAudioWorkflowOperationHandler.class.getResource("/sox-encode-track.xml").toURI();
75  
76      String soxTrackXml = IOUtils.toString(soxTrackUri.toURL().openStream());
77      String encodeTrackXml = IOUtils.toString(soxEncodeUri.toURL().openStream());
78  
79      instance = EasyMock.createNiceMock(WorkflowInstance.class);
80      EasyMock.expect(instance.getMediaPackage()).andReturn(mp).anyTimes();
81      EasyMock.expect(instance.getCurrentOperation()).andReturn(operationInstance).anyTimes();
82      EasyMock.replay(instance);
83  
84      DefaultOrganization org = new DefaultOrganization();
85      User anonymous = new JaxbUser("anonymous", "test", org, new JaxbRole(
86              DefaultOrganization.DEFAULT_ORGANIZATION_ANONYMOUS, org));
87      UserDirectoryService userDirectoryService = EasyMock.createMock(UserDirectoryService.class);
88      EasyMock.expect(userDirectoryService.loadUser((String) EasyMock.anyObject())).andReturn(anonymous).anyTimes();
89      EasyMock.replay(userDirectoryService);
90  
91      Organization organization = new DefaultOrganization();
92      OrganizationDirectoryService organizationDirectoryService = EasyMock.createMock(OrganizationDirectoryService.class);
93      EasyMock.expect(organizationDirectoryService.getOrganization((String) EasyMock.anyObject()))
94              .andReturn(organization).anyTimes();
95      EasyMock.replay(organizationDirectoryService);
96  
97      SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
98      EasyMock.expect(securityService.getUser()).andReturn(anonymous).anyTimes();
99      EasyMock.expect(securityService.getOrganization()).andReturn(organization).anyTimes();
100     EasyMock.replay(securityService);
101 
102     IncidentService incidentService = EasyMock.createNiceMock(IncidentService.class);
103     EasyMock.replay(incidentService);
104 
105     ServiceRegistry serviceRegistry = new ServiceRegistryInMemoryImpl(null, securityService, userDirectoryService,
106             organizationDirectoryService, incidentService);
107 
108     Job analyzeJob = serviceRegistry.createJob(SoxService.JOB_TYPE, "Analyze", null, soxTrackXml, false);
109     analyzeJob.setStatus(Status.FINISHED);
110     analyzeJob = serviceRegistry.updateJob(analyzeJob);
111 
112     Job encodeJob = serviceRegistry.createJob(ComposerService.JOB_TYPE, "Encode", null, encodeTrackXml, false);
113     encodeJob.setStatus(Status.FINISHED);
114     encodeJob = serviceRegistry.updateJob(encodeJob);
115 
116     SoxService sox = EasyMock.createNiceMock(SoxService.class);
117     EasyMock.expect(sox.analyze((Track) EasyMock.anyObject())).andReturn(analyzeJob).anyTimes();
118     EasyMock.replay(sox);
119 
120     ComposerService composer = EasyMock.createNiceMock(ComposerService.class);
121     EasyMock.expect(composer.encode((Track) EasyMock.anyObject(), (String) EasyMock.anyObject())).andReturn(encodeJob);
122     EasyMock.replay(composer);
123 
124     Workspace workspace = EasyMock.createNiceMock(Workspace.class);
125     EasyMock.replay(workspace);
126 
127     // set up the handler
128     operationHandler = new AnalyzeAudioWorkflowOperationHandler();
129     operationHandler.setJobBarrierPollingInterval(0);
130     operationHandler.setComposerService(composer);
131     operationHandler.setSoxService(sox);
132     operationHandler.setWorkspace(workspace);
133     operationHandler.setServiceRegistry(serviceRegistry);
134   }

136   @Test
137   public void testAudioVideo() throws Exception {
138     operationInstance.setConfiguration("source-tags", "");
139     operationInstance.setConfiguration("source-flavor", "*/video-audio");
140     operationInstance.setConfiguration("source-flavors", "");
141     operationInstance.setConfiguration("force-transcode", "false");
142 
143     WorkflowOperationResult result = operationHandler.start(instance, null);
144     Assert.assertEquals(Action.CONTINUE, result.getAction());
145 
146     Assert.assertEquals("Resulting mediapackage has the wrong number of tracks", 3, result.getMediaPackage()
147             .getElements().length);
148     Track[] tracks = result.getMediaPackage().getTracks(new MediaPackageElementFlavor("presentation", "video-audio"));
149     Assert.assertEquals("Resulting mediapackage has the wrong number of tracks", 1, tracks.length);
150     TrackImpl audioVideo = (TrackImpl) tracks[0];
151     Assert.assertEquals(-20f, audioVideo.getAudio().get(0).getRmsLevDb().floatValue(), 0.001d);
152   }
