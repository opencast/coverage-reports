130   /**
131    * Tests "normal" behavior, where the media package contains at least one element with the given flavor and tags
132    */
133   @Test
134   public void testHandleEvent() throws Exception {
135     Catalog episodeCatalog = CatalogImpl.newInstance();
136     episodeCatalog.setURI(URI.create("/episode.xml"));
137     episodeCatalog.setFlavor(MediaPackageElementFlavor.parseFlavor("dublincore/episode"));
138     episodeCatalog.addTag("archive");
139     MediaPackage updatedMp = createMediaPackage(episodeCatalog);
140 
141     // these are the interactions we expect with the asset manager
142     mockAssetManager(updatedMp);
143 
144     // these are the interactions we expect with the security service
145     mockSecurityService();
146 
147     // these are the interactions we expect for the OAI-PMH database
148     mockOaiPmhDatabase();
149 
150     Capture<MediaPackage> mpCapture = Capture.newInstance();
151     Capture<String> repositoryCapture = Capture.newInstance();
152     Capture<Set<String>> flavorsCapture = Capture.newInstance();
153     Capture<Set> tagsCapture = Capture.newInstance();
154     expect(oaiPmhPublicationService.updateMetadata(capture(mpCapture), capture(repositoryCapture),
155             capture(flavorsCapture), capture(tagsCapture), anyBoolean()))
156             .andAnswer(() -> mock(Job.class)).times(1);
157 
158     replayAll();
159 
160     cut.handleEvent(createSnapshot(updatedMp));
161 
162     assertEquals(updatedMp.getIdentifier().toString(), mpCapture.getValue().getIdentifier().toString());
163     assertEquals(OAIPMH_REPOSITORY, repositoryCapture.getValue());
164     Assert.assertNotNull(flavorsCapture.getValue());
165     Assert.assertTrue(flavorsCapture.getValue().contains("dublincore/*"));
166     Assert.assertTrue(flavorsCapture.getValue().contains("security/*"));
167     Assert.assertTrue(tagsCapture.getValue().contains("archive"));
168     Assert.assertTrue(orgIdCapture.hasCaptured());
169     Assert.assertEquals(new DefaultOrganization().getId(), orgIdCapture.getValue());
170     Assert.assertTrue(mpIdCapture.hasCaptured());
171     Assert.assertEquals(updatedMp.getIdentifier().toString(), mpIdCapture.getValue());
172     Assert.assertTrue(snapshotVersionCapture.hasCaptured());
173     Assert.assertEquals("3", snapshotVersionCapture.getValue());
174   }

176   /**
177    * Tests if publishing to OAI-PMH is skipped, if the episode is not known by OAI-PMH.
178    */
179   @Test
180   public void testEpisodeNotKnownByOaiPmh() throws Exception {
181     Catalog episodeCatalog = CatalogImpl.newInstance();
182     episodeCatalog.setURI(URI.create("/episode.xml"));
183     episodeCatalog.setFlavor(MediaPackageElementFlavor.parseFlavor("dublincore/episode"));
184     episodeCatalog.addTag("archive");
185     MediaPackage updatedMp = createMediaPackage(episodeCatalog);
186 
187     // these are the interactions we expect with the asset manager
188     mockAssetManager(updatedMp);
189     // these are the interactions we expect with the security service
190     mockSecurityService();
191 
192     // mock the OAI-PMH database
193     SearchResult searchResultMock = mock(MockType.NICE, SearchResult.class);
194     expect(searchResultMock.getItems()).andReturn(Collections.EMPTY_LIST).anyTimes();
195     queryCapture = Capture.newInstance();
196     expect(oaiPmhDatabaseMock.search(capture(queryCapture))).andReturn(searchResultMock);
197 
198     replayAll();
199 
200     cut.handleEvent(createSnapshot(updatedMp));
201 
202     // the OAI-PMH publication service should not be called as the media package isn't mocked in the OAI-PMH database
203     verifyAll();
204   }

250   private AssetManagerItem.TakeSnapshot createSnapshot(MediaPackage mediaPackage) throws Exception {
251     AccessControlList acl = new AccessControlList();
252     AssetManagerItem.TakeSnapshot result = AssetManagerItem.add(workspace, mediaPackage, acl, 3L, new Date());
253     return result;
254   }
255 
256   private MediaPackage createMediaPackage(MediaPackageElement... elements) throws MediaPackageException {
257     MediaPackage result = MP_BUILDER.createNew();
258     result.setTitle(MEDIA_PACKAGE_TITLE);
259     for (MediaPackageElement mpe : elements) {
260       result.add(mpe);
261     }
262     return result;
263   }

265   private void mockAssetManager(MediaPackage mediaPackage) {
266     AQueryBuilder queryBuilder = mock(AQueryBuilder.class);
267     ASelectQuery selectQuery = mock(ASelectQuery.class);
268     expect(queryBuilder.select(EasyMock.anyObject())).andReturn(selectQuery);
269     Target target = mock(Target.class);
270     expect(queryBuilder.snapshot()).andReturn(target);
271     Predicate queryPredicate1 = mock(Predicate.class);
272     Predicate queryPredicate2 = mock(Predicate.class);
273     Predicate queryPredicate3 = mock(Predicate.class);
274     expect(queryPredicate1.and(EasyMock.anyObject())).andReturn(queryPredicate2);
275     expect(queryPredicate2.and(EasyMock.anyObject())).andReturn(queryPredicate3);
276     orgIdCapture = Capture.newInstance();
277     Field<String> orgIdField = mock(Field.class);
278     expect(orgIdField.eq(capture(orgIdCapture))).andReturn(queryPredicate1);
279     expect(queryBuilder.organizationId()).andReturn(orgIdField);
280     mpIdCapture = Capture.newInstance();
281     expect(queryBuilder.mediaPackageId(capture(mpIdCapture))).andReturn(queryPredicate2);
282     VersionField versionField = mock(VersionField.class);
283     expect(versionField.eq(EasyMock.anyObject(Version.class))).andReturn(queryPredicate3);
284     expect(queryBuilder.version()).andReturn(versionField);
285     ASelectQuery selectQuery2 = mock(ASelectQuery.class);
286     expect(selectQuery.where(EasyMock.anyObject())).andReturn(selectQuery2);
287     AResult queryResult = mock(AResult.class);
288     expect(selectQuery2.run()).andReturn(queryResult);
289     expect(assetManagerMock.createQuery()).andReturn(queryBuilder);
290     snapshotVersionCapture = EasyMock.newCapture();
291     expect(assetManagerMock.toVersion(capture(snapshotVersionCapture))).andReturn(Optional.of(mock(Version.class)));
292     ARecord record = mock(ARecord.class);
293     expect(queryResult.getRecords()).andReturn(new LinkedHashSet<>(Arrays.asList(record)));
294     Snapshot snapshot = mock(Snapshot.class);
295     expect(record.getSnapshot()).andReturn(Optional.of(snapshot));
296     expect(snapshot.getMediaPackage()).andReturn(mediaPackage);
297   }
