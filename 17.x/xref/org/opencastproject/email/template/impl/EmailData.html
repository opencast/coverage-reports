21  package org.opencastproject.email.template.impl;
22  
23  import org.opencastproject.job.api.Incident;
24  import org.opencastproject.mediapackage.MediaPackage;
25  import org.opencastproject.util.doc.DocData;
26  import org.opencastproject.workflow.api.WorkflowInstance;
27  import org.opencastproject.workflow.api.WorkflowOperationInstance;
28  
29  import java.util.HashMap;
30  import java.util.LinkedHashMap;
31  import java.util.List;
32  import java.util.Map;
33  
34  /**
35   * Holds data to be displayed in an email message. The following data will be available: mediaPackage, workflow,
36   * workflowConfig: workflow configuration as key-value pairs, catalogs: hash of catalogs whose key is the catalog flavor
37   * sub-type e.g. "series", "episode", failedOperation: the last operation marked as "failOnError" that failed.
38   *
39   */
40  public class EmailData extends DocData {
41  
42    private final WorkflowInstance workflow;
43    private final Map<String, String> workflowConfig;
44    private final MediaPackage mediaPackage;
45    // The hash map below looks like this:
46    // "dublincore/episode" --> { "creator": "John Harvard", "type": "L05", "isPartOf": "20140224038"... }
47    // "dublincore/series" --> { "creator": "Harvard", "identifier": "20140224038"... }
48    private final HashMap<String, HashMap<String, String>> catalogs;
49    private final WorkflowOperationInstance failed;
50    private final List<Incident> incidents;
51    private final Map<String, String> orgProperties;
52  
53    /**
54     * Create the base data object for populating email fields.
55     *
56     * @param name
57     *          a name for this object
58     * @param workflow
59     *          workflow instance
60     * @param catalogs
61     *          hash map of media package catalogs
62     * @param failed
63     *          workflow operation that caused the workflow to fail
64     * @param incidents
65     *          incidents
66     * @param orgProperties
67     *          organization properties
68     */
69    public EmailData(String name, WorkflowInstance workflow, HashMap<String, HashMap<String, String>> catalogs,
70            WorkflowOperationInstance failed, List<Incident> incidents, Map<String, String> orgProperties) {
71      super(name, null, null);
72  
73      this.workflow = workflow;
74      this.workflowConfig = new HashMap<String, String>();
75      for (String key : workflow.getConfigurationKeys()) {
76        workflowConfig.put(key, workflow.getConfiguration(key));
77      }
78      this.mediaPackage = workflow.getMediaPackage();
79      this.catalogs = catalogs;
80      this.failed = failed;
81      this.incidents = incidents;
82      this.orgProperties = orgProperties;
83    }
84  
85    /**
86     * Returns a string representation of this object.
87     *
88     * @return a string representation of this object
89     */
90    @Override
91    public String toString() {
92      return "EmailDOC:name=" + meta.get("name") + ", notes=" + notes + ", workflow id=" + workflow.getId()
93              + ", media package id=" + mediaPackage.getIdentifier();
94    }
95  
96    @Override
97    public Map<String, Object> toMap() throws IllegalStateException {
98      LinkedHashMap<String, Object> m = new LinkedHashMap<String, Object>();
99      m.put("meta", meta);
100     m.put("notes", notes);
101     m.put("mediaPackage", mediaPackage);
102     m.put("workflow", workflow);
103     m.put("workflowConfig", workflowConfig);
104     m.put("catalogs", catalogs);
105     m.put("failedOperation", failed); // Will be null if no errors
106     m.put("incident", incidents); // Will be null if no incidents
107     m.put("organization", orgProperties);
108     return m;
109   }
110 
111 }
