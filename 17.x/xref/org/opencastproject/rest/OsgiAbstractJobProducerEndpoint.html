22  package org.opencastproject.rest;
23  
24  import static java.lang.String.format;
25  
26  import org.opencastproject.job.api.JobProducer;
27  import org.opencastproject.serviceregistry.api.ServiceRegistry;
28  
29  /**
30   * Refined implementation of {@link org.opencastproject.rest.AbstractJobProducerEndpoint} suitable for use in an
31   * OSGi environment.
32   * <p>
33   * OSGi dependency injection methods are provided to reduce the amount of boilerplate code needed per
34   * service implementation.
35   * <p>
36   * Declare as type variable the {@linkplain AbstractJobProducerEndpoint#getService() job producing service} on which the
37   * endpoint depends.
38   * <p>
39   * <strong>Example:</strong> The endpoint for the WorkflowService can be declared like this:
40   * <pre>
41   *   public final class WorkflowServiceEndpoint extends OsgiAbstractJobProducerEndpoint&lt;WorkflowServiceImpl&gt;
42   * </pre>
43   * <p>
44   * <strong>Implementation note:</strong> Type variable <code>A</code> <em>cannot</em> have upper bound
45   * {@link org.opencastproject.job.api.JobProducer}. Even though this may seem reasonable it will cause trouble
46   * with OSGi dependency injection. The dependency will most likely be declared on the service's <em>interface</em>
47   * and <em>not</em> the concrete implementation. But only the concrete implementation is a <code>JobProducer</code>.
48   * With <code>A</code> having an upper bound of <code>JobProducer</code> the signature of {@link #setService(Object)}
49   * will be fixed to that bound: <code>setService(JobProducer)</code>. Now the service cannot be injected anymore.
50   */
51  public abstract class OsgiAbstractJobProducerEndpoint<A> extends AbstractJobProducerEndpoint {
52    private A service;
53    private ServiceRegistry serviceRegistry;
54  
55    @Override
56    public JobProducer getService() {
57      if (service instanceof JobProducer) {
58        return (JobProducer) service;
59      } else {
60        throw new RuntimeException(format("Service %s is expected to be of type JobProducer", service));
61      }
62    }
63  
64    public void setService(A service) {
65      this.service = service;
66    }
67  
68    public A getSvc() {
69      return service;
70    }
71  
72    @Override
73    public ServiceRegistry getServiceRegistry() {
74      return serviceRegistry;
75    }
76  
77    public void setServiceRegistry(ServiceRegistry serviceRegistry) {
78      this.serviceRegistry = serviceRegistry;
79    }
80  }
