64  /**
65   * The workflow definition creating a video from a still image.
66   */
67  @Component(
68      immediate = true,
69      service = WorkflowOperationHandler.class,
70      property = {
71          "service.description=Image to Video Workflow Operation Handler",
72          "workflow.operation=image-to-video"
73      }
74  )
75  public class ImageToVideoWorkflowOperationHandler extends AbstractWorkflowOperationHandler {
76    private static final String OPT_DURATION = "duration";
77    private static final String OPT_PROFILE = "profile";
78  
79    /** The logging facility */
80    private static final Logger logger = LoggerFactory.getLogger(ImageToVideoWorkflowOperationHandler.class);
81  
82    /** The composer service */
83    private ComposerService composerService = null;
84  
85    /** The local workspace */
86    private Workspace workspace = null;

111   @Override
112   public WorkflowOperationResult start(final WorkflowInstance workflowInstance, JobContext context)
113           throws WorkflowOperationException {
114     logger.debug("Running image to video workflow operation on workflow {}", workflowInstance.getId());
115     try {
116       return imageToVideo(workflowInstance.getMediaPackage(), workflowInstance);
117     } catch (WorkflowOperationException e) {
118       throw e;
119     } catch (Exception e) {
120       throw new WorkflowOperationException(e);
121     }
122   }
123 
124   private WorkflowOperationResult imageToVideo(MediaPackage mp, WorkflowInstance wi) throws Exception {
125     // read cfg
126     ConfiguredTagsAndFlavors tagsAndFlavors = getTagsAndFlavors(wi,
127         Configuration.many, Configuration.many, Configuration.many, Configuration.many);
128     final List<String> sourceTags = tagsAndFlavors.getSrcTags();
129     List<MediaPackageElementFlavor> sourceFlavors = tagsAndFlavors.getSrcFlavors();
130 
131     if (sourceFlavors.isEmpty() && sourceTags.isEmpty()) {
132       logger.warn("No source tags or flavor are given to determine the image to use");
133       return createResult(mp, Action.SKIP);
134     }
135     final Option<MediaPackageElementFlavor> sourceFlavor = Option.option(sourceFlavors.get(0));
136 
137     final List<String> targetTags = tagsAndFlavors.getTargetTags();
138     List<MediaPackageElementFlavor> targetFlavors = tagsAndFlavors.getTargetFlavors();
139     final Option<MediaPackageElementFlavor> targetFlavor = Option.option(targetFlavors.get(0));
140     final double duration = getCfg(wi, OPT_DURATION).bind(Strings.toDouble).getOrElse(
141             this.<Double> cfgKeyMissing(OPT_DURATION));
142     final String profile = getCfg(wi, OPT_PROFILE).getOrElse(this.<String> cfgKeyMissing(OPT_PROFILE));
143     // run image to video jobs
144     final List<Job> jobs = Monadics.<MediaPackageElement> mlist(mp.getAttachments())
145             .filter(sourceFlavor.map(Filters.matchesFlavor).getOrElse(Booleans.<MediaPackageElement> yes()))
146             .filter(Filters.hasTagAny(sourceTags)).map(Misc.<MediaPackageElement, Attachment> cast())
147             .map(imageToVideo(profile, duration)).value();
148     if (JobUtil.waitForJobs(serviceRegistry, jobs).isSuccess()) {
149       for (final Job job : jobs) {
150         if (job.getPayload().length() > 0) {
151           Track track = (Track) MediaPackageElementParser.getFromXml(job.getPayload());
152           track.setURI(workspace.moveTo(track.getURI(), mp.getIdentifier().toString(), track.getIdentifier(),
153                   FilenameUtils.getName(track.getURI().toString())));
154           // Adjust the target tags
155           for (String tag : targetTags) {
156             track.addTag(tag);
157           }
158           // Adjust the target flavor.
159           for (MediaPackageElementFlavor flavor : targetFlavor) {
160             track.setFlavor(flavor);
161           }
162           // store new tracks to mediaPackage
163           mp.add(track);
164           logger.debug("Image to video operation completed");
165         } else {
166           logger.info("Image to video operation unsuccessful, no payload returned: {}", job);
167           return createResult(mp, Action.SKIP);
168         }
169       }
170       return createResult(mp, Action.CONTINUE, mlist(jobs).foldl(0L, new Function2<Long, Job, Long>() {
171         @Override
172         public Long apply(Long max, Job job) {
173           return Math.max(max, job.getQueueTime());
174         }
175       }));
176     } else {
177       throw new WorkflowOperationException("The image to video encoding jobs did not return successfully");
178     }
179   }
180 
181   /** Returned function may throw exceptions. */
182   private Function<Attachment, Job> imageToVideo(final String profile, final double duration) {
183     return new Function.X<Attachment, Job>() {
184       @Override
185       protected Job xapply(Attachment attachment) throws MediaPackageException, EncoderException {
186         logger.info("Converting image {} to a video of {} sec", attachment.getURI().toString(), duration);
187         return composerService.imageToVideo(attachment, profile, duration);
188       }
189     };
190   }
