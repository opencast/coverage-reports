124   public WorkflowOperationResult start(final WorkflowInstance workflowInstance, JobContext context)
125           throws WorkflowOperationException {
126     logger.debug("Running analyze audio workflow operation on workflow {}", workflowInstance.getId());
127 
128     try {
129       return analyze(workflowInstance.getMediaPackage(), workflowInstance);
130     } catch (Exception e) {
131       throw new WorkflowOperationException(e);
132     }
133   }

135   private WorkflowOperationResult analyze(MediaPackage src, WorkflowInstance workflow) throws SoxException,
136           IOException, NotFoundException, MediaPackageException, WorkflowOperationException, EncoderException {
137     MediaPackage mediaPackage = (MediaPackage) src.clone();
138 
139     ConfiguredTagsAndFlavors tagsAndFlavors = getTagsAndFlavors(workflow, Configuration.many, Configuration.many,
140         Configuration.none, Configuration.none);
141 
142     boolean forceTranscode = BooleanUtils.toBoolean(workflow.getCurrentOperation().getConfiguration("force-transcode"));
143 
144     AbstractMediaPackageElementSelector<Track> elementSelector = new TrackSelector();
145 
146     // Make sure either one of tags or flavors are provided
147     List<MediaPackageElementFlavor> sourceFlavors = tagsAndFlavors.getSrcFlavors();
148     List<String> sourceTags = tagsAndFlavors.getSrcTags();
149     if (sourceFlavors.isEmpty() && sourceTags.isEmpty()) {
150       logger.info("No source tags or flavors have been specified, not matching anything");
151       return createResult(mediaPackage, Action.CONTINUE);
152     }
153 
154     // Select the source flavors
155     for (MediaPackageElementFlavor flavor : sourceFlavors) {
156       try {
157         elementSelector.addFlavor(flavor);
158       } catch (IllegalArgumentException e) {
159         throw new WorkflowOperationException("Source flavor '" + flavor.toString() + "' is malformed");
160       }
161     }
162 
163     // Select the source tags
164     for (String tag : sourceTags) {
165       elementSelector.addTag(tag);
166     }
167 
168     // Look for elements matching the tag
169     Collection<Track> elements = elementSelector.select(mediaPackage, false);
170 
171     // Analyze audio for all tracks found
172     long totalTimeInQueue = 0;
173     List<URI> cleanupURIs = new ArrayList<URI>();
174     Map<Job, Track> analyzeJobs = new HashMap<Job, Track>();
175     try {
176       for (Track track : elements) {
177 
178         TrackImpl audioTrack = (TrackImpl) track;
179         // Skip video only mismatches
180         if (!track.hasAudio()) {
181           logger.info("Skipping audio analysis of '{}', since it contains no audio stream", track);
182           continue;
183         } else if (track.hasVideo() || forceTranscode) {
184           audioTrack = (TrackImpl) extractAudioTrack(track);
185           audioTrack.setAudio(((TrackImpl) track).getAudio());
186           cleanupURIs.add(audioTrack.getURI());
187         }
188 
189         analyzeJobs.put(soxService.analyze(audioTrack), track);
190       }
191 
192       if (analyzeJobs.isEmpty()) {
193         logger.info("No matching tracks found");
194         return createResult(mediaPackage, Action.CONTINUE);
195       }
196 
197       // Wait for the jobs to return
198       if (!waitForStatus(analyzeJobs.keySet().toArray(new Job[analyzeJobs.size()])).isSuccess()) {
199         throw new WorkflowOperationException("One of the analyze jobs did not complete successfully");
200       }
201 
202       // Process the result
203       for (Map.Entry<Job, Track> entry : analyzeJobs.entrySet()) {
204         Job job = entry.getKey();
205         TrackImpl origTrack = (TrackImpl) entry.getValue();
206 
207         // add this receipt's queue time to the total
208         totalTimeInQueue += job.getQueueTime();
209 
210         if (job.getPayload().length() > 0) {
211           TrackImpl analyzed = (TrackImpl) MediaPackageElementParser.getFromXml(job.getPayload());
212 
213           // Set metadata on track
214           origTrack.setAudio(analyzed.getAudio());
215         } else {
216           logger.warn("Analyze audio job {} for track {} has no result!", job, origTrack);
217         }
218       }
219     } finally {
220       // Clean up temporary audio files from workspace
221       for (URI uri : cleanupURIs) {
222         workspace.delete(uri);
223       }
224     }
225 
226     WorkflowOperationResult result = createResult(mediaPackage, Action.CONTINUE, totalTimeInQueue);
227     logger.debug("Analyze audio operation completed");
228     return result;
229   }

231   /**
232    * Extract the audio track from the given video track.
233    *
234    * @param videoTrack
235    *          the track containing the audio
236    * @return the extracted audio track
237    * @throws WorkflowOperationException
238    * @throws NotFoundException
239    * @throws EncoderException
240    * @throws MediaPackageException
241    */
242   private Track extractAudioTrack(Track videoTrack) throws WorkflowOperationException, EncoderException,
243           MediaPackageException {
244     logger.info("Extract audio stream from track {}", videoTrack);
245     Job job = composerService.encode(videoTrack, SOX_AONLY_PROFILE);
246     if (!waitForStatus(job).isSuccess()) {
247       throw new WorkflowOperationException("Extracting audio track from video track " + videoTrack + " failed");
248     }
249 
250     return (Track) MediaPackageElementParser.getFromXml(job.getPayload());
251   }
