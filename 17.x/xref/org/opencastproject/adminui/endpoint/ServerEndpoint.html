101   // List of filter keys

104   private static final String KEY_STATUS = "status";

107   /** Cache time */

110   /**
111    * Comparator for the servers list
112    */
113   private class ServerComparator implements Comparator<Server> {
114 
115     private final Sort sortType;
116     private final boolean ascending;
117 
118     ServerComparator(Sort sortType, Boolean ascending) {
119       this.sortType = sortType;
120       this.ascending = ascending;
121     }
122 
123     @Override
124     public int compare(Server host1, Server host2) {
125       return (ascending ? 1 : -1) * compareByType(host1, host2);
126     }
127 
128     private int compareByType(final Server host1, final Server host2) {
129       switch (sortType) {
130         case ONLINE:
131           return Boolean.compare(host1.online, host2.online);
132         case CORES:
133           return Long.compare(host1.cores, host2.cores);
134         case QUEUED:
135           return Long.compare(host1.queued, host2.queued);
136         case MAINTENANCE:
137           return Boolean.compare(host1.maintenance, host2.maintenance);
138         case RUNNING:
139           return Long.compare(host1.running, host2.running);
140         case NODENAME:
141           return host1.nodeName.compareTo(host2.nodeName);
142         case HOSTNAME:
143         default:
144           return host1.hostname.compareTo(host2.hostname);
145       }
146     }
147   }

183   @GET
184   @Path("servers.json")
185   @Produces(MediaType.APPLICATION_JSON)
186   @RestQuery(description = "Returns the list of servers", name = "servers", restParameters = {
187           @RestParameter(name = "limit", description = "The maximum number of items to return per page", isRequired = false, type = INTEGER),
188           @RestParameter(name = "offset", description = "The offset", isRequired = false, type = INTEGER),
189           @RestParameter(name = "filter", description = "Filter results by hostname, status or free text query", isRequired = false, type = STRING),
190           @RestParameter(name = "sort", description = "The sort order.  May include any of the following: "
191                   + "CORES, HOSTNAME, MAINTENANCE, ONLINE, QUEUED (jobs), RUNNING (jobs)."
192                   + "The suffix must be :ASC for ascending or :DESC for descending sort order (e.g. HOSTNAME:DESC).", isRequired = false, type = STRING) },
193           responses = { @RestResponse(description = "Returns the list of jobs from Opencast", responseCode = HttpServletResponse.SC_OK) },
194           returnDescription = "The list of servers")
195   public Response getServers(@QueryParam("limit") int limit, @QueryParam("offset") int offset,
196           @QueryParam("filter") String filter, @QueryParam("sort") String sort)
197           throws Exception {
198 
199     final Map<String, String> filters;
200     try {
201       filters = Arrays.stream(StringUtils.split(Objects.toString(filter, ""), ","))
202           .map(f -> f.split(":", 2))
203           .collect(Collectors.toMap(f -> f[0], f -> f[1]));
204     } catch (ArrayIndexOutOfBoundsException e) {
205       return badRequest("Invalid filter string", e);
206     }
207 
208     List<Server> servers = new ArrayList<>();
209     for (Server server: getServerData()) {
210       if (!filters.getOrDefault(KEY_HOSTNAME, server.hostname).equalsIgnoreCase(server.hostname)) {
211         continue;
212       }
213 
214       if (!StringUtils.equalsIgnoreCase(filters.getOrDefault(KEY_NODE_NAME, server.nodeName), server.nodeName)) {
215         continue;
216       }
217 
218       if (filters.containsKey(KEY_STATUS)) {
219         final Status status = Status.valueOf(filters.get(KEY_STATUS).toUpperCase());
220         if (Status.ONLINE.equals(status) && !server.online) {
221           continue;
222         }
223         if (Status.OFFLINE.equals(status) && server.online) {
224           continue;
225         }
226         if (Status.MAINTENANCE.equals(status) && !server.maintenance) {
227           continue;
228         }
229       }
230 
231       final String text = filters.getOrDefault(KEY_TEXT_FILTER, "");
232       if (Stream.of(server.hostname, server.nodeName).noneMatch(v -> StringUtils.containsIgnoreCase(v, text))) {
233         continue;
234       }
235 
236       servers.add(server);
237     }
238 
239     // Sorting
240     Sort sortKey = Sort.HOSTNAME;
241     boolean ascending = true;
242     if (StringUtils.isNotBlank(sort)) {
243       try {
244         SortCriterion sortCriterion = RestUtils.parseSortQueryParameter(sort).iterator().next();
245         sortKey = Sort.valueOf(sortCriterion.getFieldName().toUpperCase());
246         ascending = Order.Ascending == sortCriterion.getOrder() || Order.None == sortCriterion.getOrder();
247       } catch (WebApplicationException | IllegalArgumentException e) {
248         return badRequest(String.format("Invalid sort parameter `%s`", sort), e);
249       }
250     }
251     servers.sort(new ServerComparator(sortKey, ascending));
252 
253     offset = Math.min(offset, servers.size());
254     final List<Server> serverResults;
255     try {
256       serverResults = servers.subList(offset, Math.min(servers.size(), limit + offset));
257     } catch (IllegalArgumentException | IndexOutOfBoundsException e) {
258       return badRequest("Invalid offset and limit", e);
259     }
260 
261     Map<String, Object> result = new HashMap<>();
262     result.put("total", servers.size());
263     result.put("offset", offset);
264     result.put("count", serverResults.size());
265     result.put("limit", limit);
266     result.put("results", serverResults);
267     return Response.ok(gson.toJson(result)).build();
268   }

270   /**
271    * Get service statistics for all hosts and services
272    * @return List of all servers
273    * @throws ServiceRegistryException
274    *          If the host data could not be retrieved
275    */
276   private synchronized List<Server> getServerData() throws ServiceRegistryException {
277     List<Server> serverData = new ArrayList<>();
278 
279     for (HostRegistration host : serviceRegistry.getHostRegistrations()) {
280       // Calculate statistics per server
281       Server server = new Server();
282       server.id = host.getId();
283       server.online = host.isOnline();
284       server.maintenance = host.isMaintenanceMode();
285       server.hostname = host.getBaseUrl();
286       server.nodeName = host.getNodeName();
287       server.cores = host.getCores();
288       serverData.add(server);
289     }
290 
291     // Check if cache is still valid
292     if (lastUpdated + CACHE_SECONDS < Instant.now().getEpochSecond()) {
293       // Update cache
294       cachedServerData.clear();
295       logger.debug("Updating server data");
296       HostStatistics statistics = serviceRegistry.getHostStatistics();
297       for (HostRegistration host : serviceRegistry.getHostRegistrations()) {
298         // Calculate statistics per server
299         CachedServer server = new CachedServer();
300         server.running = statistics.runningJobs(host.getId());
301         server.queued = statistics.queuedJobs(host.getId());
302         cachedServerData.put(host.getId(), server);
303       }
304       lastUpdated = Instant.now().getEpochSecond();
305     }
306 
307     // Add info from cache
308     for (int i = 0; i < serverData.size(); i++) {
309       Optional<CachedServer> cachedServer = Optional.ofNullable(cachedServerData.get(serverData.get(i).id));
310       if (cachedServer.isPresent()) {
311         Server server = serverData.get(i);
312         server.running = cachedServer.get().running;
313         server.queued = cachedServer.get().queued;
314         serverData.set(i, server);
315       }
316     }
317 
318     return serverData;
319   }

323    *
324    * @param message
325    *          Message to send
326    * @param e
327    *          Exception to log. If <pre>null</pre>, a new exception is created to log a stack trace.
328    * @return 400 BAD REQUEST HTTP response
329    */
330   private Response badRequest(final String message, final Exception e) {
331     logger.debug(message, e == null && logger.isDebugEnabled() ? new IllegalArgumentException(message) : e);
332     return Response.status(Response.Status.BAD_REQUEST)
333         .entity(gson.toJson(message))
334         .build();
335   }
