94    private final Counter requests = Counter.build()
95        .name("requests_total")
96        .help("Total requests.")
97        .register();
98    private final Gauge jobLoadMax = Gauge.build()
99        .name("opencast_job_load_max")
100       .help("Maximum job load")
101       .labelNames("host")
102       .register();
103   private final Gauge jobLoadCurrent = Gauge.build()
104       .name("opencast_job_load_current")
105       .help("Maximum job load")
106       .labelNames("host")
107       .register();
108   private final Gauge jobsActive = Gauge.build()
109       .name("opencast_job_active")
110       .help("Active jobs")
111       .labelNames("host", "organization")
112       .register();
113   private final Gauge workflowsActive = Gauge.build()
114       .name("opencast_workflow_active")
115       .help("Active workflows")
116       .labelNames("organization")
117       .register();
118   private final Gauge servicesTotal = Gauge.build()
119       .name("opencast_services_total")
120       .help("Number of services in a cluster")
121       .labelNames("state")
122       .register();
123   private final Gauge version = Gauge.build()
124       .name("opencast_version")
125       .help("Version of Opencast (based on metrics module)")
126       .labelNames("part")
127       .register();
128   private Gauge eventsInAssetManager;

130   /** OSGi services */
131   private ServiceRegistry serviceRegistry;
132   private OrganizationDirectoryService organizationDirectoryService;
133   private AssetManager assetManager;

135   @Activate
136   public void activate(BundleContext bundleContext) {
137     final Version version = bundleContext.getBundle().getVersion();
138     this.version.labels("major").set(version.getMajor());
139     this.version.labels("minor").set(version.getMinor());
140     DefaultExports.initialize();
141   }
142 
143   @GET
144   @Path("/")
145   @Produces(TextFormat.CONTENT_TYPE_004)
146   @RestQuery(name = "metrics",
147       description = "Metrics about Opencast",
148       responses = {@RestResponse(description = "Metrics", responseCode = HttpServletResponse.SC_OK)},
149       returnDescription = "OpenMetrics about Opencast.")
150   public Response metrics() throws Exception {
151     // track requests
152     requests.inc();
153 
154     // track service states
155     final List<ServiceState> serviceStates = serviceRegistry.getServiceRegistrations().parallelStream()
156         .map(ServiceRegistration::getServiceState)
157         .collect(Collectors.toList());
158     final long error = serviceStates.parallelStream().filter(ServiceState.ERROR::equals).count();
159     final long warn = serviceStates.parallelStream().filter(ServiceState.WARNING::equals).count();
160     servicesTotal.labels(ServiceState.NORMAL.name()).set(serviceStates.size() - error - warn);
161     servicesTotal.labels(ServiceState.WARNING.name()).set(warn);
162     servicesTotal.labels(ServiceState.ERROR.name()).set(error);
163 
164     // track host loads
165     for (SystemLoad.NodeLoad nodeLoad: serviceRegistry.getCurrentHostLoads().getNodeLoads()) {
166       jobLoadCurrent.labels(nodeLoad.getHost()).set(nodeLoad.getCurrentLoad());
167       jobLoadMax.labels(nodeLoad.getHost()).set(nodeLoad.getMaxLoad());
168     }
169 
170     // set workflows by organization
171     for (var entry: serviceRegistry.countActiveTypeByOrganization("START_WORKFLOW").entrySet()) {
172       workflowsActive.labels(entry.getKey()).set(entry.getValue());
173     }
174 
175     // set jobs by organization and host
176     for (var entry: serviceRegistry.countActiveByOrganizationAndHost().entrySet()) {
177       final var org = entry.getKey();
178       for (Map.Entry<String, Long> orgEntry: entry.getValue().entrySet()) {
179         final var host = orgEntry.getKey();
180         final var count = orgEntry.getValue();
181         jobsActive.labels(host, org).set(count);
182       }
183     }
184 
185     // Get numbers from asset manager
186     if (assetManager != null && organizationDirectoryService.getOrganizations().size() == 1) {
187       var org = organizationDirectoryService.getOrganizations().get(0).getId();
188       eventsInAssetManager
189           .labels(org)
190           .set(assetManager.countEvents(null));
191     } else if (assetManager != null) {
192       for (Organization organization: organizationDirectoryService.getOrganizations()) {
193         eventsInAssetManager
194             .labels(organization.getId())
195             .set(assetManager.countEvents(organization.getId()));
196       }
197     }
198 
199     // collect metrics
200     final StringWriter writer = new StringWriter();
201     TextFormat.write004(writer, registry.metricFamilySamples());
202     return Response.ok().entity(writer.toString()).build();
203   }

205   @Reference
206   public void setServiceRegistry(ServiceRegistry service) {
207     this.serviceRegistry = service;
208   }
209 
210   @Reference
211   public void setOrganizationDirectoryService(OrganizationDirectoryService organizationDirectoryService) {
212     this.organizationDirectoryService = organizationDirectoryService;
213   }
214 
215   @Reference(
216       policy = ReferencePolicy.DYNAMIC,
217       cardinality = ReferenceCardinality.OPTIONAL,
218       unbind = "unsetAssetManager"
219   )
220   public void setAssetManager(AssetManager assetManager) {
221     this.assetManager = assetManager;
222     eventsInAssetManager = Gauge.build()
223         .name("opencast_asset_manager_events")
224         .help("Events in Asset Manager")
225         .labelNames("organization")
226         .register();
227   }
228 
229   public void unsetAssetManager(AssetManager assetManager) {
230     if (this.assetManager == assetManager) {
231       this.assetManager = null;
232       registry.unregister(eventsInAssetManager);
233     }
234   }
