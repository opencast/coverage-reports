112   // CALLBACKS with status - default events: recognitions.started, recognitions.completed, and recognitions.failed.

114   @GET
115   @Path("results")
116   @Produces(MediaType.TEXT_PLAIN)
117   @RestQuery(
118       name = "results",
119       description = "Called by the speech-to-text service when registering the callback url",
120       returnDescription = "Echo the string sent.",
121       restParameters = {
122           @RestParameter(
123               name = "challenge_string",
124               description = "String to be echoed in the response body",
125               isRequired = true,
126               type = Type.STRING
127           ),
128       },
129       responses = {
130           @RestResponse(responseCode = HttpServletResponse.SC_OK, description = "If no errors"),
131           @RestResponse(responseCode = HttpServletResponse.SC_INTERNAL_SERVER_ERROR, description = "An error occurred"),
132       }
133   )
134   // GET /transcripts/watson/results?challenge_string=51WRISEyAbuJq5fE HTTP/1.1" 302 5 "-" "Jersey/2.22.1 (Apache
135   // HttpClient 4.5)"
136   public Response checkCallbackUrl(@QueryParam("challenge_string") String challengeString,
137           @Context HttpServletRequest request) {
138     // Registering callback: just echo the challenge string
139     logger.debug("==== checkCallback got called: challenge_string is '{}'", challengeString);
140     Enumeration en = request.getHeaderNames();
141     while (en.hasMoreElements()) {
142       String name = (String) en.nextElement();
143       logger.debug("==== {}: {}", name, request.getHeader(name));
144     }
145     // return Response.ok(challengeString).build();
146     return Response.ok(challengeString).type(MediaType.TEXT_PLAIN).build();
147   }

149   /**
150    * { "id": "{job_id}", "event": "{recognitions_status}", "user_token": "{user_token}" } If the event is
151    * recognitions.completed_with_results, the object includes a results field that provides the results of the
152    * recognition request. The client should respond to the callback notification with status code 200.
153    *
154    */
155   @POST
156   @Path("results")
157   @Produces(MediaType.TEXT_PLAIN)
158   @RestQuery(
159       name = "results",
160       description = "Called by the speech-to-text service to report status.",
161       returnDescription = "",
162       responses = {
163           @RestResponse(responseCode = HttpServletResponse.SC_OK, description = "Got notification!"),
164       }
165   )
166   public Response reportStatus(String body) {
167     logger.trace("Body is: " + body);
168 
169     JSONObject jsonObj = null;
170     try {
171       JSONParser parser = new JSONParser();
172       jsonObj = (JSONObject) parser.parse(body);
173       // jsonObj = (JSONObject) parser.parse(request.getReader());
174       String mpId = (String) jsonObj.get("user_token");
175       String event = (String) jsonObj.get("event");
176       logger.info("Transcription notification for mp {} is {}", mpId, event);
177 
178       if (IBMWatsonTranscriptionService.JobEvent.COMPLETED_WITH_RESULTS.equals(event)) {
179         service.transcriptionDone(mpId, jsonObj);
180       } else if (IBMWatsonTranscriptionService.JobEvent.FAILED.equals(event)) {
181         service.transcriptionError(mpId, jsonObj);
182       }
183 
184       // return Response.ok().build();
185       return Response.ok().type(MediaType.APPLICATION_JSON).build();
186     } catch (ParseException e) {
187       logger.warn("{} occurred. Notification results could not be parsed: {}", e.getClass(),
188               jsonObj == null ? jsonObj : jsonObj.toJSONString());
189       return Response.status(HttpServletResponse.SC_INTERNAL_SERVER_ERROR).entity(e.getMessage()).build();
190     } catch (Exception e) {
191       logger.warn("Error handling notification results: {}", e.getMessage());
192       return Response.status(HttpServletResponse.SC_INTERNAL_SERVER_ERROR).entity(e.getMessage()).build();
193     }
194   }
