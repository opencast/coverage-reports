31  import org.osgi.framework.BundleContext;
32  import org.osgi.service.cm.ManagedService;
33  import org.osgi.service.component.annotations.Activate;
34  import org.osgi.service.component.annotations.Component;

36  import org.slf4j.LoggerFactory;
37  
38  import java.util.Dictionary;
39  import java.util.Hashtable;
40  import java.util.LinkedList;
41  import java.util.regex.Matcher;
42  import java.util.regex.Pattern;
43  
44  /**
45   * This dictionary service implementation applies a pattern
46   * to an input string - as many times as it matches - and
47   * returns the matches, separated by a space character.
48   */
49  @Component(
50      immediate = true,
51      service = { DictionaryService.class,ManagedService.class },
52      property = {
53          "service.description=Dictionary Service (Regular Expressions)",
54          "service.pid=org.opencastproject.dictionary.regexp.DictionaryServiceImpl"
55      }
56  )
57  public class DictionaryServiceImpl implements DictionaryService, ManagedService {
58  
59    /** The logging facility */
60    private static final Logger logger =
61        LoggerFactory.getLogger(DictionaryServiceImpl.class);
62  
63    public static final String PATTERN_CONFIG_KEY = "pattern";
64  
65    /* The regular expression to use for string matching */
66    private String pattern = "\\w+";
67  
68    /* The compiles pattern to use for matching */
69    private Pattern compilesPattern = Pattern.compile(pattern);
70  
71    public void setPattern(String p) {
72      try {
73        compilesPattern = Pattern.compile(p);
74        pattern = p;
75      } catch (RuntimeException e) {
76        logger.error("Failed to compile pattern '{}'", p);
77      }
78    }
79  
80    public String getPattern() {
81      return pattern;
82    }
83  
84    /**
85     * Load configuration
86     */
87    @Override
88    public synchronized void updated(Dictionary<String, ?> properties) {
89      if (properties != null && properties.get(PATTERN_CONFIG_KEY) != null) {
90        String pattern = properties.get(PATTERN_CONFIG_KEY).toString();
91        logger.info("Setting pattern for regexp based DictionaryService to '{}'", pattern);
92        setPattern(pattern);
93      }
94    }
95  
96    /**
97     * OSGi callback on component activation.
98     *
99     * @param  ctx  the bundle context
100    */
101   @Activate
102   void activate(BundleContext ctx) {
103     logger.info("Activating regexp based DictionaryService");
104     Dictionary<String, String> properties = new Hashtable<String, String>();
105     properties.put(ARTIFACT, "dictionary");
106     ctx.registerService(ReadinessIndicator.class.getName(),
107         new ReadinessIndicator(), properties);
108   }
109 
110   /**
111    * Filter the text according to the rules defined by the dictionary
112    * implementation used. This implementation uses a regular expression to find
113    * matching terms.
114    *
115    * @return filtered text
116    **/
117   @Override
118   public Textual cleanUpText(String text) {
119 
120     logger.debug("Text input: “{}”", text);
121     LinkedList<String> words = new LinkedList<String>();
122     Matcher matcher = compilesPattern.matcher(text);
123     while (matcher.find()) {
124       words.add(matcher.group());
125     }
126     String result = org.apache.commons.lang3.StringUtils.join(words, " ");
127     logger.debug("Resulting text: “{}”", result);
128     if ("".equals(result)) {
129       return null;
130     }
131     return new TextualImpl(result);
132   }
133 
134 }
