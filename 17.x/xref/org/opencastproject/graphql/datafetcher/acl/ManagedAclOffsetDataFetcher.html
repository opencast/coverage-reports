22  package org.opencastproject.graphql.datafetcher.acl;
23  
24  import org.opencastproject.authorization.xacml.manager.api.AclService;
25  import org.opencastproject.authorization.xacml.manager.api.AclServiceFactory;
26  import org.opencastproject.authorization.xacml.manager.api.ManagedAcl;
27  import org.opencastproject.graphql.acl.GqlManagedAclCatalogue;
28  import org.opencastproject.graphql.datafetcher.ParameterDataFetcher;
29  import org.opencastproject.graphql.execution.context.OpencastContext;
30  import org.opencastproject.security.api.SecurityService;
31  
32  import java.util.List;
33  
34  import graphql.schema.DataFetchingEnvironment;
35  
36  public class ManagedAclOffsetDataFetcher extends ParameterDataFetcher<GqlManagedAclCatalogue> {
37  
38    @Override
39    public GqlManagedAclCatalogue get(OpencastContext opencastContext, DataFetchingEnvironment dataFetchingEnvironment) {
40      SecurityService securityService = opencastContext.getService(SecurityService.class);
41      AclServiceFactory aclServiceFactory = opencastContext.getService(AclServiceFactory.class);
42  
43      AclService aclService = aclServiceFactory.serviceFor(securityService.getOrganization());
44  
45      List<ManagedAcl> acls = aclService.getAcls();
46  
47      return new GqlManagedAclCatalogue(acls);
48    }
49  
50  }
