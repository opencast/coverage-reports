22  package org.opencastproject.graphql.type.output;
23  
24  import org.opencastproject.security.api.AccessControlEntry;
25  import org.opencastproject.security.api.AccessControlList;
26  import org.opencastproject.security.api.Group;
27  import org.opencastproject.userdirectory.UserIdRoleProvider;
28  
29  import java.util.List;
30  import java.util.Map;
31  import java.util.Set;
32  import java.util.stream.Collectors;
33  
34  import graphql.annotations.annotationTypes.GraphQLDescription;
35  import graphql.annotations.annotationTypes.GraphQLField;
36  import graphql.annotations.annotationTypes.GraphQLName;
37  
38  @GraphQLName(GqlAccessControlList.TYPE_NAME)
39  @GraphQLDescription("Access control list")
40  public class GqlAccessControlList {
41  
42    public static final String TYPE_NAME = "AccessControlList";
43  
44    private final AccessControlList accessControlList;
45  
46    private final List<GqlAccessControlItem> items;
47  
48    public GqlAccessControlList(AccessControlList accessControlList) {
49      this.accessControlList = accessControlList;
50      this.items = getItems();
51    }
52  
53    @GraphQLField
54    public GqlAccessControlUserItem[] users() {
55      return items.stream()
56          .filter(i -> i instanceof GqlAccessControlUserItem)
57          .toArray(GqlAccessControlUserItem[]::new);
58    }
59  
60    @GraphQLField
61    public GqlAccessControlItem[] entries() {
62      return items.toArray(new GqlAccessControlItem[0]);
63    }
64  
65    private List<GqlAccessControlItem> getItems() {
66      String groupPrefix = Group.ROLE_PREFIX;
67      String userPrefix = UserIdRoleProvider.getUserIdRole("");
68  
69      Map<String, Set<AccessControlEntry>> entries = accessControlList.getEntries().stream()
70          .collect(Collectors.groupingBy(AccessControlEntry::getRole, Collectors.toSet()));
71  
72      return entries.entrySet().stream()
73          .map(e -> {
74            String role = e.getKey();
75            Set<AccessControlEntry> ace = e.getValue();
76            if (role.startsWith(groupPrefix)) {
77              return new GqlAccessControlGroupItem(ace);
78            } else if (role.startsWith(userPrefix)) {
79              return new GqlAccessControlUserItem(ace);
80            } else {
81              return new GqlAccessControlGenericItem(ace);
82            }
83          })
84          .collect(Collectors.toList());
85    }
86  
87  }
