61  /**
62   * Serves UI configuration files via REST
63   */
64  @Path("/ui/config")
65  @RestService(name = "UIConfigEndpoint",
66      title = "UI Config Endpoint",
67      abstractText = "Serves the configuration of the UI",
68      notes = {
69          "All paths above are relative to the REST endpoint base (something like http://your.server/files)",
70          "If the service is down or not working it will return a status 503, this means the the "
71              + "underlying service is not working and is either restarting or has failed",
72          "A status code 500 means a general failure has occurred which is not recoverable and was "
73              + "not anticipated.In other words, there is a bug! You should file an error report "
74              + "with your server logs from the timewhen the error occurred: "
75              + "<a href=\"https://github.com/opencast/opencast/issues\">Opencast Issue Tracker</a>"
76      }
77  )
78  @Component(
79      property = {
80          "service.description=UI Config REST Endpoint",
81          "opencast.service.type=org.opencastproject.uiconfig",
82          "opencast.service.path=/ui/config"
83      },
84      immediate = true,
85      service = UIConfigRest.class
86  )
87  @JaxrsResource
88  public class UIConfigRest {
89    /** The logger */
90    private static final Logger logger = LoggerFactory.getLogger(UIConfigRest.class);
91  
92    /** Configuration key for the ui config folder */
93    static final String UI_CONFIG_FOLDER_PROPERTY = "org.opencastproject.uiconfig.folder";
94    static final String X_ACCEL_REDIRECT_PROPERTY = "X-Accel-Redirect";
95  
96    /** Default Path for the ui configuration folder (relative to ${karaf.etc}) */
97    private static final String UI_CONFIG_FOLDER_DEFAULT = "ui-config";
98  
99    /** The currently used path to the configuration folder */
100   private String uiConfigFolder = null;
101   private String xAccelRedirect = null;
102 
103   /** The used SecurityService */
104   private SecurityService securityService;
105 
106   @Reference
107   protected void setSecurityService(SecurityService securityService) {
108     this.securityService = securityService;
109   }
110 
111   /**
112    * OSGI callback for activating this component
113    *
114    * @param context
115    *          the osgi component context
116    */
117   @Activate
118   @Modified
119   public void activate(BundleContext context, Map<String, String> properties)
120           throws ConfigurationException, IOException {
121     uiConfigFolder = properties.get(UI_CONFIG_FOLDER_PROPERTY);
122     logger.debug("UI configuration folder configured as '{}'", uiConfigFolder);
123     if (StringUtils.isNotEmpty(uiConfigFolder)) {
124       uiConfigFolder = new File(uiConfigFolder).getCanonicalPath();
125     } else {
126       final String karafEtc = context.getProperty("karaf.etc");
127       if (StringUtils.isBlank(karafEtc)) {
128         throw new ConfigurationException(String.format(
129             "%s not set and unable to fall back to default location based on ${karaf.etc}",
130             UI_CONFIG_FOLDER_PROPERTY));
131       }
132       uiConfigFolder = new File(karafEtc, UI_CONFIG_FOLDER_DEFAULT).getCanonicalPath();
133     }
134     logger.debug("UI configuration folder set to '{}'", uiConfigFolder);
135 
136     xAccelRedirect = properties.get(X_ACCEL_REDIRECT_PROPERTY);
137     logger.debug("X-Accel-Redirect path set to {}", xAccelRedirect);
138   }
139 
140   @GET
141   @Path("{component}/{filename : .+}")
142   @Produces(MediaType.WILDCARD)
143   @RestQuery(name = "getConfigFile",
144       description = "Returns the requested configuration file (json, css, etc..)",
145       pathParameters = {
146           @RestParameter(description = "Name of the component, which the configuration file belongs to",
147               isRequired = true, name = "component", type = RestParameter.Type.STRING),
148           @RestParameter(description = "Name of the configuration file", isRequired = true,
149               name = "filename", type = RestParameter.Type.STRING)
150       },
151       responses = {
152           @RestResponse(
153               description = "the requested configuration file",
154               responseCode = HttpServletResponse.SC_OK
155           ),
156           @RestResponse(
157               description = "if the configuration file doesn't exist",
158               responseCode = HttpServletResponse.SC_NOT_FOUND
159           ),
160       },
161       returnDescription = ""
162   )
163   public Response getConfigFile(@PathParam("component") String component, @PathParam("filename") String filename)
164           throws IOException, NotFoundException {
165     final String orgId = securityService.getOrganization().getId();
166     File configFile = Paths.get(uiConfigFolder, orgId, component, filename).toFile();
167 
168     try {
169       final String basePath = new File(uiConfigFolder, orgId).getCanonicalPath() + File.separator;
170       final String configFileCanPath = configFile.getCanonicalPath();
171 
172       // is configFile a subdirectory of basePath (additional directory traversal protection), if not stop
173       if (!configFileCanPath.startsWith(basePath)) {
174         logger.debug("Directory traversal prevented (trying to access '{}')", configFile.getPath());
175         throw new AccessDeniedException(configFileCanPath);
176       }
177 
178       // Falling back to default organization if file does not exist
179       if (!configFile.exists()) {
180         logger.debug("Falling back to default organization");
181         configFile = Paths.get(uiConfigFolder, DEFAULT_ORGANIZATION_ID, component, filename).toFile();
182       }
183 
184       if (xAccelRedirect != null) {
185         final String relative = Paths.get(uiConfigFolder).relativize(configFile.toPath()).toString();
186         return Response.noContent()
187             .header("X-Accel-Redirect", Paths.get(xAccelRedirect, relative).toString())
188             .build();
189       }
190 
191       // It is safe to pass the InputStream without closing it, JAX-RS takes care of that
192       return Response.ok(new FileInputStream(configFile))
193               .header("Content-Length", configFile.length())
194               .header("Content-Type", MimeTypes.getMimeType(filename))
195               .build();
196     } catch (FileNotFoundException e) {
197       logger.debug("Could not find requested configuration file '{}'", configFile.getPath(), e);
198       throw new NotFoundException();
199     }
200   }
