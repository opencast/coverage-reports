108   @Activate
109   protected synchronized void activate(ComponentContext cc) throws Exception {
110     /* Ensure a working directory is set */
111     if (workRoot == null) {
112       String dir = cc.getBundleContext().getProperty(PROPKEY_KARAF_DATA);
113       if (dir == null) {
114         throw new RuntimeException("Storage directory not defined.");
115       }
116       workRoot = new File(dir, DEFAULT_UPLOAD_WORKDIR);
117       logger.info("Chunk file upload directory set to {}.", workRoot.getAbsolutePath());
118     }
119 
120     // set up de-/serialization
121     ClassLoader cl = FileUploadJob.class.getClassLoader();
122     JAXBContext jctx = JAXBContext.newInstance("org.opencastproject.fileupload.api.job", cl);
123     jobMarshaller = jctx.createMarshaller();
124     jobUnmarshaller = jctx.createUnmarshaller();
125 
126     cleaner = new FileUploadServiceCleaner(this);
127     cleaner.schedule();
128 
129     logger.info("File Upload Service activated.");
130   }

133   protected void deactivate(ComponentContext cc) {
134     logger.info("File Upload Service deactivated");
135     cleaner.shutdown();

141     // try to get time-to-live threshold for jobs, use default if not configured

143     if (dir != null) {
144       workRoot = new File(dir);

148       jobMaxTTL = Integer.parseInt(((String) properties.get(PROPKEY_CLEANER_MAXTTL)).trim());

153     logger.info("Configuration updated. Jobs older than {} hours are deleted.", jobMaxTTL);

174   @Override
175   public FileUploadJob createJob(String filename, long filesize, int chunksize, MediaPackage mp,
176           MediaPackageElementFlavor flavor) throws FileUploadException {
177     FileUploadJob job = new FileUploadJob(filename, filesize, chunksize, mp, flavor);
178     logger.info("Creating new upload job: {}", job);
179 
180     try {
181       File jobDir = getJobDir(job.getId()); // create working dir
182       FileUtils.forceMkdir(jobDir);
183       ensureExists(getPayloadFile(job.getId())); // create empty payload file
184       storeJob(job); // create job file
185     } catch (FileUploadException e) {
186       deleteJob(job.getId());
187       throw fileUploadException(Severity.error, "Could not create job file in " + workRoot.getAbsolutePath(), e);
188     } catch (IOException e) {
189       deleteJob(job.getId());
190       throw fileUploadException(Severity.error,
191               "Could not create upload job directory in " + workRoot.getAbsolutePath(), e);
192     }
193     return job;
194   }

202   public boolean hasJob(String id) {
203     try {
204       if (jobCache.containsKey(id)) {
205         return true;
206       } else {
207         File jobFile = getJobFile(id);
208         return jobFile.exists();
209       }
210     } catch (Exception e) {
211       logger.warn("Error while looking for upload job: " + e.getMessage());
212       return false;
213     }

216   /**
217    * {@inheritDoc}
218    *
219    * @see org.opencastproject.fileupload.api.FileUploadService#getJob(String id)
220    */
221   @Override
222   public FileUploadJob getJob(String id) throws FileUploadException {
223     if (jobCache.containsKey(id)) { // job already cached?
224       return jobCache.get(id);
225     } else { // job not in cache?
226       try { // try to load job from filesystem
227         synchronized (this) {
228           File jobFile = getJobFile(id);
229           FileUploadJob job = null;
230           try (FileInputStream jobFileStream = new FileInputStream(jobFile)) {
231             job = (FileUploadJob) jobUnmarshaller.unmarshal(XmlSafeParser.parse(jobFileStream));
232           }
233           job.setLastModified(jobFile.lastModified()); // get last modified time from job file
234           return job;
235         } // if loading from fs also fails
236       } catch (Exception e) { // we could not find the job and throw an Exception
237         throw fileUploadException(Severity.warn, "Failed to load job " + id + " from file.", e);
238       }
239     }
240   }

242   /**
243    * {@inheritDoc}
244    *
245    * @see org.opencastproject.fileupload.api.FileUploadService#cleanOutdatedJobs()
246    */
247   @Override
248   public void cleanOutdatedJobs() throws IOException {
249     File[] workRootFiles = workRoot.listFiles();
250     if (workRootFiles == null) {
251       logger.trace("No outdated files found in {}", workRoot.getAbsolutePath());
252       return;
253     }
254     for (File dir : workRoot.listFiles()) {
255       if (dir.getParentFile().equals(workRoot) && dir.isDirectory()) {
256         try {
257           String id = dir.getName(); // assuming that the dir name is the ID of a job..
258           if (!isLocked(id)) { // ..true if not in cache or job is in cache and not locked
259             FileUploadJob job = getJob(id);
260             Calendar cal = Calendar.getInstance();
261             cal.add(Calendar.HOUR, -jobMaxTTL);
262             if (job.lastModified() < cal.getTimeInMillis()) {
263               FileUtils.forceDelete(dir);
264               jobCache.remove(id);
265               logger.info("Deleted outdated job {}", id);
266             }
267           }
268         } catch (Exception e) { // something went wrong, so we assume the dir is corrupted
269           FileUtils.forceDelete(dir); // ..and delete it right away
270           logger.info("Deleted corrupted job {}", dir.getName());
271         }
272       }
273     }
274   }

276   private void storeJob(FileUploadJob job) throws FileUploadException {
277     try {
278       synchronized (this) {
279         logger.debug("Attempting to store job {}", job.getId());
280         File jobFile = ensureExists(getJobFile(job.getId()));
281         jobMarshaller.marshal(job, jobFile);
282       }
283     } catch (Exception e) {
284       throw fileUploadException(Severity.error, "Failed to write job file.", e);
285     }
286   }

307   @Override
308   public void acceptChunk(FileUploadJob job, long chunkNumber, InputStream content) throws FileUploadException {
309     // job already completed?
310     if (job.getState().equals(FileUploadJob.JobState.COMPLETE)) {
311       removeFromCache(job);
312       throw fileUploadException(Severity.warn, "Job is already complete.");
313     }
314 
315     // job ready to receive data?
316     if (isLocked(job.getId())) {
317       throw fileUploadException(Severity.error,
318               "Job is locked. Seems like a concurrent upload to this job is in progress.");
319     } else {
320       lock(job);
321     }
322 
323     // right chunk offered?
324     int supposedChunk = job.getCurrentChunk().getNumber() + 1;
325     if (chunkNumber != supposedChunk) {
326       removeFromCache(job);
327       throw fileUploadException(Severity.error,
328               format("Wrong chunk number. Awaiting #%d but #%d was offered.", supposedChunk, chunkNumber));
329     }
330     logger.debug("Receiving chunk #" + chunkNumber + " of job {}", job);
331 
332     // write chunk to temp file
333     job.getCurrentChunk().incrementNumber();
334     File chunkFile = null;
335     try {
336       chunkFile = ensureExists(getChunkFile(job.getId()));
337     } catch (IOException e) {
338       throw fileUploadException(Severity.error, "Cannot create chunk file", e);
339     }
340     OutputStream out = null;
341     try {
342       byte[] readBuffer = new byte[READ_BUFFER_LENGTH];
343       out = new FileOutputStream(chunkFile, false);
344       int bytesRead = 0;
345       long bytesReadTotal = 0L;
346       Chunk currentChunk = job.getCurrentChunk(); // copy manually (instead of using IOUtils.copy()) so we can count the
347       // number of bytes
348       do {
349         bytesRead = content.read(readBuffer);
350         if (bytesRead > 0) {
351           out.write(readBuffer, 0, bytesRead);
352           bytesReadTotal += bytesRead;
353           currentChunk.setReceived(bytesReadTotal);
354         }
355       } while (bytesRead != -1);
356       if (job.getPayload().getTotalSize() == -1 && job.getChunksTotal() == 1) { // set totalSize in case of ordinary
357         // from submit
358         job.getPayload().setTotalSize(bytesReadTotal);
359       }
360     } catch (Exception e) {
361       removeFromCache(job);
362       throw fileUploadException(Severity.error, "Failed to store chunk data", e);
363     } finally {
364       IOUtils.closeQuietly(content);
365       IOUtils.closeQuietly(out);
366     }
367 
368     // check if chunk has right size
369     long actualSize = chunkFile.length();
370     long supposedSize;
371     if (chunkNumber == job.getChunksTotal() - 1) {
372       supposedSize = job.getPayload().getTotalSize() % job.getChunksize();
373       supposedSize = supposedSize == 0 ? job.getChunksize() : supposedSize; // a not so nice workaround for the rare
374       // case that file size is a multiple of the
375       // chunk size
376     } else {
377       supposedSize = job.getChunksize();
378     }
379     if (actualSize == supposedSize || (job.getChunksTotal() == 1 && job.getChunksize() == -1)) {
380 
381       // append chunk to payload file
382       FileInputStream in = null;
383       try {
384         File payloadFile = getPayloadFile(job.getId());
385         in = new FileInputStream(chunkFile);
386         out = new FileOutputStream(payloadFile, true);
387         IOUtils.copy(in, out);
388         Payload payload = job.getPayload();
389         payload.setCurrentSize(payload.getCurrentSize() + actualSize);
390 
391       } catch (IOException e) {
392         removeFromCache(job);
393         throw fileUploadException(Severity.error, "Failed to append chunk data", e);
394 
395       } finally {
396         IOUtils.closeQuietly(in);
397         IOUtils.closeQuietly(out);
398         deleteChunkFile(job.getId());
399       }
400 
401     } else {
402       removeFromCache(job);
403       throw fileUploadException(Severity.warn,
404               format("Chunk has wrong size. Awaited: %d bytes, received: %d bytes.", supposedSize, actualSize));
405     }
406 
407     // update job
408     if (chunkNumber == job.getChunksTotal() - 1) { // upload is complete
409       finalizeJob(job);
410       logger.info("Upload job completed: {}", job);
411     } else {
412       job.setState(FileUploadJob.JobState.READY); // upload still incomplete
413     }
414     storeJob(job);
415     removeFromCache(job);
416   }

418   @Override
419   public InputStream getPayload(FileUploadJob job) throws FileUploadException {
420     // job not locked?
421     if (isLocked(job.getId())) {
422       throw fileUploadException(Severity.warn,
423               "Job is locked. Download is only permitted while no upload to this job is in progress.");
424     }
425 
426     try {
427       FileInputStream payload = new FileInputStream(getPayloadFile(job.getId()));
428       return payload;
429     } catch (FileNotFoundException e) {
430       throw fileUploadException(Severity.error, "Failed to retrieve file from job " + job.getId(), e);
431     }
432   }

445   /**
446    * Returns true if the job with the given ID is currently locked.
447    *
448    * @param id
449    *          ID of the job in question
450    * @return true if job is locked, false otherwise
451    */
452   private boolean isLocked(String id) {
453     if (jobCache.containsKey(id)) {
454       FileUploadJob job = jobCache.get(id);
455       return job.getState().equals(FileUploadJob.JobState.INPROGRESS)
456               || job.getState().equals(FileUploadJob.JobState.FINALIZING);
457     } else {
458       return false;
459     }
460   }

472   /**
473    * Unlocks an finalizes an upload job.
474    *
475    * @param job
476    *          job to finalize
477    * @throws FileUploadException
478    */
479   private void finalizeJob(FileUploadJob job) throws FileUploadException {
480     job.setState(FileUploadJob.JobState.FINALIZING);
481 
482     if (job.getPayload().getMediaPackage() == null) { // do we have a target mediaPackge ?
483       job.getPayload().setUrl(putPayloadIntoCollection(job)); // if not, put file into upload collection in WFR
484     } else {
485       job.getPayload().setUrl(putPayloadIntoMediaPackage(job)); // else add file to target MP
486     }
487     deletePayloadFile(job.getId()); // delete payload in temp directory
488 
489     job.setState(FileUploadJob.JobState.COMPLETE);
490   }

492   /** Function that writes the given file to the uploaded collection. */
493   private Function2<InputStream, File, Option<URI>> putInCollection = new Function2<InputStream, File, Option<URI>>() {
494 
495     @Override
496     public Option<URI> apply(InputStream is, File f) {
497       try {
498         URI uri = workspace.putInCollection(UPLOAD_COLLECTION, f.getName(), is); // storing file with jod id as name
499         // instead of original filename to
500         // avoid collisions (original filename
501         // can be obtained from upload job)
502         return Option.some(uri);
503       } catch (IOException e) {
504         logger.error("Could not add file to collection.", e);
505         return Option.none();
506       }
507     }
508   };

510   /**
511    * Puts the payload of an upload job into the upload collection in the WFR and returns the URL to the file in the WFR.
512    *
513    * @param job
514    * @return URL of the file in the WFR
515    * @throws FileUploadException
516    */
517   private URL putPayloadIntoCollection(FileUploadJob job) throws FileUploadException {
518     logger.info("Moving payload of job " + job.getId() + " to collection " + UPLOAD_COLLECTION);
519     Option<URI> result = IoSupport.withFile(getPayloadFile(job.getId()), putInCollection).flatMap(
520             Functions.<Option<URI>> identity());
521     if (result.isSome()) {
522       try {
523         return result.get().toURL();
524       } catch (MalformedURLException e) {
525         throw fileUploadException(Severity.error, "Unable to return URL of payloads final destination.", e);
526       }
527     } else {
528       throw fileUploadException(Severity.error, "Failed to put payload in collection.");
529     }
530   }

532   /**
533    * Puts the payload of an upload job into a MediaPackage in the WFR, adds the files as a track to the MediaPackage and
534    * returns the files URL in the WFR.
535    *
536    * @param job
537    * @return URL of the file in the WFR
538    * @throws FileUploadException
539    */
540   private URL putPayloadIntoMediaPackage(FileUploadJob job) throws FileUploadException {
541     MediaPackage mediaPackage = job.getPayload().getMediaPackage();
542     MediaPackageElementFlavor flavor = job.getPayload().getFlavor();
543     List<Track> excludeTracks = Arrays.asList(mediaPackage.getTracks(flavor));
544 
545     FileInputStream fileInputStream = null;
546     try {
547       fileInputStream = new FileInputStream(getPayloadFile(job.getId()));
548       MediaPackage mp = ingestService.addTrack(fileInputStream, job.getPayload().getFilename(), job.getPayload()
549               .getFlavor(), mediaPackage);
550 
551       List<Track> tracks = new ArrayList<Track>(Arrays.asList(mp.getTracks(flavor)));
552       tracks.removeAll(excludeTracks);
553       if (tracks.size() != 1) {
554         throw new FileUploadException("Ingested track not found");
555       }
556 
557       return tracks.get(0).getURI().toURL();
558     } catch (Exception e) {
559       throw fileUploadException(Severity.error, "Failed to add payload to MediaPackage.", e);
560     } finally {
561       IOUtils.closeQuietly(fileInputStream);
562     }
563   }

654   private FileUploadException fileUploadException(Severity severity, String msg) throws FileUploadException {
655     switch (severity) {
656       case warn:
