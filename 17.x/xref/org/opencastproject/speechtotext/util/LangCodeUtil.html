22  package org.opencastproject.speechtotext.util;
23  
24  import java.util.HashMap;
25  import java.util.Locale;
26  import java.util.Map;
27  
28  /**
29   * Utility class for language codes
30   */
31  public final class LangCodeUtil {
32  
33    // Prevent instantiation because of Checkstyle: "HideUtilityClassConstructor"
34    private LangCodeUtil() {
35      throw new UnsupportedOperationException("Utility class should not have a public or default constructor");
36    }
37  
38    /** Map to get ISO 639 language code for language name in English */
39    private static final Map<String, String> iso3ToIso2Map = new HashMap<>();
40  
41    /** Map to get ISO 639 language code for 3-letter language code */
42    private static final Map<String, String> langNameToIso2Map = new HashMap<>();
43  
44    static {
45      for (String languageCode : Locale.getISOLanguages()) {
46        Locale locale = new Locale(languageCode);
47        String languageName = locale.getDisplayLanguage(new Locale("en"));
48        langNameToIso2Map.put(languageName, languageCode);
49        String languageISO3 = locale.getISO3Language();
50        iso3ToIso2Map.put(languageISO3, languageCode);
51      }
52    }
53  
54    /**
55     * Convert ISO 639-3 language code to ISO 639-2 language code.
56     *
57     * @param langIso3Code The ISO 639-3 language code to convert.
58     * @param defaultValue The default value to return if the conversion fails.
59     * @return The ISO 639-2 language code, or the default value if the conversion fails.
60     */
61    public static String iso3ToIso2(String langIso3Code, String defaultValue) {
62      return iso3ToIso2Map.getOrDefault(langIso3Code, defaultValue);
63    }
64  
65    /**
66     * Get the ISO 639-2 language code for a given language name in English.
67     *
68     * @param langNameInEnglish The language name in English.
69     * @param defaultValue      The default value to return if the conversion fails.
70     * @return The ISO 639-2 language code, or the default value if the conversion fails.
71     */
72    public static String getIso2FromLang(String langNameInEnglish, String defaultValue) {
73      return langNameToIso2Map.getOrDefault(langNameInEnglish, defaultValue);
74    }
75  
76  }
