47  /**
48   * A remote youtube service invoker.
49   */
50  @Component(
51      immediate = true,
52      service = YouTubePublicationService.class,
53      property = {
54          "service.description=Publication (youTube) Remote Service Proxy"
55      }
56  )
57  public class YouTubePublicationServiceRemoteImpl extends RemoteBase implements YouTubePublicationService {
58  
59    /** The logger */
60    private static final Logger logger = LoggerFactory.getLogger(YouTubePublicationServiceRemoteImpl.class);
61  
62    public YouTubePublicationServiceRemoteImpl() {
63      super(JOB_TYPE);
64    }
65  
66    @Override
67    public Job publish(MediaPackage mediaPackage, Track track) throws PublicationException {
68      final String trackId = track.getIdentifier();
69      List<BasicNameValuePair> params = new ArrayList<BasicNameValuePair>();
70      params.add(new BasicNameValuePair("mediapackage", MediaPackageParser.getAsXml(mediaPackage)));
71      params.add(new BasicNameValuePair("elementId", trackId));
72      HttpPost post = new HttpPost();
73      HttpResponse response = null;
74      try {
75        post.setEntity(new UrlEncodedFormEntity(params, "UTF-8"));
76        response = getResponse(post);
77        if (response != null) {
78          logger.info("Publishing {} to youtube", trackId);
79          return JobParser.parseJob(response.getEntity().getContent());
80        }
81      } catch (Exception e) {
82        throw new PublicationException("Unable to publish track " + trackId + " from mediapackage "
83                + mediaPackage + " using a remote youtube publication service", e);
84      } finally {
85        closeConnection(response);
86      }
87      throw new PublicationException("Unable to publish track " + trackId + " from mediapackage "
88              + mediaPackage + " using a remote youtube publication service");
89    }
90  
91    @Override
92    public Job retract(MediaPackage mediaPackage) throws PublicationException {
93      List<BasicNameValuePair> params = new ArrayList<BasicNameValuePair>();
94      params.add(new BasicNameValuePair("mediapackage", MediaPackageParser.getAsXml(mediaPackage)));
95      HttpPost post = new HttpPost("/retract");
96      HttpResponse response = null;
97      try {
98        post.setEntity(new UrlEncodedFormEntity(params, "UTF-8"));
99        response = getResponse(post);
100       if (response != null) {
101         logger.info("Retracting {} from youtube", mediaPackage);
102         return JobParser.parseJob(response.getEntity().getContent());
103       }
104     } catch (Exception e) {
105       throw new PublicationException("Unable to retract mediapackage " + mediaPackage
106               + " using a remote youtube publication service", e);
107     } finally {
108       closeConnection(response);
109     }
110     throw new PublicationException("Unable to retract mediapackage " + mediaPackage
111             + " using a remote youtube publication service");
112   }
113 
114   @Reference
115   @Override
116   public void setTrustedHttpClient(TrustedHttpClient trustedHttpClient) {
117     super.setTrustedHttpClient(trustedHttpClient);
118   }
119 
120   @Reference
121   @Override
122   public void setRemoteServiceManager(ServiceRegistry serviceRegistry) {
123     super.setRemoteServiceManager(serviceRegistry);
124   }
125 
126 }
