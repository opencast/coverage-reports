22  package org.opencastproject.index.service.resources.list.query;
23  
24  import org.opencastproject.index.service.resources.list.provider.GroupsListProvider;
25  import org.opencastproject.index.service.util.FiltersUtils;
26  import org.opencastproject.list.api.ResourceListFilter;
27  import org.opencastproject.list.api.ResourceListFilter.SourceType;
28  import org.opencastproject.list.impl.ResourceListQueryImpl;
29  import org.opencastproject.util.data.Option;
30  
31  /**
32   * Query for the users list.
33   *
34   * The following filters can be used:
35   * <ul>
36   * <li>name</li>
37   * <li>role</li>
38   * <li>provider</li>
39   * </ul>
40   */
41  public class GroupsListQuery extends ResourceListQueryImpl {
42  
43    public static final String FILTER_NAME_NAME = "Name";
44    private static final String FILTER_NAME_LABEL = "FILTERS.USERS.NAME.LABEL";
45  
46    public static final String FILTER_TEXT_NAME = "textFilter";
47  
48    public GroupsListQuery() {
49      super();
50    }
51  
52    /**
53     * Add a {@link ResourceListFilter} filter to the query with the given name
54     *
55     * @param name
56     *          the name to filter for
57     */
58    public void withName(String name) {
59      this.addFilter(createNameFilter(Option.option(name)));
60    }
61  
62    /**
63     * Returns an {@link Option} containing the name used to filter if set
64     *
65     * @return an {@link Option} containing the name or none.
66     */
67    public Option<String> getName() {
68      return this.getFilterValue(FILTER_NAME_NAME);
69    }
70  
71    /**
72     * Create a new {@link ResourceListFilter} based on a name
73     *
74     * @param name
75     *          the name to filter on wrapped in an {@link Option} or {@link Option#none()}
76     * @return a new {@link ResourceListFilter} for a name based query
77     */
78    public static ResourceListFilter<String> createNameFilter(Option<String> name) {
79      return FiltersUtils.generateFilter(name, FILTER_NAME_NAME, FILTER_NAME_LABEL, SourceType.SELECT,
80              Option.some(GroupsListProvider.NAME));
81    }
82  
83  }
