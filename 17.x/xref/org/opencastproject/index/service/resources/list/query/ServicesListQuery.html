26  import org.opencastproject.index.service.resources.list.provider.ServersListProvider;

34  /**
35   * Query for the services list.
36   *
37   * The following filters can be used:
38   * <ul>
39   * <li>name</li>
40   * <li>host</li>
41   * <li>status</li>
42   * <li>actions</li>
43   * </ul>
44   */
45  public class ServicesListQuery extends ResourceListQueryImpl {
46  
47    /** Prefix for the filter labels. */
48    private static final String FILTER_PREFIX = "FILTERS.SERVICES";
49    /** Name filter name. */
50    public static final String FILTER_NAME_NAME = "name";
51    /** Name filter label. */
52    public static final String FILTER_LABEL_NAME = FILTER_PREFIX + ".NAME.LABEL";
53    /** Hostname filter name. */
54    public static final String FILTER_NAME_HOSTNAME = "hostname";
55    /** Hostname filter label. */
56    public static final String FILTER_LABEL_HOSTNAME = FILTER_PREFIX + ".HOSTNAME.LABEL";
57    /** NodeName filter name. */
58    public static final String FILTER_NAME_NODE_NAME = "nodeName";
59    /** NodeName filter label. */
60    public static final String FILTER_LABEL_NODE_NAME = FILTER_PREFIX + ".NODE_NAME.LABEL";
61    /** Status filter name. */
62    public static final String FILTER_NAME_STATUS = "status";
63    /** Status filter label. */
64    public static final String FILTER_LABEL_STATUS = FILTER_PREFIX + ".STATUS.LABEL";
65    /** Actions filter name. */
66    public static final String FILTER_NAME_ACTIONS = "actions";
67    /** Actions filter label. */
68    public static final String FILTER_LABEL_ACTIONS = FILTER_PREFIX + ".ACTIONS.LABEL";
69  
70    /** Default constructor. */
71    public ServicesListQuery() {
72      super();
73      this.availableFilters.add(createNameFilter(Option.<String> none()));
74      this.availableFilters.add(createHostnameFilter(Option.<String> none()));
75      this.availableFilters.add(createNodeNameFilter(Option.<String> none()));
76      this.availableFilters.add(createStatusFilter(Option.<String> none()));
77      this.availableFilters.add(createActionsFilter(Option.<Boolean> none()));
78    }
79  
80    /**
81     * Copy constructor for the base class {@code ResourceListQuery}.
82     *
83     * @param query copy values from the given query
84     */
85    public ServicesListQuery(ResourceListQuery query) {
86      this();
87      availableFilters.addAll(query.getAvailableFilters());
88  
89      for (ResourceListFilter filter : query.getFilters())
90        addFilter(filter);
91  
92      sortBy = query.getSortBy();
93      if (query.getOffset().isSome())
94        setOffset(query.getOffset().get());
95      if (query.getLimit().isSome())
96        setLimit(query.getLimit().get());
97    }
98

153   /**
154    * Returns an {@link Option} containing the name used to filter if set.
155    * {@link Option#none()} otherwise.
156    *
157    * @return an {@link Option} containing the name or none.
158    */
159   public Option<String> getName() {
160     return getFilterValue(FILTER_NAME_NAME);
161   }
162 
163   /**
164    * Returns an {@link Option} containing the hostname used to filter if set.
165    * {@link Option#none()} otherwise.
166    *
167    * @return an {@link Option} containing the hostname or none.
168    */
169   public Option<String> getHostname() {
170     return getFilterValue(FILTER_NAME_HOSTNAME);
171   }
172 
173   /**
174    * Returns an {@link Option} containing the node name used to filter if set.
175    * {@link Option#none()} otherwise.
176    *
177    * @return an {@link Option} containing the node name or none.
178    */
179   public Option<String> getNodeName() {
180     return getFilterValue(FILTER_NAME_NODE_NAME);
181   }
182 
183   /**
184    * Returns an {@link Option} containing the status used to filter if set.
185    * {@link Option#none()} otherwise.
186    *
187    * @return an {@link Option} containing the status or none.
188    */
189   public Option<String> getStatus() {
190     return getFilterValue(FILTER_NAME_STATUS);
191   }
192 
193   /**
194    * Returns an {@link Option} containing boolean value used to filter any actions if set.
195    * {@link Option#none()} otherwise.
196    *
197    * @return an {@link Option} containing the boolean value for any actions or none.
198    */
199   public Option<Boolean> getActions() {
200     Option result = getFilterValue(FILTER_NAME_ACTIONS);
201     if (result.isSome()) {
202       if (result.get() instanceof String) {
203         return BooleanListProvider.parseValue((String)result.get());
204       }
205     }
206     return result;
207   }
208 
209   /**
210    * Returns an {@link Option} containing the free text used to filter if set.
211    * {@link Option#none()} otherwise.
212    *
213    * @return an {@link Option} containing the free text or none.
214    */
215   public Option<String> getFreeText() {
216     return getFilterValue(ResourceListFilter.FREETEXT);
217   }

219   /**
220    * Create a new {@link ResourceListFilter} based on a name.
221    *
222    * @param value the name to filter on wrapped in an {@link Option} or {@link Option#none()}
223    * @return a new {@link ResourceListFilter} for a name based query
224    */
225   public static <String> ResourceListFilter<String> createNameFilter(Option<String> value) {
226     return FiltersUtils.generateFilter(
227             value,
228             FILTER_NAME_NAME,
229             FILTER_LABEL_NAME,
230             ResourceListFilter.SourceType.SELECT,
231             Option.some(ServicesListProvider.LIST_NAME));
232   }
