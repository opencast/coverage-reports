22  package org.opencastproject.index.service.catalog.adapter.series;
23  
24  import static java.util.Objects.requireNonNull;
25  
26  import org.opencastproject.mediapackage.MediaPackageElementFlavor;
27  import org.opencastproject.mediapackage.MediaPackageElements;
28  import org.opencastproject.metadata.dublincore.DublinCoreCatalog;
29  import org.opencastproject.security.api.UnauthorizedException;
30  import org.opencastproject.series.api.SeriesException;
31  import org.opencastproject.util.NotFoundException;
32  
33  import com.entwinemedia.fn.data.Opt;
34  
35  import org.osgi.service.cm.ManagedService;
36  import org.slf4j.Logger;
37  import org.slf4j.LoggerFactory;
38  
39  public class CommonSeriesCatalogUIAdapter extends ConfigurableSeriesDCCatalogUIAdapter implements ManagedService {
40  
41    private static final Logger logger = LoggerFactory.getLogger(CommonSeriesCatalogUIAdapter.class);
42  
43    @Override
44    public MediaPackageElementFlavor getFlavor() {
45      return MediaPackageElements.SERIES;
46    }
47  
48    @Override
49    public String getUITitle() {
50      return "Opencast Series DublinCore";
51    }
52  
53    @Override
54    protected Opt<DublinCoreCatalog> loadDublinCoreCatalog(String seriesId) {
55      try {
56        return Opt.nul(getSeriesService().getSeries(requireNonNull(seriesId)));
57      } catch (SeriesException e) {
58        logger.error("Error while loading DublinCore catalog of series '{}':", seriesId, e);
59        return Opt.none();
60      } catch (NotFoundException e) {
61        logger.debug("No DublinCore metadata catalog for series '{}' found", seriesId);
62        return Opt.none();
63      } catch (UnauthorizedException e) {
64        logger.warn(
65                "The current user does not have sufficient permissions to load the DublinCore metadata catalog of the series '{}'",
66                seriesId);
67        return Opt.none();
68      }
69    }
70  
71    @Override
72    protected boolean saveDublinCoreCatalog(String seriesId, DublinCoreCatalog dc) {
73      try {
74        getSeriesService().updateSeries(dc);
75      } catch (SeriesException e) {
76        logger.warn("Error while updating series DublinCore", e);
77        return false;
78      } catch (UnauthorizedException e) {
79        logger.warn("User is not authorized to change series DublinCore");
80        return false;
81      }
82      return true;
83    }
84  
85  }
