161    * Test of addClip(s) methods, of class SmilServiceImpl.

163   @Test
164   public void testAddClip() throws Exception {
165     TrackImpl videoTrack = new TrackImpl();
166     videoTrack.setIdentifier("track-1");
167     videoTrack.setFlavor(new MediaPackageElementFlavor("source", "presentation"));
168     videoTrack.setURI(new URI("http://hostname/video.mp4"));
169     videoTrack.addStream(new VideoStreamImpl());
170     videoTrack.setDuration(1000000000000L);
171 
172     SmilResponse smilResponse = smilService.createNewSmil();
173     smilResponse = smilService.addParallel(smilResponse.getSmil());
174     SmilMediaContainer par = (SmilMediaContainer) smilResponse.getEntity();
175     // add video track into parallel element
176     smilResponse = smilService.addClip(smilResponse.getSmil(), par.getId(), videoTrack, 1000L, 1000000L);
177     // logger.info(smilResponse.getSmil().toXML());
178     SmilMediaObject media = null;
179     for (SmilObject entity : smilResponse.getEntities()) {
180       if (entity instanceof SmilMediaObject) {
181         media = (SmilMediaObject) entity;
182         break;
183       }
184     }
185     assertNotNull(media);
186     assertEquals(media.getId(), ((SmilMediaContainer) smilResponse.getSmil().getBody().getMediaElements().get(0))
187             .getElements().get(0).getId());
188     assertTrue(media instanceof SmilMediaVideoImpl);
189     assertSame(((SmilMediaElement) media).getMediaType(), SmilMediaElement.MediaType.VIDEO);
190     // 1000 milliseconds = 1 second
191     assertEquals(1000L, ((SmilMediaElement) media).getClipBeginMS());
192     // duration is 1000000 milliseconds = 1000 soconds
193     // start + duration = 1s + 1000s = 1001s
194     assertEquals(1001000L, ((SmilMediaElement) media).getClipEndMS());
195 
196     TrackImpl audioTrack = new TrackImpl();
197     audioTrack.setIdentifier("track-2");
198     audioTrack.setFlavor(new MediaPackageElementFlavor("source", "presenter"));
199     audioTrack.setURI(new URI("http://hostname/audio.mp3"));
200     audioTrack.addStream(new AudioStreamImpl());
201     audioTrack.setDuration(1000000000000L);
202 
203     // add audio track into parallel element
204     smilResponse = smilService.addClip(smilResponse.getSmil(), par.getId(), audioTrack, 1000L, 1000000L);
205     // logger.info(smilResponse.getSmil().toXML());
206     media = null;
207     for (SmilObject entity : smilResponse.getEntities()) {
208       if (entity instanceof SmilMediaObject) {
209         media = (SmilMediaObject) entity;
210         break;
211       }
212     }
213     assertNotNull(media);
214     assertEquals(media.getId(), ((SmilMediaContainer) smilResponse.getSmil().getBody().getMediaElements().get(0))
215         .getElements().get(1).getId());
216     assertTrue(media instanceof SmilMediaAudioImpl);
217     assertSame(((SmilMediaElement) media).getMediaType(), SmilMediaElement.MediaType.AUDIO);
218     // 1000 milliseconds = 1 second
219     assertEquals(1000L, ((SmilMediaElement) media).getClipBeginMS());
220     // duration is 1000000 milliseconds = 1000 soconds
221     // start + duration = 1s + 1000s = 1001s
222     assertEquals(1001000L, ((SmilMediaElement) media).getClipEndMS());
223 
224     // add new par
225     smilResponse = smilService.addParallel(smilResponse.getSmil());
226     par = (SmilMediaContainer) smilResponse.getEntity();
227     // add tracks (as array) to par
228     smilResponse = smilService.addClips(smilResponse.getSmil(), par.getId(),
229             new Track[]{audioTrack, videoTrack}, 15000L, 1000L);
230     // logger.info(smilResponse.getSmil().toXML());
231     assertSame(2, smilResponse.getEntitiesCount());
232     assertTrue(smilResponse.getEntities()[0] instanceof SmilMediaElement);
233     // get audio element
234     SmilMediaElement mediaElement = (SmilMediaElement) smilResponse.getEntities()[0];
235     assertTrue(mediaElement.getMediaType() == SmilMediaElement.MediaType.AUDIO);
236     // 15000ms = 15s
237     assertEquals(15000L, mediaElement.getClipBeginMS());
238     // start + duration = 15s + 1s = 16s
239     assertEquals(16000L, mediaElement.getClipEndMS());
240     // get video element
241     mediaElement = (SmilMediaElement) smilResponse.getEntities()[1];
242     assertTrue(mediaElement.getMediaType() == SmilMediaElement.MediaType.VIDEO);
243     // 15000ms = 15s
244     assertEquals(15000L, mediaElement.getClipBeginMS());
245     // start + duration = 15s + 1s = 16s
246     assertEquals(16000L, mediaElement.getClipEndMS());
247   }

301   /**
302    * passing a previously created paramGroupId
303    */
304   @Test
305   public void testAddClipWithParamGroupId() throws Exception {
306     // first, create SMIL with 1 par, 1 video
307     TrackImpl videoTrack = new TrackImpl();
308     videoTrack.setIdentifier("presenter-track-1");
309     videoTrack.setFlavor(new MediaPackageElementFlavor("source", "presenter"));
310     videoTrack.setURI(new URI("http://hostname/video.mp4"));
311     videoTrack.addStream(new VideoStreamImpl());
312     videoTrack.setDuration(1000000000000L);
313 
314     SmilResponse smilResponse = smilService.createNewSmil();
315     smilResponse = smilService.addParallel(smilResponse.getSmil());
316     SmilMediaContainer par = (SmilMediaContainer) smilResponse.getEntity();
317     // add video track into parallel element
318     smilResponse = smilService.addClip(smilResponse.getSmil(), par.getId(), videoTrack, 1000L, 1000000L);
319 
320     SmilMediaObject media = null;
321     for (SmilObject entity : smilResponse.getEntities()) {
322       if (entity instanceof SmilMediaObject) {
323         media = (SmilMediaObject) entity;
324         break;
325       }
326     }
327     assertNotNull(media);
328     assertEquals(media.getId(), ((SmilMediaContainer) smilResponse.getSmil().getBody().getMediaElements().get(0))
329             .getElements().get(0).getId());
330     assertTrue(media instanceof SmilMediaVideoImpl);
331     assertSame(((SmilMediaElement) media).getMediaType(), SmilMediaElement.MediaType.VIDEO);
332     // 1000 milliseconds = 1 second
333     assertEquals(1000L, ((SmilMediaElement) media).getClipBeginMS());
334     // duration is 1000000 milliseconds = 1000 soconds
335     // start + duration = 1s + 1000s = 1001s
336     assertEquals(1001000L, ((SmilMediaElement) media).getClipEndMS());
337 
338     // get param group id
339     List<SmilMediaParamGroup> groups = smilResponse.getSmil().getHead().getParamGroups();
340     assertEquals(1, groups.size());
341     String paramGroupId = groups.get(0).getId();
342 
343     // then, create a 2nd par with 1 video with the same param group id
344     TrackImpl videoTrack2 = new TrackImpl();
345     videoTrack2.setIdentifier("presenter-track-2");
346     videoTrack2.setFlavor(new MediaPackageElementFlavor("source2", "presenter"));
347     videoTrack2.setURI(new URI("http://hostname/video2.mp4"));
348     videoTrack2.addStream(new VideoStreamImpl());
349     videoTrack2.setDuration(2000000000000L);
350 
351     smilResponse = smilService.addParallel(smilResponse.getSmil());
352     SmilMediaContainer par2 = null;
353     for (SmilObject entity : smilResponse.getEntities()) {
354       if (entity instanceof SmilMediaContainer && !entity.getId().equals(par.getId())) {
355         par2 = (SmilMediaContainer) entity;
356         break;
357       }
358     }
359     // add video track into parallel element
360     smilResponse = smilService
361             .addClip(smilResponse.getSmil(), par2.getId(), videoTrack2, 2000L, 2000000L, paramGroupId);
362 
363     SmilMediaObject media2 = null;
364     for (SmilObject entity : smilResponse.getEntities()) {
365       if (entity instanceof SmilMediaObject && !entity.getId().equals(media.getId())) {
366         media2 = (SmilMediaObject) entity;
367         break;
368       }
369     }
370     assertNotNull(media2);
371     SmilMediaElement mediaElement2 = (SmilMediaElement) ((SmilMediaContainer) smilResponse.getSmil().getBody()
372             .getMediaElements().get(1)).getElements().get(0);
373     assertEquals(media2.getId(), mediaElement2.getId());
374     assertTrue(media2 instanceof SmilMediaVideoImpl);
375     assertSame(((SmilMediaElement) media2).getMediaType(), SmilMediaElement.MediaType.VIDEO);
376     // 1000 milliseconds = 1 second
377     assertEquals(2000L, ((SmilMediaElement) media2).getClipBeginMS());
378     // duration is 2000000 milliseconds = 2000 seconds
379     // start + duration = 2s + 2000s = 2002s
380     assertEquals(2002000L, ((SmilMediaElement) media2).getClipEndMS());
381     // make sure there's still 1 param group
382     groups = smilResponse.getSmil().getHead().getParamGroups();
383     assertEquals(1, groups.size());
384     // make sure that the media element has the previous param groupo id
385     assertEquals(paramGroupId, mediaElement2.getParamGroup());
386   }

388   /**
389    * Test of removeSmilElement method, of class SmilServiceImpl.
390    */
391   @Test
392   public void testRemoveSmilElement() throws Exception {
393     SmilResponse smilResponse = smilService.fromXml(TEST_SMIL);
394     assertNotNull(smilResponse.getSmil());
395     SmilMediaContainer par = (SmilMediaContainer) smilResponse.getSmil().getBody().getMediaElements().get(0);
396     assertSame(2, par.getElements().size());
397 
398     // remove first element from parallel
399     smilResponse = smilService.removeSmilElement(smilResponse.getSmil(), par.getElements().get(0).getId());
400     assertTrue(smilResponse.getEntity() instanceof SmilMediaElement);
401     assertSame(2, smilResponse.getSmil().getBody().getMediaElements().size());
402     par = (SmilMediaContainer) smilResponse.getSmil().getBody().getMediaElements().get(0);
403     assertSame(1, par.getElements().size());
404 
405     // remove parallel from smil
406     smilResponse = smilService.removeSmilElement(smilResponse.getSmil(), par.getId());
407     assertSame(1, smilResponse.getSmil().getBody().getMediaElements().size());
408     assertEquals(par.getId(), smilResponse.getEntity().getId());
409 
410     // remove the same parallel again
411     // should not fail
412     // response should return the same smil without entities
413     smilResponse = smilService.removeSmilElement(smilResponse.getSmil(), par.getId());
414     assertSame(1, smilResponse.getSmil().getBody().getMediaElements().size());
415     assertSame(0, smilResponse.getEntitiesCount());
416   }
