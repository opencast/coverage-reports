118   @Test
119   public void testToJson() throws ParseException, IOException {
120     Event event = new Event(id, defaultOrganization);
121     event.setTitle(title);
122     event.setDescription(description);
123     event.setSubject(subject);
124     event.setLocation(location);
125     event.setPresenters(presenters);
126     event.setContributors(contributors);
127     event.setAgentConfiguration(agentConfiguration);
128     logger.info(event.toJSON());
129     JSONObject parse = (JSONObject) new JSONParser().parse(event.toJSON());
130     if (parse.get(EVENT_JSON_KEY) == null || !(parse.get(EVENT_JSON_KEY) instanceof JSONObject)) {
131       fail("There must be an event object returned.");
132     }
133     JSONObject eventJsonObject = (JSONObject) parse.get(EVENT_JSON_KEY);
134     assertEquals(id, eventJsonObject.get(IDENTIFIER_JSON_KEY));
135     assertEquals(defaultOrganization, eventJsonObject.get(ORGANIZATION_JSON_KEY));
136     assertEquals(title, eventJsonObject.get(TITLE_JSON_KEY));
137     assertEquals(description, eventJsonObject.get(DESCRIPTION_JSON_KEY));
138     assertEquals(subject, eventJsonObject.get(SUBJECT_JSON_KEY));
139     assertEquals(location, eventJsonObject.get(LOCATION_JSON_KEY));
140 
141     JSONArray presentersArray = (JSONArray) ((JSONObject) eventJsonObject.get(PRESENTERS_JSON_KEY))
142             .get(PRESENTER_JSON_KEY);
143     // Ordering not important, just a convenient shorthand.
144     assertEquals(presenter1, presentersArray.get(0));
145     assertEquals(presenter2, presentersArray.get(1));
146     assertEquals(presenter3, presentersArray.get(2));
147 
148     JSONArray contributorsArray = (JSONArray) ((JSONObject) eventJsonObject.get(CONTRIBUTORS_JSON_KEY))
149             .get(CONTRIBUTOR_JSON_KEY);
150     // Ordering not important, just a convenient shorthand.
151     assertEquals(contributor1, contributorsArray.get(0));
152     assertEquals(contributor2, contributorsArray.get(1));
153     assertEquals(contributor3, contributorsArray.get(2));
154 
155     JSONObject agentConfigurationObject = (JSONObject) eventJsonObject.get(AGENT_CONFIGURATION_KEY);
156     JSONArray entryArray = (JSONArray) agentConfigurationObject.get(ENTRY_KEY);
157     // Ordering not important
158     assertThat(eventCAConfigJson, SameJSONAs.sameJSONAs(entryArray.toJSONString()).allowingAnyArrayOrdering());
159   }

161   @Test
162   public void testHasRecordingStarted() {
163     Event event = new Event(id, defaultOrganization);
164 
165     // This is not a scheduled event so the recording cannot have started
166     assertFalse(event.hasRecordingStarted());
167 
168     event.setAgentId("test");
169 
170     // With a valid capture agent ID, it is a scheduled event, but the recording cannot have started
171     // as we don't have a start time
172     assertFalse(event.hasRecordingStarted());
173 
174     event.setRecordingStatus(RecordingState.CAPTURING);
175 
176     // The event is a scheduled event and has a valid recording state, so the recording has started
177     assertTrue(event.hasRecordingStarted());
178   }
