57    /**
58     * @throws java.lang.Exception
59     */
60    @Before
61    public void setUp() throws Exception {
62      // Mock up a security service
63      SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
64      User user = new JaxbUser("admin", "test", new DefaultOrganization(), new JaxbRole(
65              SecurityConstants.GLOBAL_ADMIN_ROLE, new DefaultOrganization()));
66      EasyMock.expect(securityService.getOrganization()).andReturn(new DefaultOrganization()).anyTimes();
67      EasyMock.expect(securityService.getUser()).andReturn(user).anyTimes();
68      EasyMock.replay(securityService);
69  
70      UserDirectoryService userDirectoryService = EasyMock.createNiceMock(UserDirectoryService.class);
71      EasyMock.expect(userDirectoryService.loadUser(EasyMock.anyString())).andReturn(user).anyTimes();
72      EasyMock.replay(userDirectoryService);
73  
74      themesDatabase = new ThemesServiceDatabaseImpl();
75      themesDatabase.setEntityManagerFactory(newEntityManagerFactory(ThemesServiceDatabaseImpl.PERSISTENCE_UNIT));
76      themesDatabase.setDBSessionFactory(getDbSessionFactory());
77      themesDatabase.setSecurityService(securityService);
78      themesDatabase.setUserDirectoryService(userDirectoryService);
79      themesDatabase.activate(null);
80    }
81  
82    @Test
83    public void testStoreUpdateAndDelete() throws Exception {
84      JaxbOrganization org = new DefaultOrganization();
85      JaxbUser creator = new JaxbUser("admin", "test", org);
86  
87      Theme theme = new Theme(Option.<Long> none(), new Date(), true, creator, "New");
88      Theme updateTheme = themesDatabase.updateTheme(theme);
89      Assert.assertEquals("New", updateTheme.getName());
90      Assert.assertEquals(1, themesDatabase.countThemes());
91      theme = new Theme(updateTheme.getId(), new Date(), true, creator, "Updated");
92  
93      updateTheme = themesDatabase.updateTheme(theme);
94      Assert.assertEquals("Updated", updateTheme.getName());
95      Assert.assertEquals(1, themesDatabase.countThemes());
96  
97      try {
98        theme = themesDatabase.getTheme(updateTheme.getId().get());
99        Assert.assertNotNull(theme);
100     } catch (NotFoundException e) {
101       Assert.fail("Existing theme has not been found");
102     }
103 
104     themesDatabase.deleteTheme(updateTheme.getId().get());
105     Assert.assertEquals(0, themesDatabase.countThemes());
106 
107     try {
108       themesDatabase.getTheme(updateTheme.getId().get());
109       Assert.fail("Deleted theme has been found");
110     } catch (NotFoundException e) {
111       Assert.assertNotNull(e);
112     }
113   }

115   @Test
116   public void testFindQuery() throws Exception {
117     JaxbOrganization org = new DefaultOrganization();
118     JaxbUser user1 = new JaxbUser("admin", "test", org);
119     JaxbUser user2 = new JaxbUser("bdmin", "test", org);
120     JaxbUser user3 = new JaxbUser("cdmin", "test", org);
121 
122     Theme theme1 = new Theme(Option.<Long> none(), new Date(1), true, user1, "A theme");
123     Theme theme2 = new Theme(Option.<Long> none(), new Date(2), true, user2, "B theme");
124     Theme theme3 = new Theme(Option.<Long> none(), new Date(2), true, user3, "C theme");
125 
126     themesDatabase.updateTheme(theme1);
127     themesDatabase.updateTheme(theme2);
128     themesDatabase.updateTheme(theme3);
129 
130     List<Theme> themes;
131 
132     // Empty query
133     themes = themesDatabase.findThemes(
134         Optional.empty(),
135         Optional.empty(),
136         new ArrayList<>(),
137         Optional.empty(),
138         Optional.empty()
139     );
140     Assert.assertEquals(3, themes.size());
141 
142     // With limit
143     themes = themesDatabase.findThemes(
144         Optional.ofNullable(1),
145         Optional.empty(),
146         new ArrayList<>(),
147         Optional.empty(),
148         Optional.empty()
149     );
150     Assert.assertEquals(1, themes.size());
151 
152     themes = themesDatabase.findThemes(
153         Optional.ofNullable(Integer.MAX_VALUE),
154         Optional.empty(),
155         new ArrayList<>(),
156         Optional.empty(),
157         Optional.empty()
158     );
159     Assert.assertEquals(3, themes.size());
160 
161     // With offset
162     themes = themesDatabase.findThemes(
163         Optional.ofNullable(Integer.MAX_VALUE),
164         Optional.ofNullable(1),
165         new ArrayList<>(),
166         Optional.empty(),
167         Optional.empty()
168     );
169     Assert.assertEquals(2, themes.size());
170 
171     // With sort
172     SortCriterion sortCriterion1 = new SortCriterion("creation_date", SortCriterion.Order.Descending);
173     SortCriterion sortCriterion2 = new SortCriterion("name", SortCriterion.Order.Ascending);
174     ArrayList<SortCriterion> sortCriteria = new ArrayList<>();
175     sortCriteria.add(sortCriterion1);
176     sortCriteria.add(sortCriterion2);
177     themes = themesDatabase.findThemes(
178         Optional.ofNullable(Integer.MAX_VALUE),
179         Optional.ofNullable(0),
180         sortCriteria,
181         Optional.empty(),
182         Optional.empty()
183     );
184     Assert.assertEquals("B theme", themes.get(0).getName());
185     Assert.assertEquals("C theme", themes.get(1).getName());
186     Assert.assertEquals("A theme", themes.get(2).getName());
187 
188     // With filter
189     themes = themesDatabase.findThemes(
190         Optional.ofNullable(Integer.MAX_VALUE),
191         Optional.ofNullable(0),
192         new ArrayList<>(),
193         Optional.ofNullable("admin"),
194         Optional.empty()
195     );
196     Assert.assertEquals(1, themes.size());
197 
198     themes = themesDatabase.findThemes(
199         Optional.ofNullable(Integer.MAX_VALUE),
200         Optional.ofNullable(0),
201         new ArrayList<>(),
202         Optional.empty(),
203         Optional.ofNullable("admin")
204     );
205     Assert.assertEquals(1, themes.size());
206   }
