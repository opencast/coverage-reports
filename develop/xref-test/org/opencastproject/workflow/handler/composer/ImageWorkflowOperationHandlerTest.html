52    @Test
53    public void testMediaPositionParserSuccess() throws WorkflowOperationException {
54      test("75900000000001", sec(75900000000001d));
55      test("10, 20.3", sec(10), sec(20.3));
56      test("10 20.3", sec(10), sec(20.3));
57      test("1020.3", sec(1020.3));
58      test("10 , 20.3", sec(10), sec(20.3));
59      test("10    20.3", sec(10), sec(20.3));
60      test("15", sec(15));
61      test("11%", percent(11));
62      test(" 19.70% , 20.3", percent(19.7), sec(20.3));
63      test(" 10.62% 20.3 14.2     ", percent(10.62), sec(20.3), sec(14.2));
64      test("10.62% 20.3 ,14.213%", percent(10.62), sec(20.3), percent(14.213));
65    }

73    @Test
74    public void testFileNameGeneration() {
75      final ImageWorkflowOperationHandler dummy = new ImageWorkflowOperationHandler();
76      assertEquals("thumbnail_12.5p_small.jpg",
77              new Extractor(dummy, cfg(Optional.<String> empty(), Optional.of("thumbnail_%.1fp%s"))).createFileName("_small.jpg",
78                      uri("http://localhost/path/filename.mp4"), new MediaPosition(PositionType.Percentage, 12.5)));
79      assertEquals("thumbnail_0p.jpg",
80              new Extractor(dummy, cfg(Optional.<String> empty(), Optional.of("thumbnail_%.0fp%s"))).createFileName(".jpg",
81                      uri("http://localhost/path/filename.mp4"), new MediaPosition(PositionType.Percentage, 0)));
82      assertEquals("video_14.200s.jpg",
83              new Extractor(dummy, cfg(Optional.<String> empty(), Optional.<String> empty())).createFileName(".jpg",
84                      uri("http://localhost/path/video.mp4"), new MediaPosition(PositionType.Seconds, 14.2)));
85      assertEquals("video_15.110s_medium.jpg",
86              new Extractor(dummy, cfg(Optional.<String> empty(), Optional.<String> empty())).createFileName("_medium.jpg",
87                      uri("http://localhost/path/video.mp4"), new MediaPosition(PositionType.Seconds, 15.1099)));
88      assertEquals("thumbnail_15.110s_large.jpg",
89              new Extractor(dummy, cfg(Optional.of("thumbnail_%.3fs%s"), Optional.<String> empty())).createFileName("_large.jpg",
90                      uri("http://localhost/path/video.mp4"), new MediaPosition(PositionType.Seconds, 15.1099)));
91      assertEquals("thumbnail", new Extractor(dummy, cfg(Optional.of("thumbnail"), Optional.<String> empty())).createFileName(
92              "_large.jpg", uri("http://localhost/path/video.mp4"), new MediaPosition(PositionType.Seconds, 15.1099)));
93      assertEquals("thumbnail_large.jpg",
94              new Extractor(dummy, cfg(Optional.of("thumbnail%2$s"), Optional.<String> empty())).createFileName("_large.jpg",
95                      uri("http://localhost/path/video.mp4"), new MediaPosition(PositionType.Seconds, 15.1099)));
96    }

98    @Test
99    public void testValidateTargetBaseNameFormat() {
100     validateTargetBaseNameFormat("format-a").accept("thumbnail_%.1fs%s");
101     validateTargetBaseNameFormat("format-a").accept("thumbnail_%1$.1fs%2$s");
102     validateTargetBaseNameFormat("format-a").accept("%2$s_thumbnail_%1$.1fs%1$.1fs%2$s");
103     try {
104       validateTargetBaseNameFormat("format-a").accept("thumbnail_%.1fs%.3f");
105       fail("Invalid format passed check. Suffix format %s is missing.");
106     } catch (Exception ignore) {
107     }
108     try {
109       validateTargetBaseNameFormat("format-a").accept("thumbnail_%.3f");
110       fail("Invalid format passed check. Suffix format %s is missing.");
111     } catch (Exception ignore) {
112     }
113     try {
114       validateTargetBaseNameFormat("format-a").accept("thumbnail_%s");
115       fail("Invalid format passed check. Suffix is missing since %s does not have a positional parameter.");
116     } catch (Exception ignore) {
117     }
118     // omitting the position should pass
119     validateTargetBaseNameFormat("format-a").accept("thumbnail_%2$s");
120   }

122   @Test
123   public void testLimit() {
124     assertTrue(limit(track(1511), Arrays.asList(sec(-1), sec(1511), sec(1512))).isEmpty());
125     assertTrue(limit(track(1511), Arrays.asList(percent(-0.2), percent(101))).isEmpty());
126     assertEquals(4, limit(track(1511), Arrays.asList(percent(0), percent(100), sec(0), sec(1510))).size());
127     assertEquals(4, limit(track(1511), Arrays.asList(percent(0), percent(10), sec(10), sec(1500), percent(200))).size());
128   }

130   @Test
131   public void testToSeconds() {
132     assertEquals(0.985, toSeconds(track(1970), percent(50), 0), 0);
133     assertEquals(0, toSeconds(track(1970), percent(0), 0), 0);
134     assertEquals(1.87, toSeconds(track(1970), percent(100), 100), 0);
135     assertEquals(0, toSeconds(track(100), percent(100), 200), 0);
136     assertEquals(1.77, toSeconds(track(1970), sec(1.970), 200), 0);
137     assertEquals(1.77, toSeconds(track(1970), sec(1.870), 200), 0);
138     assertEquals(1.77, toSeconds(track(1970), sec(1.770), 200), 0);
139     assertEquals(1.769, toSeconds(track(1970), sec(1.769), 200), 0);
140   }

144   private Cfg cfg(Optional<String> targetBaseNamePatternSecond, Optional<String> targetBaseNamePatternPercent) {
145     return new Cfg(Collections.<Track> emptyList(), Collections.<MediaPosition> emptyList(), Collections.<EncodingProfile> emptyList(),
146             Collections.<MediaPackageElementFlavor> emptyList(), Collections.<String> emptyList(), targetBaseNamePatternSecond,
147             targetBaseNamePatternPercent, 0);
148   }
149 
150   private MediaPosition sec(double a) {
151     return new MediaPosition(PositionType.Seconds, a);
152   }
153 
154   private MediaPosition percent(double a) {
155     return new MediaPosition(PositionType.Percentage, a);
156   }
157 
158   private Track track(long duration) {
159     final Track t = EasyMock.createNiceMock(Track.class);
160     EasyMock.expect(t.getDuration()).andReturn(duration).anyTimes();
161     EasyMock.replay(t);
162     return t;
163   }
164 
165   private void test(String expr, MediaPosition... expected) throws WorkflowOperationException {
166     List<MediaPosition> result = MediaPositionParser.parsePositions(expr);
167     assertEquals(Arrays.asList(expected), result);
168   }
169 
170 
171   private boolean testSuccess(String expr) {
172     try {
173       List<MediaPosition> result = MediaPositionParser.parsePositions(expr);
174       return result != null && !result.isEmpty();
175     } catch (WorkflowOperationException e) {
176       return false;
177     }
178   }
