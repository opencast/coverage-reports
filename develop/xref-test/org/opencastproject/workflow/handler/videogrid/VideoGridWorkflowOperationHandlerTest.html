78    private static final String ENCODING_PROFILE = "concat-encoding-profile";
79    private static final String ENCODING_PROFILE_ID = "concat.work";
80    private static final String TARGET_FLAVOR_RETURN = "a/b";
81    private static final String SOURCE_FLAVOR = "source-flavors";
82    private static final String SOURCE_FLAVOR_KEY = "source/*";
83    private static final String SOURCE_SMIL_FLAVOR = "source-smil-flavor";
84    private static final String SOURCE_SMIL_FLAVOR_KEY = "smil/partial";

86    @Before
87    public void setUp() throws Exception {
88  
89      handler = new VideoGridWorkflowOperationHandler() {
90        @Override
91        protected JobBarrier.Result waitForStatus(Job... jobs) throws IllegalStateException, IllegalArgumentException {
92          JobBarrier.Result result = EasyMock.createNiceMock(JobBarrier.Result.class);
93          EasyMock.expect(result.isSuccess()).andReturn(true).anyTimes();
94          EasyMock.replay(result);
95          return result;
96        }
97      };
98  
99      /** Prepare flavor contents **/
100     // Smil
101     MediaPackageBuilder mpBuilder = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder();
102     mpSmilURI = getClass().getResource("/smil_ingest.xml").toURI();
103 
104     // Track
105     track = new TrackImpl();
106     track.setIdentifier("d79699b2-d683-4f0d-95ff-2dc4da3c9c40");                // To match the ID defined in the SMIL
107     track.setFlavor(MediaPackageElementFlavor.parseFlavor("source/presenter"));
108     VideoStreamImpl videoStream = new VideoStreamImpl("test1");
109     videoStream.setFrameWidth(80);
110     videoStream.setFrameHeight(30);
111     track.setVideo(Arrays.asList(videoStream, null));
112     URI trackURI = getClass().getResource("/af3bd8bb-2cd4-4d79-8e64-8d54a96876fe.mp4").toURI();   // Absolute URI
113     track.setURI(trackURI);
114 
115     // MediaPackage
116     MediaPackage mediaPackage = new MediaPackageBuilderImpl().createNew();
117     mediaPackage.setIdentifier(new IdImpl("123-456"));
118     mediaPackage.add(track);
119     mediaPackage.add(mpSmilURI, MediaPackageElement.Type.Catalog,
120         MediaPackageElementFlavor.parseFlavor(SOURCE_SMIL_FLAVOR_KEY));
121 
122     /** Create Mocks **/
123     instance = EasyMock.createNiceMock(WorkflowOperationInstance.class);
124     EasyMock.expect(instance.getConfiguration("target-flavor")).andReturn(TARGET_FLAVOR_RETURN).anyTimes();
125 
126     workspace = EasyMock.createNiceMock(Workspace.class);
127     // To avoid NullPointerEx when the SMIL is parsed
128     EasyMock.expect(workspace.get(mpSmilURI)).andReturn(new File(mpSmilURI)).anyTimes();
129     // To avoid NullPointerEx when grabbing Absolute Track Path
130     EasyMock.expect(workspace.get(trackURI)).andReturn(new File(trackURI)).anyTimes();
131 
132     workflow = EasyMock.createNiceMock(WorkflowInstance.class);
133     EasyMock.expect(workflow.getMediaPackage()).andReturn(mediaPackage).anyTimes();
134     EasyMock.expect(workflow.getCurrentOperation()).andReturn(instance).anyTimes();
135 
136     /** Create Service Mocks **/
137     Job videoGridJob = new JobImpl(0);
138     videoGridJob.setPayload(gson.toJson(trackURI));
139     VideoGridService videoGridService = EasyMock.createMock(VideoGridService.class);
140     // Mocking the same function multiple times because of varargs
141     EasyMock.expect(videoGridService.createPartialTrack(anyObject())).andReturn(videoGridJob).anyTimes();
142     EasyMock.expect(videoGridService.createPartialTrack(anyObject(), anyObject())).andReturn(videoGridJob).anyTimes();
143 
144     Job inspectJob = new JobImpl(1);
145     inspectTrack = new TrackImpl();
146     inspectTrack.setIdentifier("d79699b2-d683-4f0d-95ff-2dc4da3c9c40");
147     inspectTrack.setFlavor(MediaPackageElementFlavor.parseFlavor("source/presenter"));
148     inspectTrack.setURI(trackURI);
149     inspectJob.setPayload(MediaPackageElementParser.getAsXml(inspectTrack));
150     MediaInspectionService mediaInspectionService = EasyMock.createMock(MediaInspectionService.class);
151     EasyMock.expect(mediaInspectionService.enrich(anyObject(), anyBoolean())).andReturn(inspectJob).anyTimes();
152 
153     Job concatJob = new JobImpl(2);
154     concatTrack = new TrackImpl();
155     concatJob.setPayload(MediaPackageElementParser.getAsXml(concatTrack));
156     profile = EasyMock.createNiceMock(EncodingProfile.class);
157     EasyMock.expect(profile.getIdentifier()).andReturn(ENCODING_PROFILE_ID).anyTimes();
158     EasyMock.expect(profile.getApplicableMediaType()).andReturn(EncodingProfile.MediaType.Stream).anyTimes();
159     EasyMock.expect(profile.getOutputType()).andReturn(EncodingProfile.MediaType.AudioVisual).anyTimes();
160     EasyMock.replay(profile);
161     ComposerService composerService = EasyMock.createMock(ComposerService.class);
162     EasyMock.expect(composerService.getProfile(ENCODING_PROFILE_ID)).andReturn(profile).anyTimes();
163     EasyMock.expect(composerService.getProfile("nothing")).andReturn(null).anyTimes();
164     // Mocking a function with varargs (triple dot notation) is not trivial
165     // So this will only works for exactly three tracks
166     // If it needs to work for another number of tracks, you'll have to mock the function again
167     EasyMock.expect(
168         composerService.concat(anyString(), anyObject(), anyBoolean(), anyObject(), anyObject(), anyObject()))
169         .andReturn(concatJob)
170         .anyTimes();
171 
172     EasyMock.replay(videoGridService, workspace, workflow, mediaInspectionService, composerService);
173 
174     handler.setVideoGridService(videoGridService);
175     handler.setWorkspace(workspace);
176     handler.setMediaInspectionService(mediaInspectionService);
177     handler.setComposerService(composerService);
178   }

180   @Test
181   public void testComplete() throws Exception {
182     EasyMock.expect(instance.getConfiguration(SOURCE_FLAVOR)).andReturn(SOURCE_FLAVOR_KEY).anyTimes();
183     EasyMock.expect(instance.getConfiguration(SOURCE_SMIL_FLAVOR)).andReturn(SOURCE_SMIL_FLAVOR_KEY).anyTimes();
184     EasyMock.expect(instance.getConfiguration(ENCODING_PROFILE)).andReturn(ENCODING_PROFILE_ID).anyTimes();
185     EasyMock.replay(instance);
186     WorkflowOperationResult result = handler.start(workflow, null);
187     Assert.assertEquals(WorkflowOperationResult.Action.CONTINUE, result.getAction());
188 
189     MediaPackage mpNew = result.getMediaPackage();
190     Track[] tracks = mpNew.getTracks(MediaPackageElementFlavor.parseFlavor(TARGET_FLAVOR_RETURN));
191     Assert.assertEquals(1, tracks.length);
192   }
