22  package org.opencastproject.event.comment;
23  
24  import static org.junit.Assert.assertThat;
25  import static org.opencastproject.util.IoSupport.loadFileFromClassPathAsString;
26  import static org.xmlmatchers.XmlMatchers.similarTo;
27  import static org.xmlmatchers.transform.XmlConverters.the;
28  
29  import org.opencastproject.security.api.DefaultOrganization;
30  import org.opencastproject.security.api.JaxbUser;
31  import org.opencastproject.security.api.UserDirectoryService;
32  import org.opencastproject.util.DateTimeSupport;
33  
34  import org.apache.commons.io.IOUtils;
35  import org.easymock.EasyMock;
36  import org.junit.Assert;
37  import org.junit.Before;
38  import org.junit.Test;
39  
40  public class CommentParserTest {
41  
42    private UserDirectoryService userDirectoryService;
43    private JaxbUser testUser;
44  
45    @Before
46    public void setUp() throws Exception {
47      userDirectoryService = EasyMock.createNiceMock(UserDirectoryService.class);
48      testUser = new JaxbUser("test", "test", new DefaultOrganization());
49      EasyMock.expect(userDirectoryService.loadUser(EasyMock.anyObject(String.class))).andReturn(testUser).anyTimes();
50      EasyMock.replay(userDirectoryService);
51    }
52  
53    @Test
54    public void testParseAndSerializeComment() throws Exception {
55      String serializedCommentXml = IOUtils.toString(getClass().getResource("/comment.xml").toURI());
56  
57      EventComment comment = EventCommentParser.getCommentFromXml(serializedCommentXml, userDirectoryService);
58      Assert.assertNotNull(comment);
59      Assert.assertEquals(2L, comment.getId().get().longValue());
60      Assert.assertEquals("Test comment", comment.getText());
61      Assert.assertEquals("Test", comment.getReason());
62      Assert.assertTrue(comment.isResolvedStatus());
63      Assert.assertEquals(DateTimeSupport.fromUTC("2014-05-21T10:09:45Z"), comment.getCreationDate().getTime());
64      Assert.assertEquals(DateTimeSupport.fromUTC("2014-05-21T12:09:45Z"), comment.getModificationDate().getTime());
65      Assert.assertEquals(testUser, comment.getAuthor());
66      Assert.assertEquals(2, comment.getReplies().size());
67  
68      // Reply 1
69      EventCommentReply reply = comment.getReplies().get(0);
70      Assert.assertEquals(1L, reply.getId().get().longValue());
71      Assert.assertEquals("Test reply", reply.getText());
72      Assert.assertEquals(DateTimeSupport.fromUTC("2014-06-21T10:09:45Z"), reply.getCreationDate().getTime());
73      Assert.assertEquals(DateTimeSupport.fromUTC("2014-07-21T12:09:45Z"), reply.getModificationDate().getTime());
74      Assert.assertEquals(testUser, reply.getAuthor());
75  
76      // Reply 2
77      reply = comment.getReplies().get(1);
78      Assert.assertTrue(reply.getId().isNone());
79      Assert.assertEquals("Test reply 2", reply.getText());
80      Assert.assertEquals(DateTimeSupport.fromUTC("2014-08-21T10:09:45Z"), reply.getCreationDate().getTime());
81      Assert.assertEquals(DateTimeSupport.fromUTC("2014-09-21T12:09:45Z"), reply.getModificationDate().getTime());
82      Assert.assertEquals(testUser, reply.getAuthor());
83  
84      try {
85        String serializedComment = EventCommentParser.getAsXml(comment);
86        assertThat(the(serializedComment), similarTo(the(loadFileFromClassPathAsString("/comment.xml").get())));
87      } catch (EventCommentException e) {
88        Assert.fail(e.getMessage());
89      }
90  
91      try {
92        String serializedReply = EventCommentParser.getAsXml(comment.getReplies().get(0));
93        assertThat(the(serializedReply), similarTo(the(loadFileFromClassPathAsString("/comment-reply.xml").get())));
94      } catch (EventCommentException e) {
95        Assert.fail(e.getMessage());
96      }
97    }

99  }
