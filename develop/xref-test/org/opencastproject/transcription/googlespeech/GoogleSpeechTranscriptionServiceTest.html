280   public void testTranscriptionDone() throws Exception {
281     InputStream stream
282         = GoogleSpeechTranscriptionServiceTest.class.getResourceAsStream("/" + PULLED_TRANSCRIPTION_FILE);
283     database.storeJobControl(MP_ID, TRACK_ID, JOB_ID,
284         TranscriptionJobControl.Status.InProgress.name(), TRACK_DURATION, DATE_EXPECTED, PROVIDER);
285     JSONObject obj = (JSONObject) jsonParser.parse(new InputStreamReader(stream));
286 
287     Capture<String> capturedCollection = Capture.newInstance();
288     Capture<String> capturedFileName = Capture.newInstance();
289     EasyMock.expect(workspace.putInCollection(EasyMock.capture(capturedCollection), EasyMock.capture(capturedFileName),
290         EasyMock.anyObject(InputStream.class))).andReturn(new URI("http://anything"));
291     EasyMock.replay(workspace);
292 
293     long before = System.currentTimeMillis();
294     service.transcriptionDone(MP_ID, obj);
295     long after = System.currentTimeMillis();
296     // Check if status and date in db was updated
297     TranscriptionJobControl job = database.findByJob(JOB_ID);

299     Assert.assertEquals(TranscriptionJobControl.Status.TranscriptionComplete.name(), job.getStatus());
300     Assert.assertNotNull(job.getDateCompleted());
301     Assert.assertTrue(before <= job.getDateCompleted().getTime() && job.getDateCompleted().getTime() <= after);
302     // Check if results were saved into a collection
303     Assert.assertEquals(GoogleSpeechTranscriptionService.TRANSCRIPT_COLLECTION, capturedCollection.getValue());
304     Assert.assertEquals(JOB_ID + ".json", capturedFileName.getValue());
