62    @Test
63    public void testEncodeDate() {
64      Date now = new Date();
65      assertEquals(4, encodeDate(now, Precision.Year).getValue().length());
66      assertEquals(3, encodeDate(now, Precision.Day).getValue().split("-").length);
67      assertEquals("2009-01-01T00:00:00Z".length(), encodeDate(now, Precision.Second).getValue().length());
68      assertEquals(Optional.of(DublinCore.ENC_SCHEME_W3CDTF), encodeDate(now, Precision.Year).getEncodingScheme());
69      // Test symmetry
70      assertEquals(decodeDate(encodeDate(now, Precision.Second)), precisionSecond(now));
71      assertEquals(decodeDate(encodeDate(createDate(1999, 3, 21, 14, 0, 0), Precision.Day)), precisionDay(createDate(
72              1999, 3, 21, 14, 0, 0)));
73      assertEquals("1724-04-22", encodeDate(createDate(1724, 4, 22, 18, 30, 0), Precision.Day).getValue());
74      assertEquals("1724-04-22T18:30:00Z", encodeDate(createDate(1724, 4, 22, 18, 30, 0, "UTC"), Precision.Second)
75              .getValue());
76      assertEquals("1724-04-22T17:30:10Z", encodeDate(createDate(1724, 4, 22, 17, 30, 10, "UTC"), Precision.Second)
77              .getValue());
78      assertEquals("1724-04-22T17:30Z", encodeDate(createDate(1724, 4, 22, 17, 30, 25, "UTC"), Precision.Minute)
79              .getValue());
80      assertEquals("1999-03-21", encodeDate(createDate(1999, 3, 21, 18, 30, 25), Precision.Day).getValue());
81      //
82      logger.info(encodeDate(now, Precision.Day).getValue());
83      logger.info(encodeDate(now, Precision.Second).getValue());
84    }

121   @Test
122   public void testDecodePeriod() {
123     DCMIPeriod a = decodePeriod(DublinCoreValue.mk("start=2008-10-01; end=2009-01-01;"));
124     assertEquals(createDate(2008, 10, 1, 0, 0, 0), a.getStart());
125     assertEquals(createDate(2009, 1, 1, 0, 0, 0), a.getEnd());
126     DCMIPeriod b = decodePeriod(DublinCoreValue.mk("start=2008-10-01; end=2009-01-01"));
127     assertEquals(createDate(2008, 10, 1, 0, 0, 0), b.getStart());
128     assertEquals(createDate(2009, 1, 1, 0, 0, 0), b.getEnd());
129     DCMIPeriod c = decodePeriod(DublinCoreValue.mk("start=2008-10-01"));
130     assertEquals(createDate(2008, 10, 1, 0, 0, 0), c.getStart());
131     assertNull(c.getEnd());
132     DCMIPeriod d = decodePeriod(DublinCoreValue.mk("start=2008-10-01T10:20Z; end=2009-01-01; scheme=UNKNOWN"));
133     assertNull(d);
134     DCMIPeriod e = decodePeriod(DublinCoreValue.mk("start=2008-10-01T10:20Z; end=2009-01-01; scheme=W3C-DTF"));
135     assertNotNull(e);
136     DCMIPeriod f = decodePeriod(DublinCoreValue.mk("start=2008-10-01ERR; end=2009-01-01; scheme=W3C-DTF"));
137     assertNull(f);
138   }

140   @Test
141   public void testDecodeTemporal() {
142     Temporal.Match<Integer> match = new Temporal.Match<Integer>() {
143       @Override
144       public Integer period(DCMIPeriod period) {
145         return 1;
146   }
147 
148       @Override
149       public Integer instant(Date instant) {
150         return 2;
151       }
152 
153       @Override
154       public Integer duration(long duration) {
155         return 3;
156       }
157     };
158     Temporal.Match<Long> durationMatch = new Temporal.Match<Long>() {
159       @Override
160       public Long period(DCMIPeriod period) {
161         throw new RuntimeException();
162       }
163 
164       @Override
165       public Long instant(Date instant) {
166         throw new RuntimeException();
167       }
168 
169       @Override
170       public Long duration(long duration) {
171         return duration;
172       }
173     };
174     assertSame(1, decodeTemporal(DublinCoreValue.mk("start=2008-10-01; end=2009-01-01;")).fold(match));
175     assertSame(2, decodeTemporal(DublinCoreValue.mk("2008-10-01")).fold(match));
176     assertSame(2, decodeTemporal(DublinCoreValue.mk("2008-10-01T10:30:05Z")).fold(match));
177     assertSame(1, decodeTemporal(DublinCoreValue.mk("start=2008-10-01T10:20Z; end=2009-01-01; scheme=W3C-DTF")).fold(match));
178     assertSame(3, decodeTemporal(DublinCoreValue.mk("PT10H5M")).fold(match));
179     assertEquals(10L * 60 * 60 * 1000 + 5 * 60 * 1000,
180             (long) decodeTemporal(DublinCoreValue.mk("PT10H5M")).fold(durationMatch));
181     assertEquals(10L * 60 * 60 * 1000 + 5 * 60 * 1000 + 28 * 1000,
182             (long) decodeTemporal(DublinCoreValue.mk("PT10H5M28S")).fold(durationMatch));
183   }

193     assertEquals(new Long(1 * 1000 * 60 * 60 + 10 * 1000 * 60 + 5 * 1000), decodeDuration("01:10:05"));
194 
195     assertEquals(Optional.of(DublinCore.ENC_SCHEME_ISO8601), encodeDuration(d3).getEncodingScheme());
196 
197     assertNull(decodeDuration(DublinCoreValue.mk("bla")));
198     assertNull(decodeDuration(DublinCoreValue.mk(encodeDuration(d1).getValue(), DublinCore.LANGUAGE_UNDEFINED,
199             DublinCore.ENC_SCHEME_BOX)));
