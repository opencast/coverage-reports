175   @GET
176   @Path("acls.json")
177   @Produces(MediaType.APPLICATION_JSON)
178   @RestQuery(name = "allaclasjson", description = "Returns a list of acls", returnDescription = "Returns a JSON representation of the list of acls available the current user's organization", restParameters = {
179           @RestParameter(name = "filter", isRequired = false, description = "The filter used for the query. They should be formated like that: 'filter1:value1,filter2:value2'", type = STRING),
180           @RestParameter(name = "sort", isRequired = false, description = "The sort order. May include any of the following: NAME. Add '_DESC' to reverse the sort order (e.g. NAME_DESC).", type = STRING),
181           @RestParameter(defaultValue = "100", description = "The maximum number of items to return per page.", isRequired = false, name = "limit", type = RestParameter.Type.STRING),
182           @RestParameter(defaultValue = "0", description = "The page number.", isRequired = false, name = "offset", type = RestParameter.Type.STRING) }, responses = { @RestResponse(responseCode = SC_OK, description = "The list of ACL's has successfully been returned") })
183   public Response getAclsAsJson(@QueryParam("filter") String filter, @QueryParam("sort") String sort,
184           @QueryParam("offset") int offset, @QueryParam("limit") int limit) throws IOException {
185     if (limit < 1)
186       limit = 100;
187     Optional<String> optSort = Optional.ofNullable(trimToNull(sort));
188     Option<String> filterName = Option.none();
189     Option<String> filterText = Option.none();
190 
191     Map<String, String> filters = RestUtils.parseFilter(filter);
192     for (String name : filters.keySet()) {
193       String value = filters.get(name);
194       if (AclsListQuery.FILTER_NAME_NAME.equals(name)) {
195         filterName = Option.some(value);
196       } else if ((AclsListQuery.FILTER_TEXT_NAME.equals(name)) && (StringUtils.isNotBlank(value))) {
197         filterText = Option.some(value);
198       }
199     }
200 
201     // Filter acls by filter criteria
202     List<ManagedAcl> filteredAcls = new ArrayList<>();
203     for (ManagedAcl acl : aclService().getAcls()) {
204       // Filter list
205       if ((filterName.isSome() && !filterName.get().equals(acl.getName()))
206               || (filterText.isSome() && !TextFilter.match(filterText.get(), acl.getName()))) {
207         continue;
208       }
209       filteredAcls.add(acl);
210     }
211     int total = filteredAcls.size();
212 
213     // Sort by name, description or role
214     if (optSort.isPresent()) {
215       final ArrayList<SortCriterion> sortCriteria = RestUtils.parseSortQueryParameter(optSort.get());
216       Collections.sort(filteredAcls, new Comparator<ManagedAcl>() {
217         @Override
218         public int compare(ManagedAcl acl1, ManagedAcl acl2) {
219           for (SortCriterion criterion : sortCriteria) {
220             Order order = criterion.getOrder();
221             switch (criterion.getFieldName()) {
222               case "name":
223                 if (order.equals(Order.Descending))
224                   return ObjectUtils.compare(acl2.getName(), acl1.getName());
225                 return ObjectUtils.compare(acl1.getName(), acl2.getName());
226               default:
227                 logger.info("Unkown sort type: {}", criterion.getFieldName());
228                 return 0;
229             }
230           }
231           return 0;
232         }
233       });
234     }
235 
236     int start = Math.min(offset, filteredAcls.size());
237     int end = Math.min(start + limit, filteredAcls.size());
238 
239     // Apply Limit and offset
240     List<ManagedAcl> subList = filteredAcls.subList(start, end);
241 
242     // Convert each ManagedAcl to JsonObject using a helper method
243     List<JsonObject> aclJSON = new ArrayList<>();
244     for (ManagedAcl acl : subList) {
245       aclJSON.add(full(acl));
246     }
247 
248     return okJsonList(aclJSON, offset, limit, total);
249   }

251   @GET
252   @Path("roles.json")
253   @Produces(MediaType.APPLICATION_JSON)
254   @RestQuery(name = "getRoles", description = "Returns a list of roles",
255              returnDescription = "Returns a JSON representation of the roles with the given parameters under the "
256                                + "current user's organization.",
257              restParameters = {
258                @RestParameter(name = "query", isRequired = false, description = "The query.", type = STRING),
259                @RestParameter(name = "target", isRequired = false, description = "The target of the roles.",
260                               type = STRING),
261                @RestParameter(name = "limit", defaultValue = "100",
262                               description = "The maximum number of items to return per page.", isRequired = false,
263                               type = RestParameter.Type.STRING),
264                @RestParameter(name = "offset", defaultValue = "0", description = "The page number.", isRequired = false,
265                               type = RestParameter.Type.STRING) },
266              responses = { @RestResponse(responseCode = SC_OK, description = "The list of roles.") })
267   public Response getRoles(@QueryParam("query") String query, @QueryParam("target") String target,
268                            @QueryParam("offset") int offset, @QueryParam("limit") int limit) {
269 
270     String roleQuery = "%";
271     if (StringUtils.isNotBlank(query)) {
272       roleQuery = query.trim() + "%";
273     }
274 
275     Role.Target roleTarget = Role.Target.ALL;
276 
277     if (StringUtils.isNotBlank(target)) {
278       try {
279         roleTarget = Role.Target.valueOf(target.trim());
280       } catch (Exception e) {
281         logger.warn("Invalid target filter value {}", target);
282       }
283     }
284 
285     List<Role> roles = roleDirectoryService.findRoles(roleQuery, roleTarget, offset, limit);
286 
287     JSONArray jsonRoles = new JSONArray();
288     for (Role role: roles) {
289       JSONObject jsonRole = new JSONObject();
290       jsonRole.put("name", role.getName());
291       jsonRole.put("type", role.getType().toString());
292       jsonRole.put("description", role.getDescription());
293       jsonRole.put("organization", role.getOrganizationId());
294       jsonRole.put("isSanitize", isSanitize());
295       if (!isSanitize()) {
296         User user = userDirectoryService.loadUser(role.getName().replaceFirst(getUserRolePrefix(), ""));
297         if (user != null) {
298           jsonRole.put("user", generateJsonUser(user));
299         }
300       }
301       jsonRoles.add(jsonRole);
302     }
303 
304     return Response.ok(jsonRoles.toJSONString()).build();
305   }

307   @DELETE
308   @Path("{id}")
309   @RestQuery(name = "deleteacl", description = "Delete an ACL", returnDescription = "Delete an ACL", pathParameters = { @RestParameter(name = "id", isRequired = true, description = "The ACL identifier", type = INTEGER) }, responses = {
310           @RestResponse(responseCode = SC_OK, description = "The ACL has successfully been deleted"),
311           @RestResponse(responseCode = SC_NOT_FOUND, description = "The ACL has not been found"),
312           @RestResponse(responseCode = SC_CONFLICT, description = "The ACL could not be deleted, there are still references on it") })
313   public Response deleteAcl(@PathParam("id") long aclId) throws NotFoundException {
314     try {
315       if (!aclService().deleteAcl(aclId))
316         return conflict();
317     } catch (AclServiceException e) {
318       logger.warn("Error deleting manged acl with id '{}'", aclId, e);
319       throw new WebApplicationException(Response.Status.INTERNAL_SERVER_ERROR);
320     }
321     return noContent();
322   }

324   @POST
325   @Path("")
326   @Produces(MediaType.APPLICATION_JSON)
327   @RestQuery(name = "createacl", description = "Create an ACL", returnDescription = "Create an ACL", restParameters = {
328           @RestParameter(name = "name", isRequired = true, description = "The ACL name", type = STRING),
329           @RestParameter(name = "acl", isRequired = true, description = "The access control list", type = STRING) }, responses = {
330           @RestResponse(responseCode = SC_OK, description = "The ACL has successfully been added"),
331           @RestResponse(responseCode = SC_CONFLICT, description = "An ACL with the same name already exists"),
332           @RestResponse(responseCode = SC_BAD_REQUEST, description = "Unable to parse the ACL") })
333   public Response createAcl(@FormParam("name") String name, @FormParam("acl") String accessControlList) {
334     final AccessControlList acl = parseAcl(accessControlList);
335     Optional<ManagedAcl> managedAcl = aclService().createAcl(acl, name);
336     if (managedAcl.isEmpty()) {
337       logger.info("An ACL with the same name '{}' already exists", name);
338       throw new WebApplicationException(Response.Status.CONFLICT);
339     }
340     return RestUtils.okJson(full(managedAcl.get()));
341   }

343   @PUT
344   @Path("{id}")
345   @Produces(MediaType.APPLICATION_JSON)
346   @RestQuery(name = "updateacl", description = "Update an ACL", returnDescription = "Update an ACL", pathParameters = { @RestParameter(name = "id", isRequired = true, description = "The ACL identifier", type = INTEGER) }, restParameters = {
347           @RestParameter(name = "name", isRequired = true, description = "The ACL name", type = STRING),
348           @RestParameter(name = "acl", isRequired = true, description = "The access control list", type = STRING) }, responses = {
349           @RestResponse(responseCode = SC_OK, description = "The ACL has successfully been updated"),
350           @RestResponse(responseCode = SC_NOT_FOUND, description = "The ACL has not been found"),
351           @RestResponse(responseCode = SC_BAD_REQUEST, description = "Unable to parse the ACL") })
352   public Response updateAcl(@PathParam("id") long aclId, @FormParam("name") String name,
353           @FormParam("acl") String accessControlList) throws NotFoundException {
354     final Organization org = securityService.getOrganization();
355     final AccessControlList acl = parseAcl(accessControlList);
356     final ManagedAclImpl managedAcl = new ManagedAclImpl(aclId, name, org.getId(), acl);
357     if (!aclService().updateAcl(managedAcl)) {
358       logger.info("No ACL with id '{}' could be found under organization '{}'", aclId, org.getId());
359       throw new NotFoundException();
360     }
361     return RestUtils.okJson(full(managedAcl));
362   }

366   @Produces(MediaType.APPLICATION_JSON)
367   @RestQuery(name = "getacl", description = "Return the ACL by the given id", returnDescription = "Return the ACL by the given id", pathParameters = { @RestParameter(name = "id", isRequired = true, description = "The ACL identifier", type = INTEGER) }, responses = {
368           @RestResponse(responseCode = SC_OK, description = "The ACL has successfully been returned"),

371     Optional<ManagedAcl> managedAcl = aclService().getAcl(aclId);
372     if (managedAcl.isPresent()) {

375     logger.info("No ACL with id '{}' could by found", aclId);
376     throw new NotFoundException();
377   }

379   private static AccessControlList parseAcl(String acl) {
380     try {
381       return AccessControlParser.parseAcl(acl);
382     } catch (Exception e) {
383       logger.warn("Unable to parse ACL", e);
384       throw new WebApplicationException(Response.Status.BAD_REQUEST);
385     }
386   }

388   public JsonObject full(AccessControlEntry ace) {
389     JsonObject json = new JsonObject();
390     json.addProperty(JsonConv.KEY_ROLE, ace.getRole());
391     json.addProperty(JsonConv.KEY_ACTION, ace.getAction());
392     json.addProperty(JsonConv.KEY_ALLOW, ace.isAllow());
393     return json;
394   }
395 
396   private JsonObject fullAccessControlEntry(AccessControlEntry ace) {
397     return full(ace);
398   }
399 
400   public JsonObject full(AccessControlList acl) {
401     JsonObject json = new JsonObject();
402     JsonArray aceArray = new JsonArray();
403 
404     List<AccessControlEntry> entries = acl.getEntries();
405     if (entries != null) {
406       for (AccessControlEntry entry : entries) {
407         aceArray.add(fullAccessControlEntry(entry));
408       }
409     }
410 
411     json.add(JsonConv.KEY_ACE, aceArray);
412     return json;
413   }
414 
415   public JsonObject full(ManagedAcl acl) {
416     JsonObject json = new JsonObject();
417     json.addProperty(JsonConv.KEY_ID, acl.getId());
418     json.addProperty(JsonConv.KEY_NAME, acl.getName());
419     json.addProperty(JsonConv.KEY_ORGANIZATION_ID, acl.getOrganizationId());
420     json.add("acl", full(acl.getAcl()));
421     return json;
422   }
