42  /**
43   * This is the document model class which holds the data about a set of rest endpoints.
44   */
45  public class RestDocData extends DocData {
46  
47    /**
48     * The name to identify the endpoint holder for read endpoints (get/head).
49     */
50    private static final String READ_ENDPOINT_HOLDER_NAME = "READ";
51  
52    /**
53     * The name to identify the endpoint holder for write endpoints (delete,post,put).
54     */
55    private static final String WRITE_ENDPOINT_HOLDER_NAME = "WRITE";
56  
57    /**
58     * Regular expression used to count the number of path parameters in a path.
59     */
60    public static final String PATH_PARAM_COUNTING_REGEX = "\\{(.+?)\\}";
61  
62    /**
63     * List of RestEndpointHolderData which each stores a group of endpoints. Currently there are 2 groups, READ group and
64     * WRITE group.
65     */
66    protected List<RestEndpointHolderData> holders;
67  
68    /**
69     * Create the base data object for creating REST documentation.
70     *
71     * @param name
72     *          the name of the set of rest endpoints (must be alphanumeric (includes _) and no spaces or special chars)
73     * @param title
74     *          [OPTIONAL] the title of the documentation
75     * @param url
76     *          this is the absolute base URL for this endpoint, do not include the trailing slash (e.g. /workflow)
77     * @param notes
78     *          [OPTIONAL] an array of notes to add into the end of the documentation
79     * @throws IllegalArgumentException
80     *           if the url is null or empty
81     */
82    public RestDocData(String name, String title, String url, String[] notes) throws IllegalArgumentException {
83      super(name, title, notes);
84      if (url == null || "".equals(url)) {
85        throw new IllegalArgumentException("URL cannot be blank.");
86      }
87      meta.put("url", url);
88      // create the endpoint holders
89      holders = new Vector<RestEndpointHolderData>(2);
90      holders.add(new RestEndpointHolderData(READ_ENDPOINT_HOLDER_NAME, "Read"));
91      holders.add(new RestEndpointHolderData(WRITE_ENDPOINT_HOLDER_NAME, "Write"));
92    }
93  
94    /**
95     * Verify the integrity of this object. If its data is verified to be okay, it return a map representation of this
96     * RestDocData object.
97     *
98     * @return a map representation of this RestDocData object if this object passes the verification
99     *
100    * @throws IllegalStateException
101    *           if any path parameter is not present in the endpoint's path
102    */
103   @Override
104   public Map<String, Object> toMap() throws IllegalStateException {
105     LinkedHashMap<String, Object> m = new LinkedHashMap<String, Object>();
106     m.put("meta", meta);
107     m.put("notes", notes);
108     // only pass through the holders with things in them
109     ArrayList<RestEndpointHolderData> holdersList = new ArrayList<RestEndpointHolderData>();
110     for (RestEndpointHolderData holder : holders) {
111       if (!holder.getEndpoints().isEmpty()) {
112         for (RestEndpointData endpoint : holder.getEndpoints()) {
113           // Validate the endpoint path matches the specified path parameters.
114           // First, it makes sure that every path parameter is present in the endpoint's path.
115           if (!endpoint.getPathParams().isEmpty()) {
116             for (RestParamData param : endpoint.getPathParams()) {
117               // Some endpoints allow for arbitrary characters, including slashes, in their path parameters, so we
118               // must check for both {param} and {param:.*}.
119               if (!endpoint.getPath().contains("{" + param.getName() + "}")
120                       && !endpoint.getPath().contains("{" + param.getName() + ":")) {
121                 throw new IllegalStateException("Path (" + endpoint.getPath() + ") does not match path parameter ("
122                         + param.getName() + ") for endpoint (" + endpoint.getName()
123                         + "), the path must contain all path parameter names.");
124               }
125             }
126           }
127           // Then, it makes sure that the number of path parameter patterns in the path is the same as the number of
128           // path parameters in the endpoint.
129           // The following part uses a regular expression to find patterns like {something}.
130           Pattern pattern = Pattern.compile(PATH_PARAM_COUNTING_REGEX);
131           Matcher matcher = pattern.matcher(endpoint.getPath());
132 
133           int count = 0;
134           while (matcher.find()) {
135             count++;
136           }
137           if (count != endpoint.getPathParams().size()) {
138             throw new IllegalStateException("Path (" + endpoint.getPath() + ") does not match path parameters ("
139                     + endpoint.getPathParams() + ") for endpoint (" + endpoint.getName()
140                     + "), the path must contain the same number of path parameters (" + count
141                     + ") as the pathParams list (" + endpoint.getPathParams().size() + ").");
142           }
143         }
144         holdersList.add(holder);
145       }
146     }
147     m.put("endpointHolders", holdersList);
148     return m;
149   }

161   /**
162    * Returns a string representation of this object.
163    *
164    * @return a string representation of this object
165    */
166   @Override
167   public String toString() {
168     return "DOC:meta=" + meta + ", notes=" + notes + ", " + holders;

171   /**
172    * Add an endpoint to this documentation using and assign it to the correct type group (read/write).
173    *
174    * @param type
175    *          the type of this endpoint (RestEndpointData.Type.READ or RestEndpointData.Type.WRITE)
176    * @param endpoint
177    *          the endpoint to be added
178    * @throws IllegalStateException
179    *           if the endpoint cannot be assigned to a group
180    */
181   private void addEndpoint(String type, RestEndpointData endpoint) throws IllegalStateException {
182     RestEndpointHolderData currentHolder = null;
183     for (RestEndpointHolderData holder : holders) {
184       if (type.equalsIgnoreCase(holder.getName())) {
185         currentHolder = holder;
186         break;
187       }
188     }
189     if (currentHolder == null) {
190       throw new IllegalStateException("Could not find holder of type: " + type + ".");
191     }
192     currentHolder.addEndPoint(endpoint);
193   }

195   /**
196    * Creates an abstract section which is displayed at the top of the documentation page.
197    *
198    * @param abstractText
199    *          any text to place at the top of the document, can be html markup but must be valid
200    */
201   public void setAbstract(String abstractText) {
202     if (isBlank(abstractText)) {
203       meta.remove("abstract");
204     } else {
205       meta.put("abstract", abstractText);
206     }
207   }

209   /**
210    * Validates paths: VALID: /sample , /sample/{thing} , /{my}/{path}.xml , /my/fancy_path/is/{awesome}.{FORMAT}
211    * INVALID: sample, /sample/, /sa#$%mple/path
212    *
213    * @param path
214    *          the path value to check
215    * @return true if this path is valid, false otherwise
216    */
217   public static boolean isValidPath(String path) {
218     return path != null && path.matches("^/$|^/[\\w/{}|:.*+]*[\\w}.]$");
219   }

221   /**
222    * Add an endpoint to the Rest documentation.
223    *
224    * @param restQuery
225    *          the RestQuery annotation type storing information of an endpoint
226    * @param returnType
227    *          the return type for this endpoint. If this is {@link javax.xml.bind.annotation.XmlRootElement} or
228    *          {@link javax.xml.bind.annotation.XmlRootElement}, the XML schema for the class will be made available to
229    *          clients
230    * @param produces
231    *          the return type(s) of this endpoint, values should be constants from <a
232    *          href="http://jackson.codehaus.org/javadoc/jax-rs/1.0/javax/ws/rs/core/MediaType.html"
233    *          >javax.ws.rs.core.MediaType</a> or ExtendedMediaType
234    *          (org.opencastproject.util.doc.rest.ExtendedMediaType).
235    * @param httpMethodString
236    *          the HTTP method of this endpoint (e.g. GET, POST)
237    * @param path
238    *          the path of this endpoint
239    */
240   public void addEndpoint(RestQuery restQuery, Class<?> returnType, Produces produces, String httpMethodString,
241           Path path) {
242     String pathValue = path.value().startsWith("/") ? path.value() : "/" + path.value();
243     RestEndpointData endpoint = new RestEndpointData(returnType, restQuery.name(), httpMethodString, pathValue,
244             restQuery.description());
245     // Add return description if needed
246     if (!restQuery.returnDescription().isEmpty()) {
247       endpoint.addNote("Return value description: " + restQuery.returnDescription());
248     }
249 
250     // Add formats
251     if (produces != null) {
252       for (String format : produces.value()) {
253         endpoint.addFormat(new RestFormatData(format));
254       }
255     }
256 
257     // Add responses
258     for (RestResponse restResp : restQuery.responses()) {
259       endpoint.addStatus(restResp);
260     }
261 
262     // Add body parameter
263     if (restQuery.bodyParameter().type() != RestParameter.Type.NO_PARAMETER) {
264       endpoint.addBodyParam(restQuery.bodyParameter());
265     }
266 
267     // Add path parameter
268     for (RestParameter pathParam : restQuery.pathParameters()) {
269       endpoint.addPathParam(new RestParamData(pathParam));
270     }
271 
272     // Add query parameter (required and optional)
273     for (RestParameter restParam : restQuery.restParameters()) {
274       if (restParam.isRequired()) {
275         endpoint.addRequiredParam(new RestParamData(restParam));
276       } else {
277         endpoint.addOptionalParam(new RestParamData(restParam));
278       }
279     }
280 
281     // Set the test form after all parameters are added.
282     endpoint.setTestForm(new RestFormData(endpoint));
283 
284     // Add the endpoint to the corresponding group based on its HTTP method
285     if ("GET".equalsIgnoreCase(httpMethodString) || "HEAD".equalsIgnoreCase(httpMethodString)) {
286       addEndpoint(READ_ENDPOINT_HOLDER_NAME, endpoint);
287     } else if ("DELETE".equalsIgnoreCase(httpMethodString) || "POST".equalsIgnoreCase(httpMethodString)
288             || "PUT".equalsIgnoreCase(httpMethodString)) {
289       addEndpoint(WRITE_ENDPOINT_HOLDER_NAME, endpoint);
290     }
291   }
