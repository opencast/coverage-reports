102   /**
103    * {@inheritDoc}
104    */
105   @Override
106   public WorkflowOperationResult start(WorkflowInstance workflowInstance, JobContext context)
107           throws WorkflowOperationException {
108 
109     // The current workflow operation instance
110     WorkflowOperationInstance operation = workflowInstance.getCurrentOperation();
111 
112     // MediaPackage from previous workflow operations
113     MediaPackage srcPackage = workflowInstance.getMediaPackage();
114 
115     // "To", "CC", "BCC", subject, body can be Freemarker templates
116     String[] to = processDestination(workflowInstance, operation, TO_PROPERTY);
117     String[] cc = processDestination(workflowInstance, operation, CC_PROPERTY);
118     String[] bcc = processDestination(workflowInstance, operation, BCC_PROPERTY);
119 
120     if (to.length + cc.length + bcc.length == 0) {
121       logger.info("No recipients. Skipping operation.");
122       return createResult(srcPackage, Action.SKIP);
123     }
124 
125     String subject = applyTemplateIfNecessary(workflowInstance, operation, SUBJECT_PROPERTY);
126 
127     String bodyCfg = operation.getConfiguration(BODY_PROPERTY);
128     String bodyTemplateFileCfg = operation.getConfiguration(BODY_TEMPLATE_FILE_PROPERTY);
129     String bodyHtmlCfg = operation.getConfiguration(BODY_HTML_PROPERTY);
130     String bodyHtmlTemplateFileCfg = operation.getConfiguration(BODY_HTML_TEMPLATE_FILE_PROPERTY);
131 
132     String bodyText = null;
133     String bodyHTML = null;
134 
135     // Body informed? If not, use the default.
136     if (bodyCfg == null && bodyHtmlCfg == null && bodyTemplateFileCfg == null && bodyHtmlTemplateFileCfg == null) {
137       // Set the body of the message to be the ID of the media package
138       bodyText = String.format("%s (%s)", srcPackage.getTitle(), srcPackage.getIdentifier());
139     }
140 
141     if (bodyCfg != null) {
142       bodyText = applyTemplateIfNecessary(workflowInstance, operation, BODY_PROPERTY);
143     } else if (bodyTemplateFileCfg != null) {
144       bodyText = applyTemplateIfNecessary(workflowInstance, operation, BODY_TEMPLATE_FILE_PROPERTY);
145     }
146 
147     if (bodyHtmlCfg != null) {
148       bodyHTML = applyTemplateIfNecessary(workflowInstance, operation, BODY_HTML_PROPERTY);
149     } else if (bodyHtmlTemplateFileCfg != null) {
150       bodyHTML = applyTemplateIfNecessary(workflowInstance, operation, BODY_HTML_TEMPLATE_FILE_PROPERTY);
151     }
152 
153     try {
154       logger.debug("Sending e-mail notification with subject '{}' and text body '{}' / HTML body '{}' "
155               + "to '{}', CC '{}' and BCC '{}'",
156               subject, bodyText, bodyHTML, to, cc, bcc);
157       // "To", "CC" and "BCC" can be comma- or space-separated lists of emails
158       smtpService.send(to, cc, bcc, subject, bodyText, bodyHTML);
159       logger.info("E-mail notification sent to {}, CC addresses {} and BCC addresses {}", to, cc, bcc);
160     } catch (MessagingException e) {
161       throw new WorkflowOperationException(e);
162     }
163 
164     // Return the source mediapackage and tell processing to continue
165     return createResult(srcPackage, Action.CONTINUE);
166   }

168   private String[] processDestination(
169       final WorkflowInstance workflowInstance,
170       final WorkflowOperationInstance operation,
171       final String emailHeader) throws WorkflowOperationException {
172     final String separator = Objects.toString(
173         operation.getConfiguration(ADDRESS_SEPARATOR_PROPERTY),
174         ADDRESS_SEPARATOR_DEFAULT);
175     // First apply the template if there's one
176     final String templateApplied = applyTemplateIfNecessary(workflowInstance, operation, emailHeader, separator);
177     if (templateApplied == null) {
178       return new String[0];
179     }
180 
181     // Are these valid email addresses?
182     final ArrayList<String> recipients = new ArrayList<>();
183     final boolean skipInvalid = BooleanUtils.toBoolean(operation.getConfiguration(SKIP_INVALID_ADDRESS_PROPERTY));
184     for (final String part : StringUtils.split(templateApplied, separator)) {
185       // Is this a user name? Look for that user via user directory service.
186       final User user = userDirectoryService.loadUser(part);
187       if (user != null && StringUtils.isNotEmpty(user.getEmail())) {
188         // Yes, this is a user name and the user has an email registered. Use it.
189         recipients.add(user.getEmail());
190       } else {
191         // Either not a user name or user doesn't have an email registered.
192         try {
193           // Validate it as an email address
194           new InternetAddress(part).validate();
195           recipients.add(part);
196         } catch (Exception e) {
197           // Otherwise, log an error
198           if (skipInvalid) {
199             logger.debug("Skip sending mail to invalid email address {}", part);
200           } else {
201             throw new WorkflowOperationException(
202                 String.format("Email address invalid or user doesn't have an email: %s.", part), e);
203           }
204         }
205       }
206     }
207     return recipients.toArray(new String[0]);
208   }

210   private String applyTemplateIfNecessary(WorkflowInstance workflowInstance, WorkflowOperationInstance operation,
211       String configName) {
212     return applyTemplateIfNecessary(workflowInstance, operation, configName, null);
213   }
214 
215   private String applyTemplateIfNecessary(WorkflowInstance workflowInstance, WorkflowOperationInstance operation,
216           String configName, String separator) {
217     String configValue = operation.getConfiguration(configName);
218 
219     // Templates are cached, use as template name: the template name or, if
220     // in-line, the
221     // workflow name + the operation number + body/to/subject
222     String templateName = null;
223     String templateContent = null;
224 
225     if (BODY_TEMPLATE_FILE_PROPERTY.equals(configName) || BODY_HTML_TEMPLATE_FILE_PROPERTY.equals(configName)) {
226       templateName = configValue; // Use body template file name
227     } else if (configValue != null && configValue.contains("${")) {
228       // If value contains a "${", it may be a template so apply it
229       // Give a name to the inline template
230       templateName = workflowInstance.getTemplate() + "_" + operation.getId() + "_" + configName;
231       // Only alphanumeric and _
232       templateName = templateName.replaceAll("[^A-Za-z0-9 ]", "_");
233       templateContent = configValue;
234     } else {
235       // If value doesn't contain a "${", assume it is NOT a Freemarker
236       // template and thus return the value as it is
237       return configValue;
238     }
239 
240     // Apply the template
241     if (separator == null) {
242       return emailTemplateService.applyTemplate(templateName, templateContent, workflowInstance);
243     }
244     return emailTemplateService.applyTemplate(templateName, templateContent, workflowInstance, separator);
245   }
