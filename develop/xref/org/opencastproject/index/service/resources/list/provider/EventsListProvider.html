22  package org.opencastproject.index.service.resources.list.provider;
23  
24  import org.opencastproject.elasticsearch.index.ElasticsearchIndex;
25  import org.opencastproject.elasticsearch.index.objects.event.Event;
26  import org.opencastproject.elasticsearch.index.objects.event.EventIndexSchema;
27  import org.opencastproject.list.api.ResourceListProvider;
28  import org.opencastproject.list.api.ResourceListQuery;
29  import org.opencastproject.workflow.api.WorkflowInstance.WorkflowState;
30  
31  import org.osgi.framework.BundleContext;
32  import org.osgi.service.component.annotations.Activate;
33  import org.osgi.service.component.annotations.Component;
34  import org.osgi.service.component.annotations.Reference;
35  import org.slf4j.Logger;
36  import org.slf4j.LoggerFactory;
37  
38  import java.util.HashMap;
39  import java.util.Map;
40  
41  @Component(
42      service = ResourceListProvider.class,
43      property = {
44          "service.description=Events list provider",
45          "opencast.service.type=org.opencastproject.index.service.resources.list.provider.EventsListProvider"
46      }
47  )
48  public class EventsListProvider implements ResourceListProvider {
49  
50    private static final String PROVIDER_PREFIX = "EVENTS";
51  
52    public static final String CONTRIBUTORS = PROVIDER_PREFIX + ".CONTRIBUTORS";
53    public static final String PRESENTERS_BIBLIOGRAPHIC = PROVIDER_PREFIX + ".PRESENTERS_BIBLIOGRAPHIC";
54    public static final String PRESENTERS_TECHNICAL = PROVIDER_PREFIX + ".PRESENTERS_TECHNICAL";
55    public static final String SUBJECT = PROVIDER_PREFIX + ".SUBJECT";
56    public static final String LOCATION = PROVIDER_PREFIX + ".LOCATION";
57    public static final String PROGRESS = PROVIDER_PREFIX + ".PROGRESS";
58    public static final String STATUS = PROVIDER_PREFIX + ".STATUS";
59    public static final String COMMENTS = PROVIDER_PREFIX + ".COMMENTS";
60    public static final String PUBLISHER = PROVIDER_PREFIX + ".PUBLISHER";
61    public static final String ISPUBLISHED = PROVIDER_PREFIX + ".ISPUBLISHED";
62  
63    public enum Comments {
64      NONE, OPEN, RESOLVED;
65    }
66  
67    public enum IsPublished {
68      YES, NO;
69    }
70  
71    private static final String[] NAMES = { PROVIDER_PREFIX, CONTRIBUTORS, PRESENTERS_BIBLIOGRAPHIC, PRESENTERS_TECHNICAL,
72            SUBJECT, LOCATION, PROGRESS, STATUS, COMMENTS, PUBLISHER, ISPUBLISHED };
73  
74    private static final Logger logger = LoggerFactory.getLogger(EventsListProvider.class);
75  
76    private ElasticsearchIndex index;
77  
78    @Activate
79    protected void activate(BundleContext bundleContext) {
80      logger.info("Events list provider activated!");
81    }
82  
83    @Reference
84    public void setIndex(ElasticsearchIndex index) {
85      this.index = index;
86    }
87  
88    @Override
89    public String[] getListNames() {
90      return NAMES;
91    }
92  
93    @Override
94    public Map<String, String> getList(String listName, ResourceListQuery query) {
95      Map<String, String> list = new HashMap<String, String>();
96  
97      if (CONTRIBUTORS.equals(listName)) {
98        for (String contributor : index.getTermsForField(EventIndexSchema.CONTRIBUTOR, Event.DOCUMENT_TYPE))
99          list.put(contributor, contributor);
100     } else if (PRESENTERS_BIBLIOGRAPHIC.equals(listName)) {
101       for (String presenter : index.getTermsForField(EventIndexSchema.PRESENTER, Event.DOCUMENT_TYPE))
102         list.put(presenter, presenter);
103     } else if (PRESENTERS_TECHNICAL.equals(listName)) {
104       for (String presenter : index.getTermsForField(EventIndexSchema.TECHNICAL_PRESENTERS, Event.DOCUMENT_TYPE))
105         list.put(presenter, presenter);
106     } else if (LOCATION.equals(listName)) {
107       for (String location : index.getTermsForField(EventIndexSchema.LOCATION, Event.DOCUMENT_TYPE))
108         list.put(location, location);
109     } else if (SUBJECT.equals(listName)) {
110       for (String subject : index.getTermsForField(EventIndexSchema.SUBJECT, Event.DOCUMENT_TYPE))
111         list.put(subject, subject);
112     } else if (PROGRESS.equals(listName)) {
113       for (WorkflowState progress : WorkflowState.values())
114         list.put(progress.toString(), progress.toString());
115     } else if (STATUS.equals(listName)) {
116       list.put("EVENTS.EVENTS.STATUS.SCHEDULED", "EVENTS.EVENTS.STATUS.SCHEDULED");
117       list.put("EVENTS.EVENTS.STATUS.RECORDING", "EVENTS.EVENTS.STATUS.RECORDING");
118       list.put("EVENTS.EVENTS.STATUS.INGESTING", "EVENTS.EVENTS.STATUS.INGESTING");
119       list.put("EVENTS.EVENTS.STATUS.PENDING", "EVENTS.EVENTS.STATUS.PENDING");
120       list.put("EVENTS.EVENTS.STATUS.PROCESSING", "EVENTS.EVENTS.STATUS.PROCESSING");
121       list.put("EVENTS.EVENTS.STATUS.PAUSED", "EVENTS.EVENTS.STATUS.PAUSED");
122       list.put("EVENTS.EVENTS.STATUS.PROCESSED", "EVENTS.EVENTS.STATUS.PROCESSED");
123       list.put("EVENTS.EVENTS.STATUS.RECORDING_FAILURE", "EVENTS.EVENTS.STATUS.RECORDING_FAILURE");
124       list.put("EVENTS.EVENTS.STATUS.PROCESSING_FAILURE", "EVENTS.EVENTS.STATUS.PROCESSING_FAILURE");
125       list.put("EVENTS.EVENTS.STATUS.PROCESSING_CANCELLED", "EVENTS.EVENTS.STATUS.PROCESSING_CANCELLED");
126     } else if (COMMENTS.equals(listName)) {
127       for (Comments comments : Comments.values())
128         list.put(comments.toString(), "FILTERS.EVENTS.COMMENTS." + comments.toString());
129     } else if (PUBLISHER.equals(listName)) {
130       for (String publisher : index.getTermsForField(EventIndexSchema.PUBLISHER, Event.DOCUMENT_TYPE))
131         list.put(publisher, publisher);
132     } else if (ISPUBLISHED.equals(listName)) {
133       for (IsPublished isPublished : IsPublished.values())
134         list.put(isPublished.toString(), "FILTERS.EVENTS.ISPUBLISHED." + isPublished.toString());
135     }
136 
137     return list;
138   }
139 
140   @Override
141   public boolean isTranslatable(String listName) {
142     return STATUS.equals(listName) || COMMENTS.equals(listName);
143   }
144 
145   @Override
146   public String getDefault() {
147     return null;
148   }
149 }
