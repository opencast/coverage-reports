22  package org.opencastproject.util;
23  
24  import java.io.BufferedReader;
25  import java.io.IOException;
26  import java.io.InputStream;
27  import java.io.InputStreamReader;
28  import java.util.concurrent.CountDownLatch;
29  import java.util.function.Function;
30  
31  /**
32   * A StreamConsumer helps to asynchronously consume a text input stream line by line.
33   * The consumer guarantees the closing of the stream.
34   */
35  public class StreamConsumer implements Runnable {
36    private final CountDownLatch running = new CountDownLatch(1);
37    private final CountDownLatch ready = new CountDownLatch(1);
38    private final CountDownLatch finished = new CountDownLatch(1);
39  
40    private final Function<String, Boolean> consumer;
41  
42    private boolean stopped = false;
43    private InputStream stream;
44    private BufferedReader reader;
45  
46    /**
47     * Create a new stream consumer.
48     *
49     * @param consumer
50     *         a predicate function that may stop reading further lines by returning <code>false</code>
51     */
52    public StreamConsumer(Function<String, Boolean> consumer) {
53      this.consumer = consumer;
54    }
55  
56    @Override public void run() {
57      try {
58        running.countDown();
59        ready.await();
60        // also save a reference to the reader to able to close it in stopReading
61        // otherwise the read loop may continue reading from the buffer
62        reader = new BufferedReader(new InputStreamReader(stream));
63        IoSupport.<Void, BufferedReader>withResource(reader, consumeBufferedFunction);
64        finished.countDown();
65      } catch (InterruptedException e) {
66        throw new RuntimeException(e);
67      }
68    }
69  
70    /** Wait for the executing thread to run. */
71    public void waitUntilRunning() {
72      try {
73        running.await();
74      } catch (InterruptedException e) {
75        throw new RuntimeException(e);
76      }
77    }
78  
79    /** Wait until the stream has been fully consumed. */
80    public void waitUntilFinished() {
81      try {
82        finished.await();
83      } catch (InterruptedException e) {
84        throw new RuntimeException(e);
85      }
86    }
87  
88    /** Forcibly stop consuming the stream. */
89    public void stopConsuming() {
90      if (stream != null) {
91        stopped = true;
92        IoSupport.closeQuietly(stream);
93        IoSupport.closeQuietly(reader);
94      }
95    }
96  
97    /** Start consuming <code>stream</code>. It is guaranteed that the stream gets closed. */
98    public void consume(InputStream stream) {
99      waitUntilRunning();
100     this.stream = stream;
101     ready.countDown();
102   }
103 
104   private final Function<BufferedReader, Void> consumeBufferedFunction = new Function<BufferedReader, Void>() {
105     @Override
106     public Void apply(BufferedReader bufferedReader) {
107       consumeBuffered(bufferedReader);
108       return null; // Void return
109     }
110   };
111 
112   private void consumeBuffered(BufferedReader reader) {
113     try {
114       String line;
115       while ((line = reader.readLine()) != null) {
116         if (!consumer.apply(line)) {
117           stopConsuming();
118           break;
119         }
120       }
121     } catch (IOException e) {
122       if (!stopped) {
123         throw new RuntimeException(e);
124       }
125     }
126   }
127 }
