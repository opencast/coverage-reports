57  /**
58   * A remote publication service invoker.
59   */
60  @Component(
61      immediate = true,
62      service = ConfigurablePublicationService.class,
63      property = {
64          "service.description=Publication (Configurable) Remote Service Proxy"
65      }
66  )
67  public class ConfigurablePublicationServiceRemoteImpl extends RemoteBase implements ConfigurablePublicationService {
68  
69    /** The logger */
70    private static final Logger logger = LoggerFactory.getLogger(ConfigurablePublicationServiceRemoteImpl.class);
71  
72    /* Gson is thread-safe so we use a single instance */
73    private Gson gson = new Gson();
74  
75    public ConfigurablePublicationServiceRemoteImpl() {
76      super(JOB_TYPE);
77    }
78  
79    @Override
80    public Job replace(final MediaPackage mediaPackage, final String channelId,
81            final Collection<? extends MediaPackageElement> addElements, final Set<String> retractElementIds)
82                throws PublicationException, MediaPackageException {
83  
84      final List<BasicNameValuePair> params = new ArrayList<>();
85      params.add(new BasicNameValuePair("mediapackage", MediaPackageParser.getAsXml(mediaPackage)));
86      params.add(new BasicNameValuePair("channel", channelId));
87      params.add(new BasicNameValuePair("addElements", MediaPackageElementParser.getArrayAsXml(addElements)));
88      params.add(new BasicNameValuePair("retractElements", gson.toJson(retractElementIds)));
89  
90      final HttpPost post = new HttpPost("/replace");
91      HttpResponse response = null;
92      try {
93        post.setEntity(new UrlEncodedFormEntity(params, UTF_8));
94        response = getResponse(post);
95        if (response != null) {
96          logger.info("Publishing media package {} to channel {} using a remote publication service",
97                  mediaPackage, channelId);
98          try {
99            return JobParser.parseJob(response.getEntity().getContent());
100         } catch (final Exception e) {
101           throw new PublicationException(
102                   "Unable to publish media package '" + mediaPackage + "' using a remote publication service", e);
103         }
104       }
105     } catch (final Exception e) {
106       throw new PublicationException(
107               "Unable to publish media package " + mediaPackage + " using a remote publication service.", e);
108     } finally {
109       closeConnection(response);
110     }
111     throw new PublicationException(
112             "Unable to publish mediapackage " + mediaPackage + " using a remote publication service.");
113   }
114 
115   @Override
116   public Publication replaceSync(
117       MediaPackage mediaPackage, String channelId, Collection<? extends MediaPackageElement> addElements,
118       Set<String> retractElementIds) throws PublicationException, MediaPackageException {
119     final List<BasicNameValuePair> params = new ArrayList<>();
120     params.add(new BasicNameValuePair("mediapackage", MediaPackageParser.getAsXml(mediaPackage)));
121     params.add(new BasicNameValuePair("channel", channelId));
122     params.add(new BasicNameValuePair("addElements", MediaPackageElementParser.getArrayAsXml(addElements)));
123     params.add(new BasicNameValuePair("retractElements", gson.toJson(retractElementIds)));
124 
125     final HttpPost post = new HttpPost("/replacesync");
126     HttpResponse response = null;
127     try {
128       post.setEntity(new UrlEncodedFormEntity(params, UTF_8));
129       response = getResponse(post);
130       if (response != null) {
131         logger.info("Publishing media package {} to channel {} using a remote publication service",
132             mediaPackage, channelId);
133         try {
134           final String xml = IOUtils.toString(response.getEntity().getContent(), Charset.forName("utf-8"));
135           return (Publication) MediaPackageElementParser.getFromXml(xml);
136         } catch (final Exception e) {
137           throw new PublicationException(
138               "Unable to publish media package '" + mediaPackage + "' using a remote publication service", e);
139         }
140       }
141     } catch (final Exception e) {
142       throw new PublicationException(
143           "Unable to publish media package " + mediaPackage + " using a remote publication service.", e);
144     } finally {
145       closeConnection(response);
146     }
147     throw new PublicationException(
148         "Unable to publish mediapackage " + mediaPackage + " using a remote publication service.");
149   }
