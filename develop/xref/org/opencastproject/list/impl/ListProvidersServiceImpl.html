26  import org.opencastproject.list.api.ResourceListProvider;
27  import org.opencastproject.list.api.ResourceListQuery;
28  import org.opencastproject.list.util.ListProviderUtil;
29  import org.opencastproject.security.api.SecurityService;

31  import org.osgi.service.component.annotations.Component;
32  import org.osgi.service.component.annotations.Reference;
33  import org.osgi.service.component.annotations.ReferenceCardinality;
34  import org.osgi.service.component.annotations.ReferencePolicy;
35  import org.osgi.service.component.annotations.ReferencePolicyOption;
36  import org.slf4j.Logger;
37  import org.slf4j.LoggerFactory;
38  
39  import java.util.ArrayList;
40  import java.util.List;
41  import java.util.Map;
42  import java.util.Objects;
43  import java.util.concurrent.ConcurrentHashMap;
44  
45  @Component(
46      service = ListProvidersService.class,
47      property = {
48          "service.description=Resources list providers service",
49          "opencast.service.type=org.opencastproject.list.api.ListProvidersService"
50      }
51  )
52  public class ListProvidersServiceImpl implements ListProvidersService {
53  
54    static final String ALL_ORGANIZATIONS = "*";
55  
56    private SecurityService securityService;
57    private Map<ResourceTuple, ResourceListProvider> providers = new ConcurrentHashMap<>();
58    private static final Logger logger = LoggerFactory.getLogger(ListProvidersServiceImpl.class);
59  
60    /**
61     * Instances of this class represent unique keys for the parent {@link ConcurrentHashMap},
62     * made up of both a provider ID and its associated organisation ID.
63     */
64    static class ResourceTuple {
65      private final String resourceName;
66      private final String organizationId;
67  
68      ResourceTuple(String resourceName, String organizationId) {
69        this.resourceName = resourceName;
70        this.organizationId = organizationId;
71      }
72  
73  
74      String getResourceName() {
75        return resourceName;
76      }
77  
78      String getOrganizationId() {
79        return organizationId;
80      }
81  
82      @Override
83      public boolean equals(Object o) {
84        if (this == o) {
85          return true;
86        }
87  
88        return o instanceof ResourceTuple
89                && Objects.equals(resourceName, ((ResourceTuple) o).resourceName)
90                && Objects.equals(organizationId, ((ResourceTuple) o).organizationId);
91      }
92  
93      @Override
94      public int hashCode() {
95        return Objects.hash(resourceName, organizationId);
96      }
97    }
98  
99      /** OSGi callback for security service */
100   @Reference
101   public void setSecurityService(SecurityService securityService) {
102     this.securityService = securityService;
103   }
104 
105   /** OSGi callback for provider. */
106   @Reference(
107       cardinality = ReferenceCardinality.MULTIPLE,
108       policy = ReferencePolicy.DYNAMIC,
109       policyOption = ReferencePolicyOption.GREEDY,
110       unbind = "removeProvider"
111   )
112   public void addProvider(ResourceListProvider provider) {
113     for (String listName : provider.getListNames()) {
114       addProvider(listName, provider);
115     }
116   }
117 
118   /** OSGi callback for provider. */
119   public void removeProvider(ResourceListProvider provider) {
120     for (String listName : provider.getListNames()) {
121       removeProvider(listName);
122     }
123   }
124 
125   /**
126    *
127    * @param resourceName the name of the provider
128    * @return the queried list provider
129    * @throws ListProviderNotFoundException if no list provider matches query
130    */
131   private ResourceListProvider getProvider(String resourceName)
132           throws ListProviderNotFoundException {
133     ResourceListProvider provider;
134     String organizationId;
135 
136     if (securityService.getOrganization() != null) {
137       organizationId = securityService.getOrganization().getId();
138       provider = providers.get(new ResourceTuple(resourceName, organizationId));
139       if (provider != null) {
140         return provider;
141       }
142       // use default if no specific provider is set
143       provider = providers.get(new ResourceTuple(resourceName, ALL_ORGANIZATIONS));
144     } else {
145       organizationId = ALL_ORGANIZATIONS;
146       provider = providers.get(new ResourceTuple(resourceName, ALL_ORGANIZATIONS));
147     }
148     if (provider != null) {
149       return provider;
150     } else {
151       throw new ListProviderNotFoundException("No provider found for organisation <"
152             + organizationId + "> with the name " + resourceName);
153     }
154   }
155 
156   @Override
157   public Map<String, String> getList(String listName, ResourceListQuery query, boolean inverseValueKey)
158           throws ListProviderException {
159     ResourceListProvider provider = getProvider(listName);
160     Map<String, String> list = provider.getList(listName, query);
161     return inverseValueKey ? ListProviderUtil.invertMap(list) : list;
162   }
163 
164   @Override
165   public boolean isTranslatable(String listName) throws ListProviderNotFoundException {
166     ResourceListProvider provider = getProvider(listName);
167     return provider.isTranslatable(listName);
168   }
169 
170   @Override
171   public String getDefault(String listName) throws ListProviderNotFoundException {
172     ResourceListProvider provider = getProvider(listName);
173     return provider.getDefault();
174   }
175 
176   @Override
177   public void addProvider(String listName, ResourceListProvider provider) {
178     addProvider(listName, provider, ALL_ORGANIZATIONS);
179   }
180 
181   @Override
182   public void addProvider(String listName, ResourceListProvider provider, String organizationId) {
183     providers.put(new ResourceTuple(listName, organizationId), provider);
184   }
185 
186   @Override
187   public void removeProvider(String name) {
188     removeProvider(name, ALL_ORGANIZATIONS);
189   }
190 
191   @Override
192   public void removeProvider(String name, String organizationId) {
193     providers.remove(new ResourceTuple(name, organizationId));
194   }
195 
196   @Override
197   public boolean hasProvider(String name) {
198     return hasProvider(name, ALL_ORGANIZATIONS);
199   }
200 
201   @Override
202   public boolean hasProvider(String name, String organizationId) {
203     return providers.containsKey(new ResourceTuple(name, organizationId));
204   }
205 
206   @Override
207   public List<String> getAvailableProviders() {
208     List<String> sources = new ArrayList<>();
209     for (ResourceTuple key : providers.keySet()) {
210       sources.add(key.getResourceName());
211     }
212     return sources;
213   }
214 }
