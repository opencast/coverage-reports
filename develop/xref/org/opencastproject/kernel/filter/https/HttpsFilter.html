22  package org.opencastproject.kernel.filter.https;
23  
24  import org.osgi.service.component.annotations.Component;
25  import org.osgi.service.component.annotations.ServiceScope;
26  import org.osgi.service.component.propertytypes.ServiceRanking;
27  import org.osgi.service.http.whiteboard.propertytypes.HttpWhiteboardContextSelect;
28  import org.osgi.service.http.whiteboard.propertytypes.HttpWhiteboardFilterName;
29  import org.osgi.service.http.whiteboard.propertytypes.HttpWhiteboardFilterPattern;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import java.io.IOException;
34  
35  import javax.servlet.Filter;
36  import javax.servlet.FilterChain;
37  import javax.servlet.FilterConfig;
38  import javax.servlet.ServletException;
39  import javax.servlet.ServletRequest;
40  import javax.servlet.ServletResponse;
41  import javax.servlet.http.HttpServletRequest;
42  
43  /**
44   * This filter is wrapping <code>HttpServletRequest</code>s in such a way that they feature the https scheme.
45   */
46  @Component(
47      service = Filter.class,
48      scope = ServiceScope.PROTOTYPE,
49      property = {
50          "service.description=Https Filter",
51      }
52  )
53  @ServiceRanking(990)
54  @HttpWhiteboardFilterName("HttpsFilter")
55  @HttpWhiteboardFilterPattern("/*")
56  @HttpWhiteboardContextSelect("(osgi.http.whiteboard.context.name=opencast)")
57  public class HttpsFilter implements Filter {
58    /** The logger */
59    private static final Logger logger = LoggerFactory.getLogger(HttpsFilter.class);
60  
61    /** Request header that is set when behind an SSL proxy */
62    private static final String X_FORWARDED_SSL = "X-Forwarded-SSL";
63  
64    /** Alternative request header set by proxy */
65    private static final String X_FORWARDED_PROTO = "X-Forwarded-Proto";
66  
67    /** Value of the X-Forwarded-SSL header that activates request wrapping */
68    private static final String X_FORWARDED_SSL_VALUE = "on";
69  
70    /** Value of the X-Forwarded-Proto header that activates request wrapping */
71    private static final String X_FORWARDED_PROTO_VALUE = "https";
72  
73    /**
74     * @see javax.servlet.Filter#doFilter(javax.servlet.ServletRequest, javax.servlet.ServletResponse,
75     *      javax.servlet.FilterChain)
76     */
77    @Override
78    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException,
79            ServletException {
80      HttpServletRequest httpReqquest = (HttpServletRequest) request;
81  
82      // Check if the forwarded SSL header is set
83      if (X_FORWARDED_SSL_VALUE.equalsIgnoreCase(httpReqquest.getHeader(X_FORWARDED_SSL))) {
84        logger.debug("Found forwarded SSL header");
85        httpReqquest = new HttpsRequestWrapper(httpReqquest);
86      } else if (X_FORWARDED_PROTO_VALUE.equalsIgnoreCase(httpReqquest.getHeader(X_FORWARDED_PROTO))) {
87        logger.debug("Found forwarded proto HTTPS header");
88        httpReqquest = new HttpsRequestWrapper(httpReqquest);
89      }
90      chain.doFilter(httpReqquest, response);
91    }
92  
93    @Override
94    public void init(FilterConfig filterConfig) throws ServletException {
95    }
96  
97    @Override
98    public void destroy() {
99    }
100 
101 }
