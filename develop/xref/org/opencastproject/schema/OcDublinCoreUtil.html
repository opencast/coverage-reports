87    /**
88     * Create a generic DublinCoreCatalog from an OcDublinCore.
89     * Fields are encoded according to the Opencast rules. This class provides functions for each DublinCore
90     * property that needs special encoding.
91     */
92    public static DublinCoreCatalog toCatalog(final OcDublinCore source) {
93      // completeness assured by unit test
94      final DublinCoreCatalog target = DublinCores.mkOpencastEpisode().getCatalog();
95      for (String a : source.getAbstract()) target.set(DublinCore.PROPERTY_ABSTRACT, a);
96      for (String a : source.getAccessRights()) target.set(DublinCore.PROPERTY_ACCESS_RIGHTS, a);
97      for (String a : source.getAccrualMethod()) target.set(DublinCore.PROPERTY_ACCRUAL_METHOD, a);
98      for (String a : source.getAccrualPeriodicity()) target.set(DublinCore.PROPERTY_ACCRUAL_PERIODICITY, a);
99      for (String a : source.getAccrualPolicy()) target.set(DublinCore.PROPERTY_ACCRUAL_POLICY, a);
100     for (String a : source.getAlternative()) target.set(DublinCore.PROPERTY_ALTERNATIVE, a);
101     for (String a : source.getAudience()) target.set(DublinCore.PROPERTY_AUDIENCE, a);
102     for (String a : source.getAvailable()) target.set(DublinCore.PROPERTY_AVAILABLE, a);
103     for (String a : source.getBibliographicCitation()) target.set(DublinCore.PROPERTY_BIBLIOGRAPHIC_CITATION, a);
104     for (String a : source.getConformsTo()) target.set(DublinCore.PROPERTY_CONFORMS_TO, a);
105     for (String a : source.getContributor()) target.set(DublinCore.PROPERTY_CONTRIBUTOR, a);
106     for (String a : source.getCoverage()) target.set(DublinCore.PROPERTY_COVERAGE, a);
107     target.set(DublinCore.PROPERTY_CREATED, encodeCreated(source.getCreated()));
108     for (String a : source.getCreator()) target.set(DublinCore.PROPERTY_CREATOR, a);
109     for (Date a : source.getDate()) target.set(DublinCore.PROPERTY_DATE, encodeDate(a));
110     for (Date a : source.getDateAccepted()) target.set(DublinCore.PROPERTY_DATE_ACCEPTED, encodeDateAccepted(a));
111     for (Date a : source.getDateCopyrighted())
112       target.set(DublinCore.PROPERTY_DATE_COPYRIGHTED, encodeDateCopyrighted(a));
113     for (Date a : source.getDateSubmitted()) target.set(DublinCore.PROPERTY_DATE_SUBMITTED, encodeDateSubmitted(a));
114     for (String a : source.getDescription()) target.set(DublinCore.PROPERTY_DESCRIPTION, a);
115     for (String a : source.getEducationLevel()) target.set(DublinCore.PROPERTY_EDUCATION_LEVEL, a);
116     for (Long a : source.getExtent()) target.set(DublinCore.PROPERTY_EXTENT, encodeExtent(a));
117     for (String a : source.getFormat()) target.set(DublinCore.PROPERTY_FORMAT, a);
118     for (String a : source.getHasFormat()) target.set(DublinCore.PROPERTY_HAS_FORMAT, a);
119     for (String a : source.getHasPart()) target.set(DublinCore.PROPERTY_HAS_PART, a);
120     for (String a : source.getHasVersion()) target.set(DublinCore.PROPERTY_HAS_VERSION, a);
121     for (String a : source.getIdentifier()) target.set(DublinCore.PROPERTY_IDENTIFIER, a);
122     for (String a : source.getInstructionalMethod()) target.set(DublinCore.PROPERTY_INSTRUCTIONAL_METHOD, a);
123     for (String a : source.getIsFormatOf()) target.set(DublinCore.PROPERTY_IS_FORMAT_OF, a);
124     for (String a : source.getIsPartOf()) target.set(DublinCore.PROPERTY_IS_PART_OF, a);
125     for (String a : source.getIsReferencedBy()) target.set(DublinCore.PROPERTY_IS_REFERENCED_BY, a);
126     for (String a : source.getIsReplacedBy()) target.set(DublinCore.PROPERTY_IS_REPLACED_BY, a);
127     for (String a : source.getIsRequiredBy()) target.set(DublinCore.PROPERTY_IS_REQUIRED_BY, a);
128     for (String a : source.getIssued()) target.set(DublinCore.PROPERTY_ISSUED, a);
129     for (String a : source.getIsVersionOf()) target.set(DublinCore.PROPERTY_IS_VERSION_OF, a);
130     for (String a : source.getLanguage()) target.set(DublinCore.PROPERTY_LANGUAGE, a);
131     for (String a : source.getLicense()) target.set(DublinCore.PROPERTY_LICENSE, a);
132     for (String a : source.getMediator()) target.set(DublinCore.PROPERTY_MEDIATOR, a);
133     for (String a : source.getMedium()) target.set(DublinCore.PROPERTY_MEDIUM, a);
134     for (String a : source.getModified()) target.set(DublinCore.PROPERTY_MODIFIED, a);
135     for (String a : source.getProvenance()) target.set(DublinCore.PROPERTY_PROVENANCE, a);
136     for (String a : source.getPublisher()) target.set(DublinCore.PROPERTY_PUBLISHER, a);
137     for (String a : source.getReferences()) target.set(DublinCore.PROPERTY_REFERENCES, a);
138     for (String a : source.getRelation()) target.set(DublinCore.PROPERTY_RELATION, a);
139     for (String a : source.getReplaces()) target.set(DublinCore.PROPERTY_REPLACES, a);
140     for (String a : source.getRequires()) target.set(DublinCore.PROPERTY_REQUIRES, a);
141     for (String a : source.getRights()) target.set(DublinCore.PROPERTY_RIGHTS, a);
142     for (String a : source.getRightsHolder()) target.set(DublinCore.PROPERTY_RIGHTS_HOLDER, a);
143     for (String a : source.getSource()) target.set(DublinCore.PROPERTY_SOURCE, a);
144     for (String a : source.getSpatial()) target.set(DublinCore.PROPERTY_SPATIAL, a);
145     for (String a : source.getSubject()) target.set(DublinCore.PROPERTY_SUBJECT, a);
146     for (String a : source.getTableOfContents()) target.set(DublinCore.PROPERTY_TABLE_OF_CONTENTS, a);
147     for (String a : source.getTemporal()) target.set(DublinCore.PROPERTY_TEMPORAL, a);
148     target.set(DublinCore.PROPERTY_TITLE, source.getTitle());
149     for (String a : source.getType()) target.set(DublinCore.PROPERTY_TYPE, a);
150     for (String a : source.getValid()) target.set(DublinCore.PROPERTY_VALID, a);
151     return target;
152   }
