97      EasyMock.replay(statusLine);
98  
99      HttpResponse response = EasyMock.createNiceMock(HttpResponse.class);
100     EasyMock.expect(response.getStatusLine()).andReturn(statusLine).anyTimes();
101     EasyMock.replay(response);
102 
103     final TrustedHttpClient httpClient = EasyMock.createNiceMock(TrustedHttpClient.class);
104     EasyMock.expect(httpClient.execute((HttpUriRequest) EasyMock.anyObject())).andReturn(response).anyTimes();
105     EasyMock.replay(httpClient);
106 
107     defaultOrganization = new DefaultOrganization();
108     User anonymous = new JaxbUser("anonymous", "test", defaultOrganization,
109             new JaxbRole(DefaultOrganization.DEFAULT_ORGANIZATION_ANONYMOUS, defaultOrganization));
110     UserDirectoryService userDirectoryService = EasyMock.createMock(UserDirectoryService.class);
111     EasyMock.expect(userDirectoryService.loadUser((String) EasyMock.anyObject())).andReturn(anonymous).anyTimes();
112     EasyMock.replay(userDirectoryService);
113     service.setUserDirectoryService(userDirectoryService);
114 
115     Organization organization = new DefaultOrganization();
116     OrganizationDirectoryService organizationDirectoryService = EasyMock.createMock(OrganizationDirectoryService.class);
117     EasyMock.expect(organizationDirectoryService.getOrganization((String) EasyMock.anyObject())).andReturn(organization)
118             .anyTimes();
119     EasyMock.replay(organizationDirectoryService);
120     service.setOrganizationDirectoryService(organizationDirectoryService);
121 
122     SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
123     EasyMock.expect(securityService.getUser()).andReturn(anonymous).anyTimes();
124     EasyMock.expect(securityService.getOrganization()).andReturn(organization).anyTimes();
125     EasyMock.replay(securityService);
126     service.setSecurityService(securityService);

128     serviceRegistry = new ServiceRegistryInMemoryImpl(service, securityService, userDirectoryService,
129             organizationDirectoryService, EasyMock.createNiceMock(IncidentService.class));

133     final Workspace workspace = EasyMock.createNiceMock(Workspace.class);
134     service.setWorkspace(workspace);
135 
136     EasyMock.expect(workspace.get((URI) EasyMock.anyObject())).andAnswer(new IAnswer<File>() {
137       @Override

139         final URI uri = (URI) EasyMock.getCurrentArguments()[0];
140         final String[] pathElems = uri.getPath().split("/");
141         final String file = pathElems[pathElems.length - 1];
142         return new File(mediaPackageRoot, file);
143       }
144     }).anyTimes();
145     final Capture<String> myId1 = Capture.newInstance();
146     final Capture<String> myId2 = Capture.newInstance();
147     final Capture<String> name = Capture.newInstance();
148     final Capture<InputStream> in = Capture.newInstance();
149     EasyMock.expect(workspace.put(EasyMock.capture(myId1), EasyMock.capture(myId2), EasyMock.capture(name),

151               @Override
152               public URI answer() throws Throwable {
153                 File file = new File(distributionRoot,
154                         myId1.getValue() + "/" + myId2.getValue() + "/" + name.getValue());
155                 FileUtils.copyInputStreamToFile(in.getValue(), file);
156                 return file.toURI();
157               }
158             }).anyTimes();
159 
160     EasyMock.replay(workspace);
161 
162     BundleContext bc = EasyMock.createNiceMock(BundleContext.class);
163     EasyMock.expect(bc.getProperty("org.opencastproject.download.directory"))
164         .andReturn(distributionRoot.toString()).anyTimes();
165     EasyMock.expect(bc.getProperty("org.opencastproject.download.url"))
166         .andReturn(UrlSupport.DEFAULT_BASE_URL).anyTimes();
167     EasyMock.expect(bc.getProperty(DIGEST_USER_PROPERTY))
168         .andReturn("opencast_system_account").anyTimes();
169     ComponentContext cc = EasyMock.createNiceMock(ComponentContext.class);
170     Dictionary<String, Object> p = new Hashtable<String, Object>();
171     p.put(DistributionService.CONFIG_KEY_STORE_TYPE, "download");
172     EasyMock.expect(cc.getProperties()).andReturn(p).anyTimes();
173     EasyMock.expect(cc.getBundleContext()).andReturn(bc).anyTimes();
174     EasyMock.replay(bc, cc);
175     service.activate(cc);
176   }

184   @Test
185   public void testPartialDistribution() throws Exception {
186     // Distribute only some of the elements in the mediapackage
187     Job job1 = service.distribute("engage-player", mp, "track-1");
188     Job job2 = service.distribute("engage-player", mp, "catalog-1");
189     Job job3 = service.distribute("oai-pmh", mp, "catalog-1");
190     JobBarrier jobBarrier = new JobBarrier(null, serviceRegistry, 500, job1, job2, job3);
191     jobBarrier.waitForJobs();
192 
193     File mpDir = new File(distributionRoot,
194             PathSupport.path(defaultOrganization.getId(), "engage-player", mp.getIdentifier().toString()));
195     Assert.assertTrue(mpDir.exists());
196     File mediaDir = new File(mpDir, "track-1");
197     File metadataDir = new File(mpDir, "catalog-1");
198     File attachmentsDir = new File(mpDir, "notes");
199     Assert.assertTrue(mediaDir.exists());
200     Assert.assertTrue(metadataDir.exists());
201     Assert.assertFalse(attachmentsDir.exists());
202     Assert.assertTrue(new File(mediaDir, "media.mov").exists()); // the filenames are changed to reflect the element ID
203     Assert.assertTrue(new File(metadataDir, "dublincore.xml").exists());
204     Assert.assertFalse(new File(metadataDir, "mpeg7.xml").exists());
205     Assert.assertFalse(new File(attachmentsDir, "attachment.txt").exists());
206   }

208   @Test
209   public void testRetractByOriginal() throws Exception {
210     // Distribute the mediapackage and all of its elements
211     Job job1 = service.distribute("engage-player", mp, "track-1");
212     Job job2 = service.distribute("engage-player", mp, "catalog-1");
213     Job job3 = service.distribute("engage-player", mp, "catalog-2");
214     Job job4 = service.distribute("engage-player", mp, "notes");
215     Job job5 = service.distribute("oai-pmh", mp, "track-1");
216     JobBarrier jobBarrier = new JobBarrier(null, serviceRegistry, 500, job1, job2, job3, job4, job5);
217     jobBarrier.waitForJobs();
218 
219     // Add the new elements to the mediapackage
220     mp.add(MediaPackageElementParser.getFromXml(job1.getPayload()));
221     mp.add(MediaPackageElementParser.getFromXml(job2.getPayload()));
222     mp.add(MediaPackageElementParser.getFromXml(job3.getPayload()));
223     mp.add(MediaPackageElementParser.getFromXml(job4.getPayload()));
224     mp.add(MediaPackageElementParser.getFromXml(job5.getPayload()));
225 
226     File mpDir = new File(distributionRoot,
227             PathSupport.path(defaultOrganization.getId(), "engage-player", mp.getIdentifier().toString()));
228     File mediaDir = new File(mpDir, "track-1");
229     File metadata1Dir = new File(mpDir, "catalog-1");
230     File metadata2Dir = new File(mpDir, "catalog-2");
231     File attachmentsDir = new File(mpDir, "notes");
232     Assert.assertTrue(mediaDir.exists());
233     Assert.assertTrue(metadata1Dir.exists());
234     Assert.assertTrue(metadata2Dir.exists());
235     Assert.assertTrue(attachmentsDir.exists());
236     Assert.assertTrue(new File(mediaDir, "media.mov").exists()); // the filenames are changed to reflect the element ID
237     Assert.assertTrue(new File(metadata1Dir, "dublincore.xml").exists());
238     Assert.assertTrue(new File(metadata2Dir, "mpeg7.xml").exists());
239     Assert.assertTrue(new File(attachmentsDir, "attachment.txt").exists());
240 
241     // Now retract the mediapackage and ensure that the distributed files have been removed
242     Job job6 = service.retract("engage-player", mp, "track-1");
243     Job job7 = service.retract("engage-player", mp, "catalog-1");
244     Job job8 = service.retract("engage-player", mp, "catalog-2");
245     Job job9 = service.retract("engage-player", mp, "notes");
246     jobBarrier = new JobBarrier(null, serviceRegistry, 500, job6, job7, job8, job9);
247     jobBarrier.waitForJobs();
248 
249     Assert.assertFalse(service.getDistributionFile("engage-player", mp, mp.getElementById("track-1")).isFile());
250     Assert.assertFalse(service.getDistributionFile("engage-player", mp, mp.getElementById("catalog-1")).isFile());
251     Assert.assertFalse(service.getDistributionFile("engage-player", mp, mp.getElementById("catalog-2")).isFile());
252     Assert.assertFalse(service.getDistributionFile("engage-player", mp, mp.getElementById("notes")).isFile());
253     Assert.assertTrue(service.getDistributionFile("oai-pmh", mp, mp.getElementById("track-1")).isFile());
254   }

256   @Test
257   public void testRetractByDistributed() throws Exception {
258     int elementCount = mp.getElements().length;
259 
260     // Distribute the mediapackage and all of its elements
261     Job job1 = service.distribute("engage-player", mp, "track-1");
262     Job job2 = service.distribute("engage-player", mp, "catalog-1");
263     Job job3 = service.distribute("engage-player", mp, "catalog-2");
264     Job job4 = service.distribute("engage-player", mp, "notes");
265     Job job5 = service.distribute("oai-pmh", mp, "notes");
266     JobBarrier jobBarrier = new JobBarrier(null, serviceRegistry, 500, job1, job2, job3, job4, job5);
267     jobBarrier.waitForJobs();
268 
269     // Add the new elements to the mediapackage
270     MediaPackageElement job1Element = MediaPackageElementParser.getFromXml(job1.getPayload());
271     mp.add(job1Element);
272     MediaPackageElement job2Element = MediaPackageElementParser.getFromXml(job2.getPayload());
273     mp.add(job2Element);
274     MediaPackageElement job3Element = MediaPackageElementParser.getFromXml(job3.getPayload());
275     mp.add(job3Element);
276     MediaPackageElement job4Element = MediaPackageElementParser.getFromXml(job4.getPayload());
277     mp.add(job4Element);
278     MediaPackageElement job5Element = MediaPackageElementParser.getFromXml(job5.getPayload());
279     mp.add(job5Element);
280 
281     File mpDir = new File(distributionRoot,
282             PathSupport.path(defaultOrganization.getId(), "engage-player", mp.getIdentifier().toString()));
283     File mediaDir = new File(mpDir, "track-1");
284     File metadata1Dir = new File(mpDir, "catalog-1");
285     File metadata2Dir = new File(mpDir, "catalog-2");
286     File attachmentsDir = new File(mpDir, "notes");
287     Assert.assertTrue(mediaDir.exists());
288     Assert.assertTrue(metadata1Dir.exists());
289     Assert.assertTrue(metadata2Dir.exists());
290     Assert.assertTrue(attachmentsDir.exists());
291     Assert.assertTrue(new File(mediaDir, "media.mov").exists()); // the filenames are changed to reflect the element ID
292     Assert.assertTrue(new File(metadata1Dir, "dublincore.xml").exists());
293     Assert.assertTrue(new File(metadata2Dir, "mpeg7.xml").exists());
294     Assert.assertTrue(new File(attachmentsDir, "attachment.txt").exists());
295 
296     // Now retract the mediapackage and ensure that the distributed files have been removed
297     Job job6 = service.retract("engage-player", mp, job1Element.getIdentifier());
298     Job job7 = service.retract("engage-player", mp, job2Element.getIdentifier());
299     Job job8 = service.retract("engage-player", mp, job3Element.getIdentifier());
300     Job job9 = service.retract("engage-player", mp, job4Element.getIdentifier());
301     jobBarrier = new JobBarrier(null, serviceRegistry, 500, job6, job7, job8, job9);
302     jobBarrier.waitForJobs();
303 
304     // Remove the distributed elements from the mediapackage
305     mp.remove(MediaPackageElementParser.getFromXml(job6.getPayload()));
306     mp.remove(MediaPackageElementParser.getFromXml(job7.getPayload()));
307     mp.remove(MediaPackageElementParser.getFromXml(job8.getPayload()));
308     mp.remove(MediaPackageElementParser.getFromXml(job9.getPayload()));
309 
310     // +1 since the oai-pmh distribution has not been retracted
311     Assert.assertEquals(elementCount + 1, mp.getElements().length);
312     Assert.assertNotNull(mp.getElementById("track-1"));
313     Assert.assertNotNull(mp.getElementById("catalog-1"));
314     Assert.assertNotNull(mp.getElementById("catalog-2"));
315     Assert.assertNotNull(mp.getElementById("notes"));
316 
317     Assert.assertFalse(service.getDistributionFile("engage-player", mp, mp.getElementById("track-1")).isFile());
318     Assert.assertFalse(service.getDistributionFile("engage-player", mp, mp.getElementById("catalog-1")).isFile());
319     Assert.assertFalse(service.getDistributionFile("engage-player", mp, mp.getElementById("catalog-2")).isFile());
320     Assert.assertFalse(service.getDistributionFile("engage-player", mp, mp.getElementById("notes")).isFile());
321     Assert.assertTrue(service.getDistributionFile("oai-pmh", mp, mp.getElementById("notes")).isFile());
322   }

324   @Test
325   public void testHLSDistribution() throws Exception {
326     // Distribute only some of the elements in the mediapackage
327     mp = MediaPackageParser
328             .getFromXml(IOUtils.toString(getClass().getResourceAsStream("/hls_1_var_mediapackage.xml"), "UTF-8"));
329 
330     String[] elementIds = { "master", "variant", "segments", "dublincore", "mpeg7" };
331     Job job = service.distribute("engage-player", mp, Arrays.stream(elementIds).collect(Collectors.toSet()), true);
332     JobBarrier jobBarrier = new JobBarrier(null, serviceRegistry, 500, job);
333     jobBarrier.waitForJobs();
334 
335     File mpDir = new File(distributionRoot,
336             PathSupport.path(defaultOrganization.getId(), "engage-player", mp.getIdentifier().toString()));
337     Assert.assertTrue(mpDir.exists());
338     File mediaDir = new File(mpDir, "master");
339     File variantDir = new File(mpDir, "variant");
340     File segmentDir = new File(mpDir, "segments");
341     File metadataDir = new File(mpDir, "dublincore");
342     File attachmentsDir = new File(mpDir, "mpeg7");
343     Assert.assertTrue(mediaDir.exists());
344     Assert.assertTrue(variantDir.exists());
345     Assert.assertTrue(segmentDir.exists());
346     Assert.assertTrue(metadataDir.exists());
347     Assert.assertTrue(attachmentsDir.exists());
348     Assert.assertTrue(new File(mediaDir, "master.m3u8").exists()); // the filenames are changed to reflect the element
349                                                                    // ID
350     Assert.assertTrue(new File(variantDir, "variant_0.m3u8").exists()); // the filenames are changed to reflect the
351                                                                       // element ID
352     Assert.assertTrue(new File(segmentDir, "file_0.mp4").exists()); // the filenames are changed to reflect the element
353                                                                     // ID
354     Assert.assertTrue(new File(metadataDir, "dublincore.xml").exists());
355     Assert.assertTrue(new File(attachmentsDir, "mpeg7.xml").exists()); // the filenames are changed to reflect the
356                                                                        // element ID
357     Assert.assertFalse(new File(metadataDir, "mpeg7.xml").exists());
358     Assert.assertFalse(new File(attachmentsDir, "attachment.txt").exists());
359     File playlist = new File(mediaDir, "master.m3u8");
360     logger.info("Reading playlist " + playlist);
361     Optional<String> hasRelativeLink = Files.lines(playlist.toPath())
362             .filter(s -> s.contains("../variant/variant_0.m3u8"))
363             .findFirst();
364     Assert.assertTrue(hasRelativeLink.isPresent());
365     playlist = new File(variantDir, "variant_0.m3u8");
366     hasRelativeLink = Files.lines(playlist.toPath()).filter(s -> s.contains("../segments/file_0.mp4"))
367             .findFirst();
368     Assert.assertTrue(hasRelativeLink.isPresent());
369   }
