39  import java.io.IOException;
40  import java.util.ArrayList;
41  import java.util.Date;
42  import java.util.List;

44  public class SeriesTest {
45    private static final Logger logger = LoggerFactory.getLogger(SeriesTest.class);
46    private static final String ACCESS_POLICY_KEY = "access_policy";
47    private static final String CONTRIBUTORS_JSON_KEY = "contributors";
48    private static final String CONTRIBUTOR_JSON_KEY = "contributor";
49    private static final String CREATED_DATE_TIME = "createdDateTime";
50    private static final String CREATOR_JSON_KEY = "creator";
51    private static final String DESCRIPTION_JSON_KEY = "description";
52    private static final String IDENTIFIER_JSON_KEY = "identifier";
53    private static final String LANGUAGE_JSON_KEY = "language";
54    private static final String LICENSE_JSON_KEY = "license";
55    private static final String ORGANIZATION_JSON_KEY = "organization";
56    private static final String ORGANIZERS_JSON_KEY = "organizers";
57    private static final String ORGANIZER_JSON_KEY = "organizer";
58    private static final String SERIES_JSON_KEY = "series";
59    private static final String SUBJECT_JSON_KEY = "subject";
60    private static final String TITLE_JSON_KEY = "title";
61  
62    private String id = "10.0000-1";
63    private String title = "Land and Vegetation: Key players on the Climate Scene";
64    private String description = "This is the description for this series.";
65    private String subject = "This is the subject";
66    private String organization = "Organization";
67    private String language = "Klingon";
68    private String creator = "Creator Name";
69    private String license = "Creative Commons 2.0";
70    private String accessPolicy = "ROLE_ADMIN";
71    private Date createdDateTime;
72    private List<String> organizers = new ArrayList<String>();
73    private String organizer1 = "organizer-one";
74    private String organizer2 = "organizer-two";
75    private String organizer3 = "organizer-three";
76  
77    private List<String> contributors = new ArrayList<String>();
78    private String contributor1 = "contributor-one";
79    private String contributor2 = "contributor-two";
80    private String contributor3 = "contributor-three";
81  
82    private String seriesJson;
83    private String seriesXml;
84  
85    @Before
86    public void setUp() throws IOException, IllegalStateException, java.text.ParseException {
87      createdDateTime = new Date(DateTimeSupport.fromUTC("2011-07-16T20:39:05Z"));
88      // Setup presenter collection
89      organizers.add(organizer1);
90      organizers.add(organizer2);
91      organizers.add(organizer3);
92      // Setup contributors
93      contributors.add(contributor1);
94      contributors.add(contributor2);
95      contributors.add(contributor3);
96      // Setup results
97      seriesJson = IOUtils.toString(getClass().getResource("/adminui_series_metadata.json"));
98      seriesXml = IOUtils.toString(getClass().getResource("/adminui_series_metadata.xml"));
99    }
100 
101   @Test
102   public void testToJson() throws ParseException {
103     // Initialize series
104     Series series = new Series(id, organization);
105     series.setTitle(title);
106     series.setDescription(description);
107     series.setSubject(subject);
108     series.setLanguage(language);
109     series.setCreator(creator);
110     series.setLicense(license);
111     series.setAccessPolicy(accessPolicy);
112     series.setCreatedDateTime(createdDateTime);
113     series.setOrganizers(organizers);
114     series.setContributors(contributors);
115     logger.info(series.toJSON());
116 
117     // Check that generated JSON has proper values
118     JSONObject parse = (JSONObject) new JSONParser().parse(series.toJSON());
119     if (parse.get(SERIES_JSON_KEY) == null || !(parse.get(SERIES_JSON_KEY) instanceof JSONObject)) {
120       fail("There must be an series object returned.");
121     }
122     JSONObject seriesJsonObject = (JSONObject) parse.get(SERIES_JSON_KEY);
123     assertEquals(id, seriesJsonObject.get(IDENTIFIER_JSON_KEY));
124     assertEquals(title, seriesJsonObject.get(TITLE_JSON_KEY));
125     assertEquals(description, seriesJsonObject.get(DESCRIPTION_JSON_KEY));
126     assertEquals(subject, seriesJsonObject.get(SUBJECT_JSON_KEY));
127     assertEquals(organization, seriesJsonObject.get(ORGANIZATION_JSON_KEY));
128     assertEquals(language, seriesJsonObject.get(LANGUAGE_JSON_KEY));
129     assertEquals(creator, seriesJsonObject.get(CREATOR_JSON_KEY));
130     assertEquals(license, seriesJsonObject.get(LICENSE_JSON_KEY));
131     assertEquals(accessPolicy, seriesJsonObject.get(ACCESS_POLICY_KEY));
132     assertEquals(DateTimeSupport.toUTC(createdDateTime.getTime()), seriesJsonObject.get(CREATED_DATE_TIME));
133 
134     JSONObject organizersJsonObject = (JSONObject) seriesJsonObject.get(ORGANIZERS_JSON_KEY);
135     if (organizersJsonObject == null || !(organizersJsonObject.get(ORGANIZER_JSON_KEY) instanceof JSONArray)) {
136       fail("There should be an array of organizers returned.");
137     }
138     JSONArray organizersArray = (JSONArray) organizersJsonObject.get(ORGANIZER_JSON_KEY);
139     // Ordering not important, just a convenient shorthand.
140     assertEquals(organizer1, organizersArray.get(0));
141     assertEquals(organizer2, organizersArray.get(1));
142     assertEquals(organizer3, organizersArray.get(2));
143 
144     JSONObject contributorsJsonObject = (JSONObject) seriesJsonObject.get(CONTRIBUTORS_JSON_KEY);
145     if (contributorsJsonObject == null || !(contributorsJsonObject.get(CONTRIBUTOR_JSON_KEY) instanceof JSONArray)) {
146       fail("There should be an array of contributors returned.");
147     }
148     JSONArray contributorsArray = (JSONArray) contributorsJsonObject.get(CONTRIBUTOR_JSON_KEY);
149     // Ordering not important, just a convenient shorthand.
150     assertEquals(contributor1, contributorsArray.get(0));
151     assertEquals(contributor2, contributorsArray.get(1));
152     assertEquals(contributor3, contributorsArray.get(2));
153   }
154 }
