185   @Override
186   public WorkflowOperationResult start(final WorkflowInstance workflowInstance, JobContext context)
187           throws WorkflowOperationException {
188     logger.debug("Running partial import workflow operation on workflow {}", workflowInstance.getId());
189 
190     List<MediaPackageElement> elementsToClean = new ArrayList<MediaPackageElement>();
191 
192     try {
193       return concat(workflowInstance.getMediaPackage(), workflowInstance.getCurrentOperation(), elementsToClean);
194     } catch (Exception e) {
195       throw new WorkflowOperationException(e);
196     } finally {
197       for (MediaPackageElement elem : elementsToClean) {
198         try {
199           workspace.delete(elem.getURI());
200         } catch (Exception e) {
201           logger.warn("Unable to delete element {}", elem, e);
202         }
203       }
204     }
205   }

207   private WorkflowOperationResult concat(MediaPackage src, WorkflowOperationInstance operation,
208           List<MediaPackageElement> elementsToClean) throws EncoderException, IOException, NotFoundException,
209           MediaPackageException, WorkflowOperationException, ServiceRegistryException {
210     final MediaPackage mediaPackage = (MediaPackage) src.clone();
211     final Long operationId = operation.getId();
212     //
213     // read config options
214     final Optional<String> presenterFlavor = getOptConfig(operation, SOURCE_PRESENTER_FLAVOR);
215     final Optional<String> presentationFlavor = getOptConfig(operation, SOURCE_PRESENTATION_FLAVOR);
216     final MediaPackageElementFlavor smilFlavor = MediaPackageElementFlavor.parseFlavor(getConfig(operation, SOURCE_SMIL_FLAVOR));
217     final String concatEncodingProfile = getConfig(operation, CONCAT_ENCODING_PROFILE);
218     final Optional<String> concatOutputFramerate = getOptConfig(operation, CONCAT_OUTPUT_FRAMERATE);
219     final String trimEncodingProfile = getConfig(operation, TRIM_ENCODING_PROFILE);
220     final MediaPackageElementFlavor targetPresenterFlavor = parseTargetFlavor(
221             getConfig(operation, TARGET_PRESENTER_FLAVOR), "presenter");
222     final MediaPackageElementFlavor targetPresentationFlavor = parseTargetFlavor(
223             getConfig(operation, TARGET_PRESENTATION_FLAVOR), "presentation");
224     final boolean forceEncoding = BooleanUtils.toBoolean(getOptConfig(operation, FORCE_ENCODING).orElse("false"));
225     final Optional<EncodingProfile> forceProfile = getForceEncodingProfile(operation, forceEncoding);
226     final boolean forceDivisible = BooleanUtils.toBoolean(getOptConfig(operation, ENFORCE_DIVISIBLE_BY_TWO).orElse("false"));
227     final List<String> requiredExtensions = getRequiredExtensions(operation);
228     final String preencodeEncodingProfile = getConfig(operation, PREENCODE_ENCODING_PROFILE);
229 
230     //
231     // further checks on config options
232     // Skip the worklow if no presenter and presentation flavor has been configured
233     if (presenterFlavor.isEmpty() && presentationFlavor.isEmpty()) {
234       logger.warn("No presenter and presentation flavor has been set.");
235       return createResult(mediaPackage, Action.SKIP);
236     }
237 
238     final EncodingProfile preencodeProfile = composerService.getProfile(preencodeEncodingProfile);
239     if (preencodeProfile == null) {
240       throw new WorkflowOperationException("Preencode encoding profile '" + preencodeEncodingProfile + "' was not found");
241     }
242 
243     final EncodingProfile concatProfile = composerService.getProfile(concatEncodingProfile);
244     if (concatProfile == null) {
245       throw new WorkflowOperationException("Concat encoding profile '" + concatEncodingProfile + "' was not found");
246     }
247 
248     float outputFramerate = -1.0f;
249     if (concatOutputFramerate.isPresent()) {
250       if (NumberUtils.isNumber(concatOutputFramerate.get())) {
251         logger.info("Using concat output framerate");
252         outputFramerate = NumberUtils.toFloat(concatOutputFramerate.get());
253       } else {
254         throw new WorkflowOperationException("Unable to parse concat output frame rate!");
255       }
256     }
257 
258     final EncodingProfile trimProfile = composerService.getProfile(trimEncodingProfile);
259     if (trimProfile == null) {
260       throw new WorkflowOperationException("Trim encoding profile '" + trimEncodingProfile + "' was not found");
261     }
262 
263     //
264     // get tracks
265     final TrackSelector presenterTrackSelector = mkTrackSelector(presenterFlavor);
266     final TrackSelector presentationTrackSelector = mkTrackSelector(presentationFlavor);
267     List<Track> originalTracks = new ArrayList<Track>();
268     // Collecting presenter tracks
269     for (Track t : presenterTrackSelector.select(mediaPackage, false)) {
270       logger.info("Found partial presenter track {}", t);
271       originalTracks.add(t);
272     }
273     // Collecting presentation tracks
274     for (Track t : presentationTrackSelector.select(mediaPackage, false)) {
275       logger.info("Found partial presentation track {}", t);
276       originalTracks.add(t);
277     }
278 
279     // Encode all tracks to same format to enable use of ffmpeg concat-demuxer
280     logger.info("Starting preencoding");
281     originalTracks = preencode(preencodeProfile, originalTracks);
282 
283 
284     // flavor_type -> job
285     final Map<String, Job> jobs = new HashMap<String, Job>();
286     // get SMIL catalog
287     final SMILDocument smilDocument;
288     try {
289       smilDocument = SmilUtil.getSmilDocumentFromMediaPackage(mediaPackage, smilFlavor, workspace);
290     } catch (SAXException e) {
291       throw new WorkflowOperationException(e);
292     }
293     final SMILParElement parallel = (SMILParElement) smilDocument.getBody().getChildNodes().item(0);
294     final NodeList sequences = parallel.getTimeChildren();
295     final float trackDurationInSeconds = parallel.getDur();
296     final long trackDurationInMs = Math.round(trackDurationInSeconds * 1000f);
297     for (int i = 0; i < sequences.getLength(); i++) {
298       final SMILElement item = (SMILElement) sequences.item(i);
299 
300       for (final String mediaType : new String[] { NODE_TYPE_AUDIO, NODE_TYPE_VIDEO }) {
301         final List<Track> tracks = new ArrayList<Track>();
302         final VCell<String> sourceType = VCell.cell(EMPTY_VALUE);
303 
304         final long position = processChildren(0, tracks, item.getChildNodes(), originalTracks, sourceType, mediaType,
305                 elementsToClean, operationId);
306 
307         if (tracks.isEmpty()) {
308           logger.debug("The tracks list was empty.");
309           continue;
310         }
311         final Track lastTrack = tracks.get(tracks.size() - 1);
312 
313         if (position < trackDurationInMs) {
314           final double extendingTime = (trackDurationInMs - position) / 1000d;
315           if (extendingTime > 0) {
316             if (!lastTrack.hasVideo()) {
317               logger.info("Extending {} audio track end by {} seconds with silent audio", sourceType.get(),
318                       extendingTime);
319               tracks.add(getSilentAudio(extendingTime, elementsToClean, operationId));
320             } else {
321               logger.info("Extending {} track end with last image frame by {} seconds", sourceType.get(), extendingTime);
322               Attachment tempLastImageFrame = extractLastImageFrame(lastTrack, elementsToClean);
323               tracks.add(createVideoFromImage(tempLastImageFrame, extendingTime, elementsToClean));
324             }
325           }
326         }
327 
328         if (tracks.size() < 2) {
329           logger.debug("There were less than 2 tracks, copying track...");
330           if (sourceType.get().startsWith(PRESENTER_KEY)) {
331             createCopyOfTrack(mediaPackage, tracks.get(0), targetPresenterFlavor);
332           } else if (sourceType.get().startsWith(PRESENTATION_KEY)) {
333             createCopyOfTrack(mediaPackage, tracks.get(0), targetPresentationFlavor);
334           } else {
335             logger.warn("Can't handle unkown source type '{}' for unprocessed track", sourceType.get());
336           }
337           continue;
338         }
339 
340         for (final Track t : tracks) {
341           if (!t.hasVideo() && !t.hasAudio()) {
342             logger.error("No audio or video stream available in the track with flavor {}! {}", t.getFlavor(), t);
343             throw new WorkflowOperationException("No audio or video stream available in the track " + t.toString());
344           }
345         }
346 
347         if (sourceType.get().startsWith(PRESENTER_KEY)) {
348           logger.info("Concatenating {} track", PRESENTER_KEY);
349           jobs.put(sourceType.get(), startConcatJob(concatProfile, tracks, outputFramerate, forceDivisible));
350         } else if (sourceType.get().startsWith(PRESENTATION_KEY)) {
351           logger.info("Concatenating {} track", PRESENTATION_KEY);
352           jobs.put(sourceType.get(), startConcatJob(concatProfile, tracks, outputFramerate, forceDivisible));
353         } else {
354           logger.warn("Can't handle unknown source type '{}'!", sourceType.get());
355         }
356       }
357     }
358 
359     // Wait for the jobs to return
360     if (jobs.size() > 0) {
361       if (!JobUtil.waitForJobs(serviceRegistry, jobs.values()).isSuccess()) {
362         throw new WorkflowOperationException("One of the concat jobs did not complete successfully");
363       }
364     } else {
365       logger.info("No concatenating needed for presenter and presentation tracks, took partial source elements");
366     }
367 
368     // All the jobs have passed, let's update the media package
369     long queueTime = 0L;
370     MediaPackageElementFlavor adjustedTargetPresenterFlavor = targetPresenterFlavor;
371     MediaPackageElementFlavor adjustedTargetPresentationFlavor = targetPresentationFlavor;
372     for (final Entry<String, Job> job : jobs.entrySet()) {
373       final Optional<Job> concatJob = JobUtil.update(serviceRegistry, job.getValue());
374       if (concatJob.isPresent()) {
375         final String concatPayload = concatJob.get().getPayload();
376         if (concatPayload != null) {
377           final Track concatTrack;
378           try {
379             concatTrack = (Track) MediaPackageElementParser.getFromXml(concatPayload);
380           } catch (MediaPackageException e) {
381             throw new WorkflowOperationException(e);
382           }
383 
384           final String fileName;
385 
386           // Adjust the target flavor.
387           if (job.getKey().startsWith(PRESENTER_KEY)) {
388             if (!concatTrack.hasVideo()) {
389               fileName = PRESENTER_KEY.concat(FLAVOR_AUDIO_SUFFIX);
390               adjustedTargetPresenterFlavor = deriveAudioFlavor(targetPresenterFlavor);
391             } else {
392               fileName = PRESENTER_KEY;
393               adjustedTargetPresenterFlavor = targetPresenterFlavor;
394             }
395             concatTrack.setFlavor(adjustedTargetPresenterFlavor);
396           } else if (job.getKey().startsWith(PRESENTATION_KEY)) {
397             if (!concatTrack.hasVideo()) {
398               fileName = PRESENTATION_KEY.concat(FLAVOR_AUDIO_SUFFIX);
399               adjustedTargetPresentationFlavor = deriveAudioFlavor(targetPresentationFlavor);
400             } else {
401               fileName = PRESENTATION_KEY;
402               adjustedTargetPresentationFlavor = targetPresentationFlavor;
403             }
404             concatTrack.setFlavor(adjustedTargetPresentationFlavor);
405           } else {
406             fileName = UNKNOWN_KEY;
407           }
408 
409           concatTrack.setURI(workspace.moveTo(concatTrack.getURI(), mediaPackage.getIdentifier().toString(),
410                   concatTrack.getIdentifier(),
411                   fileName + "." + FilenameUtils.getExtension(concatTrack.getURI().toString())));
412 
413           logger.info("Concatenated track {} got flavor '{}'", concatTrack, concatTrack.getFlavor());
414 
415           mediaPackage.add(concatTrack);
416           queueTime += concatJob.get().getQueueTime();
417         } else {
418           // If there is no payload, then the item has not been distributed.
419           logger.warn("Concat job {} does not contain a payload", concatJob);
420         }
421       } else {
422         logger.warn("Concat job {} could not be updated since it cannot be found", job.getValue());
423       }
424     }
425 
426     // Trim presenter and presentation source track if longer than the duration from the SMIL catalog
427     queueTime += checkForTrimming(mediaPackage, trimProfile, targetPresentationFlavor, trackDurationInSeconds,
428             elementsToClean);
429     queueTime += checkForTrimming(mediaPackage, trimProfile, deriveAudioFlavor(targetPresentationFlavor),
430             trackDurationInSeconds, elementsToClean);
431     queueTime += checkForTrimming(mediaPackage, trimProfile, targetPresenterFlavor, trackDurationInSeconds,
432             elementsToClean);
433     queueTime += checkForTrimming(mediaPackage, trimProfile, deriveAudioFlavor(targetPresenterFlavor),
434             trackDurationInSeconds, elementsToClean);
435 
436     // New: Mux within presentation and presenter
437     queueTime += checkForMuxing(mediaPackage, targetPresenterFlavor, deriveAudioFlavor(targetPresenterFlavor), false, elementsToClean);
438     queueTime += checkForMuxing(mediaPackage, targetPresentationFlavor, deriveAudioFlavor(targetPresentationFlavor), false, elementsToClean);
439 
440     adjustAudioTrackTargetFlavor(mediaPackage, targetPresenterFlavor);
441     adjustAudioTrackTargetFlavor(mediaPackage, targetPresentationFlavor);
442 
443     // Mux between presentation and presenter? Why?
444     queueTime += checkForMuxing(mediaPackage, targetPresenterFlavor, targetPresentationFlavor, false, elementsToClean);
445 
446     queueTime += checkForEncodeToStandard(mediaPackage, forceEncoding, forceProfile, requiredExtensions,
447             targetPresenterFlavor, targetPresentationFlavor, elementsToClean);
448 
449     final WorkflowOperationResult result = createResult(mediaPackage, Action.CONTINUE, queueTime);
450     logger.debug("Partial import operation completed");
451     return result;
452   }

454   protected long checkForEncodeToStandard(MediaPackage mediaPackage, boolean forceEncoding,
455           Optional<EncodingProfile> forceProfile, List<String> requiredExtensions,
456           MediaPackageElementFlavor targetPresenterFlavor, MediaPackageElementFlavor targetPresentationFlavor,
457           List<MediaPackageElement> elementsToClean) throws EncoderException, IOException, MediaPackageException,
458           NotFoundException, ServiceRegistryException, WorkflowOperationException {
459     long queueTime = 0;
460     if (forceProfile.isPresent()) {
461       Track[] targetPresenterTracks = mediaPackage.getTracks(targetPresenterFlavor);
462       for (Track track : targetPresenterTracks) {
463         if (forceEncoding || trackNeedsTobeEncodedToStandard(track, requiredExtensions)) {
464           logger.debug("Encoding '{}' flavored track '{}' with standard encoding profile {}",
465                   targetPresenterFlavor, track.getURI(), forceProfile.get());
466           queueTime += encodeToStandard(mediaPackage, forceProfile.get(), targetPresenterFlavor, track);
467           elementsToClean.add(track);
468           mediaPackage.remove(track);
469         }
470       }
471       // Skip presentation target if it is the same as the presenter one.
472       if (!targetPresenterFlavor.toString().equalsIgnoreCase(targetPresentationFlavor.toString())) {
473         Track[] targetPresentationTracks = mediaPackage.getTracks(targetPresentationFlavor);
474         for (Track track : targetPresentationTracks) {
475           if (forceEncoding || trackNeedsTobeEncodedToStandard(track, requiredExtensions)) {
476             logger.debug("Encoding '{}' flavored track '{}' with standard encoding profile {}",
477                     targetPresentationFlavor, track.getURI(), forceProfile.get());
478             queueTime += encodeToStandard(mediaPackage, forceProfile.get(), targetPresentationFlavor, track);
479             elementsToClean.add(track);
480             mediaPackage.remove(track);
481           }
482         }
483       }
484     }
485     return queueTime;
486   }

489    * This function creates a copy of a given track in the media package

491    * @param mediaPackage
492    *          The media package being processed.
493    * @param track
494    *          The track we want to create a copy from.
495    * @param targetFlavor

498   private void createCopyOfTrack(MediaPackage mediaPackage, Track track, MediaPackageElementFlavor targetFlavor)
499              throws IllegalArgumentException, NotFoundException,IOException {
500 
501     MediaPackageElementFlavor targetCopyFlavor = null;
502     if (track.hasVideo()) {
503       targetCopyFlavor = targetFlavor;
504     } else {
505       targetCopyFlavor = deriveAudioFlavor(targetFlavor);
506     }
507     logger.debug("Copying track {} with flavor {} using target flavor {}", track.getURI(), track.getFlavor(), targetCopyFlavor);
508     copyPartialToSource(mediaPackage, targetCopyFlavor, track);
509   }

511   /**
512    * This functions adjusts the target flavor for audio tracks.
513    * While processing audio tracks, an audio suffix is appended to the type of the audio tracks target flavor.
514    * This functions essentially removes that suffix again and therefore ensures that the target flavor of
515    * audio tracks is set correctly.
516    *
517    * @param mediaPackage
518    *          The media package to look for audio tracks.
519    * @param targetFlavor
520    *          The target flavor for the audio tracks.
521    */
522   private void adjustAudioTrackTargetFlavor(MediaPackage mediaPackage, MediaPackageElementFlavor targetFlavor)
523              throws IllegalArgumentException, NotFoundException,IOException {
524 
525     Track[] targetAudioTracks = mediaPackage.getTracks(deriveAudioFlavor(targetFlavor));
526     for (Track track : targetAudioTracks) {
527       logger.debug("Adding {} to finished audio tracks.", track.getURI());
528       mediaPackage.remove(track);
529       track.setFlavor(targetFlavor);
530       mediaPackage.add(track);
531     }
532   }

534   private TrackSelector mkTrackSelector(Optional<String> flavor) throws WorkflowOperationException {
535     final TrackSelector s = new TrackSelector();
536     if (flavor.isPresent()) {
537       try {
538         final MediaPackageElementFlavor f = MediaPackageElementFlavor.parseFlavor(flavor.get());
539         s.addFlavor(f);
540         s.addFlavor(deriveAudioFlavor(f));
541       } catch (IllegalArgumentException e) {
542         throw new WorkflowOperationException("Flavor '" + flavor.get() + "' is malformed");
543       }
544     }
545     return s;
546   }

548   /**
549    * Start job to concatenate a list of tracks.
550    *
551    * @param profile
552    *          the encoding profile to use
553    * @param tracks
554    *          non empty track list
555    * @param forceDivisible
556    *          Whether to enforce the track's dimension to be divisible by two
557    */
558   protected Job startConcatJob(EncodingProfile profile, List<Track> tracks, float outputFramerate, boolean forceDivisible)
559           throws MediaPackageException, EncoderException {
560     final Dimension dim = determineDimension(tracks, forceDivisible);
561     if (outputFramerate > 0.0) {
562       return composerService.concat(profile.getIdentifier(), dim, outputFramerate, true, Collections.toArray(Track.class, tracks));
563     } else {
564       return composerService.concat(profile.getIdentifier(), dim, true, Collections.toArray(Track.class, tracks));    }
565   }

663   /**
664    * Determine the largest dimension of the given list of tracks
665    *
666    * @param tracks
667    *          the list of tracks
668    * @param forceDivisible
669    *          Whether to enforce the track's dimension to be divisible by two
670    * @return the largest dimension from the list of track
671    */
672   private Dimension determineDimension(List<Track> tracks, boolean forceDivisible) {
673     Tuple<Track, Dimension> trackDimension = getLargestTrack(tracks);
674     if (trackDimension == null)
675       return null;
676 
677     if (forceDivisible && (trackDimension.getB().getHeight() % 2 != 0 || trackDimension.getB().getWidth() % 2 != 0)) {
678       Dimension scaledDimension = Dimension.dimension((trackDimension.getB().getWidth() / 2) * 2, (trackDimension
679               .getB().getHeight() / 2) * 2);
680       logger.info("Determined output dimension {} scaled down from {} for track {}", scaledDimension,
681               trackDimension.getB(), trackDimension.getA());
682       return scaledDimension;
683     } else {
684       logger.info("Determined output dimension {} for track {}", trackDimension.getB(), trackDimension.getA());
685       return trackDimension.getB();
686     }
687   }

689   /**
690    * Returns the track with the largest resolution from the list of tracks
691    *
692    * @param tracks
693    *          the list of tracks
694    * @return a {@link Tuple} with the largest track and it's dimension
695    */
696   private Tuple<Track, Dimension> getLargestTrack(List<Track> tracks) {
697     Track track = null;
698     Dimension dimension = null;
699     for (Track t : tracks) {
700       if (!t.hasVideo())
701         continue;
702 
703       VideoStream[] videoStreams = TrackSupport.byType(t.getStreams(), VideoStream.class);
704       int frameWidth = videoStreams[0].getFrameWidth();
705       int frameHeight = videoStreams[0].getFrameHeight();
706       if (dimension == null || (frameWidth * frameHeight) > (dimension.getWidth() * dimension.getHeight())) {
707         dimension = Dimension.dimension(frameWidth, frameHeight);
708         track = t;
709       }
710     }
711     if (track == null || dimension == null)
712       return null;
713 
714     return Tuple.tuple(track, dimension);
715   }

717   private long checkForTrimming(MediaPackage mediaPackage, EncodingProfile trimProfile,
718           MediaPackageElementFlavor targetFlavor, Float videoDuration, List<MediaPackageElement> elementsToClean)
719           throws EncoderException, MediaPackageException, WorkflowOperationException, NotFoundException,
720           ServiceRegistryException, IOException {
721     MediaPackageElement[] elements = mediaPackage.getElementsByFlavor(targetFlavor);
722     if (elements.length == 0)
723       return 0;
724 
725     Track trackToTrim = (Track) elements[0];
726     if (elements.length == 1 && trackToTrim.getDuration() / 1000 > videoDuration) {
727       Long trimSeconds = (long) (trackToTrim.getDuration() / 1000 - videoDuration);
728       logger.info("Shorten track {} to target duration {} by {} seconds",
729               trackToTrim.toString(), videoDuration.toString(), trimSeconds.toString());
730       return trimEnd(mediaPackage, trimProfile, trackToTrim, videoDuration, elementsToClean);
731     } else if (elements.length > 1) {
732       logger.warn("Multiple tracks with flavor {} found! Trimming not possible!", targetFlavor);
733     }
734     return 0;
735   }

753   protected long checkForMuxing(MediaPackage mediaPackage, MediaPackageElementFlavor targetPresentationFlavor,
754           MediaPackageElementFlavor targetPresenterFlavor, boolean useSuffix, List<MediaPackageElement> elementsToClean)
755           throws EncoderException, MediaPackageException, WorkflowOperationException, NotFoundException,
756           ServiceRegistryException, IOException {
757 
758     long queueTime = 0L;
759 
760     List<Track> videoElements = getPureVideoTracks(mediaPackage, targetPresentationFlavor);
761     List<Track> audioElements;
762     if (useSuffix) {
763       audioElements = getPureAudioTracks(mediaPackage, deriveAudioFlavor(targetPresentationFlavor));
764     } else {
765       audioElements = getPureAudioTracks(mediaPackage, targetPresentationFlavor);
766     }
767 
768     Track videoTrack = null;
769     Track audioTrack = null;
770 
771     if (videoElements.size() == 1 && audioElements.size() == 0) {
772       videoTrack = videoElements.get(0);
773     } else if (videoElements.size() == 0 && audioElements.size() == 1) {
774       audioTrack = audioElements.get(0);
775     }
776 
777     videoElements = getPureVideoTracks(mediaPackage, targetPresenterFlavor);
778     if (useSuffix) {
779       audioElements = getPureAudioTracks(mediaPackage, deriveAudioFlavor(targetPresenterFlavor));
780     } else {
781       audioElements = getPureAudioTracks(mediaPackage, targetPresenterFlavor);
782     }
783 
784     if (videoElements.size() == 1 && audioElements.size() == 0) {
785       videoTrack = videoElements.get(0);
786     } else if (videoElements.size() == 0 && audioElements.size() == 1) {
787       audioTrack = audioElements.get(0);
788     }
789 
790     logger.debug("Check for mux between '{}' and '{}' flavors and found video track '{}' and audio track '{}'",
791             targetPresentationFlavor, targetPresenterFlavor, videoTrack, audioTrack);
792     if (videoTrack != null && audioTrack != null) {
793       queueTime += mux(mediaPackage, videoTrack, audioTrack, elementsToClean);
794       return queueTime;
795     } else {
796       return queueTime;
797     }
798   }

801    * Mux a video and an audio track. Add the result to media package <code>mediaPackage</code> with the same flavor as
802    * the <code>video</code>.
803    *
804    * @return the mux job's queue time
805    */
806   protected long mux(MediaPackage mediaPackage, Track video, Track audio, List<MediaPackageElement> elementsToClean)
807           throws EncoderException, MediaPackageException, WorkflowOperationException, NotFoundException,
808           ServiceRegistryException, IOException {
809     logger.debug("Muxing video {} and audio {}", video.getURI(), audio.getURI());
810     Job muxJob = composerService.mux(video, audio, PrepareAVWorkflowOperationHandler.MUX_AV_PROFILE);
811     if (!waitForStatus(muxJob).isSuccess()) {
812       throw new WorkflowOperationException("Muxing of audio " + audio + " and video " + video + " failed");
813     }
814     muxJob = serviceRegistry.getJob(muxJob.getId());
815 
816     final Track muxed = (Track) MediaPackageElementParser.getFromXml(muxJob.getPayload());
817     if (muxed == null) {
818       throw new WorkflowOperationException("Muxed job " + muxJob + " returned no payload!");
819     }
820     muxed.setFlavor(video.getFlavor());
821     muxed.setURI(workspace.moveTo(muxed.getURI(), mediaPackage.getIdentifier().toString(), muxed.getIdentifier(),
822             FilenameUtils.getName(video.getURI().toString())));
823     elementsToClean.add(audio);
824     mediaPackage.remove(audio);
825     elementsToClean.add(video);
826     mediaPackage.remove(video);
827     mediaPackage.add(muxed);
828     return muxJob.getQueueTime();
829   }

884    * Encode <code>track</code> using encoding profile <code>profile</code> and add the result to media package
885    * <code>mp</code> under the given <code>targetFlavor</code>.
886    *
887    * @return the encoder job's queue time
888    */
889   private long encodeToStandard(MediaPackage mp, EncodingProfile profile, MediaPackageElementFlavor targetFlavor,
890           Track track) throws EncoderException, MediaPackageException, WorkflowOperationException, NotFoundException,
891           ServiceRegistryException, IOException {
892     Job encodeJob = composerService.encode(track, profile.getIdentifier());
893     if (!waitForStatus(encodeJob).isSuccess()) {
894       throw new WorkflowOperationException("Encoding of track " + track + " failed");
895     }
896     encodeJob = serviceRegistry.getJob(encodeJob.getId());
897     Track encodedTrack = (Track) MediaPackageElementParser.getFromXml(encodeJob.getPayload());
898     if (encodedTrack == null) {
899       throw new WorkflowOperationException("Encoded track " + track + " failed to produce a track");
900     }
901     URI uri;
902     if (FilenameUtils.getExtension(encodedTrack.getURI().toString()).equalsIgnoreCase(
903             FilenameUtils.getExtension(track.getURI().toString()))) {
904       uri = workspace.moveTo(encodedTrack.getURI(), mp.getIdentifier().toString(), encodedTrack.getIdentifier(),
905               FilenameUtils.getName(track.getURI().toString()));
906     } else {
907       // The new encoded file has a different extension.
908       uri = workspace.moveTo(
909               encodedTrack.getURI(),
910               mp.getIdentifier().toString(),
911               encodedTrack.getIdentifier(),
912               FilenameUtils.getBaseName(track.getURI().toString()) + "."
913                       + FilenameUtils.getExtension(encodedTrack.getURI().toString()));
914     }
915     encodedTrack.setURI(uri);
916     encodedTrack.setFlavor(targetFlavor);
917     mp.add(encodedTrack);
918     return encodeJob.getQueueTime();
919   }

946   private long processChildren(long position, List<Track> tracks, NodeList children, List<Track> originalTracks,
947           VCell<String> type, String mediaType, List<MediaPackageElement> elementsToClean, Long operationId)
948           throws EncoderException, MediaPackageException, WorkflowOperationException, NotFoundException, IOException {
949     for (int j = 0; j < children.getLength(); j++) {
950       Node item = children.item(j);
951       if (item.hasChildNodes()) {
952         position = processChildren(position, tracks, item.getChildNodes(), originalTracks, type, mediaType,
953                 elementsToClean, operationId);
954       } else {
955         SMILMediaElement e = (SMILMediaElement) item;
956         if (mediaType.equals(e.getNodeName())) {
957           Track track;
958           try {
959             track = getFromOriginal(e.getId(), originalTracks, type);
960           } catch (IllegalStateException exception) {
961             logger.debug("Skipping smil entry, reason: " + exception.getMessage());
962             continue;
963           }
964           double beginInSeconds = e.getBegin().item(0).getResolvedOffset();
965           long beginInMs = Math.round(beginInSeconds * 1000d);
966           // Fill out gaps with first or last frame from video
967           if (beginInMs > position) {
968             double positionInSeconds = position / 1000d;
969             if (position == 0) {
970               if (NODE_TYPE_AUDIO.equals(e.getNodeName())) {
971                 logger.info("Extending {} audio track start by {} seconds silent audio", type.get(), beginInSeconds);
972                 tracks.add(getSilentAudio(beginInSeconds, elementsToClean, operationId));
973               } else {
974                 logger.info("Extending {} track start image frame by {} seconds", type.get(), beginInSeconds);
975                 Attachment tempFirstImageFrame = extractImage(track, 0, elementsToClean);
976                 tracks.add(createVideoFromImage(tempFirstImageFrame, beginInSeconds, elementsToClean));
977               }
978               position += beginInMs;
979             } else {
980               double fillTime = (beginInMs - position) / 1000d;
981               if (NODE_TYPE_AUDIO.equals(e.getNodeName())) {
982                 logger.info("Fill {} audio track gap from {} to {} with silent audio", type.get(),
983                         Double.toString(positionInSeconds), Double.toString(beginInSeconds));
984                 tracks.add(getSilentAudio(fillTime, elementsToClean, operationId));
985               } else {
986                 logger.info("Fill {} track gap from {} to {} with image frame",
987                         type.get(), Double.toString(positionInSeconds), Double.toString(beginInSeconds));
988                 Track previousTrack = tracks.get(tracks.size() - 1);
989                 Attachment tempLastImageFrame = extractLastImageFrame(previousTrack, elementsToClean);
990                 tracks.add(createVideoFromImage(tempLastImageFrame, fillTime, elementsToClean));
991               }
992               position = beginInMs;
993             }
994           }
995           tracks.add(track);
996           position += Math.round(e.getDur() * 1000f);
997         }
998       }
999     }
1000     return position;
1001   }

1003   private Track getFromOriginal(String trackId, List<Track> originalTracks, VCell<String> type) {
1004     for (Track t : originalTracks) {
1005       if (t.getIdentifier().contains(trackId)) {
1006         logger.debug("Track-Id from smil found in Mediapackage ID: " + t.getIdentifier());
1007         if (EMPTY_VALUE.equals(type.get())) {
1008           String suffix = (t.hasAudio() && !t.hasVideo()) ? FLAVOR_AUDIO_SUFFIX : "";
1009           type.set(t.getFlavor().getType() + suffix);
1010         }
1011         originalTracks.remove(t);
1012         return t;
1013       }
1014     }
1015     throw new IllegalStateException("No track matching smil Track-id: " + trackId);
1016   }
