129   @Override
130   public WorkflowOperationResult start(final WorkflowInstance wi, JobContext ctx)
131           throws WorkflowOperationException {
132     logger.debug("Running image workflow operation on {}", wi);
133     try {
134       MediaPackage mp = wi.getMediaPackage();
135       final Extractor e = new Extractor(this, configure(mp, wi));
136       return e.main(MediaPackageSupport.copy(mp));
137     } catch (Exception e) {
138       throw new WorkflowOperationException(e);
139     }
140   }

142   /**
143    * Computation within the context of a {@link Cfg}.
144    */
145   static final class Extractor {
146     private final ImageWorkflowOperationHandler handler;
147     private final Cfg cfg;
148 
149     Extractor(ImageWorkflowOperationHandler handler, Cfg cfg) {
150       this.handler = handler;
151       this.cfg = cfg;
152     }
153 
154     /** Run the extraction. */
155     WorkflowOperationResult main(final MediaPackage mp) throws WorkflowOperationException {
156       if (cfg.sourceTracks.size() == 0) {
157         logger.info("No source tracks found in media package {}, skipping operation", mp.getIdentifier());
158         return handler.createResult(mp, Action.SKIP);
159       }
160       // start image extraction jobs
161       final List<Extraction> extractions = cfg.sourceTracks.stream().flatMap(track -> {
162           final List<MediaPosition> positions = limit(track, cfg.positions);
163           if (positions.size() != cfg.positions.size()) {
164             logger.warn("Could not apply all configured positions to track {}", track);
165           }
166           logger.info("Extracting images from {} at position {}", track, positions);
167           // create one extraction per encoding profile
168           return cfg.profiles.stream()
169                   .map(profile -> new Extraction(extractImages(track, profile, positions), track, profile, positions));
170         }).collect(Collectors.toList());
171       final List<Job> extractionJobs = concatJobs(extractions);
172       final JobBarrier.Result extractionResult = JobUtil.waitForJobs(handler.serviceRegistry, extractionJobs);
173       if (extractionResult.isSuccess()) {
174         // all extractions were successful; iterate them
175         for (final Extraction extraction : extractions) {
176           final List<Attachment> images = getImages(extraction.job);
177           final int expectedNrOfImages = extraction.positions.size();
178           if (images.size() == expectedNrOfImages) {
179             // post process images
180             for (int i = 0; i < images.size(); i++) {
181               Attachment image = images.get(i);
182               MediaPosition position = extraction.positions.get(i);
183 
184               adjustMetadata(extraction, image);
185               if (image.getIdentifier() == null) {
186                 image.setIdentifier(UUID.randomUUID().toString());
187               }
188               mp.addDerived(image, extraction.track);
189               String fileName = createFileName(
190                   extraction.profile.getSuffix(),
191                   extraction.track.getURI(),
192                   position
193               );
194               moveToWorkspace(mp, image, fileName);
195             }
196           } else {
197             // less images than expected have been extracted
198             throw new WorkflowOperationException(
199                     format("Only %s of %s images have been extracted from track %s",
200                            images.size(), expectedNrOfImages, extraction.track));
201           }
202         }
203         return handler.createResult(mp, Action.CONTINUE, JobUtil.sumQueueTime(extractionJobs));
204       } else {
205         throw new WorkflowOperationException("Image extraction failed");
206       }
207     }

209     /**
210      * Adjust flavor, tags, mime type of <code>image</code> according to the
211      * configuration and the extraction.
212      */
213     void adjustMetadata(Extraction extraction, Attachment image) {
214       // Adjust the target flavor. Make sure to account for partial updates
215       for (final MediaPackageElementFlavor flavor : cfg.targetImageFlavor) {
216         final String flavorType = eq("*", flavor.getType())
217                 ? extraction.track.getFlavor().getType()
218                 : flavor.getType();
219         final String flavorSubtype = eq("*", flavor.getSubtype())
220                 ? extraction.track.getFlavor().getSubtype()
221                 : flavor.getSubtype();
222         image.setFlavor(new MediaPackageElementFlavor(flavorType, flavorSubtype));
223         logger.debug("Resulting image has flavor '{}'", image.getFlavor());
224       }
225       // Set the mime type
226       try {
227         image.setMimeType(MimeTypes.fromURI(image.getURI()));
228       } catch (UnknownFileTypeException e) {
229         logger.warn("Mime type unknown for file {}. Setting none.", image.getURI(), e);
230       }
231       // Add tags
232       for (final String tag : cfg.targetImageTags) {
233         logger.trace("Tagging image with '{}'", tag);
234         image.addTag(tag);
235       }
236     }

238     /** Create a file name for the extracted image. */
239     String createFileName(final String suffix, final URI trackUri, final MediaPosition pos) {
240       final String trackBaseName = FilenameUtils.getBaseName(trackUri.getPath());
241       final String format;
242       switch (pos.type) {
243         case Seconds:
244           format = cfg.targetBaseNameFormatSecond.orElse(trackBaseName + "_%.3fs%s");
245           break;
246         case Percentage:
247           format = cfg.targetBaseNameFormatPercent.orElse(trackBaseName + "_%.1fp%s");
248           break;
249         default:
250           throw new IllegalArgumentException("Unhandled MediaPosition type: " + pos.type);
251       }
252       return formatFileName(format, pos.position, suffix);
253     }
254 
255     /** Move the extracted <code>image</code> to its final location in the workspace and rename it to <code>fileName</code>. */
256     private void moveToWorkspace(final MediaPackage mp, final Attachment image, final String fileName) {
257       try {
258         image.setURI(handler.workspace.moveTo(
259                 image.getURI(),
260                 mp.getIdentifier().toString(),
261                 image.getIdentifier(),
262                 fileName));
263       } catch (Exception e) {
264         throw new RuntimeException(new WorkflowOperationException(e));
265       }
266     }

268     /** Start a composer job to extract images from a track at the given positions. */
269     private Job extractImages(final Track track, final EncodingProfile profile, final List<MediaPosition> positions) {
270       List<Double> seconds = positions.stream()
271           .map(position -> toSeconds(track, position, cfg.endMargin))
272           .collect(Collectors.toList());
273 
274       try {
275         return handler.composerService.image(track, profile.getIdentifier(), Collections.toDoubleArray(seconds));
276       } catch (Exception e) {
277         throw new RuntimeException(new WorkflowOperationException("Error starting image extraction job", e));
278       }
279     }
280   }

292   /** Concat the jobs of a list of extraction objects. */
293   private static List<Job> concatJobs(List<Extraction> extractions) {
294     return extractions.stream()

299   /** Get the images (payload) from a job. */
300   @SuppressWarnings("unchecked")
301   private static List<Attachment> getImages(Job job) {
302     final List<Attachment> images;
303     try {
304       images = (List<Attachment>) MediaPackageElementParser.getArrayFromXml(job.getPayload());
305     } catch (MediaPackageException e) {
306       throw new RuntimeException(e);
307     }
308     if (!images.isEmpty()) {
309       return images;
310     } else {
311       throw new RuntimeException(new WorkflowOperationException("Job did not extract any images"));
312     }
313   }

315   /** Limit the list of media positions to those that fit into the length of the track. */
316   static List<MediaPosition> limit(Track track, List<MediaPosition> positions) {
317     final Long duration = track.getDuration();
318     // if the video has just one frame (e.g.: MP3-Podcasts) it makes no sense to go to a certain position
319     // as the video has only one image at position 0
320     if (duration == null || (track.getStreams() != null && Arrays.stream(track.getStreams())
321             .filter(stream -> stream instanceof VideoStream)
322             .map(org.opencastproject.mediapackage.Stream::getFrameCount)
323             .allMatch(frameCount -> frameCount == null || frameCount == 1))) {
324       return java.util.Collections.singletonList(new MediaPosition(PositionType.Seconds, 0));
325     }
326 
327     return positions.stream()
328         .filter(p -> (PositionType.Seconds.equals(p.type) && p.position >= 0 && p.position < duration)
329                 || (PositionType.Percentage.equals(p.type) && p.position >= 0 && p.position <= 100))
330         .collect(Collectors.toList());
331   }

333   /**
334    * Convert a <code>position</code> into seconds in relation to the given track.
335    * <em>Attention:</em> The function does not check if the calculated absolute position is within
336    * the bounds of the tracks length.
337    */
338   static double toSeconds(Track track, MediaPosition position, double endMarginMs) {
339     final long durationMs = track.getDuration() == null ? 0 : track.getDuration();
340     final double posMs;
341     switch (position.type) {
342       case Percentage:
343         posMs = durationMs * position.position / 100.0;
344         break;
345       case Seconds:
346         posMs = position.position * 1000.0;
347         break;
348       default:
349         throw new IllegalArgumentException("Unhandled MediaPosition type: " + position.type);
350     }
351     // limit maximum position to Xms before the end of the video
352     return Math.abs(durationMs - posMs) >= endMarginMs
353             ? posMs / 1000.0
354             : Math.max(0, ((double) durationMs - endMarginMs)) / 1000.0;
355   }

358   /**
359    * Fetch a profile from the composer service. Throw a WorkflowOperationException in case the profile
360    * does not exist.
361    */
362   public static Function<String, EncodingProfile> fetchProfile(final ComposerService composerService) {
363     return new Function<String, EncodingProfile>() {
364       @Override
365       public EncodingProfile apply(String profileName) {
366         final EncodingProfile profile = composerService.getProfile(profileName);
367         if (profile != null) {
368           return profile;
369         } else {
370           throw new RuntimeException(new WorkflowOperationException("Encoding profile '" + profileName + "' was not found"));
371         }
372       }
373     };

433   /** Get and parse the configuration options. */
434   private Cfg configure(MediaPackage mp, WorkflowInstance wi) throws WorkflowOperationException {
435     WorkflowOperationInstance woi = wi.getCurrentOperation();
436     ConfiguredTagsAndFlavors tagsAndFlavors = getTagsAndFlavors(wi,
437         Configuration.many, Configuration.many, Configuration.many, Configuration.one);
438     final List<EncodingProfile> profiles = getOptConfig(woi, OPT_PROFILES)
439         .map(configString -> asList(configString))
440         .orElseGet(java.util.Collections::emptyList)
441         .stream()
442         .map(fetchProfile(composerService))
443         .collect(Collectors.toList());
444     final List<String> targetImageTags = tagsAndFlavors.getTargetTags();
445     final List<MediaPackageElementFlavor> targetImageFlavor = tagsAndFlavors.getTargetFlavors();
446     final List<Track> sourceTracks;
447     {
448       // get the source tags
449       final List<String> sourceTags = tagsAndFlavors.getSrcTags();
450       final List<MediaPackageElementFlavor> sourceFlavors = tagsAndFlavors.getSrcFlavors();
451       TrackSelector trackSelector = new TrackSelector();
452 
453       //add tags and flavors to TrackSelector
454       for (String tag : sourceTags) {
455         trackSelector.addTag(tag);
456       }
457       for (MediaPackageElementFlavor flavor : sourceFlavors) {
458         trackSelector.addFlavor(flavor);
459       }
460 
461       // select the tracks based on source flavors and tags and skip those that don't have video
462       sourceTracks = trackSelector.select(mp, true).stream()
463           .filter(Track::hasVideo)
464           .collect(Collectors.toList());
465     }
466     final List<MediaPosition> positions = MediaPositionParser.parsePositions(getConfig(woi, OPT_POSITIONS));
467     final long endMargin = getOptConfig(woi, OPT_END_MARGIN)
468         .flatMap(s -> {
469           try {
470             return Optional.of(Long.parseLong(s));
471           } catch (NumberFormatException e) {
472             return Optional.empty();
473           }
474         })
475         .orElse(END_MARGIN_DEFAULT);
476     //
477     return new Cfg(sourceTracks,
478                    positions,
479                    profiles,
480                    targetImageFlavor,
481                    targetImageTags,
482                    getTargetBaseNameFormat(woi, OPT_TARGET_BASE_NAME_FORMAT_SECOND),
483                    getTargetBaseNameFormat(woi, OPT_TARGET_BASE_NAME_FORMAT_PERCENT),
484                    endMargin);
485   }
486 
487   /** Validate a target base name format. */
488   private Optional<String> getTargetBaseNameFormat(WorkflowOperationInstance woi, final String formatName) {
489     Optional<String> opt = getOptConfig(woi, formatName);
490     opt.ifPresent(validateTargetBaseNameFormat(formatName));
491     return opt;
492   }
493 
494   static Consumer<String> validateTargetBaseNameFormat(final String formatName) {
495     return format -> {
496       boolean valid;
497       try {
498         final String name = formatFileName(format, 15.11, ".png");
499         valid = name.contains(".") && name.contains(".png");
500       } catch (IllegalFormatException e) {
501         valid = false;
502       }
503       if (!valid) {
504         throw new RuntimeException(new WorkflowOperationException(String.format(
505             "%s is not a valid format string for config option %s",
506             format, formatName)));
507       }
508     };
509   }

513   /**
514    * Parse media position parameter strings.
515    */
516   public final class MediaPositionParser {
517     private MediaPositionParser() { }
518 
519     public static List<MediaPosition> parsePositions(String input) throws WorkflowOperationException {
520       if (input == null || input.trim().isEmpty()) {
521         throw new WorkflowOperationException("Cannot parse empty or blank time string");
522       }
523 
524       List<MediaPosition> positions = new ArrayList<>();
525       String[] tokens = input.trim().split("[,\\s]+");
526 
527       for (String token : tokens) {
528         MediaPosition position = parseToken(token);
529         if (position == null) {
530           throw new WorkflowOperationException("Cannot parse time string: " + input);
531         }
532         positions.add(position);
533       }
534       return positions;
535     }
536 
537     private static MediaPosition parseToken(String token) {
538       token = token.trim();
539       if (token.endsWith("%")) {
540         try {
541           double value = Double.parseDouble(token.substring(0, token.length() - 1));
542           return new MediaPosition(PositionType.Percentage, value);
543         } catch (NumberFormatException e) {
544           return null;
545         }
546       } else {
547         try {
548           double value = Double.parseDouble(token);
549           return new MediaPosition(PositionType.Seconds, value);
550         } catch (NumberFormatException e) {
551           return null;
552         }
553       }
554     }
555   }
556 
557   enum PositionType {
558     Percentage, Seconds
559   }

561   /**
562    * A position in time in a media file.
563    */
564   static final class MediaPosition {
565     private double position;
566     private final PositionType type;
567 
568     MediaPosition(PositionType type, double position) {
569       this.position = position;
570       this.type = type;
571     }
572 
573     public void setPosition(double position) {
574       this.position = position;
575     }
576 
577     @Override public int hashCode() {
578       return Objects.hash(position, type);
579     }
580 
581     @Override public boolean equals(Object that) {
582       return (this == that) || (that instanceof MediaPosition && eqFields((MediaPosition) that));
583     }
584 
585     private boolean eqFields(MediaPosition that) {
586       return position == that.position && eq(type, that.type);
587     }
588 
589     @Override public String toString() {
590       return format("MediaPosition(%s, %s)", type, position);
591     }
592   }
