21  package org.opencastproject.metadata.dublincore;
22  
23  import java.util.Date;
24  import java.util.function.Function;
25  
26  import javax.annotation.Nonnull;
27  import javax.annotation.ParametersAreNonnullByDefault;
28  
29  /**
30   * Collection of metadata encoding functions following Opencast rules.
31   */
32  @ParametersAreNonnullByDefault
33  public final class OpencastMetadataCodec {
34    private OpencastMetadataCodec() {
35    }
36  
37    /** Encode a date with a given precision. */
38    @Nonnull public static DublinCoreValue encodeDate(Date date, Precision p) {
39      return EncodingSchemeUtils.encodeDate(date, p);
40    }
41  
42    /** Decode a W3C-DTF encoded date. */
43    @Nonnull public static Date decodeDate(String date) {
44      return EncodingSchemeUtils.decodeMandatoryDate(date);
45    }
46  
47    /** {@link OpencastMetadataCodec#decodeDate(java.lang.String)} as a function. */
48    public static final Function<String, Date> decodeDate = new Function<String, Date>() {
49      @Override public Date apply(String a) {
50        return decodeDate(a);
51      }
52    };
53  
54    /** Encode a duration. */
55    @Nonnull public static DublinCoreValue encodeDuration(long ms) {
56      return EncodingSchemeUtils.encodeDuration(ms);
57    }
58  
59    /** Decode a duration. */
60    public static long decodeDuration(String ms) {
61      return EncodingSchemeUtils.decodeMandatoryDuration(ms);
62    }
63  
64    /** {@link OpencastMetadataCodec#decodeDuration(String)} as a function. */
65    public static final Function<String, Long> decodeDuration = new Function<String, Long>() {
66      @Override public Long apply(String a) {
67        return decodeDuration(a);
68      }
69    };
70  
71    /** Encode a period with a given precision. */
72    @Nonnull public static DublinCoreValue encodePeriod(Date from, Date to, Precision precision) {
73      return EncodingSchemeUtils.encodePeriod(new DCMIPeriod(from, to), precision);
74    }
75  
76    /** Decode a temporal value. */
77    @Nonnull public static Temporal decodeTemporal(DublinCoreValue temporal) {
78      return EncodingSchemeUtils.decodeMandatoryTemporal(temporal);
79    }
80  
81    /** {@link OpencastMetadataCodec#decodeTemporal(DublinCoreValue)} as a function. */
82    @Nonnull public static final Function<DublinCoreValue, Temporal> decodeTemporal = new Function<DublinCoreValue, Temporal>() {
83      @Override public Temporal apply(DublinCoreValue a) {
84        return decodeTemporal(a);
85      }
86    };
87  }
