21  package org.opencastproject.assetmanager.impl;
22  
23  import static java.lang.String.format;
24  import static org.junit.Assert.assertEquals;
25  import static org.opencastproject.db.Queries.nativeQuery;
26  
27  import org.opencastproject.assetmanager.impl.persistence.Database;
28  import org.opencastproject.db.DBSession;
29  import org.opencastproject.db.DBTestEnv;
30  
31  public class AssetManagerDeleteTestBase extends AssetManagerTestBase {
32  
33    protected DBSession db;
34  
35    @Override
36    public AssetManagerImpl makeAssetManager() throws Exception {
37      db = DBTestEnv.newDBSession(PERSISTENCE_UNIT);
38      // empty database
39      db.execTx(em -> {
40        nativeQuery.delete("delete from oc_assets_asset").apply(em);
41        nativeQuery.delete("delete from oc_assets_properties").apply(em);
42        nativeQuery.delete("delete from oc_assets_snapshot").apply(em);
43        nativeQuery.delete("delete from oc_assets_version_claim").apply(em);
44      });
45  
46      final Database database = new Database(db);
47      am = super.makeAssetManager();
48      am.setDatabase(database);
49      return am;
50    }
51  
52    void assertPropertiesTotal(long count) {
53      assertEquals(format("There should be %d properties total", count),
54                   count,
55                   am.countProperties());
56    }
57  
58    void assertAssetsTotal(long count) {
59      assertEquals(format("There should be %d assets total", count),
60                   count,
61                   am.countAssets());
62    }
63  
64    void assertSnapshotsTotal(long count) {
65      assertEquals(format("There should be %d snapshots total", count),
66                   count,
67                   am.countSnapshots(null));
68    }
69  
70    void assertTotals(long snapshots, long assets, long properties) {
71      assertSnapshotsTotal(snapshots);
72      assertAssetsTotal(assets);
73      assertPropertiesTotal(properties);
74    }
75  }
