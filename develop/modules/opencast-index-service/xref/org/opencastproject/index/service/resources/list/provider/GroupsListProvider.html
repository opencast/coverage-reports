22  package org.opencastproject.index.service.resources.list.provider;
23  
24  import org.opencastproject.list.api.ResourceListProvider;
25  import org.opencastproject.list.api.ResourceListQuery;
26  import org.opencastproject.security.api.JaxbGroup;
27  import org.opencastproject.security.api.Role;
28  import org.opencastproject.userdirectory.JpaGroupRoleProvider;
29  
30  import org.osgi.framework.BundleContext;
31  import org.osgi.service.component.annotations.Activate;
32  import org.osgi.service.component.annotations.Component;
33  import org.osgi.service.component.annotations.Reference;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  import java.io.IOException;
38  import java.util.HashMap;
39  import java.util.List;
40  import java.util.Map;
41  
42  @Component(
43      service = ResourceListProvider.class,
44      property = {
45          "service.description=Groups list provider",
46          "opencast.service.type=org.opencastproject.index.service.resources.list.provider.GroupsListProvider"
47      }
48  )
49  public class GroupsListProvider implements ResourceListProvider {
50  
51    private static final String PROVIDER_PREFIX = "GROUPS";
52  
53    public static final String DESCRIPTION = PROVIDER_PREFIX + ".DESCRIPTION";
54    public static final String NAME = PROVIDER_PREFIX + ".NAME";
55    public static final String ROLE_ONLY = PROVIDER_PREFIX + ".ROLE.ONLY"; // Role: Role
56  
57    protected static final String[] NAMES = { PROVIDER_PREFIX, NAME, DESCRIPTION, ROLE_ONLY };
58    private static final Logger logger = LoggerFactory.getLogger(GroupsListProvider.class);
59  
60    private JpaGroupRoleProvider groupRoleProvider;
61  
62    @Activate
63    protected void activate(BundleContext bundleContext) {
64      logger.info("Groups list provider activated!");
65    }
66  
67    /** OSGi callback for groups services. */
68    @Reference
69    public void setGroupProvider(JpaGroupRoleProvider groupRoleProvider) {
70      this.groupRoleProvider = groupRoleProvider;
71    }
72  
73    @Override
74    public String[] getListNames() {
75      return NAMES;
76    }
77  
78    @Override
79    public Map<String, String> getList(String listName, ResourceListQuery query) {
80      Map<String, String> groupsList = new HashMap<String, String>();
81      List<JaxbGroup> groups = null;
82      int limit = 0;
83      int offset = 0;
84  
85      if (query != null) {
86        if (query.getLimit().isSome())
87          limit = query.getLimit().get();
88  
89        if (query.getOffset().isSome())
90          offset = query.getOffset().get();
91      }
92  
93      try {
94        groups = groupRoleProvider.getGroups(limit, offset).getGroups();
95      } catch (IOException e) {
96        logger.error("Not able to get the group list: " + e);
97        return groupsList;
98      }
99  
100     for (JaxbGroup g : groups) {
101       if (NAME.equals(listName)) {
102         groupsList.put(g.getName(), g.getName());
103       } else if (DESCRIPTION.equals(listName)) {
104         groupsList.put(g.getDescription(), g.getDescription());
105       } else if (ROLE_ONLY.equals(listName) && g.getRoles().size() > 0) {
106         for (Role role : g.getRoles()) {
107           groupsList.put(role.getName(), role.getName());
108         }
109       } else {
110         groupsList.put(g.getGroupId(), g.getName());
111       }
112     }
113 
114     return groupsList;
115   }
116 
117   @Override
118   public boolean isTranslatable(String listName) {
119     return false;
120   }
121 
122   @Override
123   public String getDefault() {
124     return null;
125   }
126 }
