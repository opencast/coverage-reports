88    @Test
89    public void testInsertRepoSecond() throws Exception {
90      final Date now = new Date();
91      final MediaPackage mp1 = MediaPackageSupport.loadFromClassPath("/mp1.xml");
92      final MediaPackage mp2 = MediaPackageSupport.loadFromClassPath("/mp2.xml");
93      final MediaPackage mp3 = MediaPackageSupport.loadFromClassPath("/mp3.xml");
94      final OaiPmhRepository repo = repo(oaiPmhDatabase(mp1, mp2, mp3), Granularity.SECOND);
95      assertThat(
96              "List all records yields 3 records",
97              s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, null, null, null))),
98              allOf(hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(),
99                      equalTo(3.0)),
100                     hasXPath("//oai20:ListRecords/oai20:record/oai20:header[oai20:identifier='10.0000/11']", NS_CTX),
101                     hasXPath("//oai20:ListRecords/oai20:record/oai20:header[oai20:identifier='10.0000/12']", NS_CTX),
102                     hasXPath("//oai20:ListRecords/oai20:record/oai20:header[oai20:identifier='10.0000/13']", NS_CTX)));
103     assertThat("List records from time in the past yields 3 records",
104             s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, enc(now), null, null))),
105             hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(), equalTo(3.0)));
106     final Date future = new Date(System.currentTimeMillis() + 1000);
107     assertThat("List records from time in the future yields no records",
108             s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, enc(future), null, null))),
109             hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(), equalTo(0.0)));
110   }

112   @Test
113   public void testInsertRepoDay() throws Exception {
114     final Date ref = new Date();
115     final MediaPackage mp1 = MediaPackageSupport.loadFromClassPath("/mp1.xml");
116     final MediaPackage mp2 = MediaPackageSupport.loadFromClassPath("/mp2.xml");
117     final MediaPackage mp3 = MediaPackageSupport.loadFromClassPath("/mp3.xml");
118     final OaiPmhRepository repo = repo(oaiPmhDatabase(mp1, mp2, mp3), Granularity.DAY);
119     final Date ref2 = new Date();
120     final long diff = (ref2.getTime() - ref.getTime()) / 1000;
121     assertThat("List records from time in the past",
122             s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, enc(ref), null, null))),
123             hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(), equalTo(3.0)));
124     assertThat("List records from time in the future. Attention: Test will fail during the last " + diff
125             + " seconds of a day!",
126             s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, enc(ref2), null, null))),
127             hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(), equalTo(3.0)));
128     assertThat(
129             "List records from time in the future (5 days ahead)",
130             s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, enc(new Date(ref.getTime() + 5 * 60 * 60 * 24
131                     * 1000)), null, null))),
132             hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(), equalTo(0.0)));
133   }

145   @Test
146   public void testSelectDateRange() throws Exception {
147     final Date ref1 = new Date();
148     final MediaPackage mp1 = MediaPackageSupport.loadFromClassPath("/mp1.xml");
149     final MediaPackage mp2 = MediaPackageSupport.loadFromClassPath("/mp2.xml");
150     final MediaPackage mp3 = MediaPackageSupport.loadFromClassPath("/mp3.xml");
151     final OaiPmhRepository repo = repo(oaiPmhDatabase(mp1), Granularity.SECOND);
152     // wait 1 second since the repo has a time granularity of seconds
153     Thread.sleep(1000);
154     final Date ref2 = new Date();
155     repo.getPersistence().store(mp2, REPOSITORY_ID);
156     // wait 1 second since the repo has a time granularity of seconds
157     Thread.sleep(1000);
158     final Date ref3 = new Date();
159     repo.getPersistence().store(mp3, REPOSITORY_ID);
160     assertThat("List records from time 1 yields all 3 records",
161             s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, enc(ref1), null, null))),
162             hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(), equalTo(3.0)));
163     assertThat("List records from time 3 yields only the record record inserted last",
164             s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, enc(ref3), null, null))),
165             hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(), equalTo(1.0)));
166     assertThat("List records until time 3 yields the first two records",
167             s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, null, enc(ref3), null))),
168             hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(), equalTo(2.0)));
169     assertThat("List records from time 2 until time 3 yields the record inserted second",
170             s(repo.selectVerb(params("ListRecords", null, FORMAT_PREFIX, enc(ref2), enc(ref3), null))),
171             hasXPath("count(//oai20:ListRecords/oai20:record/oai20:header)", NS_CTX, returningANumber(), equalTo(1.0)));
172   }

211   /** Each param may be null. */
212   private static Params params(final String verb, final String identifier, final String metadataPrefix,
213           final String from, final String until, final String resumptionToken) {
214     return new Params() {
215       @Override
216       String getParameter(String key) {
217         if ("verb".equals(key))
218           return verb;
219         if ("identifier".equals(key))
220           return identifier;
221         if ("metadataPrefix".equals(key))
222           return metadataPrefix;
223         if ("from".equals(key))
224           return from;
225         if ("until".equals(key))
226           return until;
227         if ("resumptionToken".equals(key))
228           return resumptionToken;
229         return null;
230       }
231 
232       @Override
233       String getRepositoryUrl() {
234         return "http://localhost:8080/oaipmh";
235       }
236     };
237   }

239   private static AbstractOaiPmhDatabase oaiPmhDatabase(MediaPackage... mps) {
240     try {
241       final Organization org = new DefaultOrganization();
242       final SecurityService secSvc = EasyMock.createNiceMock(SecurityService.class);
243       // security service
244       final User user = createSystemUser("admin", org);
245       expect(secSvc.getOrganization()).andReturn(org).anyTimes();
246       expect(secSvc.getUser()).andReturn(user).anyTimes();
247       EasyMock.replay(secSvc);
248       // series service
249       final SeriesService seriesService = EasyMock.createNiceMock(SeriesService.class);
250       final String xacml = IOUtils.toString(OaiPmhRepositoryPersistenceTest.class.getResource("/xacml.xml").toURI());
251       final AccessControlList securityACL = AccessControlParser.parseAcl(xacml);
252       EasyMock.expect(seriesService.getSeriesAccessControl("10.0000/1")).andReturn(securityACL).anyTimes();
253       EasyMock.replay(seriesService);
254       // workspace
255       final Workspace workspace = EasyMock.createNiceMock(Workspace.class);
256       final File episodeDublinCore = new File(OaiPmhRepositoryPersistenceTest.class.getResource(
257               "/episode-dublincore.xml").toURI());
258       final File seriesDublinCore = new File(OaiPmhRepositoryPersistenceTest.class
259               .getResource("/series-dublincore.xml").toURI());
260       expect(workspace.read(EasyMock.anyObject())).andAnswer(() -> {
261         final String uri = getCurrentArguments()[0].toString();
262         if ("dublincore.xml".equals(uri))
263           return new FileInputStream(episodeDublinCore);
264         if ("series-dublincore.xml".equals(uri))
265           return new FileInputStream(seriesDublinCore);
266         throw new Error("Workspace mock does not know about file " + uri);
267       }).anyTimes();
268       EasyMock.replay(workspace);
269       // oai-pmh database
270       final DBSession dbSession = newDBSession(OaiPmhDatabaseImpl.PERSISTENCE_UNIT_NAME);
271       final AbstractOaiPmhDatabase db = new AbstractOaiPmhDatabase() {
272         @Override
273         public DBSession getDBSession() {
274           return dbSession;
275         }
276 
277         @Override
278         public SecurityService getSecurityService() {
279           return secSvc;
280         }
281 
282         @Override
283         public Workspace getWorkspace() {
284           return workspace;
285         }
286       };
287       for (MediaPackage mp : mps)
288         db.store(mp, REPOSITORY_ID);
289       return db;
290     } catch (Exception e) {
291       return chuck(e);
292     }
293   }

295   private static OaiPmhRepository repo(final AbstractOaiPmhDatabase persistence, final Granularity granularity) {
296     return new OaiPmhRepository() {
297       @Override
298       public Granularity getRepositoryTimeGranularity() {
299         return granularity;
300       }
301 
302       @Override
303       public String getRepositoryName() {
304         return "Test OAI Repository";
305       }
306 
307       @Override
308       public String getRepositoryId() {
309         return REPOSITORY_ID;
310       }
311 
312       @Override
313       public OaiPmhDatabase getPersistence() {
314         return persistence;
315       }
316 
317       @Override
318       public String getAdminEmail() {
319         return "admin@localhost.org";
320       }
321 
322       @Override
323       public String saveQuery(ResumableQuery query) {
324         return "r-token";
325       }
326 
327       @Override
328       public Option<ResumableQuery> getSavedQuery(String resumptionToken) {
329         return Option.some(new ResumableQuery(FORMAT_PREFIX, new Date(), new Date(), Option.<String> none()));
330       }
331 
332       @Override
333       public int getResultLimit() {
334         return 3;
335       }
336 
337       @Override
338       public List<MetadataProvider> getRepositoryMetadataProviders() {
339         return nil();
340       }
341 
342       @Override
343       public Date currentDate() {
344         return new Date();
345       }
346     };
