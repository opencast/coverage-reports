22  package org.opencastproject.capture.admin.impl;
23  
24  import static org.opencastproject.capture.admin.api.AgentState.CAPTURING;
25  import static org.opencastproject.capture.admin.api.AgentState.IDLE;
26  
27  import org.opencastproject.capture.admin.api.Agent;
28  import org.opencastproject.security.api.DefaultOrganization;
29  
30  import org.junit.After;
31  import org.junit.Assert;
32  import org.junit.Before;
33  import org.junit.Test;
34  
35  public class AgentTest {
36    private Agent agent = null;
37    private Long time = 0L;
38  
39    @Before
40    public void setUp() {
41      agent = new AgentImpl("test", DefaultOrganization.DEFAULT_ORGANIZATION_ID, IDLE, "http://localhost/", null);
42      Assert.assertNotNull(agent);
43      time = agent.getLastHeardFrom();
44    }
45  
46    @After
47    public void tearDown() {
48      agent = null;
49      time = 0L;
50    }
51  
52    @Test
53    public void correctInformation() {
54      Assert.assertEquals("test", agent.getName());
55      Assert.assertEquals(IDLE, agent.getState());
56      Assert.assertEquals("http://localhost/", agent.getUrl());
57    }
58  
59    @Test
60    public void changedInformation() throws InterruptedException {
61      Assert.assertEquals("test", agent.getName());
62      Assert.assertEquals(IDLE, agent.getState());
63      Assert.assertEquals(time, agent.getLastHeardFrom());
64  
65      Thread.sleep(1);
66      agent.setState(CAPTURING);
67  
68      Assert.assertEquals("test", agent.getName());
69      Assert.assertEquals(CAPTURING, agent.getState());
70      Thread.sleep(1);
71      if (agent.getLastHeardFrom() <= time || agent.getLastHeardFrom() > System.currentTimeMillis()) {
72        Assert.fail("Invalid checkin time");
73      }
74    }
75  }
