22  package org.opencastproject.authorization.xacml.manager.impl;
23  
24  import org.opencastproject.authorization.xacml.manager.api.AclService;
25  import org.opencastproject.authorization.xacml.manager.api.AclServiceFactory;
26  import org.opencastproject.elasticsearch.index.ElasticsearchIndex;
27  import org.opencastproject.security.api.Organization;
28  import org.opencastproject.security.api.SecurityService;
29  
30  import org.osgi.service.component.annotations.Component;
31  import org.osgi.service.component.annotations.Reference;
32  
33  /** OSGi implementation of {@link org.opencastproject.authorization.xacml.manager.api.AclServiceFactory}. */
34  @Component(
35          property = {
36                  "service.description=Factory to create ACL services"
37          },
38          immediate = true,
39          service = { AclServiceFactory.class }
40  )
41  public class OsgiAclServiceFactory implements AclServiceFactory {
42    private AclDb aclDb;
43    private SecurityService securityService;
44    protected ElasticsearchIndex index;
45  
46    @Override
47    public AclService serviceFor(Organization org) {
48      return new AclServiceImpl(org, aclDb, index, securityService);
49    }
50  
51    @Reference
52    public void setAclDb(AclDb aclDb) {
53      this.aclDb = aclDb;
54    }
55  
56    @Reference
57    public void setSecurityService(SecurityService securityService) {
58      this.securityService = securityService;
59    }
60  
61    @Reference
62    public void setIndex(ElasticsearchIndex index) {
63      this.index = index;
64    }
65  }
