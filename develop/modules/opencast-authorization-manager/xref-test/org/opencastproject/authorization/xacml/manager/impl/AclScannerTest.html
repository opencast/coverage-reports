64    @Before
65    public void setUp() throws Exception {
66      Organization org1 = new JpaOrganization("org1", "org1", new HashMap<String, Integer>(), "ADMIN", "ANONYMOUS",
67          new HashMap<>());
68      Organization org2 = new JpaOrganization("org2", "org2", new HashMap<String, Integer>(), "ADMIN", "ANONYMOUS",
69          new HashMap<>());
70      Organization org3 = new JpaOrganization("org3", "org3", new HashMap<String, Integer>(), "ADMIN", "ANONYMOUS",
71          new HashMap<>());
72  
73      List<Organization> orgs = new ArrayList<>();
74      orgs.add(org1);
75      orgs.add(org2);
76      orgs.add(org3);
77  
78      aclDb = EasyMock.createNiceMock(AclDb.class);
79  
80      orgService = EasyMock.createNiceMock(OrganizationDirectoryService.class);
81      EasyMock.expect(orgService.getOrganizations()).andReturn(orgs).anyTimes();
82  
83      User user = EasyMock.createNiceMock(User.class);
84      EasyMock.expect(user.getOrganization()).andReturn(new DefaultOrganization()).anyTimes();
85  
86      final SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
87      EasyMock.expect(securityService.getUser()).andReturn(user).anyTimes();
88  
89      SearchResultImpl<Event> eventSearchResult = EasyMock.createNiceMock(SearchResultImpl.class);
90      EasyMock.expect(eventSearchResult.getItems()).andReturn(new SearchResultItem[] {}).anyTimes();
91  
92      SearchResultImpl<Series> seriesSearchResult = EasyMock.createNiceMock(SearchResultImpl.class);
93      EasyMock.expect(seriesSearchResult.getItems()).andReturn(new SearchResultItem[] {}).anyTimes();
94  
95      final ElasticsearchIndex index = EasyMock.createNiceMock(ElasticsearchIndex.class);
96      EasyMock.expect(index.getByQuery(EasyMock.anyObject(EventSearchQuery.class))).andReturn(eventSearchResult)
97              .anyTimes();
98      EasyMock.expect(index.getByQuery(EasyMock.anyObject(SeriesSearchQuery.class))).andReturn(seriesSearchResult)
99              .anyTimes();
100 
101     EasyMock.replay(orgService, securityService, index, user, eventSearchResult, seriesSearchResult);
102 
103     AclServiceFactory aclServiceFactory = new AclServiceFactory() {
104       @Override
105       public AclService serviceFor(Organization org) {
106         return new AclServiceImpl(new DefaultOrganization(), aclDb, index, securityService);
107       }
108     };
109 
110     aclScanner = new AclScanner();
111     aclScanner.setAclServiceFactory(aclServiceFactory);
112     aclScanner.setOrganizationDirectoryService(orgService);
113     aclScanner.setSecurityService(securityService);
114   }

116   @Test
117   public void testCorrectFileInstall() throws Exception {
118     File file = new File(AclScannerTest.class.getResource("/correct.json").toURI());
119 
120     ManagedAcl acl = new ManagedAclImpl(1L, "TestAcl", "org", new AccessControlList());
121     Optional<ManagedAcl> managedAcl = Optional.of(acl);
122     EasyMock.expect(aclDb.createAcl(anyObject(Organization.class), anyObject(AccessControlList.class), anyString()))
123             .andReturn(managedAcl).times(3);
124     EasyMock.expect(aclDb.getAcls(anyObject(Organization.class))).andReturn(new ArrayList<>()).times(3);
125     EasyMock.replay(aclDb);
126 
127     aclScanner.install(file);
128 
129     EasyMock.verify(aclDb);
130   }
