405     Date start = new Date();
406     Date end = new Date(System.currentTimeMillis() + 60000);
407     String captureDeviceID = "demo";
408     String seriesId = "series1";
409     Set<String> userIds = new HashSet<>();
410     userIds.add("user1");
411     userIds.add("user2");
412     MediaPackage mp = generateEvent(Optional.empty());
413     mp.setSeries(seriesId);
414     DublinCoreCatalog event = generateEvent(captureDeviceID, start, end);
415     String catalogId = addDublinCore(Optional.empty(), mp, event);
416     Map<String, String> caProperties = generateCaptureAgentMetadata("demo");
417     EasyMock.reset(seriesService);
418     EasyMock.expect(seriesService.getSeries(seriesId)).andThrow(new NotFoundException()).once();
419     EasyMock.replay(seriesService);
420 
421     assertEquals("mod0", schedSvc.getScheduleLastModified(captureDeviceID));
422 
423     // Store event
424     schedSvc.addEvent(start, end, captureDeviceID, userIds, mp, wfProperties, caProperties, Optional.empty());
425     try {
426       MediaPackage mp2 = (MediaPackage) mp.clone();
427       mp2.setIdentifier(IdImpl.fromUUID());
428       schedSvc.addEvent(start, end, captureDeviceID, userIds, mp2, wfProperties, caProperties, Optional.empty());
429       Assert.fail();
430     } catch (SchedulerConflictException e) {
431       Assert.assertNotNull(e);
432     }
433     MediaPackage mediaPackage = schedSvc.getMediaPackage(mp.getIdentifier().toString());
434     assertEquals(seriesId, mediaPackage.getSeries());
435     DublinCoreCatalog eventLoaded = schedSvc.getDublinCore(mp.getIdentifier().toString());

437     TechnicalMetadata technicalMetadata = schedSvc.getTechnicalMetadata(mp.getIdentifier().toString());

442     assertEquals(userIds, technicalMetadata.getPresenters());
443     assertTrue(technicalMetadata.getRecording().isEmpty());
444     assertTrue(technicalMetadata.getCaptureAgentConfiguration().size() >= caProperties.size());
445 
446     assertEquals(wfProperties, new HashMap<>(schedSvc.getWorkflowConfig(mp.getIdentifier().toString())));
447     String lastModified = schedSvc.getScheduleLastModified(captureDeviceID);
448     assertNotEquals("mod0", lastModified);
449 
450     eventLoaded.set(PROPERTY_TITLE, "Something more");
451     addDublinCore(Optional.of(catalogId), mp, eventLoaded);
452 
453     userIds.add("user3");
454     userIds.remove("user1");
455     mp.setSeries("series2");
456 
457     // Update event
458     schedSvc.updateEvent(mp.getIdentifier().toString(), Optional.empty(), Optional.empty(), Optional.empty(),
459             Optional.of(userIds), Optional.of(mp), Optional.of(wfProperties), Optional.of(caProperties));
460 
461     mediaPackage = schedSvc.getMediaPackage(mp.getIdentifier().toString());
462     assertEquals("series2", mediaPackage.getSeries());
463     DublinCoreCatalog eventReloaded = schedSvc.getDublinCore(mp.getIdentifier().toString());
464     assertEquals("Something more", eventReloaded.getFirst(PROPERTY_TITLE));
465     technicalMetadata = schedSvc.getTechnicalMetadata(mp.getIdentifier().toString());
466     assertEquals(mp.getIdentifier().toString(), technicalMetadata.getEventId());
467     assertEquals(captureDeviceID, technicalMetadata.getAgentId());
468     assertEquals(start, technicalMetadata.getStartDate());
469     assertEquals(end, technicalMetadata.getEndDate());
470     assertEquals(userIds, technicalMetadata.getPresenters());
471     assertTrue(technicalMetadata.getRecording().isEmpty());
472     assertTrue(technicalMetadata.getCaptureAgentConfiguration().size() >= caProperties.size());
473     String updatedLastModified = schedSvc.getScheduleLastModified(captureDeviceID);
474     assertNotEquals("mod0", updatedLastModified);
475     assertNotEquals(lastModified, updatedLastModified);
476 
477     assertTrue(schedSvc.getCaptureAgentConfiguration(mp.getIdentifier().toString()).size() >= caProperties.size());
478   }
