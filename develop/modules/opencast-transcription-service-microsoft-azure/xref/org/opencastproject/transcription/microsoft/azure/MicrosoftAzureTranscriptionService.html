175   @Modified
176   public void modified(ComponentContext cc) {
177     logger.debug("Updating config...");
178     Option<Boolean> enabledOpt = OsgiUtil.getOptCfgAsBoolean(cc.getProperties(), KEY_ENABLED);
179     if (enabledOpt.isSome()) {
180       enabled = enabledOpt.get();
181     } else {
182       deactivate();
183     }
184 
185     if (!enabled) {
186       logger.info("Microsoft Azure transcription service disabled."
187           + " If you want to enable it, please update the service configuration.");
188       return;
189     }
190 
191     Option<String> azureStorageAccountNameKeyOpt = OsgiUtil.getOptCfg(cc.getProperties(),
192         KEY_AZURE_STORAGE_ACCOUNT_NAME);
193     if (azureStorageAccountNameKeyOpt.isSome()) {
194       azureStorageAccountName = azureStorageAccountNameKeyOpt.get();
195     } else {
196       logger.warn("Azure storage account name key was not set. Disabling Microsoft Azure transcription service.");
197       deactivate();
198       return;
199     }
200 
201     Option<String> azureAccountAccessKeyKeyOpt = OsgiUtil.getOptCfg(cc.getProperties(), KEY_AZURE_ACCOUNT_ACCESS_KEY);
202     if (azureAccountAccessKeyKeyOpt.isSome()) {
203       azureAccountAccessKey = azureAccountAccessKeyKeyOpt.get();
204     } else {
205       logger.warn("Azure storage account access key was not set. Disabling Microsoft Azure transcription service.");
206       deactivate();
207       return;
208     }
209 
210     Option<String> azureSpeechServicesKeyOpt = OsgiUtil.getOptCfg(cc.getProperties(),
211         KEY_AZURE_SPEECH_SERVICES_ENDPOINT);
212     if (azureSpeechServicesKeyOpt.isSome()) {
213       azureSpeechServicesEndpoint = azureSpeechServicesKeyOpt.get();
214     } else {
215       logger.warn("Azure speech services endpoint was not set. Disabling Microsoft Azure transcription service.");
216       deactivate();
217       return;
218     }
219 
220     Option<String> azureCognitiveServicesSubscriptionKeyKeyOpt = OsgiUtil.getOptCfg(cc.getProperties(),
221         KEY_COGNITIVE_SERVICES_SUBSCRIPTION_KEY);
222     if (azureCognitiveServicesSubscriptionKeyKeyOpt.isSome()) {
223       azureCognitiveServicesSubscriptionKey = azureCognitiveServicesSubscriptionKeyKeyOpt.get();
224     } else {
225       logger.warn("Azure cognitive services subscription key was not set. "
226           + "Disabling Microsoft Azure transcription service.");
227       deactivate();
228       return;
229     }
230 
231     // optional values
232     Option<String> workflowKeyOpt = OsgiUtil.getOptCfg(cc.getProperties(), KEY_WORKFLOW);
233     if (workflowKeyOpt.isSome()) {
234       workflowDefinitionId = workflowKeyOpt.get();
235       logger.info("Workflow is set to '{}'.", workflowDefinitionId);
236     } else {
237       workflowDefinitionId = DEFAULT_WORKFLOW_DEFINITION_ID;
238       logger.info("Default workflow '{}' will be used.", workflowDefinitionId);
239     }
240 
241     Option<String> azureBlobPathKeyOpt = OsgiUtil.getOptCfg(cc.getProperties(), KEY_AZURE_BOLB_PATH);
242     if (azureBlobPathKeyOpt.isSome()) {
243       azureBlobPath = azureBlobPathKeyOpt.get();
244     } else {
245       logger.debug("Azure blob path was not set, using default path.");
246       azureBlobPath = DEFAULT_AZURE_BLOB_PATH;
247     }
248 
249     Option<String> languageOpt = OsgiUtil.getOptCfg(cc.getProperties(), KEY_LANGUAGE);
250     if (languageOpt.isSome()) {
251       language = languageOpt.get();
252       logger.info("Default language is set to '{}'.", language);
253     } else {
254       language = DEFAULT_LANGUAGE;
255       logger.info("Default language '{}' will be used.", language);
256     }
257 
258     autodetectLanguages = new ArrayList<>();
259     Option<String> autoDetectLanguagesOpt = OsgiUtil.getOptCfg(cc.getProperties(), KEY_AUTO_DETECT_LANGUAGES);
260     if (languageOpt.isSome()) {
261       for (String lang : StringUtils.split(autoDetectLanguagesOpt.get(), ",")) {
262         if (StringUtils.isNotBlank(lang)) {
263           autodetectLanguages.add(StringUtils.trimToEmpty(lang));
264         }
265       }
266     }
267 
268     Option<String> azureContainerNameKeyOpt = OsgiUtil.getOptCfg(cc.getProperties(), KEY_AZURE_CONTAINER_NAME);
269     if (azureContainerNameKeyOpt.isSome()) {
270       azureContainerName = azureContainerNameKeyOpt.get();
271     } else {
272       logger.debug("Azure storage container name was not set, using default path.");
273       azureContainerName = DEFAULT_AZURE_CONTAINER_NAME;
274     }
275 
276     Option<String> azureSpeechRecognitionMinConfidenceKeyOpt = OsgiUtil.getOptCfg(cc.getProperties(),
277         KEY_AZURE_SPEECH_RECOGNITION_MIN_CONFIDENCE);
278     if (azureSpeechRecognitionMinConfidenceKeyOpt.isSome()) {
279       String azureSpeechRecognitionMinConfidenceStr = azureSpeechRecognitionMinConfidenceKeyOpt.get();
280       try {
281         azureSpeechRecognitionMinConfidence = Float.valueOf(azureSpeechRecognitionMinConfidenceStr);
282       } catch (NumberFormatException e) {
283         logger.error("Azure speech recognition min confidence value is not valid. "
284             + "Please set a value between 0.0 and 1.0. "
285             + "Setting to default value of {}.", DEFAULT_MIN_CONFIDENCE);
286         azureSpeechRecognitionMinConfidence = DEFAULT_MIN_CONFIDENCE;
287       }
288     } else {
289       logger.debug("Azure speech recognition min confidence value was not set. Setting to default value of {}.",
290           DEFAULT_MIN_CONFIDENCE);
291       azureSpeechRecognitionMinConfidence = DEFAULT_MIN_CONFIDENCE;
292     }
293 
294     Option<String> splitTextLineLengthOpt = OsgiUtil.getOptCfg(cc.getProperties(), KEY_SPLIT_TEXT_LINE_LENGTH);
295     if (splitTextLineLengthOpt.isSome()) {
296       try {
297         splitTextLineLength = Integer.parseInt(splitTextLineLengthOpt.get());
298       } catch (NumberFormatException e) {
299         splitTextLineLength = DEFAULT_SPLIT_TEXT_LINE_LENGTH;
300         logger.error("Invalid configuration value for '{}'. Set default value {}.", KEY_SPLIT_TEXT_LINE_LENGTH,
301             DEFAULT_SPLIT_TEXT_LINE_LENGTH);
302       }
303     } else {
304       logger.debug("Configuration value for '{}' was not set. Setting to default value of {}.",
305           KEY_SPLIT_TEXT_LINE_LENGTH, DEFAULT_MIN_CONFIDENCE);
306       splitTextLineLength = DEFAULT_SPLIT_TEXT_LINE_LENGTH;
307     }
308 
309     Option<String> outputFileFormatOpt = OsgiUtil.getOptCfg(cc.getProperties(), KEY_OUTPUT_FILE_FORMAT);
310     if (outputFileFormatOpt.isSome()) {
311       outputFileFormat = outputFileFormatOpt.get();
312       switch (outputFileFormat) {
313         case "srt":
314         case "vtt":
315           break;
316         default:
317           logger.debug("Azure output file format not valid, using default format {}.",
318               DEFAULT_OUTPUT_FILE_FORMAT);
319           outputFileFormat = DEFAULT_OUTPUT_FILE_FORMAT;
320           break;
321       }
322     } else {
323       logger.debug("Azure output file format not set, using default format {}.", DEFAULT_OUTPUT_FILE_FORMAT);
324       outputFileFormat = DEFAULT_OUTPUT_FILE_FORMAT;
325     }
326     logger.info("Transcription output format is set to '{}'.", outputFileFormat);
327 
328     //// create Azure storage client
329     try {
330       azureAuthorization = new MicrosoftAzureAuthorization(azureStorageAccountName, azureAccountAccessKey);
331       azureStorageClient = new MicrosoftAzureStorageClient(azureAuthorization);
332     } catch (MicrosoftAzureStorageClientException e) {
333       logger.error("Unable to create Microsoft Azure storage client. "
334           + "Deactivating Microsoft Azure Transcription service.", e);
335       deactivate();
336       return;
337     }
338 
339     // create Azure Speech Services client
340     azureSpeechServicesClient = new MicrosoftAzureSpeechServicesClient(
341         azureSpeechServicesEndpoint, azureCognitiveServicesSubscriptionKey);
342 
343     if (scheduledExecutorService != null) {
344       scheduledExecutorService.shutdown();
345       try {
346         scheduledExecutorService.awaitTermination(60, TimeUnit.SECONDS);
347       } catch (InterruptedException e) {
348         // pending task took to long
349         // pending task will be restarted on next run
350       }
351     }
352     scheduledExecutorService = Executors.newScheduledThreadPool(2);
353     scheduledExecutorService.scheduleWithFixedDelay(new WorkflowDispatcher(), 120, 120, TimeUnit.SECONDS);
354     logger.info("Activated.");
355   }

375   @Override
376   protected String process(Job job) throws Exception {
377     Operation op = null;
378     String operation = job.getOperation();
379     List<String> arguments = job.getArguments();
380     op = Operation.valueOf(operation);
381     switch (op) {
382       case StartTranscription:
383         long jobId = job.getId();
384         String mpId = arguments.get(0);
385         Track track = (Track) MediaPackageElementParser.getFromXml(arguments.get(1));
386         String languageCode = arguments.get(2);
387         if (StringUtils.isBlank(languageCode)) {
388           languageCode = getLanguage();
389         }
390         return createTranscriptionJob(jobId, mpId, track, languageCode);
391       default:
392         throw new IllegalStateException("Don't know how to handle operation '" + operation + "'");
393     }
394   }

401   @Override
402   public Job startTranscription(String mpId, Track track, String... args) throws TranscriptionServiceException {
403     try {
404       List<String> jobArgs = new ArrayList<>(2 + args.length);
405       jobArgs.add(mpId);
406       jobArgs.add(MediaPackageElementParser.getAsXml(track));
407       jobArgs.addAll(Arrays.asList(args));
408       return serviceRegistry.createJob(JOB_TYPE, Operation.StartTranscription.toString(), jobArgs);
409     } catch (ServiceRegistryException e) {
410       throw new TranscriptionServiceException(String.format(
411           "Unable to create transcription job for media package '%s'.", mpId), e);
412     } catch (MediaPackageException e) {
413       throw new TranscriptionServiceException(String.format(
414           "Unable to to parse track from media package '%s'.", mpId), e);
415     }
416   }

418   @Override
419   public MediaPackageElement getGeneratedTranscription(String mpId, String jobId, MediaPackageElement.Type type)
420           throws TranscriptionServiceException {
421     if (type != MediaPackageElement.Type.Track && type != MediaPackageElement.Type.Attachment) {
422       throw new IllegalArgumentException("Target type must be a Track or Attachment.");
423     }
424     MicrosoftAzureSpeechTranscription transcription;
425     try {
426       transcription = azureSpeechServicesClient.getTranscriptionById(jobId);
427     } catch (IOException | MicrosoftAzureNotAllowedException | MicrosoftAzureSpeechClientException
428              | MicrosoftAzureNotFoundException e) {
429       throw new TranscriptionServiceException(String.format(
430           "Unable to get transcription '%s' for media package '%s'.", jobId, mpId), e);
431     }
432     MicrosoftAzureSpeechTranscriptionJson transcriptionJson;
433     URI transcriptionFileUri;
434     try {
435       transcriptionJson = getTranscriptionJson(mpId, transcription);
436       transcriptionFileUri = MicrosoftAzureSpeechServicesClient.writeTranscriptionFile(transcriptionJson,
437           workspace, outputFileFormat, azureSpeechRecognitionMinConfidence, splitTextLineLength);
438     } catch (IOException | MicrosoftAzureNotFoundException e) {
439       throw new TranscriptionServiceException(String.format(
440           "Unable to download transcription file for media package '%s'.", mpId), e);
441     }
442     MediaPackageElement transcriptionElement =  MediaPackageElementBuilderFactory.newInstance().newElementBuilder()
443         .elementFromURI(transcriptionFileUri, type, new MediaPackageElementFlavor("captions", outputFileFormat));
444     if (type.equals(MediaPackageElement.Type.Track)) {
445       // apply predefined tags as documented in https://docs.opencast.org/develop/admin/#configuration/subtitles/
446       transcriptionElement.setTags(new String[] { "generator-type:auto", "generator:azure", "type:subtitle"});
447     }
448     return transcriptionElement;
449   }

451   @Override
452   public void transcriptionDone(String mpId, Object results) throws TranscriptionServiceException {
453     MicrosoftAzureSpeechTranscription transcription = (MicrosoftAzureSpeechTranscription) results;
454     logger.info("Transcription job {} for media package {} done.", transcription.getID(), mpId);
455     // delete audio source files in Azure storage
456     try {
457       deleteTranscriptionSourceFiles(mpId, transcription.getID());
458     } catch (TranscriptionServiceException e) {
459       logger.warn("Unable to delete transcription source files for media package {} after transcription job done.",
460           mpId, e);
461     }
462     try {
463       database.updateJobControl(transcription.getID(), TranscriptionJobControl.Status.TranscriptionComplete.name());
464     } catch (TranscriptionDatabaseException e) {
465       throw new TranscriptionServiceException(String.format(
466           "Transcription job for media package '%s' succeeded but storing job status in the database failed."
467           , mpId), e);
468     }
469   }

471   @Override
472   public void transcriptionError(String mpId, Object results) throws TranscriptionServiceException {
473     MicrosoftAzureSpeechTranscription transcription = (MicrosoftAzureSpeechTranscription) results;
474     String message = "";
475     if (transcription != null && transcription.properties != null && transcription.properties.containsKey("error")) {
476       Map<String, Object> errorInfo = (Map<String, Object>) transcription.properties.get("error");
477       message = String.format(" Microsoft error code %s: %s", errorInfo.getOrDefault("code", "UNKNOWN"),
478           errorInfo.getOrDefault("message", "No info"));
479     }
480     logger.info("Transcription job {} for media package {} failed.{}", transcription.getID(), mpId, message);
481     // delete audio source files in Azure storage
482     try {
483       deleteTranscriptionSourceFiles(mpId, transcription.getID());
484     } catch (TranscriptionServiceException e) {
485       logger.warn("Unable to delete transcription source files for media package {} after transcription kob failure.",
486           mpId, e);
487     }
488     try {
489       database.updateJobControl(transcription.getID(), TranscriptionJobControl.Status.Error.name());
490     } catch (TranscriptionDatabaseException e) {
491       throw new TranscriptionServiceException(String.format(
492           "Transcription job for media package '%s' failed and storing job status in the database failed too."
493           , mpId), e);
494     }
495   }

507   public String createTranscriptionJob(long jobId, String mpId, Track track, String language)
508           throws TranscriptionServiceException {
509     // load media file into workspace
510     File trackFile;
511     try {
512       trackFile = workspace.get(track.getURI());
513     } catch (NotFoundException e) {
514       throw new TranscriptionServiceException(String.format("Track %s not found.", track.getURI()), e);
515     } catch (IOException e) {
516       throw new TranscriptionServiceException(String.format(
517           "Unable to get track %s for transcription.", track.getURI()), e);
518     }
519     // upload media file to azure storage
520     //// assure azure storage container exists
521     try {
522       azureStorageClient.createContainer(azureContainerName);
523     } catch (IOException | MicrosoftAzureStorageClientException | MicrosoftAzureNotAllowedException e) {
524       throw new TranscriptionServiceException(String.format(
525           "Unable to query or create a storage container '%s' on Microsoft Azure.", azureContainerName), e);
526     }
527     //// upload file to azure storage container
528     String azureBlobUrl;
529     try {
530       String filename = String.format("%d-%s.%s", jobId, mpId, FilenameUtils.getExtension(trackFile.getName()));
531       azureBlobUrl = azureStorageClient.uploadFile(trackFile, azureContainerName, azureBlobPath, filename);
532     } catch (IOException | MicrosoftAzureNotAllowedException | MicrosoftAzureStorageClientException e) {
533       throw new TranscriptionServiceException(String.format(
534           "Unable to upload track %s from media package '%s' to Microsoft Azure storage container '%s'.",
535           track.getURI(), mpId, azureContainerName), e);
536     }
537     // start azure transcription job
538     List<String> contentUrls = Arrays.asList(azureBlobUrl);
539     String azureDestContainerUrl = String.format("%s?%s", azureStorageClient.getContainerUrl(azureContainerName),
540         azureAuthorization.generateServiceSasToken("cw", null, null, azureContainerName, "c"));
541     MicrosoftAzureSpeechTranscription transcription;
542     try {
543       transcription = azureSpeechServicesClient.createTranscription(contentUrls,
544           azureDestContainerUrl, String.format("Transcription job %d", jobId), language, autodetectLanguages,
545           null, null);
546       logger.info("Started transcription of {} from media package '{}' on Microsoft Azure Speech Services at {}",
547           track.getURI(), mpId, transcription.self);
548     } catch (MicrosoftAzureNotAllowedException | IOException | MicrosoftAzureSpeechClientException e) {
549       throw new TranscriptionServiceException(String.format(
550           "Unable to create transcription of track %s from media package '%s' "
551               + "in Microsoft Azure storage container '%s'.",
552           track.getURI(), mpId, azureContainerName), e);
553     }
554     // store transcription job ID and status
555     try {
556       database.storeJobControl(mpId, track.getIdentifier(), transcription.getID(),
557           TranscriptionJobControl.Status.InProgress.name(),
558           track.getDuration() == null ? 0 : track.getDuration(), new Date(), PROVIDER);
559     } catch (TranscriptionDatabaseException e) {
560       throw new TranscriptionServiceException(String.format(
561           "Unable to store transcription job of track %s from media package '%s' in the database.",
562           track.getURI(), mpId), e);
563     }
564     // return transcription job ID
565     return transcription.getID();
566   }

568   MicrosoftAzureSpeechTranscriptionJson getTranscriptionJson(String mpId,
569       MicrosoftAzureSpeechTranscription transcription)
570           throws TranscriptionServiceException, MicrosoftAzureNotFoundException {
571     if (!transcription.isSucceeded()) {
572       if (transcription.isRunning()) {
573         throw new TranscriptionServiceException(String.format("Unable to get generated transcription. "
574             + "Transcription job '%s' for media package '%s' is currently running.", transcription.getID(), mpId));
575       } else if (transcription.isFailed()) {
576         throw new TranscriptionServiceException(String.format("Unable to get generated transcription. "
577             + "Transcription job '%s' for media package '%s' is failed.", transcription.getID(), mpId));
578       }
579     }
580     // query transcription files
581     MicrosoftAzureSpeechTranscriptionFiles transcriptionFiles;
582     try {
583       transcriptionFiles = azureSpeechServicesClient.getTranscriptionFilesById(transcription.getID());
584     } catch (IOException | MicrosoftAzureNotAllowedException | MicrosoftAzureSpeechClientException e) {
585       throw new TranscriptionServiceException(String.format(
586           "Unable to get transcription files '%s' for media package '%s'.", transcription.getID(), mpId), e);
587     }
588     // download transcription file to workspace
589     MicrosoftAzureSpeechTranscriptionFile transcriptionFile = null;
590     for (MicrosoftAzureSpeechTranscriptionFile tf : transcriptionFiles.values) {
591       if (tf.isTranscriptionFile()) {
592         transcriptionFile = tf;
593         break;
594       }
595     }
596     if (transcriptionFile == null) {
597       // get more files with transcriptionFiles.nextLink
598       // TODO
599       throw new NotImplementedException("At least one transcription file should be provided.");
600     }
601 
602     try {
603       return MicrosoftAzureSpeechServicesClient
604           .getTranscriptionJson(transcriptionFile);
605     } catch (IOException | MicrosoftAzureNotAllowedException | MicrosoftAzureSpeechClientException
606              | MicrosoftAzureNotFoundException e) {
607       throw new TranscriptionServiceException(String.format(
608           "Unable to download transcription file '%s' for media package '%s'.", transcriptionFile.self, mpId), e);
609     }
610   }

612   String startWorkflow(String mpId, MicrosoftAzureSpeechTranscription transcription)
613           throws TranscriptionDatabaseException, NotFoundException, WorkflowDatabaseException,
614           TranscriptionServiceException, MicrosoftAzureNotFoundException {
615     MicrosoftAzureSpeechTranscriptionJson transcriptionJson = getTranscriptionJson(mpId, transcription);
616     String transcriptionLocale = transcriptionJson.getRecognizedLocale();
617 
618     DefaultOrganization defaultOrg = new DefaultOrganization();
619     securityService.setOrganization(defaultOrg);
620     securityService.setUser(SecurityUtil.createSystemUser(systemAccount, defaultOrg));
621 
622     // Find the episode
623     Optional<Snapshot> snapshot = assetManager.getLatestSnapshot(mpId);
624     if (snapshot.isEmpty()) {
625       // Media package not archived yet? Skip until next time.
626       logger.warn("Media package {} has not been archived yet. Skipped.", mpId);
627       return null;
628     }
629 
630     String org = snapshot.get().getOrganizationId();
631     Organization organization = organizationDirectoryService.getOrganization(org);
632     if (organization == null) {
633       logger.warn("Media package {} has an unknown organization {}. Skipped.", mpId, org);
634       return null;
635     }
636     securityService.setOrganization(organization);
637 
638     // Build workflow
639     Map<String, String> params = new HashMap<>();
640     params.put("transcriptionJobId", transcription.getID());
641     String locale = "";
642     String language = "";
643     if (StringUtils.isNotBlank(transcriptionLocale)) {
644       locale = transcriptionLocale;
645       language = Locale.forLanguageTag(transcriptionLocale).getLanguage();
646     }
647     params.put("transcriptionLocale", locale);
648     params.put("transcriptionLocaleSet", Boolean.toString(!StringUtils.isEmpty(locale)));
649     params.put("transcriptionLocaleSubtypeSuffix", !StringUtils.isEmpty(locale) ? "+" + locale : "");
650     params.put("transcriptionLocaleTag", !StringUtils.isEmpty(locale) ? "lang:" + locale : "");
651     params.put("transcriptionLanguage", language);
652     params.put("transcriptionLanguageSet", Boolean.toString(!StringUtils.isEmpty(language)));
653     params.put("transcriptionLanguageSubtypeSuffix", !StringUtils.isEmpty(language) ? "+" + language : "");
654     params.put("transcriptionLanguageTag", !StringUtils.isEmpty(language) ? "lang:" + language : "");
655     WorkflowDefinition wfDef = workflowService.getWorkflowDefinitionById(workflowDefinitionId);
656 
657     // Apply workflow
658     // wfUtil is only used by unit tests
659     Workflows workflows = wfUtil != null ? wfUtil : new Workflows(assetManager, workflowService);
660     Set<String> mpIds = new HashSet<>();
661     mpIds.add(mpId);
662     List<WorkflowInstance> wfList = workflows
663         .applyWorkflowToLatestVersion(mpIds, ConfiguredWorkflow.workflow(wfDef, params));
664     return wfList.size() > 0 ? Long.toString(wfList.get(0).getId()) : null;
665   }

667   public void deleteTranscription(String mpId, String transcriptionId)
668           throws TranscriptionServiceException, TranscriptionDatabaseException {
669     TranscriptionJobControl transcriptionJobControl = database.findByJob(transcriptionId);
670     TranscriptionJobControl.Status transcriptionJobControlStatus = TranscriptionJobControl.Status.valueOf(
671         transcriptionJobControl.getStatus());
672     if (transcriptionJobControlStatus != TranscriptionJobControl.Status.Closed
673         && transcriptionJobControlStatus != TranscriptionJobControl.Status.Canceled
674         && transcriptionJobControlStatus != TranscriptionJobControl.Status.Error) {
675       throw new TranscriptionServiceException(String.format("Abort deleting transcription %s with invalid status '%s'.",
676           transcriptionId, transcriptionJobControl.getStatus()));
677     }
678     deleteTranscriptionSourceFiles(mpId, transcriptionId);
679     try {
680       azureSpeechServicesClient.deleteTranscription(transcriptionId);
681     } catch (IOException | MicrosoftAzureNotAllowedException | MicrosoftAzureSpeechClientException e) {
682       throw new TranscriptionServiceException(String.format(
683           "Unable to delete transcription '%s' for media package '%s'.", transcriptionId, mpId), e);
684     }
685     database.deleteJobControl(transcriptionJobControl.getTranscriptionJobId());
686   }

688   public void deleteTranscriptionSourceFiles(String mpId, String transcriptionId)
689           throws TranscriptionServiceException {
690     MicrosoftAzureSpeechTranscriptionFiles transcriptionFiles;
691     try {
692       transcriptionFiles = azureSpeechServicesClient.getTranscriptionFilesById(transcriptionId);
693     } catch (IOException | MicrosoftAzureNotAllowedException | MicrosoftAzureSpeechClientException e) {
694       throw new TranscriptionServiceException(String.format(
695           "Unable to get for transcription '%s' from media package '%s'.", transcriptionId, mpId), e);
696     } catch (MicrosoftAzureNotFoundException e) {
697       // catch deleting non-existing file
698       logger.debug("Failed to get non existing transcription files from media package {} for deleting.", mpId, e);
699       return;
700     }
701     for (MicrosoftAzureSpeechTranscriptionFile transcriptionFile : transcriptionFiles.values) {
702       if (!transcriptionFile.isTranscriptionFile()) {
703         continue;
704       }
705       MicrosoftAzureSpeechTranscriptionJson transcriptionJson;
706       try {
707         transcriptionJson = MicrosoftAzureSpeechServicesClient
708             .getTranscriptionJson(transcriptionFile);
709       } catch (IOException | MicrosoftAzureNotAllowedException | MicrosoftAzureSpeechClientException e) {
710         throw new TranscriptionServiceException(String.format(
711             "Unable to download transcription file '%s' for media package '%s'.", transcriptionFile.self, mpId), e);
712       } catch (MicrosoftAzureNotFoundException e) {
713         // catch deleting non-existing file
714         logger.debug("Failed to get non existing transcription file {} from media package {} for deleting.",
715             transcriptionFile.self, mpId, e);
716         continue;
717       }
718       if (StringUtils.isNotBlank(transcriptionJson.source)) {
719         try {
720           azureStorageClient.deleteFile(new URL(transcriptionJson.source));
721         } catch (IOException | MicrosoftAzureNotAllowedException | MicrosoftAzureStorageClientException e) {
722           throw new TranscriptionServiceException(String.format(
723               "Unable to delete audio source file for media package %s.", mpId, e));
724         }
725       }
726     }

729   class WorkflowDispatcher implements Runnable {
730 
731     @Override
732     public void run() {
733       if (!enabled) {
734         logger.debug("Service disabled, cancel processing.");
735         return;
736       }
737       logger.debug("Run jobs handling loop for transcription provider {}.", PROVIDER);
738       long providerId;
739       try {
740         TranscriptionProviderControl providerInfo = database.findIdByProvider(PROVIDER);
741         if (providerInfo != null) {
742           providerId = providerInfo.getId();
743         } else {
744           logger.debug("No jobs yet for provider {}.", PROVIDER);
745           return;
746         }
747         // handle jobs in progress
748         for (TranscriptionJobControl jobControl : database.findByStatus(
749             TranscriptionJobControl.Status.InProgress.name())) {
750           if (providerId != jobControl.getProviderId()) {
751             continue;
752           }
753           String mpId = jobControl.getMediaPackageId();
754           String transcriptionId = jobControl.getTranscriptionJobId();
755           try {
756             MicrosoftAzureSpeechTranscription transcription = azureSpeechServicesClient.getTranscriptionById(
757                 transcriptionId);
758             // check and update job status
759             if (!transcription.isRunning()) {
760               if (transcription.isFailed()) {
761                 transcriptionError(mpId, transcription);
762               } else if (transcription.isSucceeded()) {
763                 transcriptionDone(mpId, transcription);
764               }
765             }
766           } catch (MicrosoftAzureNotAllowedException | IOException | MicrosoftAzureSpeechClientException e) {
767             logger.error("Unable to get or update transcription {} or transcription file from media package {}.",
768                 transcriptionId, mpId, e);
769           } catch (MicrosoftAzureNotFoundException e) {
770             logger.warn("Transcription {} from media package {} not found.", transcriptionId, mpId);
771             database.updateJobControl(transcriptionId, TranscriptionJobControl.Status.Error.name());
772           } catch (TranscriptionServiceException e) {
773             logger.error(e.getMessage(), e);
774           }
775         }
776         // handle completed jobs
777         for (TranscriptionJobControl jobControl : database.findByStatus(
778             TranscriptionJobControl.Status.TranscriptionComplete.name())) {
779           if (providerId != jobControl.getProviderId()) {
780             continue;
781           }
782           String mpId = jobControl.getMediaPackageId();
783           String transcriptionId = jobControl.getTranscriptionJobId();
784           try {
785             MicrosoftAzureSpeechTranscription transcription = azureSpeechServicesClient.getTranscriptionById(
786                 transcriptionId);
787             // start workflow
788             String workflowId = startWorkflow(mpId, transcription);
789             // update db
790             if (workflowId != null) {
791               database.updateJobControl(transcriptionId, TranscriptionJobControl.Status.Closed.name());
792               logger.info("Attach transcription workflow {} scheduled for mp {}, microsoft azure transcription job {}",
793                   workflowId, mpId, transcriptionId);
794             }
795           } catch (MicrosoftAzureNotAllowedException | IOException
796                    | MicrosoftAzureSpeechClientException e) {
797             logger.warn("Unable to get transcription {} or transcription file from media package {}.",
798                 transcriptionId, mpId, e);
799           } catch (MicrosoftAzureNotFoundException e) {
800             logger.warn("Transcription {} from media package {} not found.", transcriptionId, mpId);
801             database.updateJobControl(transcriptionId, TranscriptionJobControl.Status.Error.name());
802           } catch (TranscriptionServiceException e) {
803             logger.warn(e.getMessage(), e);
804           } catch (NotFoundException e) {
805             logger.warn("Unable to load organization.", e);
806           } catch (IllegalStateException e) {
807             logger.debug(e.getMessage());
808           }
809         }
810         // cleanup all old jobs
811         for (TranscriptionJobControl jobControl : database.findByStatus(
812             TranscriptionJobControl.Status.Closed.name(), TranscriptionJobControl.Status.Error.name())) {
813           if (providerId != jobControl.getProviderId()) {
814             continue;
815           }
816           String mpId = jobControl.getMediaPackageId();
817           String transcriptionId = jobControl.getTranscriptionJobId();
818           if (Instant.now().minus(7, ChronoUnit.DAYS).isAfter(jobControl.getDateCreated().toInstant())) {
819             try {
820               deleteTranscription(jobControl.getMediaPackageId(), jobControl.getTranscriptionJobId());
821             } catch (TranscriptionServiceException e) {
822               logger.error("Unable to delete transcription {} or transcription files from media package {}.",
823                   transcriptionId, mpId, e);
824             }
825           }
826         }
827       } catch (TranscriptionDatabaseException e) {
828         logger.warn("Could not read or update transcription job control database", e);
829       } catch (WorkflowDatabaseException e) {
830         logger.warn("Unable to get workflow definition.", e);
831       } catch (Throwable e) {
832         // catch all
833         logger.error("Something went wrong in transcription job processing loop. Exception unhandled!!!", e);
834       }
835     }
836   }
