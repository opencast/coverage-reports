186   @Produces({ MediaType.APPLICATION_JSON })
187   @Path("themes.json")
188   @RestQuery(name = "getThemes", description = "Return all of the known themes on the system", restParameters = {
189           @RestParameter(name = "filter", isRequired = false, description = "The filter used for the query. They should be formated like that: 'filter1:value1,filter2:value2'", type = STRING),
190           @RestParameter(defaultValue = "0", description = "The maximum number of items to return per page.", isRequired = false, name = "limit", type = RestParameter.Type.INTEGER),
191           @RestParameter(defaultValue = "0", description = "The page number.", isRequired = false, name = "offset", type = RestParameter.Type.INTEGER),
192           @RestParameter(name = "sort", isRequired = false, description = "The sort order. May include any of the following: NAME, CREATOR.  Add '_DESC' to reverse the sort order (e.g. CREATOR_DESC).", type = STRING) }, responses = { @RestResponse(description = "A JSON representation of the themes", responseCode = HttpServletResponse.SC_OK) }, returnDescription = "")
193   public Response getThemes(@QueryParam("filter") String filter, @QueryParam("limit") int limit,
194           @QueryParam("offset") int offset, @QueryParam("sort") String sort) {
195     Optional<Integer> optLimit = Optional.ofNullable(limit);
196     Optional<Integer> optOffset = Optional.ofNullable(offset);
197     Optional<String> optSort = Optional.ofNullable(trimToNull(sort));
198 
199     Map<String, String> filters = RestUtils.parseFilter(filter);
200 
201     ArrayList<SortCriterion> sortCriteria;
202     if (optSort.isPresent()) {
203       sortCriteria = RestUtils.parseSortQueryParameter(optSort.get());
204     } else {
205       sortCriteria = new ArrayList<>();
206     }
207     Optional<String> optCreatorFilter = Optional.ofNullable(filters.get(ThemesListQuery.FILTER_CREATOR_NAME));
208     Optional<String> optTextFilter = Optional.ofNullable(filters.get(ThemesListQuery.FILTER_TEXT_NAME));
209 
210     List<Theme> results = null;
211     int total;
212     try {
213       results = themesServiceDatabase.findThemes(
214           optLimit,
215           optOffset,
216           sortCriteria,
217           optCreatorFilter,
218           optTextFilter
219       );
220       total = themesServiceDatabase.countThemes();
221     } catch (Exception e) {
222       logger.error("Could not get themes from the database:", e);
223       return RestUtil.R.serverError();
224     }
225 
226 
227     List<JsonObject> themesJSON = new ArrayList<>();
228     for (Theme theme : results) {
229       themesJSON.add(themeToJSON(theme, false));
230     }
231 
232     return okJsonList(
233         themesJSON,
234         Optional.ofNullable(offset).orElse(0),
235         Optional.ofNullable(limit).orElse(0),
236         total
237     );
238   }

258   @GET
259   @Path("{themeId}/usage.json")
260   @Produces(MediaType.APPLICATION_JSON)
261   @RestQuery(name = "getThemeUsage", description = "Returns the theme usage by the given id as JSON", returnDescription = "The theme usage as JSON", pathParameters = { @RestParameter(name = "themeId", description = "The theme id", isRequired = true, type = RestParameter.Type.INTEGER) }, responses = {
262           @RestResponse(description = "Returns the theme usage as JSON", responseCode = HttpServletResponse.SC_OK),
263           @RestResponse(description = "Theme with the given id does not exist", responseCode = HttpServletResponse.SC_NOT_FOUND) })
264   public Response getThemeUsage(@PathParam("themeId") long themeId) {
265     try {
266       themesServiceDatabase.getTheme(themeId);
267 
268       SeriesSearchQuery query = new SeriesSearchQuery(securityService.getOrganization().getId(),
269               securityService.getUser()).withTheme(themeId);
270       SearchResult<Series> results = null;
271       try {
272         results = searchIndex.getByQuery(query);
273       } catch (SearchIndexException e) {
274         logger.error("The admin UI Search Index was not able to get the series with theme '{}':", themeId,
275                 e);
276         return RestUtil.R.serverError();
277       }
278       JsonArray seriesValues = new JsonArray();
279       for (SearchResultItem<Series> item : results.getItems()) {
280         Series series = item.getSource();
281         JsonObject seriesJson = new JsonObject();
282         seriesJson.addProperty("id", series.getIdentifier());
283         seriesJson.addProperty("title", series.getTitle());
284         seriesValues.add(seriesJson);
285       }
286       JsonObject responseJson = new JsonObject();
287       responseJson.add("series", seriesValues);
288 
289       return okJson(responseJson);
290     } catch (ThemesServiceDatabaseException e) {
291       return RestUtil.R.serverError();
292     } catch (NotFoundException e) {
293       return notFound("Cannot find a theme with id {}", themeId);
294     }
295   }

457   @RestQuery(name = "deleteTheme", description = "Deletes a theme", returnDescription = "The method doesn't return any content", pathParameters = { @RestParameter(name = "themeId", isRequired = true, description = "The theme identifier", type = RestParameter.Type.INTEGER) }, responses = {
458           @RestResponse(responseCode = SC_NOT_FOUND, description = "If the theme has not been found."),
459           @RestResponse(responseCode = SC_NO_CONTENT, description = "The method does not return any content"),
460           @RestResponse(responseCode = SC_UNAUTHORIZED, description = "If the current user is not authorized to perform this action") })
461   public Response deleteTheme(@PathParam("themeId") long themeId) throws NotFoundException, UnauthorizedException {
462     try {
463       Theme theme = themesServiceDatabase.getTheme(themeId);
464       try {
465         deleteReferencedFiles(theme);
466       } catch (IOException e) {
467         logger.warn("Error while deleting referenced file: {}", e.getMessage());
468         return R.serverError();
469       }
470 
471       themesServiceDatabase.deleteTheme(themeId);
472       deleteThemeOnSeries(themeId);
473 
474       return RestUtil.R.noContent();
475     } catch (NotFoundException e) {
476       logger.warn("Unable to find a theme with id " + themeId);
477       throw e;
478     } catch (ThemesServiceDatabaseException e) {
479       logger.error("Error getting theme {} during delete operation because:", themeId,
480               e);
481       return RestUtil.R.serverError();
482     }
483   }

491   private void deleteThemeOnSeries(long themeId) throws UnauthorizedException {
492     SeriesSearchQuery query = new SeriesSearchQuery(securityService.getOrganization().getId(),
493             securityService.getUser()).withTheme(themeId);
494     SearchResult<Series> results = null;
495     try {
496       results = searchIndex.getByQuery(query);
497     } catch (SearchIndexException e) {
498       logger.error("The admin UI Search Index was not able to get the series with theme '{}':", themeId, e);
499       throw new WebApplicationException(e, Status.INTERNAL_SERVER_ERROR);
500     }
501     for (SearchResultItem<Series> item : results.getItems()) {
502       String seriesId = item.getSource().getIdentifier();
503       try {
504         seriesService.deleteSeriesProperty(seriesId, SeriesEndpoint.THEME_KEY);
505       } catch (NotFoundException e) {
506         logger.warn("Theme {} already deleted on series {}", themeId, seriesId);
507       } catch (SeriesException e) {
508         logger.error("Unable to remove theme from series {}", seriesId, e);
509         throw new WebApplicationException(e, Status.INTERNAL_SERVER_ERROR);
510       }
511     }
512   }

514   private void extendStaticFileInfo(String fieldName, String staticFileId, JsonObject json) {
515     if (StringUtils.isNotBlank(staticFileId)) {

517         String fileName = staticFileService.getFileName(staticFileId);

524       }
525     }

528   /**
529    * Returns the JSON representation of this theme.
530    *
531    * @param theme
532    *          the theme
533    * @param editResponse
534    *          whether the returning representation should contain edit information
535    * @return the JSON representation of this theme.
536    */
537   private JsonObject themeToJSON(Theme theme, boolean editResponse) {
538     String creator = StringUtils.isNotBlank(theme.getCreator().getName())
539         ? theme.getCreator().getName()
540         : theme.getCreator().getUsername();
541 
542     JsonObject json = new JsonObject();
543 
544     json.addProperty("id",  theme.getId().getOrElse(-1L));
545     json.addProperty("creationDate", DateTimeSupport.toUTC(theme.getCreationDate().getTime()));
546     json.addProperty("default", theme.isDefault());
547     json.addProperty("name", theme.getName());
548     json.addProperty("creator", creator);
549     json.addProperty("description", safeString(theme.getDescription()));
550     json.addProperty("bumperActive", theme.isBumperActive());
551     json.addProperty("bumperFile", safeString(theme.getBumperFile()));
552     json.addProperty("trailerActive", theme.isTrailerActive());
553     json.addProperty("trailerFile", safeString(theme.getTrailerFile()));
554     json.addProperty("titleSlideActive", theme.isTitleSlideActive());
555     json.addProperty("titleSlideMetadata", safeString(theme.getTitleSlideMetadata()));
556     json.addProperty("titleSlideBackground", safeString(theme.getTitleSlideBackground()));
557     json.addProperty("licenseSlideActive", theme.isLicenseSlideActive());
558     json.addProperty("licenseSlideDescription", safeString(theme.getLicenseSlideDescription()));
559     json.addProperty("licenseSlideBackground", safeString(theme.getLicenseSlideBackground()));
560     json.addProperty("watermarkActive", theme.isWatermarkActive());
561     json.addProperty("watermarkFile", theme.getWatermarkFile() != null ? theme.getWatermarkFile() : "");
562     json.addProperty("watermarkPosition", theme.getWatermarkPosition() != null ? theme.getWatermarkPosition() : "");
563 
564     if (editResponse) {
565       extendStaticFileInfo("bumperFile", theme.getBumperFile(), json);
566       extendStaticFileInfo("trailerFile", theme.getTrailerFile(), json);
567       extendStaticFileInfo("titleSlideBackground", theme.getTitleSlideBackground(), json);
568       extendStaticFileInfo("licenseSlideBackground", theme.getLicenseSlideBackground(), json);
569       extendStaticFileInfo("watermarkFile", theme.getWatermarkFile(), json);
570     }
571 
572     return json;
573   }

627    *          The updated theme

635     updateReferencedFile(original.getLicenseSlideBackground(), updated.getLicenseSlideBackground());
636     updateReferencedFile(original.getTitleSlideBackground(), updated.getTitleSlideBackground());

638     updateReferencedFile(original.getWatermarkFile(), updated.getWatermarkFile());

641   /**
642    * If the file resource has changed between {@code original} and {@code updated}, the original file is deleted and the
643    * updated one persisted.
644    *
645    * @param original
646    *          The UUID of the original file
647    * @param updated
648    *          The UUID of the updated file
649    * @throws NotFoundException
650    *           If the file could not be found
651    * @throws IOException
652    *           If there was an error while persisting or deleting one of the files.
653    */
654   private void updateReferencedFile(String original, String updated) throws NotFoundException, IOException {
655     if (EqualsUtil.ne(original, updated)) {
656       if (isNotBlank(original))
657         staticFileService.deleteFile(original);
658       if (isNotBlank(updated))
659         staticFileService.persistFile(updated);
660     }
661   }
