274   /** OSGi callback if properties file is present */
275   @Modified
276   public void modified(Map<String, Object> properties) {
277     if (properties == null) {
278       logger.info("No configuration available, using defaults");
279       return;
280     }
281 
282     Object mapValue = properties.get(SERIES_HASEVENTS_DELETE_ALLOW_KEY);
283     if (mapValue != null) {
284       deleteSeriesWithEventsAllowed = BooleanUtils.toBoolean(mapValue.toString());
285     }
286 
287     mapValue = properties.get(SERIESTAB_ONLYSERIESWITHWRITEACCESS_KEY);
288     onlySeriesWithWriteAccessSeriesTab = BooleanUtils.toBoolean(Objects.toString(mapValue, "true"));
289 
290     mapValue = properties.get(EVENTSFILTER_ONLYSERIESWITHWRITEACCESS_KEY);
291     onlySeriesWithWriteAccessEventsFilter = BooleanUtils.toBoolean(Objects.toString(mapValue, "true"));
292 
293     properties.forEach((key, value) -> {
294       var matches = TOBIRA_CONFIG.matcher(key);
295       if (!matches.matches()) {
296         return;
297       }
298       var tobira = TobiraService.getTobira(matches.group("organization"));
299       switch (matches.group("key")) {
300         case "origin":
301           tobira.setOrigin((String) value);
302           break;
303         case "trustedKey":
304           tobira.setTrustedKey((String) value);
305           break;
306         default:
307           throw new RuntimeException("unhandled Tobira config key");
308       }
309     });
310 
311     logger.info("Configuration updated");
312   }

314   @GET
315   @Path("{seriesId}/access.json")
316   @SuppressWarnings("unchecked")
317   @Produces(MediaType.APPLICATION_JSON)
318   @RestQuery(name = "getseriesaccessinformation", description = "Get the access information of a series", returnDescription = "The access information", pathParameters = { @RestParameter(name = "seriesId", isRequired = true, description = "The series identifier", type = Type.STRING) }, responses = {
319           @RestResponse(responseCode = SC_BAD_REQUEST, description = "The required form params were missing in the request."),
320           @RestResponse(responseCode = SC_NOT_FOUND, description = "If the series has not been found."),
321           @RestResponse(responseCode = SC_OK, description = "The access information ") })
322   public Response getSeriesAccessInformation(@PathParam("seriesId") String seriesId) throws NotFoundException {
323     if (StringUtils.isBlank(seriesId))
324       return RestUtil.R.badRequest("Path parameter series ID is missing");
325 
326     boolean hasProcessingEvents = hasProcessingEvents(seriesId);
327 
328     // Add all available ACLs to the response
329     JSONArray systemAclsJson = new JSONArray();
330     List<ManagedAcl> acls = getAclService().getAcls();
331     for (ManagedAcl acl : acls) {
332       systemAclsJson.add(AccessInformationUtil.serializeManagedAcl(acl));
333     }
334 
335     JSONObject seriesAccessJson = new JSONObject();
336     try {
337       AccessControlList seriesAccessControl = seriesService.getSeriesAccessControl(seriesId);
338       Option<ManagedAcl> currentAcl = AccessInformationUtil.matchAclsLenient(acls, seriesAccessControl,
339               adminUIConfiguration.getMatchManagedAclRolePrefixes());
340       seriesAccessJson.put("current_acl", currentAcl.isSome() ? currentAcl.get().getId() : 0);
341       seriesAccessJson.put("privileges", AccessInformationUtil.serializePrivilegesByRole(seriesAccessControl));
342       seriesAccessJson.put("acl", transformAccessControList(seriesAccessControl, userDirectoryService));
343       seriesAccessJson.put("locked", hasProcessingEvents);
344     } catch (SeriesException e) {
345       logger.error("Unable to get ACL from series {}", seriesId, e);
346       return RestUtil.R.serverError();
347     }
348 
349     JSONObject jsonReturnObj = new JSONObject();
350     jsonReturnObj.put("system_acls", systemAclsJson);
351     jsonReturnObj.put("series_access", seriesAccessJson);
352 
353     return Response.ok(jsonReturnObj.toString()).build();
354   }

359   @RestQuery(name = "getseriesmetadata", description = "Returns the series metadata as JSON", returnDescription = "Returns the series metadata as JSON", pathParameters = { @RestParameter(name = "seriesId", isRequired = true, description = "The series identifier", type = STRING) }, responses = {
360           @RestResponse(responseCode = SC_OK, description = "The series metadata as JSON."),
361           @RestResponse(responseCode = SC_NOT_FOUND, description = "The series has not been found"),
362           @RestResponse(responseCode = SC_UNAUTHORIZED, description = "If the current user is not authorized to perform this action") })
363   public Response getSeriesMetadata(@PathParam("seriesId") String series) throws UnauthorizedException,
364           NotFoundException, SearchIndexException {
365     Optional<Series> optSeries = searchIndex.getSeries(series, securityService.getOrganization().getId(), securityService.getUser());
366     if (optSeries.isEmpty())
367       return notFound("Cannot find a series with id '%s'.", series);
368 
369     MetadataList metadataList = new MetadataList();
370     List<SeriesCatalogUIAdapter> catalogUIAdapters = indexService.getSeriesCatalogUIAdapters();
371     catalogUIAdapters.remove(indexService.getCommonSeriesCatalogUIAdapter());
372     for (SeriesCatalogUIAdapter adapter : catalogUIAdapters) {
373       final Optional<DublinCoreMetadataCollection> optSeriesMetadata = adapter.getFields(series);
374       if (optSeriesMetadata.isPresent()) {
375         metadataList.add(adapter.getFlavor().toString(), adapter.getUITitle(), optSeriesMetadata.get());
376       }
377     }
378     metadataList.add(indexService.getCommonSeriesCatalogUIAdapter(), getSeriesMetadata(optSeries.get()));
379     return okJson(MetadataJson.listToJson(metadataList, true));
380   }

399     metadata.removeField(subject);

464     metadata.addField(createdBy);
465

632   private boolean mountSeriesInTobira(String seriesId, JSONObject params) {
633     var tobira = getTobira();
634     if (!tobira.ready()) {
635       return false;
636     }
637 
638     var tobiraParams = params.get("tobira");
639     if (tobiraParams == null) {
640       return false;
641     }
642     if (!(tobiraParams instanceof JSONObject)) {
643       return false;
644     }
645     var tobiraParamsObject = (JSONObject) tobiraParams;
646 
647     var metadataCatalogs = (JSONArray) params.get("metadata");
648     var firstCatalog = (JSONObject) metadataCatalogs.get(0);
649     var metadataFields = (List<JSONObject>) firstCatalog.get("fields");
650     var title = metadataFields.stream()
651             .filter(field -> field.get("id").equals("title"))
652             .findAny()
653             .map(field -> field.get("value"))
654             .map(String.class::cast)
655             .get();
656 
657     var series = new JSONObject(Map.of(
658             "opencastId", seriesId,
659             "title", title));
660     tobiraParamsObject.put("series", series);
661 
662     try {
663       tobira.mount(tobiraParamsObject);
664     } catch (TobiraException e) {
665       return false;
666     }
667 
668     return true;
669   }

1290       accessControlList = AccessControlParser.parseAcl(acl);
