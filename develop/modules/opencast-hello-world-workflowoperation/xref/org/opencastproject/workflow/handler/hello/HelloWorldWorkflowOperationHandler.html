33  import org.osgi.service.component.annotations.Component;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  import java.util.Objects;
38  
39  /**
40   * The <code>HelloWorldWorkflowOperationHandler</code> provides a very simple example of how a workflow operation works
41   * and can be a starting point for new developments.
42   *
43   * Like the other hello-world modules, this is intentionally not included in the Opencast distributions and thus not
44   * listed in the documentation since people cannot actually use it.
45   */
46  @Component(
47      property = {
48          "service.description=Hello-World Workflow Operation Handler",
49          "workflow.operation=hello-world"
50      },
51      immediate = true,
52      service = WorkflowOperationHandler.class
53  )
54  public class HelloWorldWorkflowOperationHandler extends AbstractWorkflowOperationHandler {
55  
56    /** The logging facility */
57    private static final Logger logger = LoggerFactory.getLogger(HelloWorldWorkflowOperationHandler.class);
58  
59    @Override
60    public WorkflowOperationResult start(WorkflowInstance workflowInstance, JobContext context) {
61  
62      // Get reference to current operation to get configuration, â€¦
63      final WorkflowOperationInstance operation = workflowInstance.getCurrentOperation();
64  
65      // Read configuration key `message`, falling back to "Hello world!".
66      // Configuration for this operation could look like this:
67      //
68      //   - id: hello-world
69      //     configurations:
70      //       - message: A configured message
71      String message = Objects.toString(
72          operation.getConfiguration("message"),
73          "Hello World!");
74      logger.info("Message: {}", message);
75  
76  
77      // Most of the time, you would do something with the media package and its content
78      // We could also modify the media package, add new content or remove content
79      final MediaPackage mediaPackage = workflowInstance.getMediaPackage();
80      logger.info("The media package contains {} tracks", mediaPackage.getTracks().length);
81  
82      // Continue the workflow, passing the possibly modified media package to the next operation
83      return createResult(mediaPackage, Action.CONTINUE);
84    }
85  
86  }
