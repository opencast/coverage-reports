22  package org.opencastproject.engage.ui;
23  
24  import static org.junit.jupiter.api.Assertions.assertEquals;
25  
26  import org.opencastproject.security.api.Organization;
27  import org.opencastproject.security.api.SecurityService;
28  
29  import org.easymock.EasyMock;
30  import org.easymock.EasyMockSupport;
31  import org.junit.jupiter.api.BeforeEach;
32  import org.junit.jupiter.api.Test;
33  
34  import java.util.HashMap;
35  import java.util.Map;
36  
37  import javax.ws.rs.core.Response;
38  
39  public class PlayerRedirectTest extends EasyMockSupport {
40  
41    private SecurityService securityService;
42    private Organization organization;
43    private PlayerRedirect playerRedirect;
44  
45    @BeforeEach
46    public void setup() {
47      securityService = createMock(SecurityService.class);
48      organization = createMock(Organization.class);
49      playerRedirect = new PlayerRedirect();
50      playerRedirect.setSecurityService(securityService);
51    }
52  
53    @Test
54    public void shouldRedirectToDefaultPlayerWhenNoPlayerIsConfigured() {
55      String eventId = "event1";
56      EasyMock.expect(securityService.getOrganization()).andReturn(organization);
57      EasyMock.expect(organization.getProperties()).andReturn(new HashMap<>());
58      replayAll();
59  
60      try (Response response = playerRedirect.redirect(eventId)) {
61        assertEquals(Response.Status.TEMPORARY_REDIRECT.getStatusCode(), response.getStatus());
62        assertEquals("/paella7/ui/watch.html?id=event1", response.getHeaderString("location"));
63      }
64  
65      verifyAll();
66    }
67  
68    @Test
69    public void shouldRedirectToConfiguredPlayer() {
70      String eventId = "event2";
71      Map<String, String> properties = new HashMap<>();
72      properties.put("player", "/custom/ui/watch.html?id=#{id}");
73      EasyMock.expect(securityService.getOrganization()).andReturn(organization);
74      EasyMock.expect(organization.getProperties()).andReturn(properties);
75      replayAll();
76  
77      try (Response response = playerRedirect.redirect(eventId)) {
78        assertEquals(Response.Status.TEMPORARY_REDIRECT.getStatusCode(), response.getStatus());
79        assertEquals("/custom/ui/watch.html?id=event2", response.getHeaderString("location"));
80      }
81  
82      verifyAll();
83    }
84  
85    @Test
86    public void shouldHandleSpecialCharactersInEventId() {
87      String eventId = "event%20with%20spaces";
88      EasyMock.expect(securityService.getOrganization()).andReturn(organization);
89      EasyMock.expect(organization.getProperties()).andReturn(new HashMap<>());
90      replayAll();
91  
92      try (Response response = playerRedirect.redirect(eventId)) {
93        assertEquals(Response.Status.TEMPORARY_REDIRECT.getStatusCode(), response.getStatus());
94        assertEquals("/paella7/ui/watch.html?id=event%2520with%2520spaces", response.getHeaderString("location"));
95      }
96  
97      verifyAll();
98    }
99  
100   @Test
101   public void shouldHandleUTF8EventId() {
102     String eventId = "üäöß$%&/()=?!§";
103     EasyMock.expect(securityService.getOrganization()).andReturn(organization);
104     EasyMock.expect(organization.getProperties()).andReturn(new HashMap<>());
105     replayAll();
106 
107     try (Response response = playerRedirect.redirect(eventId)) {
108       assertEquals(Response.Status.TEMPORARY_REDIRECT.getStatusCode(), response.getStatus());
109       assertEquals("/paella7/ui/watch.html?id=%C3%BC%C3%A4%C3%B6%C3%9F%24%25%26%2F%28%29%3D%3F%21%C2%A7",
110           response.getHeaderString("location"));
111     }
112 
113     verifyAll();
114   }
115 }
