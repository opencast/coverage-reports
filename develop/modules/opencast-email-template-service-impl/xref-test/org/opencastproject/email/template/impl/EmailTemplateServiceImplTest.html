65    @Before
66    public void setUp() throws Exception {
67      MediaPackageBuilder builder = MediaPackageBuilderFactory.newInstance().newMediaPackageBuilder();
68  
69      uriMP = EmailTemplateServiceImplTest.class.getResource("/email_mediapackage.xml").toURI();
70      mp = builder.loadFromXml(uriMP.toURL().openStream());
71  
72      service = new EmailTemplateServiceImpl();
73  
74      URI episodeURI = EmailTemplateServiceImplTest.class.getResource("/episode_dublincore.xml").toURI();
75      URI seriesURI = EmailTemplateServiceImplTest.class.getResource("/series_dublincore.xml").toURI();
76      Workspace workspace = EasyMock.createMock(Workspace.class);
77      EasyMock.expect(workspace.read(new URI("episode_dublincore.xml")))
78          .andReturn(new FileInputStream(new File(episodeURI)));
79      EasyMock.expect(workspace.read(new URI("series_dublincore.xml")))
80          .andReturn(new FileInputStream(new File(seriesURI)));
81      EasyMock.replay(workspace);
82      service.setWorkspace(workspace);
83  
84      EmailTemplateScanner templateScanner = EasyMock.createMock(EmailTemplateScanner.class);
85      EasyMock.expect(templateScanner.getTemplate("templateBasic")).andReturn(
86              "Media package id: ${mediaPackage.identifier}, workflow id: ${workflow.id}, "
87                      + "title: ${mediaPackage.title}, series title: ${mediaPackage.seriesTitle}, "
88                      + "date: ${mediaPackage.date?datetime?iso_utc}");
89      EasyMock.expect(templateScanner.getTemplate("templateCatalog"))
90          .andReturn(
91              "EPISODE creator: ${catalogs[\"dublincore/episode\"][\"creator\"]}, "
92                  + "isPartOf: ${catalogs[\"dublincore/episode\"][\"isPartOf\"]}, "
93                  + "title: ${catalogs[\"dublincore/episode\"][\"title\"]}, "
94                  + "created: ${catalogs[\"dublincore/episode\"][\"created\"]}, "
95                  + "SERIES creator: ${catalogs[\"dublincore/series\"][\"creator\"]}, "
96                  + "description: ${catalogs[\"dublincore/series\"][\"description\"]}, "
97                  + "subject: ${catalogs[\"dublincore/series\"][\"subject\"]}");
98      EasyMock.expect(templateScanner.getTemplate("templateOrganization")).andReturn(
99              "Engage URL is: ${organization[\"org.opencastproject.engage.ui.url\"]}");
100     EasyMock.expect(templateScanner.getTemplate("templateFailed")).andReturn(
101             "<#if failedOperation?has_content>Workflow failed in operation: ${failedOperation.template}</#if>, "
102                     + "Workflow errors: <#list incident as inc><#list inc.details as de>${de.b} </#list></#list>");
103     EasyMock.expect(templateScanner.getTemplate("templateSyntaxError")).andReturn("${mediaPackage");
104     EasyMock.replay(templateScanner);
105     service.setEmailTemplateScanner(templateScanner);
106 
107     WorkflowDefinitionImpl def = new WorkflowDefinitionImpl();
108     def.setId("wfdef");
109     Map<String, String> props = new HashMap<String, String>();
110     props.put("emailAddress", "user@domain.com");
111 
112     IncidentService is = EasyMock.createMock(IncidentService.class);
113 
114     /*
115      * This is what we're building. EasyMock makes it ugly, of course /-------job1Tree / | / incident2 subtree |
116      * incident1
117      */
118 
119     // Create the tree and subtree
120     IncidentTree subtree = EasyMock.createNiceMock(IncidentTree.class);
121     IncidentTree job1Tree = EasyMock.createNiceMock(IncidentTree.class);
122 
123     // Create the incidents
124     Incident incident1 = EasyMock.createNiceMock(Incident.class);
125     List<Tuple<String, String>> details = new LinkedList<Tuple<String, String>>();
126     Tuple<String, String> detail = new Tuple<String, String>("detail-type", "error in operation1");
127     details.add(detail);
128     EasyMock.expect(incident1.getDetails()).andReturn(details);
129 
130     Incident incident2 = EasyMock.createNiceMock(Incident.class);
131     details = new LinkedList<Tuple<String, String>>();
132     detail = new Tuple<String, String>("detail-type", "error in operation2");
133     details.add(detail);
134     EasyMock.expect(incident2.getDetails()).andReturn(details);
135 
136     // Link the incident and the subtree
137     LinkedList<Incident> incidents = new LinkedList<Incident>();
138     incidents.add(incident1);
139     EasyMock.expect(subtree.getIncidents()).andReturn(incidents).anyTimes();
140     // This is what Entwine wrote in the IncidentTree class, so rather than null we're using Immutables.nil()
141     List<IncidentTree> subtreeDecendants = Collections.emptyList();
142     EasyMock.expect(subtree.getDescendants()).andReturn(subtreeDecendants).anyTimes();
143 
144     // Link the incident and the parent tree
145     incidents = new LinkedList<Incident>();
146     incidents.add(incident2);
147     EasyMock.expect(job1Tree.getIncidents()).andReturn(incidents).anyTimes();
148 
149     // Link the subtree and parent tree
150     LinkedList<IncidentTree> subtreeList = new LinkedList<IncidentTree>();
151     subtreeList.add(subtree);
152     EasyMock.expect(job1Tree.getDescendants()).andReturn(subtreeList).anyTimes();
153 
154     service.setIncidentService(is);
155 
156     IndexService indexService = EasyMock.createNiceMock(IndexService.class);
157     EventCatalogUIAdapter commonEventCatalogUIAdapter = EasyMock.createNiceMock(EventCatalogUIAdapter.class);
158     EasyMock.expect(commonEventCatalogUIAdapter.getFlavor())
159         .andReturn(new MediaPackageElementFlavor("dublincore", "episode"))
160         .anyTimes();
161     EasyMock.expect(indexService.getCommonEventCatalogUIAdapter())
162         .andReturn(commonEventCatalogUIAdapter)
163         .anyTimes();
164     EasyMock.expect(indexService.getEventCatalogUIAdapters())
165         .andReturn(List.of(commonEventCatalogUIAdapter))
166         .anyTimes();
167     SeriesCatalogUIAdapter commonSeriesCatalogUIAdapter = EasyMock.createNiceMock(SeriesCatalogUIAdapter.class);
168     EasyMock.expect(commonSeriesCatalogUIAdapter.getFlavor())
169         .andReturn(new MediaPackageElementFlavor("dublincore", "series"))
170         .anyTimes();
171     EasyMock.expect(indexService.getCommonSeriesCatalogUIAdapter())
172         .andReturn(commonSeriesCatalogUIAdapter)
173         .anyTimes();
174     EasyMock.expect(indexService.getSeriesCatalogUIAdapters())
175         .andReturn(List.of(commonSeriesCatalogUIAdapter))
176         .anyTimes();
177     service.setIndexService(indexService);
178 
179     SecurityService securityService = EasyMock.createNiceMock(SecurityService.class);
180     Organization organization = EasyMock.createNiceMock(Organization.class);
181     Map<String, String> orgProperties = new HashMap<>();
182     orgProperties.put("org.opencastproject.engage.ui.url", "http://engage.my.edu");
183     EasyMock.expect(securityService.getOrganization()).andReturn(organization).anyTimes();
184     EasyMock.expect(organization.getProperties()).andReturn(orgProperties).anyTimes();
185     service.setSecurityService(securityService);
186 
187     workflowInstance = new WorkflowInstance(def, null, null, null, props);
188     workflowInstance.setId(1);
189     workflowInstance.setState(WorkflowState.RUNNING);
190     workflowInstance.setMediaPackage(mp);
191 
192     WorkflowOperationInstance failedOperation1 = new WorkflowOperationInstance("operation1",
193             OperationState.FAILED);
194     failedOperation1.setFailOnError(true);
195     failedOperation1.setId(1L);
196     EasyMock.expect(is.getIncidentsOfJob(1L, true)).andReturn(subtree).anyTimes();
197     WorkflowOperationInstance failedOperation2 = new WorkflowOperationInstance("operation2",
198             OperationState.FAILED);
199     failedOperation2.setFailOnError(false);
200     failedOperation1.setId(2L);
201     EasyMock.expect(is.getIncidentsOfJob(2L, true)).andReturn(job1Tree).anyTimes();
202     WorkflowOperationInstance operation = new WorkflowOperationInstance("email", OperationState.RUNNING);
203     List<WorkflowOperationInstance> operationList = new ArrayList<WorkflowOperationInstance>();
204     operationList.add(failedOperation1);
205     operationList.add(failedOperation2);
206     operationList.add(operation);
207     workflowInstance.setOperations(operationList);
208 
209     EasyMock.replay(is, subtree, job1Tree, incident1, incident2, indexService, commonEventCatalogUIAdapter,
210         commonSeriesCatalogUIAdapter, securityService, organization);
