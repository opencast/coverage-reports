22  package org.opencastproject.kernel.security;
23  
24  import org.opencastproject.security.api.JaxbOrganization;
25  import org.opencastproject.security.api.JaxbOrganizationList;
26  import org.opencastproject.security.api.Organization;
27  import org.opencastproject.security.api.OrganizationDirectoryService;
28  import org.opencastproject.util.NotFoundException;
29  import org.opencastproject.util.doc.rest.RestParameter;
30  import org.opencastproject.util.doc.rest.RestParameter.Type;
31  import org.opencastproject.util.doc.rest.RestQuery;
32  import org.opencastproject.util.doc.rest.RestResponse;
33  import org.opencastproject.util.doc.rest.RestService;
34  
35  import org.osgi.service.component.annotations.Component;
36  import org.osgi.service.component.annotations.Reference;
37  import org.osgi.service.jaxrs.whiteboard.propertytypes.JaxrsResource;
38  
39  import javax.ws.rs.GET;
40  import javax.ws.rs.Path;
41  import javax.ws.rs.PathParam;
42  import javax.ws.rs.Produces;
43  import javax.ws.rs.core.MediaType;
44  
45  /**
46   * Provides access to the organizations served by this Opencast instance.
47   */
48  @Component(
49    property = {
50      "service.description=Organization listing REST endpoint",
51      "opencast.service.type=org.opencastproject.organization",
52      "opencast.service.path=/org",
53      "opencast.service.jobproducer=false"
54    },
55    immediate = true,
56    service = { OrganizationEndpoint.class }
57  )
58  @Path("/org")
59  @RestService(name = "organization", title = "Organizations", notes = { "" }, abstractText = "Displays the organizations served by this system")
60  @JaxrsResource
61  public class OrganizationEndpoint {
62  
63    /** The organization directory */
64    protected OrganizationDirectoryService orgDirectoryService = null;
65  
66    @GET
67    @Path("all.xml")
68    @Produces(MediaType.TEXT_XML)
69    @RestQuery(name = "orgsasxml", description = "Lists the organizations as xml", returnDescription = "The list of org as xml", responses = { @RestResponse(responseCode = 200, description = "Organizations returned") })
70    public JaxbOrganizationList getOrganizationsAsXml() {
71      JaxbOrganizationList organizationList = new JaxbOrganizationList();
72      for (Organization org : orgDirectoryService.getOrganizations()) {
73        organizationList.add(org);
74      }
75      return organizationList;
76    }
77  
78    @GET
79    @Path("all.json")
80    @Produces(MediaType.APPLICATION_JSON)
81    @RestQuery(name = "orgsasjson", description = "Lists the organizations as a json array", returnDescription = "The list of org as a json array", responses = { @RestResponse(responseCode = 200, description = "Organizations returned") })
82    public JaxbOrganizationList getOrganizationsAsJson() {
83      return getOrganizationsAsXml();
84    }
85  
86    @GET
87    @Path("{id}.xml")
88    @Produces(MediaType.TEXT_XML)
89    @RestQuery(name = "orgasxml", description = "Gets an organizations as xml", returnDescription = "The org as xml", pathParameters = { @RestParameter(name = "id", type = Type.STRING, description = "The job identifier", isRequired = true) }, responses = {
90            @RestResponse(responseCode = 200, description = "Organization returned"),
91            @RestResponse(responseCode = 404, description = "No organization with this identifier found") })
92    public JaxbOrganization getOrganizationAsXml(@PathParam("id") String id) {
93      try {
94        return JaxbOrganization.fromOrganization(orgDirectoryService.getOrganization(id));
95      } catch (NotFoundException e) {
96        return null;
97      }
98    }
99  
100   @GET
101   @Path("{id}.json")
102   @Produces(MediaType.APPLICATION_JSON)
103   @RestQuery(name = "orgasjson", description = "Gets an organizations as json", returnDescription = "The org as json", pathParameters = { @RestParameter(name = "id", type = Type.STRING, description = "The job identifier", isRequired = true) }, responses = {
104           @RestResponse(responseCode = 200, description = "Organization returned"),
105           @RestResponse(responseCode = 404, description = "No organization with this identifier found") })
106   public JaxbOrganization getOrganizationAsJson(@PathParam("id") String id) {
107     return getOrganizationAsXml(id);
108   }
109 
110   /**
111    * @param orgDirectoryService
112    *          the orgDirectoryService to set
113    */
114   @Reference
115   public void setOrgDirectoryService(OrganizationDirectoryService orgDirectoryService) {
116     this.orgDirectoryService = orgDirectoryService;
117   }
118 }
