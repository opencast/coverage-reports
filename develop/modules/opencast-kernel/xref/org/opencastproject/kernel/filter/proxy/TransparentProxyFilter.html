34  import java.io.IOException;

36  import javax.servlet.Filter;
37  import javax.servlet.FilterChain;
38  import javax.servlet.FilterConfig;
39  import javax.servlet.ServletException;
40  import javax.servlet.ServletRequest;
41  import javax.servlet.ServletResponse;
42  import javax.servlet.http.HttpServletRequest;
43  
44  /**
45   * This filter is looking for <code>X-FORWARDED-FOR</code> headers in the HTTP request and if found sets it as the
46   * original IP.
47   */
48  @Component(
49      service = Filter.class,
50      scope = ServiceScope.PROTOTYPE,
51      property = {
52          "service.description=Transparent Proxy Filter",
53      }
54  )
55  @ServiceRanking(6)
56  @HttpWhiteboardFilterName("TransparentProxyFilter")
57  @HttpWhiteboardFilterPattern("/*")
58  @HttpWhiteboardContextSelect("(osgi.http.whiteboard.context.name=opencast)")
59  public class TransparentProxyFilter implements Filter {
60    /** The logger */
61    private static final Logger logger = LoggerFactory.getLogger(TransparentProxyFilter.class);
62  
63    /** Request header that is set when behind a proxy */
64    public static final String X_FORWARDED_FOR = "X-Forwarded-For";
65  
66    /**
67     * @see javax.servlet.Filter#doFilter(javax.servlet.ServletRequest, javax.servlet.ServletResponse,
68     *      javax.servlet.FilterChain)
69     */
70    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException,
71            ServletException {
72      HttpServletRequest httpReqquest = (HttpServletRequest) request;
73  
74      // Check if the X-Forwarded-For header is set
75      if (StringUtils.isNotBlank(httpReqquest.getHeader(X_FORWARDED_FOR))) {
76        logger.debug("Found X-Forwarded-For header. Resetting source IP");
77        httpReqquest = new TransparentProxyRequestWrapper(httpReqquest);
78      }
79  
80      chain.doFilter(httpReqquest, response);
81    }
82  
83    public void init(FilterConfig filterConfig) throws ServletException {
84    }
85  
86    public void destroy() {
87    }
88  
89  }
